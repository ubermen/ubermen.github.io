138  cluster_list = self._get_cluster_list_for_project(service)

144  def _get_cluster_state(self, service):

151  def _cluster_ready(self, state, service):
152  if state == 'RUNNING':
153  return True
154  if state == 'ERROR':
155  cluster = self._get_cluster(service)
156  try:
157  error_details = cluster['status']['details']
158  except KeyError:
159  error_details = 'Unknown error in cluster creation, ' \
160  'check Google Cloud console for details.'
161  raise Exception(error_details)
162  return False

164  def _wait_for_done(self, service):
165  while True:
166  state = self._get_cluster_state(service)
167  if state is None:
168  logging.info("No state for cluster '%s'", self.cluster_name)
169  time.sleep(15)
170  else:
171  logging.info("State for cluster '%s' is %s", self.cluster_name, state)
172  if self._cluster_ready(state, service):
173  logging.info("Cluster '%s' successfully created",
174  self.cluster_name)
175  return

179  hook = DataProcHook(
180  gcp_conn_id=self.google_cloud_conn_id,
181  delegate_to=self.delegate_to
182  )
183  service = hook.get_conn()
184 
185  if self._get_cluster(service):

188  ))

197  "https://www.googleapis.com/compute/v1/projects/{}/zones/{}/machineTypes/{}".format(

199  )

203  )

205  'projectId': self.project_id,
206  'clusterName': self.cluster_name,

208  'gceClusterConfig': {
209  'zoneUri': zone_uri
210  },
211  'masterConfig': {
212  'numInstances': 1,
213  'machineTypeUri': master_type_uri,
214  'diskConfig': {
215  'bootDiskSizeGb': self.master_disk_size

217  },

221  'diskConfig': {
222  'bootDiskSizeGb': self.worker_disk_size

224  },

227  }
228  }

233  'diskConfig': {

235  },
236  'isPreemptible': True
237  }

239  cluster_data['labels'] = self.labels
240  if self.storage_bucket:

244  if self.properties:
245  cluster_data['config']['softwareConfig']['properties'] = self.properties
246  if self.init_actions_uris:

249  ]

252  try:
253  service.projects().regions().clusters().create(
254  projectId=self.project_id,
255  region=self.region,

257  ).execute()
258  except HttpError as e:

260  time.sleep(10)
261  if self._get_cluster(service):
262  logging.info('Cluster {} already exists... Checking status...'.format(
263  self.cluster_name
264  ))
265  self._wait_for_done(service)
266  return True

268  raise e

345 class DataProcPigOperator(BaseOperator):
346  """
347  Start a Pig query Job on a Cloud DataProc cluster. The parameters of the operation
348  will be passed to the cluster.
349 
350  It's a good practice to define dataproc_* parameters in the default_args of the dag
351  like the cluster name and UDFs.
352 
353  ```
354  default_args = {
355  'dataproc_cluster': 'cluster-1',
356  'dataproc_pig_jars': [
357  'gs://example/udf/jar/datafu/1.2.0/datafu.jar',
358  'gs://example/udf/jar/gpig/1.2/gpig.jar'
359  ]
360  }
361  ```
362 
363  You can pass a pig script as string or file reference. Use variables to pass on
364  variables for the pig script to be resolved on the cluster or use the parameters to
365  be resolved in the script as template parameters.
366 
367  ```
368  t1 = DataProcPigOperator(
369  task_id='dataproc_pig',
370  query='a_pig_script.pig',
371  variables={'out': 'gs://example/output/{{ds}}'},
372  dag=dag)
373  ```
374  """
375  template_fields = ['query', 'variables', 'job_name', 'dataproc_cluster']
376  template_ext = ('.pg', '.pig',)
377  ui_color = '#0273d4'
378 
379  @apply_defaults
380  def __init__(
381  self,
382  query=None,
383  query_uri=None,
384  variables=None,
385  job_name='{{task.task_id}}_{{ds_nodash}}',
386  dataproc_cluster='cluster-1',
387  dataproc_pig_properties=None,
388  dataproc_pig_jars=None,
389  gcp_conn_id='google_cloud_default',
390  delegate_to=None,
391  *args,
392  **kwargs):
393  """
394  Create a new DataProcPigOperator.
395 
396  For more detail on about job submission have a look at the reference:
397 
398  https://cloud.google.com/dataproc/reference/rest/v1/projects.regions.jobs
399 
400  :param query: The query or reference to the query file (pg or pig extension).
401  :type query: string
402  :param query_uri: The uri of a pig script on Cloud Storage.
403  :type query_uri: string
404  :param variables: Map of named parameters for the query.
405  :type variables: dict
406  :param job_name: The job name used in the DataProc cluster. This name by default
407  is the task_id appended with the execution data, but can be templated. The
408  name will always be appended with a random number to avoid name clashes.
409  :type job_name: string
410  :param dataproc_cluster: The id of the DataProc cluster.
411  :type dataproc_cluster: string
412  :param dataproc_pig_properties: Map for the Pig properties. Ideal to put in
413  default arguments
414  :type dataproc_pig_properties: dict
415  :param dataproc_pig_jars: URIs to jars provisioned in Cloud Storage (example: for
416  UDFs and libs) and are ideal to put in default arguments.
417  :type dataproc_pig_jars: list
418  :param gcp_conn_id: The connection ID to use connecting to Google Cloud Platform.
419  :type gcp_conn_id: string
420  :param delegate_to: The account to impersonate, if any.
421  For this to work, the service account making the request must have domain-wide
422  delegation enabled.
423  :type delegate_to: string
424  """

427  self.delegate_to = delegate_to
428  self.query = query
429  self.query_uri = query_uri
430  self.variables = variables
431  self.job_name = job_name
432  self.dataproc_cluster = dataproc_cluster
433  self.dataproc_properties = dataproc_pig_properties
434  self.dataproc_jars = dataproc_pig_jars
435 
436  def execute(self, context):
437  hook = DataProcHook(gcp_conn_id=self.gcp_conn_id,
438  delegate_to=self.delegate_to)
439  job = hook.create_job_template(self.task_id, self.dataproc_cluster, "pigJob",
440  self.dataproc_properties)
441 
442  if self.query is None:
443  job.add_query_uri(self.query_uri)
444  else:
445  job.add_query(self.query)
446  job.add_variables(self.variables)
447  job.add_jar_file_uris(self.dataproc_jars)
448  job.set_job_name(self.job_name)
449 
450  hook.submit(hook.project_id, job.build())
