47  # tasks. It should be refactored to let the task report to the dag run and get the
48  # aggregates from there.
49  qry = (
50  session

95  """
96  Yields a dependency status that indicate whether the given task instance's trigger
97  rule was met.
98 
99  :param ti: the task instance to evaluate the trigger rule of
100  :type ti: TaskInstance
101  :param successes: Number of successful upstream tasks
102  :type successes: boolean
103  :param skipped: Number of skipped upstream tasks
104  :type skipped: boolean
105  :param failed: Number of failed upstream tasks
106  :type failed: boolean
107  :param upstream_failed: Number of upstream_failed upstream tasks
108  :type upstream_failed: boolean
109  :param done: Number of completed upstream tasks
110  :type done: boolean
111  :param flag_upstream_failed: This is a hack to generate
112  the upstream_failed state creation while checking to see
113  whether the task instance is runnable. It was the shortest
114  path to add the feature
115  :type flag_upstream_failed: boolean
116  :param session: database session
117  :type session: Session
118  """
119 
120  TR = airflow.models.TriggerRule
121 
122  task = ti.task
123  upstream = len(task.upstream_task_ids)
124  tr = task.trigger_rule
125  upstream_done = done >= upstream
126  upstream_tasks_state = {
127  "successes": successes, "skipped": skipped, "failed": failed,
128  "upstream_failed": upstream_failed, "done": done
129  }
130  # TODO(aoen): Ideally each individual trigger rules would be it's own class, but
131  # this isn't very feasible at the moment since the database queries need to be
132  # bundled together for efficiency.
133  # handling instant state assignment based on trigger rules
134  if flag_upstream_failed:
135  if tr == TR.ALL_SUCCESS:
136  if upstream_failed or failed:
137  ti.set_state(State.UPSTREAM_FAILED, session)
138  elif skipped:
139  ti.set_state(State.SKIPPED, session)
140  elif tr == TR.ALL_FAILED:
141  if successes or skipped:
142  ti.set_state(State.SKIPPED, session)
143  elif tr == TR.ONE_SUCCESS:
144  if upstream_done and not successes:
145  ti.set_state(State.SKIPPED, session)
146  elif tr == TR.ONE_FAILED:
147  if upstream_done and not (failed or upstream_failed):
148  ti.set_state(State.SKIPPED, session)
149 
150  if tr == TR.ONE_SUCCESS:
151  if successes <= 0:
152  yield self._failing_status(
153  reason="Task's trigger rule '{0}' requires one upstream "
154  "task success, but none were found. "
155  "upstream_tasks_state={1}, upstream_task_ids={2}"
156  .format(tr, upstream_tasks_state, task.upstream_task_ids))
157  elif tr == TR.ONE_FAILED:
158  if not failed and not upstream_failed:
159  yield self._failing_status(
160  reason="Task's trigger rule '{0}' requires one upstream "
161  "task failure, but none were found. "
162  "upstream_tasks_state={1}, upstream_task_ids={2}"
163  .format(tr, upstream_tasks_state, task.upstream_task_ids))
164  elif tr == TR.ALL_SUCCESS:
165  num_failures = upstream - successes
166  if num_failures > 0:
167  yield self._failing_status(
168  reason="Task's trigger rule '{0}' requires all upstream "
169  "tasks to have succeeded, but found {1} non-success(es). "
170  "upstream_tasks_state={2}, upstream_task_ids={3}"
171  .format(tr, num_failures, upstream_tasks_state,
172  task.upstream_task_ids))
173  elif tr == TR.ALL_FAILED:
174  num_successes = upstream - failed - upstream_failed
175  if num_successes > 0:
176  yield self._failing_status(
177  reason="Task's trigger rule '{0}' requires all upstream "
178  "tasks to have failed, but found {1} non-failure(s). "
179  "upstream_tasks_state={2}, upstream_task_ids={3}"
180  .format(tr, num_successes, upstream_tasks_state,
181  task.upstream_task_ids))
182  elif tr == TR.ALL_DONE:
183  if not upstream_done:
184  yield self._failing_status(
185  reason="Task's trigger rule '{0}' requires all upstream "
186  "tasks to have completed, but found {1} task(s) that "
187  "weren't done. upstream_tasks_state={2}, "
188  "upstream_task_ids={3}"
189  .format(tr, upstream-done, upstream_tasks_state,
190  task.upstream_task_ids))
191  else:
192  yield self._failing_status(
193  reason="No strategy to evaluate trigger rule '{0}'.".format(tr))
