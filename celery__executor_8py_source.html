62 class CeleryExecutor(BaseExecutor):
63  """
64  CeleryExecutor is recommended for production use of Airflow. It allows
65  distributing the execution of task instances to multiple worker nodes.
66 
67  Celery is a simple, flexible and reliable distributed system to process
68  vast amounts of messages, while providing operations with the tools
69  required to maintain such a system.
70  """

72  def start(self):
73  self.tasks = {}
74  self.last_state = {}
75 
76  def execute_async(self, key, command, queue=DEFAULT_QUEUE):
77  self.logger.info( "[celery] queuing {key} through celery, "
78  "queue={queue}".format(**locals()))
79  self.tasks[key] = execute_command.apply_async(
80  args=[command], queue=queue)
81  self.last_state[key] = celery_states.PENDING
82 
83  def sync(self):
84 
85  self.logger.debug(
86  "Inquiring about {} celery task(s)".format(len(self.tasks)))
87  for key, async in list(self.tasks.items()):
88  state = async.state
89  if self.last_state[key] != state:
90  if state == celery_states.SUCCESS:
91  self.success(key)
92  del self.tasks[key]
93  del self.last_state[key]
94  elif state == celery_states.FAILURE:
95  self.fail(key)
96  del self.tasks[key]
97  del self.last_state[key]
98  elif state == celery_states.REVOKED:
99  self.fail(key)
100  del self.tasks[key]
101  del self.last_state[key]
102  else:
103  self.logger.info("Unexpected state: " + async.state)
104  self.last_state[key] = async.state
105 
106  def end(self, synchronous=False):
107  if synchronous:
108  while any([
109  async.state not in celery_states.READY_STATES
110  for async in self.tasks.values()]):
111  time.sleep(5)
112  self.sync()
