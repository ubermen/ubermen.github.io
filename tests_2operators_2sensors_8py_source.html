37  """
38  Sensor that always returns the return_value provided

41  :type return_value: any
42  """
43

45  def __init__(
46  self,

48  *args,
49  **kwargs):
50  self.return_value = return_value
51  super(TimeoutTestSensor, self).__init__(*args, **kwargs)
52 
53  def poke(self, context):
54  return self.return_value
55 
56  def execute(self, context):
57  started_at = datetime.now()
58  time_jump = self.params.get('time_jump')
59  while not self.poke(context):
60  if time_jump:
61  started_at -= time_jump
62  if (datetime.now() - started_at).total_seconds() > self.timeout:
63  if self.soft_fail:
64  raise AirflowSkipException('Snap. Time is OUT.')
65  else:
66  raise AirflowSensorTimeout('Snap. Time is OUT.')
67  time.sleep(self.poke_interval)
68  logging.info("Success criteria met. Exiting.")
