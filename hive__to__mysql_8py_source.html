25 class HiveToMySqlTransfer(BaseOperator):
26  """
27  Moves data from Hive to MySQL, note that for now the data is loaded
28  into memory before being pushed to MySQL, so this operator should
29  be used for smallish amount of data.
30 
31  :param sql: SQL query to execute against the MySQL database
32  :type sql: str
33  :param mysql_table: target MySQL table, use dot notation to target a
34  specific database
35  :type mysql_table: str
36  :param mysql_conn_id: source mysql connection
37  :type mysql_conn_id: str
38  :param hiveserver2_conn_id: destination hive connection
39  :type hiveserver2_conn_id: str
40  :param mysql_preoperator: sql statement to run against mysql prior to
41  import, typically use to truncate of delete in place of the data
42  coming in, allowing the task to be idempotent (running the task
43  twice won't double load data)
44  :type mysql_preoperator: str
45  :param mysql_postoperator: sql statement to run against mysql after the
46  import, typically used to move data from staging to production
47  and issue cleanup commands.
48  :type mysql_postoperator: str
49  :param bulk_load: flag to use bulk_load option. This loads mysql directly
50  from a tab-delimited text file using the LOAD DATA LOCAL INFILE command.
51  This option requires an extra connection parameter for the
52  destination MySQL connection: {'local_infile': true}.
53  :type bulk_load: bool
54  """

81  def execute(self, context):
82  hive = HiveServer2Hook(hiveserver2_conn_id=self.hiveserver2_conn_id)
83  logging.info("Extracting data from Hive")
84  logging.info(self.sql)
85 
86  if self.bulk_load:
87  tmpfile = NamedTemporaryFile()
88  hive.to_csv(self.sql, tmpfile.name, delimiter='\t',
89  lineterminator='\n', output_header=False)
90  else:
91  results = hive.get_records(self.sql)
92 
93  mysql = MySqlHook(mysql_conn_id=self.mysql_conn_id)
94  if self.mysql_preoperator:
95  logging.info("Running MySQL preoperator")
96  mysql.run(self.mysql_preoperator)
97 
98  logging.info("Inserting rows into MySQL")
99 
100  if self.bulk_load:
101  mysql.bulk_load(table=self.mysql_table, tmp_file=tmpfile.name)
102  tmpfile.close()
103  else:
104  mysql.insert_rows(table=self.mysql_table, rows=results)
105 
106  if self.mysql_postoperator:
107  logging.info("Running MySQL postoperator")
108  mysql.run(self.mysql_postoperator)
109 
110  logging.info("Done.")
