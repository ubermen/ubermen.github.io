279  @provide_session
280  def evaluate_dagrun(
281  self,
282  dag_id,
283  expected_task_states, # dict of task_id: state
284  dagrun_state,
285  run_kwargs=None,
286  advance_execution_date=False,
287  session=None):
288  """
289  Helper for testing DagRun states with simple two-task DAGS.
290  This is hackish: a dag run is created but its tasks are
291  run by a backfill.
292  """
293  if run_kwargs is None:
294  run_kwargs = {}
295 
296  scheduler = SchedulerJob(**self.default_scheduler_args)
297  dag = self.dagbag.get_dag(dag_id)
298  dag.clear()
299  dr = scheduler.create_dag_run(dag)
300 
301  if advance_execution_date:
302  # run a second time to schedule a dagrun after the start_date
303  dr = scheduler.create_dag_run(dag)
304  ex_date = dr.execution_date
305 
306  try:
307  dag.run(start_date=ex_date, end_date=ex_date, **run_kwargs)
308  except AirflowException:
309  pass
310 
311  # test tasks
312  for task_id, expected_state in expected_task_states.items():
313  task = dag.get_task(task_id)
314  ti = TI(task, ex_date)
315  ti.refresh_from_db()
316  self.assertEqual(ti.state, expected_state)
317 
318  # load dagrun
319  dr = DagRun.find(dag_id=dag_id, execution_date=ex_date)
320  dr = dr[0]
321  dr.dag = dag
322 
323  self.assertEqual(dr.state, dagrun_state)
