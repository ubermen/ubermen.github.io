31 class HDFSHook(BaseHook):
32  """
33  Interact with HDFS. This class is a wrapper around the snakebite library.
34  """
35  def __init__(self, hdfs_conn_id='hdfs_default', proxy_user=None):
36  if not snakebite_imported:
37  raise ImportError(
38  'This HDFSHook implementation requires snakebite, but '
39  'snakebite is not compatible with Python 3 '
40  '(as of August 2015). Please use Python 2 if you require '
41  'this hook -- or help by submitting a PR!')
42  self.hdfs_conn_id = hdfs_conn_id
43  self.proxy_user = proxy_user
44 
45  def get_conn(self):
46  """
47  Returns a snakebite HDFSClient object.
48  """
49  connections = self.get_connections(self.hdfs_conn_id)
50  use_sasl = False
51  if configuration.get('core', 'security') == 'kerberos':
52  use_sasl = True
53 
54  # When using HAClient, proxy_user must be the same, so is ok to always take the first.
55  effective_user = self.proxy_user or connections[0].login
56  if len(connections) == 1:
57  autoconfig = connections[0].extra_dejson.get('autoconfig', False)
58  if autoconfig:
59  client = AutoConfigClient(effective_user=effective_user, use_sasl=use_sasl)
60  else:
61  hdfs_namenode_principal = connections[0].extra_dejson.get('hdfs_namenode_principal')
62  client = Client(connections[0].host, connections[0].port,
63  effective_user=effective_user, use_sasl=use_sasl,
64  hdfs_namenode_principal=hdfs_namenode_principal)
65  elif len(connections) > 1:
66  hdfs_namenode_principal = connections[0].extra_dejson.get('hdfs_namenode_principal')
67  nn = [Namenode(conn.host, conn.port) for conn in connections]
68  client = HAClient(nn, effective_user=effective_user, use_sasl=use_sasl,
69  hdfs_namenode_principal=hdfs_namenode_principal)
70  else:
71  raise HDFSHookException("conn_id doesn't exist in the repository")
72 
73  return client
