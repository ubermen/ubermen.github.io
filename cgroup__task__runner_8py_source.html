28 class CgroupTaskRunner(BaseTaskRunner):
29  """
30  Runs the raw Airflow task in a cgroup that has containment for memory and
31  cpu. It uses the resource requirements defined in the task to construct
32  the settings for the cgroup.
33 
34  Note that this task runner will only work if the Airflow user has root privileges,
35  e.g. if the airflow user is called `airflow` then the following entries (or an even
36  less restrictive ones) are needed in the sudoers file (replacing
37  /CGROUPS_FOLDER with your system's cgroups folder, e.g. '/sys/fs/cgroup/'):
38  airflow ALL= (root) NOEXEC: /bin/chown /CGROUPS_FOLDER/memory/airflow/*
39  airflow ALL= (root) NOEXEC: !/bin/chown /CGROUPS_FOLDER/memory/airflow/*..*
40  airflow ALL= (root) NOEXEC: !/bin/chown /CGROUPS_FOLDER/memory/airflow/* *
41  airflow ALL= (root) NOEXEC: /bin/chown /CGROUPS_FOLDER/cpu/airflow/*
42  airflow ALL= (root) NOEXEC: !/bin/chown /CGROUPS_FOLDER/cpu/airflow/*..*
43  airflow ALL= (root) NOEXEC: !/bin/chown /CGROUPS_FOLDER/cpu/airflow/* *
44  airflow ALL= (root) NOEXEC: /bin/chmod /CGROUPS_FOLDER/memory/airflow/*
45  airflow ALL= (root) NOEXEC: !/bin/chmod /CGROUPS_FOLDER/memory/airflow/*..*
46  airflow ALL= (root) NOEXEC: !/bin/chmod /CGROUPS_FOLDER/memory/airflow/* *
47  airflow ALL= (root) NOEXEC: /bin/chmod /CGROUPS_FOLDER/cpu/airflow/*
48  airflow ALL= (root) NOEXEC: !/bin/chmod /CGROUPS_FOLDER/cpu/airflow/*..*
49  airflow ALL= (root) NOEXEC: !/bin/chmod /CGROUPS_FOLDER/cpu/airflow/* *
50  """

62  def _create_cgroup(self, path):
63  """
64  Create the specified cgroup.
65 
66  :param path: The path of the cgroup to create.
67  E.g. cpu/mygroup/mysubgroup
68  :return: the Node associated with the created cgroup.
69  :rtype: cgroupspy.nodes.Node
70  """
71  node = trees.Tree().root
72  path_split = path.split(os.sep)
73  for path_element in path_split:
74  name_to_node = {x.name: x for x in node.children}
75  if path_element not in name_to_node:
76  self.logger.debug("Creating cgroup {} in {}"
77  .format(path_element, node.path))
78  node = node.create_cgroup(path_element)
79  else:
80  self.logger.debug("Not creating cgroup {} in {} "
81  "since it already exists"
82  .format(path_element, node.path))
83  node = name_to_node[path_element]
84  return node

86  def _delete_cgroup(self, path):
87  """
88  Delete the specified cgroup.
89 
90  :param path: The path of the cgroup to delete.
91  E.g. cpu/mygroup/mysubgroup
92  """
93  node = trees.Tree().root
94  path_split = path.split("/")
95  for path_element in path_split:
96  name_to_node = {x.name: x for x in node.children}
97  if path_element not in name_to_node:
98  self.logger.warning("Cgroup does not exist: {}"
99  .format(path))
100  return
101  else:
102  node = name_to_node[path_element]
103  # node is now the leaf node
104  parent = node.parent
105  self.logger.debug("Deleting cgroup {}/{}".format(parent, node.name))
106  parent.delete_cgroup(node.name)

108  def start(self):
109  # Use bash if it's already in a cgroup
110  cgroups = self._get_cgroup_names()
111  if cgroups["cpu"] != "/" or cgroups["memory"] != "/":
112  self.logger.debug("Already running in a cgroup (cpu: {} memory: {} so "
113  "not creating another one"
114  .format(cgroups.get("cpu"),
115  cgroups.get("memory")))
116  self.process = self.run_command(['bash', '-c'], join_args=True)
117  return
118 
119  # Create a unique cgroup name
120  cgroup_name = "airflow/{}/{}".format(datetime.datetime.now().
121  strftime("%Y-%m-%d"),
122  str(uuid.uuid1()))
123 
124  self.mem_cgroup_name = "memory/{}".format(cgroup_name)
125  self.cpu_cgroup_name = "cpu/{}".format(cgroup_name)
126 
127  # Get the resource requirements from the task
128  task = self._task_instance.task
129  resources = task.resources
130  cpus = resources.cpus.qty
131  self._cpu_shares = cpus * 1024
132  self._mem_mb_limit = resources.ram.qty
133 
134  # Create the memory cgroup
135  mem_cgroup_node = self._create_cgroup(self.mem_cgroup_name)
136  self._created_mem_cgroup = True
137  if self._mem_mb_limit > 0:
138  self.logger.debug("Setting {} with {} MB of memory"
139  .format(self.mem_cgroup_name, self._mem_mb_limit))
140  mem_cgroup_node.controller.limit_in_bytes = self._mem_mb_limit * 1024 * 1024
141 
142  # Create the CPU cgroup
143  cpu_cgroup_node = self._create_cgroup(self.cpu_cgroup_name)
144  self._created_cpu_cgroup = True
145  if self._cpu_shares > 0:
146  self.logger.debug("Setting {} with {} CPU shares"
147  .format(self.cpu_cgroup_name, self._cpu_shares))
148  cpu_cgroup_node.controller.shares = self._cpu_shares
149 
150  # Start the process w/ cgroups
151  self.logger.debug("Starting task process with cgroups cpu,memory:{}"
152  .format(cgroup_name))
153  self.process = self.run_command(
154  ['cgexec', '-g', 'cpu,memory:{}'.format(cgroup_name)]
155  )

157  def return_code(self):

187  def _get_cgroup_names(self):
188  """
189  :return: a mapping between the subsystem name to the cgroup name
190  :rtype: dict[str, str]
191  """
192  with open("/proc/self/cgroup") as f:
193  lines = f.readlines()
194  d = {}
195  for line in lines:
196  line_split = line.rstrip().split(":")
197  subsystem = line_split[1]
198  group_name = line_split[2]
199  d[subsystem] = group_name
200  return d
