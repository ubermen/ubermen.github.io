20 class PrestoCheckOperator(CheckOperator):
21  """
22  Performs checks against Presto. The ``PrestoCheckOperator`` expects
23  a sql query that will return a single row. Each value on that
24  first row is evaluated using python ``bool`` casting. If any of the
25  values return ``False`` the check is failed and errors out.
26 
27  Note that Python bool casting evals the following as ``False``:
28 
29  * ``False``
30  * ``0``
31  * Empty string (``""``)
32  * Empty list (``[]``)
33  * Empty dictionary or set (``{}``)
34 
35  Given a query like ``SELECT COUNT(*) FROM foo``, it will fail only if
36  the count ``== 0``. You can craft much more complex query that could,
37  for instance, check that the table has the same number of rows as
38  the source table upstream, or that the count of today's partition is
39  greater than yesterday's partition, or that a set of metrics are less
40  than 3 standard deviation for the 7 day average.
41 
42  This operator can be used as a data quality check in your pipeline, and
43  depending on where you put it in your DAG, you have the choice to
44  stop the critical path, preventing from
45  publishing dubious data, or on the side and receive email alterts
46  without stopping the progress of the DAG.
47 
48  :param sql: the sql to be executed
49  :type sql: string
50  :param presto_conn_id: reference to the Presto database
51  :type presto_conn_id: string
52  """
53 
54  @apply_defaults
55  def __init__(
56  self, sql,
57  presto_conn_id='presto_default',
58  *args, **kwargs):
59  super(PrestoCheckOperator, self).__init__(sql=sql, *args, **kwargs)
60 
61  self.presto_conn_id = presto_conn_id
62  self.sql = sql
63 
64  def get_db_hook(self):
65  return PrestoHook(presto_conn_id=self.presto_conn_id)
