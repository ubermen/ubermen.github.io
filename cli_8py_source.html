623 def restart_workers(gunicorn_master_proc, num_workers_expected):
624  """
625  Runs forever, monitoring the child processes of @gunicorn_master_proc and
626  restarting workers occasionally.
627 
628  Each iteration of the loop traverses one edge of this state transition
629  diagram, where each state (node) represents
630  [ num_ready_workers_running / num_workers_running ]. We expect most time to
631  be spent in [n / n]. `bs` is the setting webserver.worker_refresh_batch_size.
632 
633  The horizontal transition at ? happens after the new worker parses all the
634  dags (so it could take a while!)
635 
636  V ────────────────────────────────────────────────────────────────────────┐
637  [n / n] ──TTIN──> [ [n, n+bs) / n + bs ] ────?───> [n + bs / n + bs] ──TTOU─┘
638  ^ ^───────────────┘
639  │
640  │ ┌────────────────v
641  └──────┴────── [ [0, n) / n ] <─── start
642 
643  We change the number of workers by sending TTIN and TTOU to the gunicorn
644  master process, which increases and decreases the number of child workers
645  respectively. Gunicorn guarantees that on TTOU workers are terminated
646  gracefully and that the oldest worker is terminated.
647  """

669  batch_size = conf.getint('webserver', 'worker_refresh_batch_size')
670  logging.debug('%s doing a refresh of %s workers',
671  state, batch_size)

673  sys.stderr.flush()

676  for _ in range(batch_size):
677  gunicorn_master_proc.send_signal(signal.SIGTTIN)
678  excess += 1
679  wait_until_true(lambda: num_workers_expected + excess ==
680  get_num_workers_running(gunicorn_master_proc))

682  wait_until_true(lambda: num_workers_expected ==
683  get_num_workers_running(gunicorn_master_proc))
684

687  num_ready_workers_running = get_num_ready_workers_running(gunicorn_master_proc)
688 
689  state = '[{0} / {1}]'.format(num_ready_workers_running, num_workers_running)
690 
691  # Whenever some workers are not ready, wait until all workers are ready
692  if num_ready_workers_running < num_workers_running:
693  logging.debug('%s some workers are starting up, waiting...', state)
694  sys.stdout.flush()
695  time.sleep(1)
696 
697  # Kill a worker gracefully by asking gunicorn to reduce number of workers
698  elif num_workers_running > num_workers_expected:
699  excess = num_workers_running - num_workers_expected
700  logging.debug('%s killing %s workers', state, excess)
701 
702  for _ in range(excess):
703  gunicorn_master_proc.send_signal(signal.SIGTTOU)
704  excess -= 1
705  wait_until_true(lambda: num_workers_expected + excess ==
706  get_num_workers_running(gunicorn_master_proc))
707 
708  # Start a new worker by asking gunicorn to increase number of workers
709  elif num_workers_running == num_workers_expected:
710  refresh_interval = conf.getint('webserver', 'worker_refresh_interval')
711  logging.debug(
712  '%s sleeping for %ss starting doing a refresh...',
713  state, refresh_interval
714  )
715  time.sleep(refresh_interval)
716  start_refresh(gunicorn_master_proc)
717 
718  else:
719  # num_ready_workers_running == num_workers_running < num_workers_expected
720  logging.error((
721  "%s some workers seem to have died and gunicorn"
722  "did not restart them as expected"
723  ), state)
724  time.sleep(10)

726  psutil.Process(gunicorn_master_proc.pid).children()
