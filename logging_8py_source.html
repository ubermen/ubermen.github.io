195  def parse_gcs_url(self, gsurl):
196  """
197  Given a Google Cloud Storage URL (gs://<bucket>/<blob>), returns a
198  tuple containing the corresponding bucket and blob.
199  """
200  # Python 3

203  # Python 2

206

210  else:
211  bucket = parsed_url.netloc
212  blob = parsed_url.path.strip('/')
