30 class SchedulerMetricsJob(SchedulerJob):
31  """
32  This class extends SchedulerJob to instrument the execution performance of
33  task instances contained in each DAG. We want to know if any DAG
34  is starved of resources, and this will be reflected in the stats printed
35  out at the end of the test run. The following metrics will be instrumented
36  for each task instance (dag_id, task_id, execution_date) tuple:
37 
38  1. Queuing delay - time taken from starting the executor to the task
39  instance to be added to the executor queue.
40  2. Start delay - time taken from starting the executor to the task instance
41  to start execution.
42  3. Land time - time taken from starting the executor to task instance
43  completion.
44  4. Duration - time taken for executing the task instance.
45 
46  The DAGs implement bash operators that call the system wait command. This
47  is representative of typical operators run on Airflow - queries that are
48  run on remote systems and spend the majority of their time on I/O wait.
49 
50  To Run:
51  $ python scripts/perf/scheduler_ops_metrics.py
52  """
53  __mapper_args__ = {
54  'polymorphic_identity': 'SchedulerMetricsJob'
55  }

57  def print_stats(self):
58  """
59  Print operational metrics for the scheduler test.
60  """
61  session = settings.Session()
62  TI = TaskInstance
63  tis = (
64  session
65  .query(TI)
66  .filter(TI.dag_id.in_(DAG_IDS))
67  .all()
68  )
69  successful_tis = filter(lambda x: x.state == State.SUCCESS, tis)
70  ti_perf = [(ti.dag_id, ti.task_id, ti.execution_date,
71  (ti.queued_dttm - self.start_date).total_seconds(),
72  (ti.start_date - self.start_date).total_seconds(),
73  (ti.end_date - self.start_date).total_seconds(),
74  ti.duration) for ti in successful_tis]
75  ti_perf_df = pd.DataFrame(ti_perf, columns=['dag_id', 'task_id',
76  'execution_date',
77  'queue_delay',
78  'start_delay', 'land_time',
79  'duration'])
80 
81  print('Performance Results')
82  print('###################')
83  for dag_id in DAG_IDS:
84  print('DAG {}'.format(dag_id))
85  print(ti_perf_df[ti_perf_df['dag_id'] == dag_id])
86  print('###################')
87  if len(tis) > len(successful_tis):
88  print("WARNING!! The following task instances haven't completed")
89  print(pd.DataFrame([(ti.dag_id, ti.task_id, ti.execution_date, ti.state)
90  for ti in filter(lambda x: x.state != State.SUCCESS, tis)],
91  columns=['dag_id', 'task_id', 'execution_date', 'state']))

93  session.commit()

95  def heartbeat(self):
96  """
97  Override the scheduler heartbeat to determine when the test is complete
98  """
99  super(SchedulerMetricsJob, self).heartbeat()
100  session = settings.Session()
101  # Get all the relevant task instances
102  TI = TaskInstance
103  successful_tis = (
104  session
105  .query(TI)
106  .filter(TI.dag_id.in_(DAG_IDS))
107  .filter(TI.state.in_([State.SUCCESS]))
108  .all()
109  )
110  session.commit()
111 
112  dagbag = DagBag(SUBDIR)
113  dags = [dagbag.dags[dag_id] for dag_id in DAG_IDS]
114  # the tasks in perf_dag_1 and per_dag_2 have a daily schedule interval.
115  num_task_instances = sum([(datetime.today() - task.start_date).days
116  for dag in dags for task in dag.tasks])
117 
118  if (len(successful_tis) == num_task_instances or
119  (datetime.now()-self.start_date).total_seconds() >
120  MAX_RUNTIME_SECS):
121  if (len(successful_tis) == num_task_instances):
122  self.logger.info("All tasks processed! Printing stats.")
123  else:
124  self.logger.info("Test timeout reached. "
125  "Printing available stats.")
126  self.print_stats()
127  set_dags_paused_state(True)
128  sys.exit()
