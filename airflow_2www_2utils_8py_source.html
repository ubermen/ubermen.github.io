187 def gzipped(f):
188  '''
189  Decorator to make a view compressed
190  '''
191  @functools.wraps(f)
192  def view_func(*args, **kwargs):
193  @after_this_request
194  def zipper(response):
195  accept_encoding = request.headers.get('Accept-Encoding', '')
196 
197  if 'gzip' not in accept_encoding.lower():
198  return response
199 
200  response.direct_passthrough = False
201 
202  if (response.status_code < 200 or
203  response.status_code >= 300 or
204  'Content-Encoding' in response.headers):
205  return response
206  gzip_buffer = IO()
207  gzip_file = gzip.GzipFile(mode='wb',
208  fileobj=gzip_buffer)
209  gzip_file.write(response.data)
210  gzip_file.close()
211 
212  response.data = gzip_buffer.getvalue()
213  response.headers['Content-Encoding'] = 'gzip'
214  response.headers['Vary'] = 'Accept-Encoding'
215  response.headers['Content-Length'] = len(response.data)
216 
217  return response
218 
219  return f(*args, **kwargs)
220 
221  return view_func
