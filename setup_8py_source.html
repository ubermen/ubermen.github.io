57 def git_version(version):
58  """
59  Return a version to identify the state of the underlying git repo. The version will
60  indicate whether the head of the current git-backed working directory is tied to a
61  release tag or not : it will indicate the former with a 'release:{version}' prefix
62  and the latter with a 'dev0' prefix. Following the prefix will be a sha of the current
63  branch head. Finally, a "dirty" suffix is appended to indicate that uncommitted changes
64  are present.
65  """
66  repo = None
67  try:
68  import git
69  repo = git.Repo('.git')
70  except ImportError:
71  logger.warning('gitpython not found: Cannot compute the git version.')
72  return ''
73  except Exception as e:
74  logger.warning('Git repo not found: Cannot compute the git version.')
75  return ''
76  if repo:
77  sha = repo.head.commit.hexsha
78  if repo.is_dirty():
79  return '.dev0+{sha}.dirty'.format(sha=sha)
80  # commit is clean
81  # is it release of `version` ?
82  try:
83  tag = repo.git.describe(
84  match='[0-9]*', exact_match=True,
85  tags=True, dirty=True)
86  assert tag == version, (tag, version)
87  return '.release:{version}+{sha}'.format(version=version,
88  sha=sha)
89  except git.GitCommandError:
90  return '.dev0+{sha}'.format(sha=sha)
91  else:
92  return 'no_git_version'
