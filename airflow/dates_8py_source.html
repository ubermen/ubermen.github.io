36 def date_range(
37  start_date,
38  end_date=None,
39  num=None,
40  delta=None):
41  """
42  Get a set of dates as a list based on a start, end and delta, delta
43  can be something that can be added to ``datetime.datetime``
44  or a cron expression as a ``str``
45 
46  :param start_date: anchor date to start the series from
47  :type start_date: datetime.datetime
48  :param end_date: right boundary for the date range
49  :type end_date: datetime.datetime
50  :param num: alternatively to end_date, you can specify the number of
51  number of entries you want in the range. This number can be negative,
52  output will always be sorted regardless
53  :type num: int
54 
55  >>> date_range(datetime(2016, 1, 1), datetime(2016, 1, 3), delta=timedelta(1))
56  [datetime.datetime(2016, 1, 1, 0, 0), datetime.datetime(2016, 1, 2, 0, 0), datetime.datetime(2016, 1, 3, 0, 0)]
57  >>> date_range(datetime(2016, 1, 1), datetime(2016, 1, 3), delta='0 0 * * *')
58  [datetime.datetime(2016, 1, 1, 0, 0), datetime.datetime(2016, 1, 2, 0, 0), datetime.datetime(2016, 1, 3, 0, 0)]
59  >>> date_range(datetime(2016, 1, 1), datetime(2016, 3, 3), delta="0 0 0 * *")
60  [datetime.datetime(2016, 1, 1, 0, 0), datetime.datetime(2016, 2, 1, 0, 0), datetime.datetime(2016, 3, 1, 0, 0)]
61  """
62  if not delta:
63  return []
64  if end_date and start_date > end_date:
65  raise Exception("Wait. start_date needs to be before end_date")
66  if end_date and num:
67  raise Exception("Wait. Either specify end_date OR num")
68  if not end_date and not num:
69  end_date = datetime.now()
70 
71  delta_iscron = False
72  if isinstance(delta, six.string_types):
73  delta_iscron = True
74  cron = croniter(delta, start_date)
75  elif isinstance(delta, timedelta):
76  delta = abs(delta)
77  l = []
78  if end_date:
79  while start_date <= end_date:
80  l.append(start_date)
81  if delta_iscron:
82  start_date = cron.get_next(datetime)
83  else:
84  start_date += delta
85  else:
86  for i in range(abs(num)):
87  l.append(start_date)
88  if delta_iscron:
89  if num > 0:
90  start_date = cron.get_next(datetime)
91  else:
92  start_date = cron.get_prev(datetime)
93  else:
94  if num > 0:
95  start_date += delta
96  else:
97  start_date -= delta
98  return sorted(l)

101 def round_time(dt, delta, start_date=datetime.min):
102  """
103  Returns the datetime of the form start_date + i * delta
104  which is closest to dt for any non-negative integer i.
105 
106  Note that delta may be a datetime.timedelta or a dateutil.relativedelta
107 
108  >>> round_time(datetime(2015, 1, 1, 6), timedelta(days=1))
109  datetime.datetime(2015, 1, 1, 0, 0)
110  >>> round_time(datetime(2015, 1, 2), relativedelta(months=1))
111  datetime.datetime(2015, 1, 1, 0, 0)
112  >>> round_time(datetime(2015, 9, 16, 0, 0), timedelta(1), datetime(2015, 9, 14, 0, 0))
113  datetime.datetime(2015, 9, 16, 0, 0)
114  >>> round_time(datetime(2015, 9, 15, 0, 0), timedelta(1), datetime(2015, 9, 14, 0, 0))
115  datetime.datetime(2015, 9, 15, 0, 0)
116  >>> round_time(datetime(2015, 9, 14, 0, 0), timedelta(1), datetime(2015, 9, 14, 0, 0))
117  datetime.datetime(2015, 9, 14, 0, 0)
118  >>> round_time(datetime(2015, 9, 13, 0, 0), timedelta(1), datetime(2015, 9, 14, 0, 0))
119  datetime.datetime(2015, 9, 14, 0, 0)
120  """
121 
122  if isinstance(delta, six.string_types):
123  # It's cron based, so it's easy
124  cron = croniter(delta, start_date)
125  prev = cron.get_prev(datetime)
126  if prev == start_date:
127  return start_date
128  else:
129  return prev
130 
131  # Ignore the microseconds of dt
132  dt -= timedelta(microseconds=dt.microsecond)
133 
134  # We are looking for a datetime in the form start_date + i * delta
135  # which is as close as possible to dt. Since delta could be a relative
136  # delta we don't know it's exact length in seconds so we cannot rely on
137  # division to find i. Instead we employ a binary search algorithm, first
138  # finding an upper and lower limit and then disecting the interval until
139  # we have found the closest match.
140 
141  # We first search an upper limit for i for which start_date + upper * delta
142  # exceeds dt.
143  upper = 1
144  while start_date + upper*delta < dt:
145  # To speed up finding an upper limit we grow this exponentially by a
146  # factor of 2
147  upper *= 2
148 
149  # Since upper is the first value for which start_date + upper * delta
150  # exceeds dt, upper // 2 is below dt and therefore forms a lower limited
151  # for the i we are looking for
152  lower = upper // 2
153 
154  # We now continue to intersect the interval between
155  # start_date + lower * delta and start_date + upper * delta
156  # until we find the closest value
157  while True:
158  # Invariant: start + lower * delta < dt <= start + upper * delta
159  # If start_date + (lower + 1)*delta exceeds dt, then either lower or
160  # lower+1 has to be the solution we are searching for
161  if start_date + (lower + 1)*delta >= dt:
162  # Check if start_date + (lower + 1)*delta or
163  # start_date + lower*delta is closer to dt and return the solution
164  if (
165  (start_date + (lower + 1) * delta) - dt <=
166  dt - (start_date + lower * delta)):
167  return start_date + (lower + 1)*delta
168  else:
169  return start_date + lower * delta
170 
171  # We intersect the interval and either replace the lower or upper
172  # limit with the candidate
173  candidate = lower + (upper - lower) // 2
174  if start_date + candidate*delta >= dt:
175  upper = candidate
176  else:
177  lower = candidate
178 
179  # in the special case when start_date > dt the search for upper will
180  # immediately stop for upper == 1 which results in lower = upper // 2 = 0
181  # and this function returns start_date.

184 def infer_time_unit(time_seconds_arr):
185  """
186  Determine the most appropriate time unit for an array of time durations
187  specified in seconds.
188 
189  e.g. 5400 seconds => 'minutes', 36000 seconds => 'hours'
190  """
191  if len(time_seconds_arr) == 0:
192  return 'hours'
193  max_time_seconds = max(time_seconds_arr)
194  if max_time_seconds <= 60*2:
195  return 'seconds'
196  elif max_time_seconds <= 60*60*2:
197  return 'minutes'
198  elif max_time_seconds <= 24*60*60*2:
199  return 'hours'
200  else:
201  return 'days'

204 def scale_time_units(time_seconds_arr, unit):
205  """
206  Convert an array of time durations in seconds to the specified time unit.
207  """
208  if unit == 'minutes':
209  return list(map(lambda x: x*1.0/60, time_seconds_arr))
210  elif unit == 'hours':
211  return list(map(lambda x: x*1.0/(60*60), time_seconds_arr))
212  elif unit == 'days':
213  return list(map(lambda x: x*1.0/(24*60*60), time_seconds_arr))
214  return time_seconds_arr
