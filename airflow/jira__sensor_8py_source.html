113  def issue_field_checker(self, context, issue):
114  result = None
115  try:
116  if issue is not None \
117  and self.field is not None \
118  and self.expected_value is not None:
119 
120  field_value = getattr(issue.fields, self.field)
121  if field_value is not None:
122  if isinstance(field_value, list):
123  result = self.expected_value in field_value
124  elif isinstance(field_value, str):
125  result = self.expected_value.lower() == field_value.lower()
126  elif isinstance(field_value, Resource) \
127  and getattr(field_value, 'name'):
128  result = self.expected_value.lower() == field_value.name.lower()
129  else:
130  logging.warning("not implemented checker for issue field {0} "
131  "which is neither string nor list nor "
132  "jira Resource".format(self.field))
133 
134  except JIRAError as jira_error:
135  logging.error("jira error while checking with expected value: {0}"
136  .format(jira_error))
137  except Exception as e:
138  logging.error("error while checking with expected value {0}, error: {1}"
139  .format(self.expected_value, e))
140  if result is True:
141  logging.info("issue field {0} has expected value {1}, returning success"
142  .format(self.field, self.expected_value))
143  else:
144  logging.info("issue field {0} dont have expected value {1} yet."
145  .format(self.field, self.expected_value))
146  return result
