148  def send_ingest_query(
149  self, datasource, static_path, ts_dim, columns, metric_spec,
150  intervals, num_shards, target_partition_size, query_granularity, segment_granularity,
151  hadoop_dependency_coordinates=None):
152  query = self.construct_ingest_query(
153  datasource, static_path, ts_dim, columns,

155  query_granularity, segment_granularity, hadoop_dependency_coordinates)
156  r = requests.post(

158  logging.info(self.ingest_post_url)

167  def load_from_hdfs(
168  self, datasource, static_path, ts_dim, columns,
169  intervals, num_shards, target_partition_size, query_granularity, segment_granularity,
170  metric_spec=None, hadoop_dependency_coordinates=None):
171  """
172  load data to druid from hdfs
173 
174  :param ts_dim: The column name to use as a timestamp
175  :param metric_spec: A list of dictionaries
176  """
177  task_id = self.send_ingest_query(
178  datasource, static_path, ts_dim, columns, metric_spec,
179  intervals, num_shards, target_partition_size, query_granularity, segment_granularity,
180  hadoop_dependency_coordinates)
181  status_url = self.get_ingest_status_url(task_id)
182  while True:
183  r = requests.get(status_url)
184  d = json.loads(r.text)
185  if d['status']['status'] == 'FAILED':
186  logging.error(d)
187  raise AirflowDruidLoadException(
188  "[Error]: Ingesting data to druid failed.")
189  elif d['status']['status'] == 'SUCCESS':
190  break
191  time.sleep(LOAD_CHECK_INTERVAL)
