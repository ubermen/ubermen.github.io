37  def __handle_rate_limit_exception(self, rate_limit_exception):
38  """
39  Sleep for the time specified in the exception. If not specified, wait
40  for 60 seconds.
41  """
42  retry_after = int(
43  rate_limit_exception.response.headers.get('Retry-After', 60))
44  logging.info(
45  "Hit Zendesk API rate limit. Pausing for {} "
46  "seconds".format(
47  retry_after))
48  time.sleep(retry_after)

50  def call(self, path, query=None, get_all_pages=True):
51  """
52  Call Zendesk API and return results
53 
54  :param path: The Zendesk API to call
55  :param query: Query parameters
56  :param get_all_pages: Accumulate results over all pages before
57  returning. Due to strict rate limiting, this can often timeout.
58  Waits for recommended period between tries after a timeout.
59  """
60  zendesk = self.get_conn()
61  first_request_successful = False
62 
63  while not first_request_successful:
64  try:
65  results = zendesk.call(path, query)
66  first_request_successful = True
67  except RateLimitError as rle:
68  self.__handle_rate_limit_exception(rle)
69 
70  # Find the key with the results
71  key = path.split("/")[-1].split(".json")[0]
72  next_page = results['next_page']
73  results = results[key]
74 
75  if get_all_pages:
76  while next_page is not None:
77  try:
78  # Need to split because the next page URL has
79  # `github.zendesk...`
80  # in it, but the call function needs it removed.
81  next_url = next_page.split(self.__url)[1]
82  logging.info("Calling {}".format(next_url))
83  more_res = zendesk.call(next_url)
84  results.extend(more_res[key])
85  if next_page == more_res['next_page']:
86  # Unfortunately zdesk doesn't always throw ZendeskError
87  # when we are done getting all the data. Sometimes the
88  # next just refers to the current set of results. Hence,
89  # need to deal with this special case
90  break
91  else:
92  next_page = more_res['next_page']
93  except RateLimitError as rle:
94  self.__handle_rate_limit_exception(rle)
95  except ZendeskError as ze:
96  if b"Use a start_time older than 5 minutes" in ze.msg:
97  # We have pretty up to date data
98  break
99  else:
100  raise ze
101 
102  return results
