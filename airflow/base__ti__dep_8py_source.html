71  @provide_session
72  def get_dep_statuses(self, ti, session, dep_context):
73  """
74  Wrapper around the private _get_dep_statuses method that contains some global
75  checks for all dependencies.
76 
77  :param ti: the task instance to get the dependency status for
78  :type ti: TaskInstance
79  :param session: database session
80  :type session: Session

82  :type dep_context: DepContext
