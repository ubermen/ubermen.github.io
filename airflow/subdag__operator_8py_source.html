37  should be prefixed by its parent and a dot. As in `parent.child`.
38 
39  :param subdag: the DAG object to run as a subdag of the current DAG.
40  :type subdag: airflow.DAG
41  :param dag: the parent DAG
42  :type subdag: airflow.DAG
43  """
44  if 'dag' not in kwargs:
45  raise AirflowException("Please pass in the `dag` param")
46  dag = kwargs['dag']
47  session = kwargs.pop('session')
48  super(SubDagOperator, self).__init__(*args, **kwargs)
49 
50  # validate subdag name
51  if dag.dag_id + '.' + kwargs['task_id'] != subdag.dag_id:
52  raise AirflowException(
53  "The subdag's dag_id should have the form "
54  "'{{parent_dag_id}}.{{this_task_id}}'. Expected "
55  "'{d}.{t}'; received '{rcvd}'.".format(
56  d=dag.dag_id, t=kwargs['task_id'], rcvd=subdag.dag_id))
57 
58  # validate that subdag operator and subdag tasks don't have a
59  # pool conflict
60  if self.pool:
61  pool = (
62  session
63  .query(Pool)
64  .filter(Pool.slots == 1)
65  .filter(Pool.pool == self.pool)
66  .first()
67  )
68  conflicts = [t for t in subdag.tasks if t.pool == self.pool]
69  if pool and any(t.pool == self.pool for t in subdag.tasks):
70  raise AirflowException(

72  'pool {p}, but the pool only has 1 slot. The subdag tasks'
73  'will never run.'.format(
74  sd=self.task_id,
75  plural=len(conflicts) > 1,
76  t=', '.join(t.task_id for t in conflicts),
