31 class TriggerDagRunOperator(BaseOperator):
32  """
33  Triggers a DAG run for a specified ``dag_id`` if a criteria is met
34 
35  :param trigger_dag_id: the dag_id to trigger
36  :type trigger_dag_id: str
37  :param python_callable: a reference to a python function that will be
38  called while passing it the ``context`` object and a placeholder
39  object ``obj`` for your callable to fill and return if you want
40  a DagRun created. This ``obj`` object contains a ``run_id`` and
41  ``payload`` attribute that you can modify in your function.
42  The ``run_id`` should be a unique identifier for that DAG run, and
43  the payload has to be a picklable object that will be made available
44  to your tasks while executing that DAG run. Your function header
45  should look like ``def foo(context, dag_run_obj):``
46  :type python_callable: python callable
47  """

62  def execute(self, context):
63  dro = DagRunOrder(run_id='trig__' + datetime.now().isoformat())
