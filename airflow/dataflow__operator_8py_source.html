23 class DataFlowJavaOperator(BaseOperator):
24  """
25  Start a Java Cloud DataFlow batch job. The parameters of the operation
26  will be passed to the job.
27 
28  It's a good practice to define dataflow_* parameters in the default_args of the dag
29  like the project, zone and staging location.
30 
31  ```
32  default_args = {
33  'dataflow_default_options': {
34  'project': 'my-gcp-project',
35  'zone': 'europe-west1-d',
36  'stagingLocation': 'gs://my-staging-bucket/staging/'
37  }
38  }
39  ```
40 
41  You need to pass the path to your dataflow as a file reference with the ``jar``
42  parameter, the jar needs to be a self executing jar. Use ``options`` to pass on
43  options to your job.
44 
45  ```
46  t1 = DataFlowOperation(
47  task_id='datapflow_example',
48  jar='{{var.value.gcp_dataflow_base}}pipeline/build/libs/pipeline-example-1.0.jar',
49  options={
50  'autoscalingAlgorithm': 'BASIC',
51  'maxNumWorkers': '50',
52  'start': '{{ds}}',
53  'partitionType': 'DAY'
54  },
55  dag=my-dag)
56  ```
57 
58  Both ``jar`` and ``options`` are templated so you can use variables in them.
59  """
60  template_fields = ['options', 'jar']
61  ui_color = '#0273d4'
62

65  self,
66  jar,
67  dataflow_default_options=None,
68  options=None,
69  gcp_conn_id='google_cloud_default',
70  delegate_to=None,
71  *args,
72  **kwargs):
73  """
74  Create a new DataFlowJavaOperator. Note that both
75  dataflow_default_options and options will be merged to specify pipeline
76  execution parameter, and dataflow_default_options is expected to save
77  high-level options, for instances, project and zone information, which
78  apply to all dataflow operators in the DAG.
79 
80  For more detail on job submission have a look at the reference:
81 
82  https://cloud.google.com/dataflow/pipelines/specifying-exec-params
83 
84  :param jar: The reference to a self executing DataFlow jar.
85  :type jar: string
86  :param dataflow_default_options: Map of default job options.
87  :type dataflow_default_options: dict
88  :param options: Map of job specific options.
89  :type options: dict
90  :param gcp_conn_id: The connection ID to use connecting to Google Cloud
91  Platform.
92  :type gcp_conn_id: string
93  :param delegate_to: The account to impersonate, if any.
94  For this to work, the service account making the request must have
95  domain-wide delegation enabled.
96  :type delegate_to: string
97  """
98  super(DataFlowJavaOperator, self).__init__(*args, **kwargs)
99 
100  dataflow_default_options = dataflow_default_options or {}
101  options = options or {}
102 
103  self.gcp_conn_id = gcp_conn_id
104  self.delegate_to = delegate_to
105  self.jar = jar
106  self.dataflow_default_options = dataflow_default_options
107  self.options = options
