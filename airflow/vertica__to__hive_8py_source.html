27 class VerticaToHiveTransfer(BaseOperator):
28  """
29  Moves data from Vertia to Hive. The operator runs
30  your query against Vertia, stores the file locally
31  before loading it into a Hive table. If the ``create`` or
32  ``recreate`` arguments are set to ``True``,
33  a ``CREATE TABLE`` and ``DROP TABLE`` statements are generated.
34  Hive data types are inferred from the cursor's metadata.
35  Note that the table generated in Hive uses ``STORED AS textfile``
36  which isn't the most efficient serialization format. If a
37  large amount of data is loaded and/or if the table gets
38  queried considerably, you may want to use this operator only to
39  stage the data into a temporary table before loading it into its
40  final destination using a ``HiveOperator``.
41 
42  :param sql: SQL query to execute against the Vertia database
43  :type sql: str
44  :param hive_table: target Hive table, use dot notation to target a
45  specific database
46  :type hive_table: str
47  :param create: whether to create the table if it doesn't exist
48  :type create: bool
49  :param recreate: whether to drop and recreate the table at every execution
50  :type recreate: bool
51  :param partition: target partition as a dict of partition columns and values
52  :type partition: dict
53  :param delimiter: field delimiter in the file
54  :type delimiter: str
55  :param vertica_conn_id: source Vertica connection
56  :type vertica_conn_id: str
57  :param hive_conn_id: destination hive connection
58  :type hive_conn_id: str
59 
60  """

91  # vertica-python datatype.py donot provied the full type mapping access.
92  # Manual hack. Reference: https://github.com/uber/vertica-python/blob/master/vertica_python/vertica/column.py
93  d = {
94  5: 'BOOLEAN',
95  6: 'INT',
96  7: 'FLOAT',

103  def execute(self, context):
104  hive = HiveCliHook(hive_cli_conn_id=self.hive_cli_conn_id)
105  vertica = VerticaHook(vertica_conn_id=self.vertica_conn_id)
106 
107  logging.info("Dumping Vertica query results to local file")
108  conn = vertica.get_conn()
109  cursor = conn.cursor()
110  cursor.execute(self.sql)
111  with NamedTemporaryFile("w") as f:
112  csv_writer = csv.writer(f, delimiter=self.delimiter, encoding='utf-8')
113  field_dict = OrderedDict()
114  col_count = 0
115  for field in cursor.description:
116  col_count += 1
117  col_position = "Column{position}".format(position=col_count)
118  field_dict[col_position if field[0] == '' else field[0]] = self.type_map(field[1])
119  csv_writer.writerows(cursor.iterate())
120  f.flush()
121  cursor.close()
122  conn.close()
