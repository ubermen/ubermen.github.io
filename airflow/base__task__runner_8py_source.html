97  def run_command(self, run_with, join_args=False):
98  """
99  Run the task command
100 
101  :param run_with: list of tokens to run the task command with
102  E.g. ['bash', '-c']
103  :type run_with: list
104  :param join_args: whether to concatenate the list of command tokens
105  E.g. ['airflow', 'run'] vs ['airflow run']
106  :param join_args: bool
107  :return: the process that was run
108  :rtype: subprocess.Popen
109  """
110  cmd = [" ".join(self._command)] if join_args else self._command
111  full_cmd = run_with + cmd
112  self.logger.info('Running: {}'.format(full_cmd))
113  proc = subprocess.Popen(
114  full_cmd,
