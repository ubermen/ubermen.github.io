59  def execute(self, **kwargs):
60  """
61  SlackAPIOperator calls will not fail even if the call is not unsuccessful.
62  It should not prevent a DAG from completing in success
63  """
64  if not self.api_params:
65  self.construct_api_call_params()
66  sc = SlackClient(self.token)
67  rc = sc.api_call(self.method, **self.api_params)
68  if not rc['ok']:
69  logging.error("Slack API call failed ({})".format(rc['error']))
70  raise AirflowException("Slack API call failed: ({})".format(rc['error']))
