28 class SSHHook(BaseHook):
29  """
30  Light-weight remote execution library and utilities.
31 
32  Using this hook (which is just a convenience wrapper for subprocess),
33  is created to let you stream data from a remotely stored file.
34 
35  As a bonus, :class:`SSHHook` also provides a really cool feature that let's you
36  set up ssh tunnels super easily using a python context manager (there is an example
37  in the integration part of unittests).
38 
39  :param key_file: Typically the SSHHook uses the keys that are used by the user
40  airflow is running under. This sets the behavior to use another file instead.
41  :type key_file: str
42  :param connect_timeout: sets the connection timeout for this connection.
43  :type connect_timeout: int
44  :param no_host_key_check: whether to check to host key. If True host keys will not
45  be checked, but are also not stored in the current users's known_hosts file.
46  :type no_host_key_check: bool
47  :param tty: allocate a tty.
48  :type tty: bool
49  :param sshpass: Use to non-interactively perform password authentication by using
50  sshpass.
51  :type sshpass: bool
52  """

67  def _host_ref(self):
68  if self.conn.login:
69  return "{0}@{1}".format(self.conn.login, self.conn.host)
70  else:

73  def _prepare_command(self, cmd):
74  connection_cmd = ["ssh", self._host_ref(), "-o", "ControlMaster=no"]
75  if self.sshpass:
76  connection_cmd = ["sshpass", "-e"] + connection_cmd
77  else:
78  connection_cmd += ["-o", "BatchMode=yes"] # no password prompts
79 
80  if self.conn.port:
81  connection_cmd += ["-p", str(self.conn.port)]
82 
83  if self.connect_timeout:
84  connection_cmd += ["-o", "ConnectionTimeout={}".format(self.connect_timeout)]
85 
86  if self.tcp_keepalive:
87  connection_cmd += ["-o", "TCPKeepAlive=yes"]
88  connection_cmd += ["-o", "ServerAliveInterval={}".format(self.server_alive_interval)]
89 
90  if self.no_host_key_check:
91  connection_cmd += ["-o", "UserKnownHostsFile=/dev/null",
92  "-o", "StrictHostKeyChecking=no"]
93 
94  if self.key_file:
95  connection_cmd += ["-i", self.key_file]
96 
97  if self.tty:
98  connection_cmd += ["-t"]
99 
100  connection_cmd += cmd
101  logging.debug("SSH cmd: {} ".format(connection_cmd))
102 
103  return connection_cmd

116  def check_output(self, cmd):
117  """
118  Executes a remote command and returns the stdout a remote process.
119  Simplified version of Popen when you only want the output as a string and detect any errors.
120 
121  :param cmd: command to remotely execute
122  :return: stdout
123  """
124  p = self.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
125  output, stderr = p.communicate()
126

130  cmd, self.conn.host, p.returncode, output, stderr))

135  def tunnel(self, local_port, remote_port=None, remote_host="localhost"):
136  """
137  Creates a tunnel between two hosts. Like ssh -L <LOCAL_PORT>:host:<REMOTE_PORT>.
138  Remember to close() the returned "tunnel" object in order to clean up
139  after yourself when you are done with the tunnel.
140 
141  :param local_port:
142  :type local_port: int
143  :param remote_port:
144  :type remote_port: int
145  :param remote_host:
146  :type remote_host: str
147  :return:
148  """
149  tunnel_host = "{0}:{1}:{2}".format(local_port, remote_host, remote_port)
150  proc = self.Popen(["-L", tunnel_host, "echo -n ready && cat"],
151  stdin=subprocess.PIPE, stdout=subprocess.PIPE)
152 
153  ready = proc.stdout.read(5)
154  assert ready == b"ready", "Did not get 'ready' from remote"
155  yield
156  proc.communicate()
157  assert proc.returncode == 0, "Tunnel process did unclean exit (returncode {}".format(proc.returncode)
