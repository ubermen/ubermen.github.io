62  :type object: list

67  :param schema_fields: If set, the schema field list as defined here:
68  https://cloud.google.com/bigquery/docs/reference/v2/jobs#configuration.load
69  :type schema_fields: list
70  :param schema_object: If set, a GCS object path pointing to a .json file that
71  contains the schema for the table.
72  :param schema_object: string
73  :param source_format: File format to export.
74  :type source_format: string
75  :param create_disposition: The create disposition if the table doesn't exist.
76  :type create_disposition: string
77  :param skip_leading_rows: Number of rows to skip when loading from a CSV.
78  :type skip_leading_rows: int
79  :param write_disposition: The write disposition if the table already exists.
80  :type write_disposition: string
81  :param field_delimiter: The delimiter to use when loading from a CSV.
82  :type field_delimiter: string
83  :param max_id_key: If set, the name of a column in the BigQuery table
84  that's to be loaded. Thsi will be used to select the MAX value from
85  BigQuery after the load occurs. The results will be returned by the
86  execute() command, which in turn gets stored in XCom for future
87  operators to use. This can be helpful with incremental loads--during
88  future executions, you can pick up from the max ID.
89  :type max_id_key: string
90  :param bigquery_conn_id: Reference to a specific BigQuery hook.

126  def execute(self, context):
127  bq_hook = BigQueryHook(bigquery_conn_id=self.bigquery_conn_id,
128  delegate_to=self.delegate_to)
129 
130  if not self.schema_fields and self.schema_object:
131  gcs_hook = GoogleCloudStorageHook(
132  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
133  delegate_to=self.delegate_to)
134  schema_fields = json.loads(gcs_hook.download(
135  self.bucket,
136  self.schema_object).decode("utf-8"))
137  else:
138  schema_fields = self.schema_fields
139 
140  source_uris = ['gs://{}/{}'.format(self.bucket, source_object)
141  for source_object in self.source_objects]
142  conn = bq_hook.get_conn()
143  cursor = conn.cursor()
144  cursor.run_load(
145  destination_project_dataset_table=self.destination_project_dataset_table,
146  schema_fields=schema_fields,
147  source_uris=source_uris,
148  source_format=self.source_format,
149  create_disposition=self.create_disposition,
150  skip_leading_rows=self.skip_leading_rows,
151  write_disposition=self.write_disposition,
152  field_delimiter=self.field_delimiter,
153  schema_update_options=self.schema_update_options)
154 
155  if self.max_id_key:
156  cursor.execute('SELECT MAX({}) FROM {}'.format(
157  self.max_id_key,
158  self.destination_project_dataset_table))
159  row = cursor.fetchone()
160  max_id = row[0] if row[0] else 0
161  logging.info('Loaded BQ data with max {}.{}={}'.format(
162  self.destination_project_dataset_table,
163  self.max_id_key, max_id))
