159 def list_py_file_paths(directory, safe_mode=True):
160  """
161  Traverse a directory and look for Python files.
162 
163  :param directory: the directory to traverse
164  :type directory: unicode
165  :param safe_mode: whether to use a heuristic to determine whether a file
166  contains Airflow DAG definitions
167  :return: a list of paths to Python files in the specified directory
168  :rtype: list[unicode]
169  """
170  file_paths = []
171  if directory is None:
172  return []
173  elif os.path.isfile(directory):
174  return [directory]
175  elif os.path.isdir(directory):
176  patterns = []
177  for root, dirs, files in os.walk(directory, followlinks=True):
178  ignore_file = [f for f in files if f == '.airflowignore']
179  if ignore_file:
180  f = open(os.path.join(root, ignore_file[0]), 'r')
181  patterns += [p for p in f.read().split('\n') if p]
182  f.close()
183  for f in files:
184  try:
185  file_path = os.path.join(root, f)
186  if not os.path.isfile(file_path):
187  continue
188  mod_name, file_ext = os.path.splitext(
189  os.path.split(file_path)[-1])
190  if file_ext != '.py':
191  continue
192  if any([re.findall(p, file_path) for p in patterns]):
193  continue
194 
195  # Heuristic that guesses whether a Python file contains an
196  # Airflow DAG definition.
197  might_contain_dag = True
198  if safe_mode:
199  with open(file_path, 'rb') as f:
200  content = f.read()
201  might_contain_dag = all(
202  [s in content for s in (b'DAG', b'airflow')])
203 
204  if not might_contain_dag:
205  continue
206 
207  file_paths.append(file_path)
208  except Exception:
209  logging.exception("Error while examining %s", f)
210  return file_paths

298 class DagFileProcessorManager(LoggingMixin):

300  Given a list of DAG definition files, this kicks off several processors
301  in parallel to process them. The parallelism is limited and as the
302  processors finish, more are launched. The files are processed over and
303  over again, but no more often than the specified interval.

450  @staticmethod

543  def heartbeat(self):
544  """
545  This should be periodically called by the scheduler. This method will
546  kick of new processes to process DAG definition files and read the
547  results from the finished processors.
548 
549  :return: a list of SimpleDags that were produced by processors that
550  have finished since the last time this was called
551  :rtype: list[SimpleDag]
552  """
553  finished_processors = {}
554  """:type : dict[unicode, AbstractDagFileProcessor]"""
555  running_processors = {}
556  """:type : dict[unicode, AbstractDagFileProcessor]"""
557 
558  for file_path, processor in self._processors.items():
559  if processor.done:
560  self.logger.info("Processor for {} finished".format(file_path))
561  now = datetime.now()
562  finished_processors[file_path] = processor
563  self._last_runtime[file_path] = (now -
564  processor.start_time).total_seconds()
565  self._last_finish_time[file_path] = now
566  self._run_count[file_path] += 1
567  else:
568  running_processors[file_path] = processor
569  self._processors = running_processors
570 
571  # Collect all the DAGs that were found in the processed files
572  simple_dags = []
573  for file_path, processor in finished_processors.items():
574  if processor.result is None:
575  self.logger.warning("Processor for {} exited with return code "
576  "{}. See {} for details."
577  .format(processor.file_path,
578  processor.exit_code,
579  processor.log_file))
580  else:
581  for simple_dag in processor.result:
582  simple_dags.append(simple_dag)
583 
584  # Generate more file paths to process if we processed all the files
585  # already.
586  if len(self._file_path_queue) == 0:
587  # If the file path is already being processed, or if a file was
588  # processed recently, wait until the next batch
589  file_paths_in_progress = self._processors.keys()
590  now = datetime.now()
591  file_paths_recently_processed = []
592  for file_path in self._file_paths:
593  last_finish_time = self.get_last_finish_time(file_path)
594  if (last_finish_time is not None and
595  (now - last_finish_time).total_seconds() <
596  self._process_file_interval):
597  file_paths_recently_processed.append(file_path)
598 
599  files_paths_at_run_limit = [file_path
600  for file_path, num_runs in self._run_count.items()
601  if num_runs == self._max_runs]
602 
603  files_paths_to_queue = list(set(self._file_paths) -
604  set(file_paths_in_progress) -
605  set(file_paths_recently_processed) -
606  set(files_paths_at_run_limit))
607 
608  for file_path, processor in self._processors.items():
609  self.logger.debug("File path {} is still being processed (started: {})"
610  .format(processor.file_path,
611  processor.start_time.isoformat()))
612 
613  self.logger.debug("Queuing the following files for processing:\n\t{}"
614  .format("\n\t".join(files_paths_to_queue)))
615 
616  self._file_path_queue.extend(files_paths_to_queue)
617 
618  # Start more processors if we have enough slots and files to process
619  while (self._parallelism - len(self._processors) > 0 and
620  len(self._file_path_queue) > 0):
621  file_path = self._file_path_queue.pop(0)
622  log_file_path = self._get_log_file_path(file_path)
623  processor = self._processor_factory(file_path, log_file_path)
624 
625  processor.start()
626  self.logger.info("Started a process (PID: {}) to generate "
627  "tasks for {} - logging into {}"
628  .format(processor.pid, file_path, log_file_path))
629 
630  self._processors[file_path] = processor
631 
632  self.symlink_latest_log_directory()
633 
634  # Update scheduler heartbeat count.
635  self._run_count[self._heart_beat_key] += 1
636 
637  return simple_dags

639  def max_runs_reached(self):
640  """
641  :return: whether all file paths have been processed max_runs times
642  """
643  if self._max_runs == -1: # Unlimited runs.
644  return False

646  if self._run_count[file_path] != self._max_runs:
647  return False
648  if self._run_count[self._heart_beat_key] < self._max_runs:
649  return False
650  return True
