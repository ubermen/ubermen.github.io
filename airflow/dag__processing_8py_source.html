159 def list_py_file_paths(directory, safe_mode=True):
160  """
161  Traverse a directory and look for Python files.
162 
163  :param directory: the directory to traverse
164  :type directory: unicode
165  :param safe_mode: whether to use a heuristic to determine whether a file
166  contains Airflow DAG definitions
167  :return: a list of paths to Python files in the specified directory
168  :rtype: list[unicode]
169  """
170  file_paths = []
171  if directory is None:
172  return []
173  elif os.path.isfile(directory):
174  return [directory]
175  elif os.path.isdir(directory):
176  patterns = []
177  for root, dirs, files in os.walk(directory, followlinks=True):
178  ignore_file = [f for f in files if f == '.airflowignore']
179  if ignore_file:
180  f = open(os.path.join(root, ignore_file[0]), 'r')
181  patterns += [p for p in f.read().split('\n') if p]
182  f.close()
183  for f in files:
184  try:
185  file_path = os.path.join(root, f)
186  if not os.path.isfile(file_path):
187  continue
188  mod_name, file_ext = os.path.splitext(
189  os.path.split(file_path)[-1])
190  if file_ext != '.py':
191  continue
192  if any([re.findall(p, file_path) for p in patterns]):
193  continue
194 
195  # Heuristic that guesses whether a Python file contains an
196  # Airflow DAG definition.
197  might_contain_dag = True
198  if safe_mode:
199  with open(file_path, 'rb') as f:
200  content = f.read()
201  might_contain_dag = all(
202  [s in content for s in (b'DAG', b'airflow')])
203 
204  if not might_contain_dag:
205  continue
206 
207  file_paths.append(file_path)
208  except Exception:
209  logging.exception("Error while examining %s", f)
210  return file_paths

298 class DagFileProcessorManager(LoggingMixin):
299  """
300  Given a list of DAG definition files, this kicks off several processors
301  in parallel to process them. The parallelism is limited and as the
302  processors finish, more are launched. The files are processed over and
303  over again, but no more often than the specified interval.
304 
305  :type _file_path_queue: list[unicode]
306  :type _processors: dict[unicode, AbstractDagFileProcessor]
307  :type _last_runtime: dict[unicode, float]
308  :type _last_finish_time: dict[unicode, datetime]
309  """
310  def __init__(self,
311  dag_directory,
312  file_paths,
313  parallelism,
314  process_file_interval,
315  child_process_log_directory,
316  max_runs,
317  processor_factory):
318  """
319  :param dag_directory: Directory where DAG definitions are kept. All
320  files in file_paths should be under this directory
321  :type dag_directory: unicode
322  :param file_paths: list of file paths that contain DAG definitions
323  :type file_paths: list[unicode]
324  :param parallelism: maximum number of simultaneous process to run at once
325  :type parallelism: int
326  :param process_file_interval: process a file at most once every this
327  many seconds
328  :type process_file_interval: float
329  :param max_runs: The number of times to parse and schedule each file. -1
330  for unlimited.
331  :type max_runs: int
332  :param child_process_log_directory: Store logs for child processes in
333  this directory
334  :type child_process_log_directory: unicode
335  :type process_file_interval: float
336  :param processor_factory: function that creates processors for DAG
337  definition files. Arguments are (dag_definition_path, log_file_path)
338  :type processor_factory: (unicode, unicode) -> (AbstractDagFileProcessor)
339 
340  """
341  self._file_paths = file_paths
342  self._file_path_queue = []
343  self._parallelism = parallelism
344  self._dag_directory = dag_directory
345  self._max_runs = max_runs
346  self._process_file_interval = process_file_interval
347  self._child_process_log_directory = child_process_log_directory
348  self._processor_factory = processor_factory
349  # Map from file path to the processor
350  self._processors = {}

449  @staticmethod

451  """
452  Return the path elements of a path as an array. E.g. /a/b/c ->
453  ['a', 'b', 'c']

542  def heartbeat(self):
543  """
544  This should be periodically called by the scheduler. This method will
545  kick of new processes to process DAG definition files and read the
546  results from the finished processors.
547 
548  :return: a list of SimpleDags that were produced by processors that
549  have finished since the last time this was called
550  :rtype: list[SimpleDag]
551  """
552  finished_processors = {}
553  """:type : dict[unicode, AbstractDagFileProcessor]"""
554  running_processors = {}
555  """:type : dict[unicode, AbstractDagFileProcessor]"""
556 
557  for file_path, processor in self._processors.items():
558  if processor.done:
559  self.logger.info("Processor for {} finished".format(file_path))
560  now = datetime.now()
561  finished_processors[file_path] = processor
562  self._last_runtime[file_path] = (now -
563  processor.start_time).total_seconds()
564  self._last_finish_time[file_path] = now
565  self._run_count[file_path] += 1
566  else:
567  running_processors[file_path] = processor
568  self._processors = running_processors
569 
570  # Collect all the DAGs that were found in the processed files
571  simple_dags = []
572  for file_path, processor in finished_processors.items():
573  if processor.result is None:
574  self.logger.warn("Processor for {} exited with return code "
575  "{}. See {} for details."
576  .format(processor.file_path,
577  processor.exit_code,
578  processor.log_file))
579  else:
580  for simple_dag in processor.result:
581  simple_dags.append(simple_dag)
582 
583  # Generate more file paths to process if we processed all the files
584  # already.
585  if len(self._file_path_queue) == 0:
586  # If the file path is already being processed, or if a file was
587  # processed recently, wait until the next batch
588  file_paths_in_progress = self._processors.keys()
589  now = datetime.now()
590  file_paths_recently_processed = []
591  for file_path in self._file_paths:
592  last_finish_time = self.get_last_finish_time(file_path)
593  if (last_finish_time is not None and
594  (now - last_finish_time).total_seconds() <
595  self._process_file_interval):
596  file_paths_recently_processed.append(file_path)
597 
598  files_paths_at_run_limit = [file_path
599  for file_path, num_runs in self._run_count.items()
600  if num_runs == self._max_runs]
601 
602  files_paths_to_queue = list(set(self._file_paths) -
603  set(file_paths_in_progress) -
604  set(file_paths_recently_processed) -
605  set(files_paths_at_run_limit))
606 
607  for file_path, processor in self._processors.items():
608  self.logger.debug("File path {} is still being processed (started: {})"
609  .format(processor.file_path,
610  processor.start_time.isoformat()))
611 
612  self.logger.debug("Queuing the following files for processing:\n\t{}"
613  .format("\n\t".join(files_paths_to_queue)))
614 
615  self._file_path_queue.extend(files_paths_to_queue)
616 
617  # Start more processors if we have enough slots and files to process
618  while (self._parallelism - len(self._processors) > 0 and
619  len(self._file_path_queue) > 0):
620  file_path = self._file_path_queue.pop(0)
621  log_file_path = self._get_log_file_path(file_path)
622  processor = self._processor_factory(file_path, log_file_path)
623 
624  processor.start()
625  self.logger.info("Started a process (PID: {}) to generate "
626  "tasks for {} - logging into {}"
627  .format(processor.pid, file_path, log_file_path))
628 
629  self._processors[file_path] = processor
630 
631  self.symlink_latest_log_directory()
632 
633  # Update scheduler heartbeat count.
634  self._run_count[self._heart_beat_key] += 1
635 
636  return simple_dags
637 
638  def max_runs_reached(self):
639  """
640  :return: whether all file paths have been processed max_runs times
641  """
642  if self._max_runs == -1: # Unlimited runs.
643  return False
644  for file_path in self._file_paths:
645  if self._run_count[file_path] != self._max_runs:
646  return False
647  if self._run_count[self._heart_beat_key] < self._max_runs:
648  return False
649  return True
