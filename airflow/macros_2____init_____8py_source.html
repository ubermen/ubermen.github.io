24 def ds_add(ds, days):
25  """
26  Add or subtract days from a YYYY-MM-DD
27 
28  :param ds: anchor date in ``YYYY-MM-DD`` format to add to
29  :type ds: str
30  :param days: number of days to add to the ds, you can use negative values
31  :type days: int
32 
33  >>> ds_add('2015-01-01', 5)
34  '2015-01-06'
35  >>> ds_add('2015-01-06', -5)
36  '2015-01-01'
37  """
38 
39  ds = datetime.strptime(ds, '%Y-%m-%d')
40  if days:
41  ds = ds + timedelta(days)

45 def ds_format(ds, input_format, output_format):
46  """
47  Takes an input string and outputs another string
48  as specified in the output format
49 
50  :param ds: input string which contains a date
51  :type ds: str
52  :param input_format: input string format. E.g. %Y-%m-%d
53  :type input_format: str
54  :param output_format: output string format E.g. %Y-%m-%d
55  :type output_format: str
56 
57  >>> ds_format('2015-01-01', "%Y-%m-%d", "%m-%d-%y")

60  '2015-01-05'
61  """
62  return datetime.strptime(ds, input_format).strftime(output_format)
