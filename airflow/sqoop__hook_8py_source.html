29 class SqoopHook(BaseHook):
30  """
31  This Hook is a wrapper around the sqoop 1 binary. To be able to use te hook
32  it is required that "sqoop" is in the PATH.
33  :param hive_home: (from json) The location of hive-site.xml
34  :type hive_home: str
35  :param job_tracker: (from json) <local|jobtracker:port> specify a job tracker
36  :type job_tracker: str
37  :param namenode: (from json) specify a namenode
38  :type namenode: str
39  :param lib_jars: (from json) specify comma separated jar files to
40  include in the classpath.
41  :type lib_jars: str
42  :param files: (from json) specify comma separated files to be copied
43  to the map reduce cluster
44  :type files: (from json) str
45  :param archives: (from json) specify comma separated archives to be
46  unarchived on the compute machines.
47  :type archives: str
48  """

55  self.files = conn.extra_dejson.get('files', None)

73  def _prepare_command(self, cmd, export=False):
74  if export:
75  connection_cmd = ["sqoop", "export", "--verbose"]
76  else:
77  connection_cmd = ["sqoop", "import", "--verbose"]
78 
79  if self.job_tracker:
80  connection_cmd += ["-jt", self.job_tracker]
81  if self.conn.login:
82  connection_cmd += ["--username", self.conn.login]
83  # todo: put this in a password file
84  if self.conn.password:
85  connection_cmd += ["--password", self.conn.password]
86  if self.lib_jars:
87  connection_cmd += ["-libjars", self.lib_jars]
88  if self.files:
89  connection_cmd += ["-files", self.files]
90  if self.namenode:
91  connection_cmd += ["-fs", self.namenode]
92  if self.archives:
93  connection_cmd += ["-archives", self.archives]
94 
95  connection_cmd += ["--connect", "{}:{}/{}".format(self.conn.host, self.conn.port,
96  self.conn.schema)]
97  connection_cmd += cmd
98 
99  return connection_cmd

128  append=False, type="text", columns=None,
129  num_mappers=None, split_by=None, where=None):
130  """
131  Imports table from remote location to target dir. Arguments are
132  copies of direct sqoop command line arguments
133  :param table: Table to read
134  :param target_dir: HDFS destination dir
135  :param append: Append data to an existing dataset in HDFS
136  :param type: "avro", "sequence", "text" Imports data to into the specified
137  format. Defaults to text.
138  :param columns: <col,col,col…> Columns to import from table
139  :param num_mappers: Use n map tasks to import in parallel
140  :param split_by: Column of the table used to split work units
141  :param where: WHERE clause to use during import
142  """
143  cmd = self._import_cmd(target_dir, append, type,
144  num_mappers, split_by)
145  cmd += ["--table", table]
146  if columns:
147  cmd += ["--columns", columns]
148  if where:
149  cmd += ["--where", where]
150 
151  p = self.Popen(cmd, export=False, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
152  output, stderr = p.communicate()
