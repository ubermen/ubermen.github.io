139  def ghe_team_check(self, username, ghe_token):
140  try:
141  # the response from ghe returns the id of the team as an integer
142  try:
143  allowed_teams = [int(team.strip())
144  for team in
145  get_config_param('allowed_teams').split(',')]
146  except ValueError:
147  # this is to deprecate using the string name for a team
148  raise ValueError('it appears that you are using the string name for a team, '
149  'please use the id number instead')
150 
151  except AirflowConfigException:
152  # No allowed teams defined, let anyone in GHE in.
153  return True
154 
155  # https://developer.github.com/v3/orgs/teams/#list-user-teams
156  resp = self.ghe_oauth.get(self.ghe_api_route('/user/teams'),
157  token=(ghe_token, ''))
158 
159  if not resp or resp.status != 200:
160  raise AuthenticationError(
161  'Bad response from GHE ({0})'.format(
162  resp.status if resp else 'None'))
163 
164  for team in resp.data:
165  # mylons: previously this line used to be if team['slug'] in teams
166  # however, teams are part of organizations. organizations are unique,
167  # but teams are not therefore 'slug' for a team is not necessarily unique.
168  # use id instead
169  if team['id'] in allowed_teams:
170  return True
171 
172  _log.debug('Denying access for user "%s", not a member of "%s"',
173  username,
174  str(allowed_teams))
175 
176  return False

178  def load_user(self, userid):

190  def oauth_callback(self):
191  _log.debug('GHE OAuth callback called')
192 
193  next_url = request.args.get('next') or url_for('admin.index')
194 
195  resp = self.ghe_oauth.authorized_response()
196 
197  try:
198  if resp is None:
199  raise AuthenticationError(
200  'Null response from GHE, denying access.'
201  )
202 
203  ghe_token = resp['access_token']
204 
205  username, email = self.get_ghe_user_profile_info(ghe_token)
206 
207  if not self.ghe_team_check(username, ghe_token):
208  return redirect(url_for('airflow.noaccess'))
209 
210  except AuthenticationError:
211  _log.exception('')
212  return redirect(url_for('airflow.noaccess'))
