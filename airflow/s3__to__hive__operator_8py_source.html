32 class S3ToHiveTransfer(BaseOperator):
33  """
34  Moves data from S3 to Hive. The operator downloads a file from S3,
35  stores the file locally before loading it into a Hive table.
36  If the ``create`` or ``recreate`` arguments are set to ``True``,
37  a ``CREATE TABLE`` and ``DROP TABLE`` statements are generated.
38  Hive data types are inferred from the cursor's metadata from.
39 
40  Note that the table generated in Hive uses ``STORED AS textfile``
41  which isn't the most efficient serialization format. If a
42  large amount of data is loaded and/or if the tables gets
43  queried considerably, you may want to use this operator only to
44  stage the data into a temporary table before loading it into its
45  final destination using a ``HiveOperator``.
46 
47  :param s3_key: The key to be retrieved from S3
48  :type s3_key: str
49  :param field_dict: A dictionary of the fields name in the file
50  as keys and their Hive types as values
51  :type field_dict: dict
52  :param hive_table: target Hive table, use dot notation to target a
53  specific database
54  :type hive_table: str
55  :param create: whether to create the table if it doesn't exist
56  :type create: bool
57  :param recreate: whether to drop and recreate the table at every
58  execution
59  :type recreate: bool
60  :param partition: target partition as a dict of partition columns
61  and values
62  :type partition: dict
63  :param headers: whether the file contains column names on the first
64  line
65  :type headers: bool
66  :param check_headers: whether the column names on the first line should be
67  checked against the keys of field_dict
68  :type check_headers: bool
69  :param wildcard_match: whether the s3_key should be interpreted as a Unix
70  wildcard pattern
71  :type wildcard_match: bool
72  :param delimiter: field delimiter in the file
73  :type delimiter: str
74  :param s3_conn_id: source s3 connection
75  :type s3_conn_id: str
76  :param hive_cli_conn_id: destination hive connection
77  :type hive_cli_conn_id: str
78  :param input_compressed: Boolean to determine if file decompression is
79  required to process headers
80  :type input_compressed: bool
81  """

198  def _get_top_row_as_list(self, file_name):
199  with open(file_name, 'rt') as f:

204  def _match_headers(self, header_list):
205  if not header_list:
206  raise AirflowException("Unable to retrieve header row from file")
207  field_names = self.field_dict.keys()
208  if len(field_names) != len(header_list):
209  logging.warning("Headers count mismatch"
210  "File headers:\n {header_list}\n"
211  "Field names: \n {field_names}\n"
212  "".format(**locals()))
213  return False
214  test_field_match = [h1.lower() == h2.lower()
215  for h1, h2 in zip(header_list, field_names)]
216  if not all(test_field_match):
217  logging.warning("Headers do not match field names"
218  "File headers:\n {header_list}\n"
219  "Field names: \n {field_names}\n"
220  "".format(**locals()))
221  return False
222  else:
223  return True

225  def _delete_top_row_and_compress(
226  self,
227  input_file_name,
228  output_file_ext,
229  dest_dir):
230  # When output_file_ext is not defined, file is not compressed
231  open_fn = open
232  if output_file_ext.lower() == '.gz':
233  open_fn = gzip.GzipFile
234  elif output_file_ext.lower() == '.bz2':
235  open_fn = bz2.BZ2File
236 
237  os_fh_output, fn_output = \
238  tempfile.mkstemp(suffix=output_file_ext, dir=dest_dir)
239  with open(input_file_name, 'rb') as f_in,\
240  open_fn(fn_output, 'wb') as f_out:
241  f_in.seek(0)
242  next(f_in)
243  for line in f_in:
244  f_out.write(line)
245  return fn_output
