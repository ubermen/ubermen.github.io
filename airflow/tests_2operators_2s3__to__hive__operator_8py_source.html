135  def _check_file_equality(self, fn_1, fn_2, ext):
136  # gz files contain mtime and filename in the header that
137  # causes filecmp to return False even if contents are identical
138  # Hence decompress to test for equality
139  if(ext == '.gz'):
140  with gzip.GzipFile(fn_1, 'rb') as f_1,\
141  NamedTemporaryFile(mode='wb') as f_txt_1,\
142  gzip.GzipFile(fn_2, 'rb') as f_2,\
143  NamedTemporaryFile(mode='wb') as f_txt_2:
144  shutil.copyfileobj(f_1, f_txt_1)
145  shutil.copyfileobj(f_2, f_txt_2)
146  f_txt_1.flush()
147  f_txt_2.flush()
148  return filecmp.cmp(f_txt_1.name, f_txt_2.name, shallow=False)
149  else:
150  return filecmp.cmp(fn_1, fn_2, shallow=False)
