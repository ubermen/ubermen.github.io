58 def expand_env_var(env_var):
59  """
60  Expands (potentially nested) env vars by repeatedly applying
61  `expandvars` and `expanduser` until interpolation stops having
62  any effect.
63  """
64  if not env_var:
65  return env_var
66  while True:
67  interpolated = os.path.expanduser(os.path.expandvars(str(env_var)))
68  if interpolated == env_var:
69  return interpolated
70  else:
71  env_var = interpolated

74 def run_command(command):
75  """
76  Runs command and returns stdout
77  """
78  process = subprocess.Popen(
79  command.split(), stdout=subprocess.PIPE, stderr=subprocess.PIPE)
80  output, stderr = process.communicate()
81 
82  if process.returncode != 0:
83  raise AirflowConfigException(
84  "Cannot execute {}. Error code is: {}. Output: {}, Stderr: {}"
85  .format(command, process.returncode, output, stderr)
86  )
87 
88  return output

542  def _validate(self):
543  if (
544  self.get("core", "executor") != 'SequentialExecutor' and
545  "sqlite" in self.get('core', 'sql_alchemy_conn')):

570  def _get_env_var_option(self, section, key):
571  # must have format AIRFLOW__{SECTION}__{KEY} (note double underscore)
572  env_var = 'AIRFLOW__{S}__{K}'.format(S=section.upper(), K=key.upper())
573  if env_var in os.environ:
574  return expand_env_var(os.environ[env_var])
575 
576  def _get_cmd_option(self, section, key):
577  fallback_key = key + '_cmd'
578  # if this is a valid command key...
579  if (section, key) in AirflowConfigParser.as_command_stdout:
580  # if the original key is present, return it no matter what
581  if self.has_option(section, key):
582  return ConfigParser.get(self, section, key)
583  # otherwise, execute the fallback key
584  elif self.has_option(section, fallback_key):
585  command = self.get(section, fallback_key)
586  return run_command(command)
587 
588  def get(self, section, key, **kwargs):
589  section = str(section).lower()
590  key = str(key).lower()
591 
592  # first check environment variables
593  option = self._get_env_var_option(section, key)
594  if option is not None:
595  return option
596 
597  # ...then the config file
598  if self.has_option(section, key):
599  return expand_env_var(
600  ConfigParser.get(self, section, key, **kwargs))
601 
602  # ...then commands
603  option = self._get_cmd_option(section, key)
604  if option:
605  return option
606 
607  else:
608  logging.warning("section/key [{section}/{key}] not found "
609  "in config".format(**locals()))
610 
611  raise AirflowConfigException(
612  "section/key [{section}/{key}] not found "
613  "in config".format(**locals()))
614 
615  def getboolean(self, section, key):
616  val = str(self.get(section, key)).lower().strip()
617  if '#' in val:
618  val = val.split('#')[0].strip()
619  if val.lower() in ('t', 'true', '1'):
620  return True
621  elif val.lower() in ('f', 'false', '0'):
622  return False
623  else:
624  raise AirflowConfigException(
625  'The value for configuration option "{}:{}" is not a '
626  'boolean (received "{}").'.format(section, key, val))
627 
628  def getint(self, section, key):
629  return int(self.get(section, key))
630 
631  def getfloat(self, section, key):
632  return float(self.get(section, key))

638  def as_dict(self, display_source=False, display_sensitive=False):
639  """
640  Returns the current configuration as an OrderedDict of OrderedDicts.
641  :param display_source: If False, the option value is returned. If True,
642  a tuple of (option_value, source) is returned. Source is either
643  'airflow.cfg' or 'default'.
644  :type display_source: bool
645  :param display_sensitive: If True, the values of options set by env
646  vars and bash commands will be displayed. If False, those options
647  are shown as '< hidden >'
648  :type display_sensitive: bool
649  """
650  cfg = copy.deepcopy(self._sections)
651 
652  # remove __name__ (affects Python 2 only)
653  for options in cfg.values():
654  options.pop('__name__', None)
655 
656  # add source
657  if display_source:
658  for section in cfg:
659  for k, v in cfg[section].items():
660  cfg[section][k] = (v, 'airflow config')
661 
662  # add env vars and overwrite because they have priority
663  for ev in [ev for ev in os.environ if ev.startswith('AIRFLOW__')]:
664  try:
665  _, section, key = ev.split('__')
666  opt = self._get_env_var_option(section, key)
667  except ValueError:
668  opt = None
669  if opt:
670  if (
671  not display_sensitive
672  and ev != 'AIRFLOW__CORE__UNIT_TEST_MODE'):
673  opt = '< hidden >'
674  if display_source:
675  opt = (opt, 'env var')
676  cfg.setdefault(section.lower(), OrderedDict()).update(
677  {key.lower(): opt})
678 
679  # add bash commands
680  for (section, key) in AirflowConfigParser.as_command_stdout:
681  opt = self._get_cmd_option(section, key)
682  if opt:
683  if not display_sensitive:
684  opt = '< hidden >'
685  if display_source:
686  opt = (opt, 'bash cmd')
687  cfg.setdefault(section, OrderedDict()).update({key: opt})
688 
689  return cfg
