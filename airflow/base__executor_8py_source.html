95  def heartbeat(self):
96 
97  # Triggering new jobs
98  if not self.parallelism:
99  open_slots = len(self.queued_tasks)
100  else:
101  open_slots = self.parallelism - len(self.running)
102 
103  self.logger.debug("{} running task instances".format(len(self.running)))
104  self.logger.debug("{} in queue".format(len(self.queued_tasks)))
105  self.logger.debug("{} open slots".format(open_slots))
106 
107  sorted_queue = sorted(
108  [(k, v) for k, v in self.queued_tasks.items()],
109  key=lambda x: x[1][1],
110  reverse=True)
111  for i in range(min((open_slots, len(self.queued_tasks)))):
112  key, (command, _, queue, ti) = sorted_queue.pop(0)
113  # TODO(jlowin) without a way to know what Job ran which tasks,
114  # there is a danger that another Job started running a task
115  # that was also queued to this executor. This is the last chance
116  # to check if that happened. The most probable way is that a
117  # Scheduler tried to run a task that was originally queued by a
118  # Backfill. This fix reduces the probability of a collision but
119  # does NOT eliminate it.
120  self.queued_tasks.pop(key)
121  ti.refresh_from_db()
122  if ti.state != State.RUNNING:
123  self.running[key] = command
124  self.execute_async(key, command=command, queue=queue)
125  else:
126  self.logger.debug(
127  'Task is already running, not sending to '
128  'executor: {}'.format(key))
129 
130  # Calling child class sync method
131  self.logger.debug("Calling the {} sync method".format(self.__class__))
132  self.sync()
