24 class SparkSubmitOperator(BaseOperator):
25  """
26  This hook is a wrapper around the spark-submit binary to kick off a spark-submit job.
27  It requires that the "spark-submit" binary is in the PATH.
28  :param application: The application that submitted as a job, either jar or py file.
29  :type application: str
30  :param conf: Arbitrary Spark configuration properties
31  :type conf: dict
32  :param conn_id: The connection id as configured in Airflow administration. When an
33  invalid connection_id is supplied, it will default to yarn.
34  :type conn_id: str
35  :param files: Upload additional files to the container running the job, separated by a
36  comma. For example hive-site.xml.
37  :type files: str
38  :param py_files: Additional python files used by the job, can be .zip, .egg or .py.
39  :type py_files: str
40  :param jars: Submit additional jars to upload and place them in executor classpath.
41  :type jars: str
42  :param executor_cores: Number of cores per executor (Default: 2)
43  :type executor_cores: int
44  :param executor_memory: Memory per executor (e.g. 1000M, 2G) (Default: 1G)
45  :type executor_memory: str
46  :param keytab: Full path to the file that contains the keytab
47  :type keytab: str
48  :param principal: The name of the kerberos principal used for keytab
49  :type principal: str
50  :param name: Name of the job (default airflow-spark)
51  :type name: str
52  :param num_executors: Number of executors to launch
53  :type num_executors: int
54  :param verbose: Whether to pass the verbose flag to spark-submit process for debugging
55  :type verbose: bool
56  """
