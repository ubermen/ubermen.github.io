23 class PrestoToMySqlTransfer(BaseOperator):
24  """
25  Moves data from Presto to MySQL, note that for now the data is loaded
26  into memory before being pushed to MySQL, so this operator should
27  be used for smallish amount of data.
28 
29  :param sql: SQL query to execute against the MySQL database
30  :type sql: str
31  :param mysql_table: target MySQL table, use dot notation to target a
32  specific database
33  :type mysql_table: str
34  :param mysql_conn_id: source mysql connection
35  :type mysql_conn_id: str
36  :param presto_conn_id: source presto connection
37  :type presto_conn_id: str
38  :param mysql_preoperator: sql statement to run against mysql prior to
39  import, typically use to truncate of delete in place of the data
40  coming in, allowing the task to be idempotent (running the task
41  twice won't double load data)
42  :type mysql_preoperator: str
43  """

67  logging.info("Extracting data from Presto")

77  logging.info("Inserting rows into MySQL")
