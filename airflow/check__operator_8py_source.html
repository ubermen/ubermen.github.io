25 class CheckOperator(BaseOperator):
26  """
27  Performs checks against a db. The ``CheckOperator`` expects
28  a sql query that will return a single row. Each value on that
29  first row is evaluated using python ``bool`` casting. If any of the
30  values return ``False`` the check is failed and errors out.
31 
32  Note that Python bool casting evals the following as ``False``:
33 
34  * ``False``
35  * ``0``
36  * Empty string (``""``)
37  * Empty list (``[]``)
38  * Empty dictionary or set (``{}``)
39 
40  Given a query like ``SELECT COUNT(*) FROM foo``, it will fail only if
41  the count ``== 0``. You can craft much more complex query that could,
42  for instance, check that the table has the same number of rows as
43  the source table upstream, or that the count of today's partition is
44  greater than yesterday's partition, or that a set of metrics are less
45  than 3 standard deviation for the 7 day average.
46 
47  This operator can be used as a data quality check in your pipeline, and
48  depending on where you put it in your DAG, you have the choice to
49  stop the critical path, preventing from
50  publishing dubious data, or on the side and receive email alerts
51  without stopping the progress of the DAG.
52 
53  Note that this is an abstract class and get_db_hook
54  needs to be defined. Whereas a get_db_hook is hook that gets a
55  single record from an external source.
56 
57  :param sql: the sql to be executed
58  :type sql: string
59  """
60 
61  template_fields = ('sql',)
62  template_ext = ('.hql', '.sql',)
63  ui_color = '#fff7e6'
64

66  def __init__(
67  self, sql,
68  conn_id=None,
69  *args, **kwargs):
70  super(CheckOperator, self).__init__(*args, **kwargs)
71  self.conn_id = conn_id
72  self.sql = sql
73 
74  def execute(self, context=None):
75  logging.info('Executing SQL check: ' + self.sql)
76  records = self.get_db_hook().get_first(self.sql)
77  logging.info("Record: " + str(records))
78  if not records:
79  raise AirflowException("The query returned None")
80  elif not all([bool(r) for r in records]):
81  exceptstr = "Test failed.\nQuery:\n{q}\nResults:\n{r!s}"
82  raise AirflowException(exceptstr.format(q=self.sql, r=records))
83  logging.info("Success.")

86  return BaseHook.get_hook(conn_id=self.conn_id)

139  records = self.get_db_hook().get_first(self.sql)
140  if not records:
141  raise AirflowException("The query returned None")
142  test_results = []
143  except_temp = ("Test failed.\nPass value:{self.pass_value}\n"
144  "Query:\n{self.sql}\nResults:\n{records!s}")
145  if not self.is_numeric_value_check:
146  tests = [str(r) == self.pass_value for r in records]
147  elif self.is_numeric_value_check:
148  try:
149  num_rec = [float(r) for r in records]
150  except (ValueError, TypeError) as e:
151  cvestr = "Converting a result to float failed.\n"
152  raise AirflowException(cvestr+except_temp.format(**locals()))
153  if self.has_tolerance:
154  tests = [
155  r / (1 + self.tol) <= self.pass_value <= r / (1 - self.tol)
156  for r in num_rec]
157  else:
158  tests = [r == self.pass_value for r in num_rec]
159  if not all(tests):
160  raise AirflowException(except_temp.format(**locals()))

166 class IntervalCheckOperator(BaseOperator):
167  """
168  Checks that the values of metrics given as SQL expressions are within
169  a certain tolerance of the ones from days_back before.
170 
171  Note that this is an abstract class and get_db_hook
172  needs to be defined. Whereas a get_db_hook is hook that gets a
173  single record from an external source.
174 
175  :param table: the table name
176  :type table: str
177  :param days_back: number of days between ds and the ds we want to check
178  against. Defaults to 7 days
179  :type days_back: int
180  :param metrics_threshold: a dictionary of ratios indexed by metrics
181  :type metrics_threshold: dict
182  """
183 
184  __mapper_args__ = {
185  'polymorphic_identity': 'IntervalCheckOperator'
186  }

188  template_ext = ('.hql', '.sql',)
189  ui_color = '#fff7e6'

191  @apply_defaults
192  def __init__(
193  self, table, metrics_thresholds,
194  date_filter_column='ds', days_back=-7,
195  conn_id=None,
196  *args, **kwargs):
197  super(IntervalCheckOperator, self).__init__(*args, **kwargs)
198  self.table = table
199  self.metrics_thresholds = metrics_thresholds
200  self.metrics_sorted = sorted(metrics_thresholds.keys())
201  self.date_filter_column = date_filter_column
202  self.days_back = -abs(days_back)
203  self.conn_id = conn_id
204  sqlexp = ', '.join(self.metrics_sorted)
205  sqlt = ("SELECT {sqlexp} FROM {table}"
206  " WHERE {date_filter_column}=").format(**locals())
207  self.sql1 = sqlt + "'{{ ds }}'"
208  self.sql2 = sqlt + "'{{ macros.ds_add(ds, "+str(self.days_back)+") }}'"

210  def execute(self, context=None):
211  hook = self.get_db_hook()
212  logging.info('Executing SQL check: ' + self.sql2)
213  row2 = hook.get_first(self.sql2)
214  logging.info('Executing SQL check: ' + self.sql1)
215  row1 = hook.get_first(self.sql1)
216  if not row2:
217  raise AirflowException("The query {q} returned None").format(q=self.sql2)
218  if not row1:
219  raise AirflowException("The query {q} returned None").format(q=self.sql1)
220  current = dict(zip(self.metrics_sorted, row1))
221  reference = dict(zip(self.metrics_sorted, row2))
222  ratios = {}
223  test_results = {}
224  rlog = "Ratio for {0}: {1} \n Ratio threshold : {2}"
225  fstr = "'{k}' check failed. {r} is above {tr}"
226  estr = "The following tests have failed:\n {0}"
227  countstr = "The following {j} tests out of {n} failed:"
228  for m in self.metrics_sorted:
229  if current[m] == 0 or reference[m] == 0:
230  ratio = None
231  else:
232  ratio = float(max(current[m], reference[m])) / \
233  min(current[m], reference[m])
234  logging.info(rlog.format(m, ratio, self.metrics_thresholds[m]))
235  ratios[m] = ratio
236  test_results[m] = ratio < self.metrics_thresholds[m]
237  if not all(test_results.values()):
238  failed_tests = [it[0] for it in test_results.items() if not it[1]]
239  j = len(failed_tests)
240  n = len(self.metrics_sorted)
241  logging.warning(countstr.format(**locals()))
242  for k in failed_tests:
243  logging.warning(fstr.format(k=k, r=ratios[k],
244  tr=self.metrics_thresholds[k]))
245  raise AirflowException(estr.format(", ".join(failed_tests)))
246  logging.info("All tests have passed")
