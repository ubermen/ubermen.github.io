29 class BashOperator(BaseOperator):
30  """
31  Execute a Bash script, command or set of commands.
32 
33  :param bash_command: The command, set of commands or reference to a
34  bash script (must be '.sh') to be executed.
35  :type bash_command: string
36  :param xcom_push: If xcom_push is True, the last line written to stdout
37  will also be pushed to an XCom when the bash command completes.
38  :type xcom_push: bool
39  :param env: If env is not None, it must be a mapping that defines the
40  environment variables for the new process; these are used instead

43  :type env: dict
44  :type output_encoding: output encoding of bash command

65  def execute(self, context):
66  """
67  Execute the bash command in a temporary directory
68  which will be cleaned afterwards
69  """
70  bash_command = self.bash_command
71  logging.info("tmp dir root location: \n" + gettempdir())
72  with TemporaryDirectory(prefix='airflowtmp') as tmp_dir:
73  with NamedTemporaryFile(dir=tmp_dir, prefix=self.task_id) as f:
74 
75  f.write(bytes(bash_command, 'utf_8'))
76  f.flush()
77  fname = f.name
78  script_location = tmp_dir + "/" + fname
79  logging.info("Temporary script "
80  "location :{0}".format(script_location))
81  logging.info("Running command: " + bash_command)
82  sp = Popen(
83  ['bash', fname],
84  stdout=PIPE, stderr=STDOUT,
85  cwd=tmp_dir, env=self.env,
86  preexec_fn=os.setsid)
87 
88  self.sp = sp
89 
90  logging.info("Output:")
91  line = ''
92  for line in iter(sp.stdout.readline, b''):
93  line = line.decode(self.output_encoding).strip()
94  logging.info(line)
95  sp.wait()
96  logging.info("Command exited with "
97  "return code {0}".format(sp.returncode))
98 
99  if sp.returncode:
100  raise AirflowException("Bash command failed")
101 
102  if self.xcom_push_flag:
103  return line
