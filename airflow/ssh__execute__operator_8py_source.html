25 class SSHTempFileContent(object):
26  """This class prvides a functionality that creates tempfile
27  with given content at remote host.
28  Use like::
29 
30  with SSHTempFileContent(ssh_hook, content) as tempfile:
31  ...
32 
33  In this case, a temporary file ``tempfile``
34  with content ``content`` is created where ``ssh_hook`` designate.
35 
36  Note that this isn't safe because other processes
37  at remote host can read and write that tempfile.
38 
39  :param ssh_hook: A SSHHook that indicates a remote host
40  where you want to create tempfile
41  :param content: Initial content of creating temporary file
42  :type content: string
43  :param prefix: The prefix string you want to use for the temporary file
44  :type prefix: string
45  """
46 
47  def __init__(self, ssh_hook, content, prefix="tmp"):
48  self._ssh_hook = ssh_hook
49  self._content = content
50  self._prefix = prefix
51 
52  def __enter__(self):
53  ssh_hook = self._ssh_hook
54  string = self._content
55  prefix = self._prefix
56 
57  pmktemp = ssh_hook.Popen(["-q",
58  "mktemp", "-t", prefix + "_XXXXXX"],
59  stdout=subprocess.PIPE,
60  stderr=STDOUT)
61  tempfile = pmktemp.communicate()[0].rstrip()
62  pmktemp.wait()
63  if pmktemp.returncode:
64  raise AirflowException("Failed to create remote temp file")
65 
66  ptee = ssh_hook.Popen(["-q", "tee", tempfile],
67  stdin=subprocess.PIPE,
68  # discard stdout
69  stderr=STDOUT)
70  ptee.stdin.write(bytes(string, 'utf_8'))
71  ptee.stdin.close()
72  ptee.wait()
73  if ptee.returncode:
74  raise AirflowException("Failed to write to remote temp file")
75 
76  self._tempfile = tempfile
77  return tempfile
78 
79  def __exit__(self, type, value, traceback):
80  sp = self._ssh_hook.Popen(["-q", "rm", "-f", "--", self._tempfile])
81  sp.communicate()
82  sp.wait()
83  if sp.returncode:
84  raise AirflowException("Failed to remove to remote temp file")
85  return False
