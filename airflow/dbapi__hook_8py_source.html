185  def insert_rows(self, table, rows, target_fields=None, commit_every=1000):
186  """
187  A generic way to insert a set of tuples into a table,
188  the whole set of inserts is treated as one transaction
189 
190  :param table: Name of the target table
191  :type table: str
192  :param rows: The rows to insert into the table
193  :type rows: iterable of tuples
194  :param target_fields: The names of the columns to fill in the table
195  :type target_fields: iterable of strings
196  :param commit_every: The maximum number of rows to insert in one
197  transaction. Set to 0 to insert all rows in one transaction.
198  :type commit_every: int
199  """
200  if target_fields:
201  target_fields = ", ".join(target_fields)
202  target_fields = "({})".format(target_fields)
203  else:
204  target_fields = ''
205  conn = self.get_conn()
206  if self.supports_autocommit:
207  self.set_autocommit(conn, False)
208  conn.commit()
209  cur = conn.cursor()
210  i = 0
211  for row in rows:
212  i += 1
213  l = []
214  for cell in row:
215  l.append(self._serialize_cell(cell, conn))
216  values = tuple(l)
217  sql = "INSERT INTO {0} {1} VALUES ({2});".format(
218  table,
219  target_fields,
220  ",".join(values))
221  cur.execute(sql)
222  if commit_every and i % commit_every == 0:
223  conn.commit()
224  logging.info(
225  "Loaded {i} into {table} rows so far".format(**locals()))
226  conn.commit()
227  cur.close()
228  conn.close()
229  logging.info(
230  "Done loading. Loaded a total of {i} rows".format(**locals()))

234  """
235  Returns the SQL literal of the cell as a string.
236 
237  :param cell: The cell to insert into the table
238  :type cell: object
239  :param conn: The database connection

242  :rtype: str
243  """
244 
245  if isinstance(cell, basestring):
246  return "'" + str(cell).replace("'", "''") + "'"
247  elif cell is None:
