28  Simple wrapper around the pig CLI.

30  Note that you can also set default pig CLI properties using the

43  def run_cli(self, pig, verbose=True):
44  """
45  Run an pig script using the pig cli
46 
47  >>> ph = PigCliHook()
48  >>> result = ph.run_cli("ls /;")
49  >>> ("hdfs://" in result)
50  True
51  """
52 
53  with TemporaryDirectory(prefix='airflow_pigop_') as tmp_dir:
54  with NamedTemporaryFile(dir=tmp_dir) as f:
55  f.write(pig)
56  f.flush()
57  fname = f.name
58  pig_bin = 'pig'
59  cmd_extra = []
60 
61  pig_cmd = [pig_bin, '-f', fname] + cmd_extra
62 
63  if self.pig_properties:
64  pig_properties_list = self.pig_properties.split()
65  pig_cmd.extend(pig_properties_list)
66  if verbose:
67  logging.info(" ".join(pig_cmd))
68  sp = subprocess.Popen(
69  pig_cmd,
70  stdout=subprocess.PIPE,
71  stderr=subprocess.STDOUT,
72  cwd=tmp_dir)
73  self.sp = sp
74  stdout = ''
75  for line in iter(sp.stdout.readline, ''):
76  stdout += line
77  if verbose:
78  logging.info(line.strip())
79  sp.wait()
80 
81  if sp.returncode:
82  raise AirflowException(stdout)
83 
84  return stdout
