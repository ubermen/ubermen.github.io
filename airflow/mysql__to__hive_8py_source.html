28 class MySqlToHiveTransfer(BaseOperator):
29  """
30  Moves data from MySql to Hive. The operator runs your query against
31  MySQL, stores the file locally before loading it into a Hive table.
32  If the ``create`` or ``recreate`` arguments are set to ``True``,
33  a ``CREATE TABLE`` and ``DROP TABLE`` statements are generated.
34  Hive data types are inferred from the cursor's metadata. Note that the
35  table generated in Hive uses ``STORED AS textfile``
36  which isn't the most efficient serialization format. If a
37  large amount of data is loaded and/or if the table gets
38  queried considerably, you may want to use this operator only to
39  stage the data into a temporary table before loading it into its
40  final destination using a ``HiveOperator``.
41 
42  :param sql: SQL query to execute against the MySQL database
43  :type sql: str
44  :param hive_table: target Hive table, use dot notation to target a
45  specific database
46  :type hive_table: str
47  :param create: whether to create the table if it doesn't exist
48  :type create: bool
49  :param recreate: whether to drop and recreate the table at every
50  execution
51  :type recreate: bool
52  :param partition: target partition as a dict of partition columns
53  and values
54  :type partition: dict
55  :param delimiter: field delimiter in the file
56  :type delimiter: str
57  :param mysql_conn_id: source mysql connection
58  :type mysql_conn_id: str
59  :param hive_conn_id: destination hive connection
60  :type hive_conn_id: str
61  """

108  mysql = MySqlHook(mysql_conn_id=self.mysql_conn_id)
109 
110  logging.info("Dumping MySQL query results to local file")
111  conn = mysql.get_conn()
