25 class S3FileTransformOperator(BaseOperator):
26  """
27  Copies data from a source S3 location to a temporary location on the
28  local filesystem. Runs a transformation on this file as specified by
29  the transformation script and uploads the output to a destination S3
30  location.
31 
32  The locations of the source and the destination files in the local
33  filesystem is provided as an first and second arguments to the
34  transformation script. The transformation script is expected to read the
35  data from source , transform it and write the output to the local
36  destination file. The operator then takes over control and uploads the
37  local destination file to S3.
38 
39  :param source_s3_key: The key to be retrieved from S3
40  :type source_s3_key: str
41  :param source_s3_conn_id: source s3 connection
42  :type source_s3_conn_id: str
43  :param dest_s3_key: The key to be written from S3
44  :type dest_s3_key: str
45  :param dest_s3_conn_id: destination s3 connection
46  :type dest_s3_conn_id: str
47  :param replace: Replace dest S3 key if it already exists
48  :type replace: bool
49  :param transform_script: location of the executable transformation script
50  :type transform_script: str
51  """

75  def execute(self, context):
76  source_s3 = S3Hook(s3_conn_id=self.source_s3_conn_id)
77  dest_s3 = S3Hook(s3_conn_id=self.dest_s3_conn_id)
78  logging.info("Downloading source S3 file {0}"
79  "".format(self.source_s3_key))
80  if not source_s3.check_for_key(self.source_s3_key):
81  raise AirflowException("The source key {0} does not exist"
82  "".format(self.source_s3_key))
83  source_s3_key_object = source_s3.get_key(self.source_s3_key)
84  with NamedTemporaryFile("w") as f_source, NamedTemporaryFile("w") as f_dest:
85  logging.info("Dumping S3 file {0} contents to local file {1}"
86  "".format(self.source_s3_key, f_source.name))
87  source_s3_key_object.get_contents_to_file(f_source)
88  f_source.flush()
89  source_s3.connection.close()
90  transform_script_process = subprocess.Popen(
91  [self.transform_script, f_source.name, f_dest.name],
92  stdout=subprocess.PIPE, stderr=subprocess.PIPE)
93  (transform_script_stdoutdata, transform_script_stderrdata) = transform_script_process.communicate()
94  logging.info("Transform script stdout "
95  "" + transform_script_stdoutdata)
96  if transform_script_process.returncode > 0:
97  raise AirflowException("Transform script failed "
98  "" + transform_script_stderrdata)
99  else:
100  logging.info("Transform script successful."
101  "Output temporarily located at {0}"
102  "".format(f_dest.name))
103  logging.info("Uploading transformed file to S3")
104  f_dest.flush()
105  dest_s3.load_file(
106  filename=f_dest.name,
107  key=self.dest_s3_key,
108  replace=self.replace
109  )
110  logging.info("Upload successful")
