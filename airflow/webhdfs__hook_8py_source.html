35 class WebHDFSHook(BaseHook):
36  """
37  Interact with HDFS. This class is a wrapper around the hdfscli library.
38  """
39  def __init__(self, webhdfs_conn_id='webhdfs_default', proxy_user=None):
40  self.webhdfs_conn_id = webhdfs_conn_id
41  self.proxy_user = proxy_user
42 
43  def get_conn(self):
44  """
45  Returns a hdfscli InsecureClient object.
46  """
47  nn_connections = self.get_connections(self.webhdfs_conn_id)
48  for nn in nn_connections:
49  try:
50  logging.debug('Trying namenode {}'.format(nn.host))
51  connection_str = 'http://{nn.host}:{nn.port}'.format(nn=nn)
52  if _kerberos_security_mode:
53  client = KerberosClient(connection_str)
54  else:
55  proxy_user = self.proxy_user or nn.login
56  client = InsecureClient(connection_str, user=proxy_user)
57  client.status('/')
58  logging.debug('Using namenode {} for hook'.format(nn.host))
59  return client
60  except HdfsError as e:
61  logging.debug("Read operation on namenode {nn.host} failed with"
62  " error: {e.message}".format(**locals()))
63  nn_hosts = [c.host for c in nn_connections]
64  no_nn_error = "Read operations failed on the namenodes below:\n{}".format("\n".join(nn_hosts))
65  raise AirflowWebHDFSHookException(no_nn_error)

76  """
77  Uploads a file to HDFS

79  :param source: Local path to file or folder. If a folder, all the files
80  inside of it will be uploaded (note that this implies that folders empty
81  of files will not be created remotely).
82  :type source: str

84  directory, files will be uploaded inside.
85  :type destination: str
86  :param overwrite: Overwrite any existing file or directory.
87  :type overwrite: bool
88  :param parallelism: Number of threads to use for parallelization. A value of

90  :type parallelism: int

92 
93 
94  """
