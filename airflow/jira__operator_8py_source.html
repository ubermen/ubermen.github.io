23 class JiraOperator(BaseOperator):
24  """
25  JiraOperator to interact and perform action on Jira issue tracking system.
26  This operator is designed to use Jira Python SDK: http://jira.readthedocs.io
27 
28  :param jira_conn_id: reference to a pre-defined Jira Connection
29  :type jira_conn_id: str
30  :param jira_method: method name from Jira Python SDK to be called
31  :type jira_method: str
32  :param jira_method_args: required method parameters for the jira_method
33  :type jira_method_args: dict
34  :param result_processor: function to further process the response from Jira
35  :type result_processor: function
36  :param get_jira_resource_method: function or operator to get jira resource
37  on which the provided jira_method will be executed
38  :type get_jira_resource_method: function
39  """

56  self.result_processor = result_processor

59  def execute(self, context):
60  try:
61  if self.get_jira_resource_method is not None:
62  # if get_jira_resource_method is provided, jira_method will be executed on
63  # resource returned by executing the get_jira_resource_method.
64  # This makes all the provided methods of JIRA sdk accessible and usable
65  # directly at the JiraOperator without additional wrappers.
66  # ref: http://jira.readthedocs.io/en/latest/api.html
67  if isinstance(self.get_jira_resource_method, JiraOperator):
68  resource = self.get_jira_resource_method.execute(**context)
69  else:
70  resource = self.get_jira_resource_method(**context)
71  else:
72  # Default method execution is on the top level jira client resource
73  hook = JiraHook(jira_conn_id=self.jira_conn_id)
74  resource = hook.client
75 
76  # Current Jira-Python SDK (1.0.7) has issue with pickling the jira response.
77  # ex: self.xcom_push(context, key='operator_response', value=jira_response)
78  # This could potentially throw error if jira_result is not picklable
79  jira_result = getattr(resource, self.method_name)(**self.jira_method_args)
80  if self.result_processor:
81  return self.result_processor(context, jira_result)
82 
83  return jira_result
84 
85  except JIRAError as jira_error:
86  raise AirflowException("Failed to execute jiraOperator, error: %s"
87  % str(jira_error))
88  except Exception as e:
89  raise AirflowException("Jira operator error: %s" % str(e))
