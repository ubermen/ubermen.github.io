41  """
42  Create new connection to Salesforce
43  and allows you to pull data out of SFDC and save it to a file.
44 
45  You can then use that file with other
46  Airflow operators to move the data into another data source
47 
48  :param conn_id: the name of the connection that has the parameters
49  we need to connect to Salesforce.

51  user's security token in the `Extras` field.
52  .. note::
53  For the HTTP connection type, you can include a
54  JSON structure in the `Extras` field.
55  We need a user's security token to connect to Salesforce.
56  So we define it in the `Extras` field as:
57  `{"security_token":"YOUR_SECRUITY_TOKEN"}`
58  """

67  def sign_in(self):
68  """
69  Sign into Salesforce.
70 
71  If we have already signed it, this will just return the original object
72  """
73  if hasattr(self, 'sf'):

155  @classmethod
156  def _to_timestamp(cls, col):
157  """
158  Convert a column of a dataframe to UNIX timestamps if applicable
159 
160  :param col: A Series object representing a column of a dataframe.
161  """
162  # try and convert the column to datetimes
163  # the column MUST have a four digit year somewhere in the string
164  # there should be a better way to do this,
165  # but just letting pandas try and convert every column without a format
166  # caused it to convert floats as well
167  # For example, a column of integers
168  # between 0 and 10 are turned into timestamps
169  # if the column cannot be converted,
170  # just return the original column untouched
171  try:
172  col = pd.to_datetime(col)
173  except ValueError:
174  logging.warning(
175  "Could not convert field to timestamps: {0}".format(col.name)
176  )
177  return col
178 
179  # now convert the newly created datetimes into timestamps
180  # we have to be careful here
181  # because NaT cannot be converted to a timestamp
182  # so we have to return NaN
183  converted = []
184  for i in col:
185  try:
186  converted.append(i.timestamp())
187  except ValueError:
188  converted.append(pd.np.NaN)
189  except AttributeError:
190  converted.append(pd.np.NaN)
191 
192  # return a new series that maintains the same index as the original
193  return pd.Series(converted, index=col.index)

195  def write_object_to_file(
196  self,
197  query_results,
198  filename,
199  fmt="csv",
200  coerce_to_timestamp=False,
201  record_time_added=False
202  ):
203  """
204  Write query results to file.
205 
206  Acceptable formats are:
207  - csv:
208  comma-seperated-values file. This is the default format.
209  - json:
210  JSON array. Each element in the array is a different row.
211  - ndjson:
212  JSON array but each element is new-line deliminated
213  instead of comman deliminated like in `json`
214 
215  This requires a significant amount of cleanup.
216  Pandas doesn't handle output to CSV and json in a uniform way.
217  This is especially painful for datetime types.
218  Pandas wants to write them as strings in CSV,
219  but as milisecond Unix timestamps.
220 
221  By default, this function will try and leave all values as
222  they are represented in Salesforce.
223  You use the `coerce_to_timestamp` flag to force all datetimes
224  to become Unix timestamps (UTC).
225  This is can be greatly beneficial as it will make all of your
226  datetime fields look the same,
227  and makes it easier to work with in other database environments
228 
229  :param query_results: the results from a SQL query
230  :param filename: the name of the file where the data
231  should be dumped to
232  :param fmt: the format you want the output in.
233  *Default:* csv.
234  :param coerce_to_timestamp: True if you want all datetime fields to be
235  converted into Unix timestamps.
236  False if you want them to be left in the
237  same format as they were in Salesforce.
238  Leaving the value as False will result
239  in datetimes being strings.
240  *Defaults to False*
241  :param record_time_added: *(optional)* True if you want to add a
242  Unix timestamp field to the resulting data
243  that marks when the data
244  was fetched from Salesforce.
245  *Default: False*.
246  """
247  fmt = fmt.lower()
248  if fmt not in ['csv', 'json', 'ndjson']:
249  raise ValueError("Format value is not recognized: {0}".format(fmt))
250 
251  # this line right here will convert all integers to floats if there are
252  # any None/np.nan values in the column
253  # that's because None/np.nan cannot exist in an integer column
254  # we should write all of our timestamps as FLOATS in our final schema
255  df = pd.DataFrame.from_records(query_results, exclude=["attributes"])
256 
257  df.columns = [c.lower() for c in df.columns]
258 
259  # convert columns with datetime strings to datetimes
260  # not all strings will be datetimes, so we ignore any errors that occur
261  # we get the object's definition at this point and only consider
262  # features that are DATE or DATETIME
263  if coerce_to_timestamp and df.shape[0] > 0:
264  # get the object name out of the query results
265  # it's stored in the "attributes" dictionary
266  # for each returned record
267  object_name = query_results[0]['attributes']['type']
268 
269  logging.info("Coercing timestamps for: {0}".format(object_name))
270 
271  schema = self.describe_object(object_name)
272 
273  # possible columns that can be convereted to timestamps
274  # are the ones that are either date or datetime types
275  # strings are too general and we risk unintentional conversion
276  possible_timestamp_cols = [
277  i['name'].lower()
278  for i in schema['fields']
279  if i['type'] in ["date", "datetime"] and
280  i['name'].lower() in df.columns
281  ]
282  df[possible_timestamp_cols] = df[possible_timestamp_cols].apply(
283  lambda x: self._to_timestamp(x)
284  )
285 
286  if record_time_added:
287  fetched_time = time.time()
288  df["time_fetched_from_salesforce"] = fetched_time
289 
290  # write the CSV or JSON file depending on the option
291  # NOTE:
292  # datetimes here are an issue.
293  # There is no good way to manage the difference
294  # for to_json, the options are an epoch or a ISO string
295  # but for to_csv, it will be a string output by datetime
296  # For JSON we decided to output the epoch timestamp in seconds
297  # (as is fairly standard for JavaScript)
298  # And for csv, we do a string
299  if fmt == "csv":
300  # there are also a ton of newline objects
301  # that mess up our ability to write to csv
302  # we remove these newlines so that the output is a valid CSV format
303  logging.info("Cleaning data and writing to CSV")
304  possible_strings = df.columns[df.dtypes == "object"]
305  df[possible_strings] = df[possible_strings].apply(
306  lambda x: x.str.replace("\r\n", "")
307  )
308  df[possible_strings] = df[possible_strings].apply(
309  lambda x: x.str.replace("\n", "")
310  )
311 
312  # write the dataframe
313  df.to_csv(filename, index=False)
314  elif fmt == "json":
315  df.to_json(filename, "records", date_unit="s")
316  elif fmt == "ndjson":
317  df.to_json(filename, "records", lines=True, date_unit="s")
318 
319  return df
