41 def validate_key(k, max_length=250):
42  if not isinstance(k, basestring):
43  raise TypeError("The key has to be a string")
44  elif len(k) > max_length:
45  raise AirflowException(
46  "The key has to be less than {0} characters".format(max_length))
47  elif not re.match(r'^[A-Za-z0-9_\-\.]+$', k):
48  raise AirflowException(
49  "The key ({k}) has to be made of alphanumeric characters, dashes, "
50  "dots and underscores exclusively".format(**locals()))
51  else:
52  return True

66  d[c.name] = value
67  return d

142 def pprinttable(rows):
143  """Returns a pretty ascii table from tuples
144 
145  If namedtuple are used, the table will have headers
146  """
147  if not rows:
148  return
149  if hasattr(rows[0], '_fields'): # if namedtuple
150  headers = rows[0]._fields
151  else:
152  headers = ["col{}".format(i) for i in range(len(rows[0]))]
153  lens = [len(s) for s in headers]
154 
155  for row in rows:
156  for i in range(len(rows[0])):
157  slenght = len("{}".format(row[i]))
158  if slenght > lens[i]:
159  lens[i] = slenght
160  formats = []
161  hformats = []
162  for i in range(len(rows[0])):
163  if isinstance(rows[0][i], int):
164  formats.append("%%%dd" % lens[i])
165  else:
166  formats.append("%%-%ds" % lens[i])
167  hformats.append("%%-%ds" % lens[i])
168  pattern = " | ".join(formats)
169  hpattern = " | ".join(hformats)
170  separator = "-+-".join(['-' * n for n in lens])
171  s = ""
172  s += separator + '\n'
173  s += (hpattern % tuple(headers)) + '\n'
174  s += separator + '\n'
175 
176  def f(t):
177  return "{}".format(t) if isinstance(t, basestring) else t
178 
179  for line in rows:
180  s += pattern % tuple(f(t) for t in line) + '\n'
181  s += separator + '\n'
182  return s

196 def kill_process_tree(logger, pid):
197  """
198  Kills the process and all of the descendants. Kills using the `kill`
199  shell command so that it can change users. Note: killing via PIDs
200  has the potential to the wrong process if the process dies and the
201  PID gets recycled in a narrow time window.
202 
203  :param logger: logger
204  :type logger: logging.Logger
205  """
206  try:
207  root_process = psutil.Process(pid)
208  except psutil.NoSuchProcess:
209  logger.warn("PID: {} does not exist".format(pid))
210  return
211 
212  # Check child processes to reduce cases where a child process died but
213  # the PID got reused.
214  descendant_processes = [x for x in root_process.children(recursive=True)
215  if x.is_running()]
216 
217  if len(descendant_processes) != 0:
218  logger.warn("Terminating descendant processes of {} PID: {}"
219  .format(root_process.cmdline(),
220  root_process.pid))
221  temp_processes = descendant_processes[:]
222  for descendant in temp_processes:
223  logger.warn("Terminating descendant process {} PID: {}"
224  .format(descendant.cmdline(), descendant.pid))
225  try:
226  kill_using_shell(descendant.pid, signal.SIGTERM)
227  except psutil.NoSuchProcess:
228  descendant_processes.remove(descendant)
229 
230  logger.warn("Waiting up to {}s for processes to exit..."
231  .format(TIME_TO_WAIT_AFTER_SIGTERM))
232  try:
233  psutil.wait_procs(descendant_processes, TIME_TO_WAIT_AFTER_SIGTERM)
234  logger.warn("Done waiting")
235  except psutil.TimeoutExpired:
236  logger.warn("Ran out of time while waiting for "
237  "processes to exit")
238  # Then SIGKILL
239  descendant_processes = [x for x in root_process.children(recursive=True)
240  if x.is_running()]
241 
242  if len(descendant_processes) > 0:
243  temp_processes = descendant_processes[:]
244  for descendant in temp_processes:
245  logger.warn("Killing descendant process {} PID: {}"
246  .format(descendant.cmdline(), descendant.pid))
247  try:
248  kill_using_shell(descendant.pid, signal.SIGTERM)
249  descendant.wait()
250  except psutil.NoSuchProcess:
251  descendant_processes.remove(descendant)
252  logger.warn("Killed all descendant processes of {} PID: {}"
253  .format(root_process.cmdline(),
254  root_process.pid))
255  else:
256  logger.debug("There are no descendant processes to kill")

259 def kill_descendant_processes(logger, pids_to_kill=None):
260  """
261  Kills all descendant processes of this process.
262 
263  :param logger: logger
264  :type logger: logging.Logger
265  :param pids_to_kill: if specified, kill only these PIDs
266  :type pids_to_kill: list[int]
267  """
268  # First try SIGTERM
269  this_process = psutil.Process(os.getpid())
270 
271  # Only check child processes to ensure that we don't have a case
272  # where a child process died but the PID got reused.
273  descendant_processes = [x for x in this_process.children(recursive=True)
274  if x.is_running()]
275  if pids_to_kill:
276  descendant_processes = [x for x in descendant_processes
277  if x.pid in pids_to_kill]
278 
279  if len(descendant_processes) == 0:
280  logger.debug("There are no descendant processes that can be killed")
281  return
282  logger.warn("Terminating descendant processes of {} PID: {}"
283  .format(this_process.cmdline(),
284  this_process.pid))
285 
286  temp_processes = descendant_processes[:]
287  for descendant in temp_processes:
288  try:
289  logger.warn("Terminating descendant process {} PID: {}"
290  .format(descendant.cmdline(), descendant.pid))
291  descendant.terminate()
292  except psutil.NoSuchProcess:
293  descendant_processes.remove(descendant)
294 
295  logger.warn("Waiting up to {}s for processes to exit..."
296  .format(TIME_TO_WAIT_AFTER_SIGTERM))
297  try:
298  psutil.wait_procs(descendant_processes, TIME_TO_WAIT_AFTER_SIGTERM)
299  logger.warn("Done waiting")
300  except psutil.TimeoutExpired:
301  logger.warn("Ran out of time while waiting for "
302  "processes to exit")
303  # Then SIGKILL
304  descendant_processes = [x for x in this_process.children(recursive=True)
305  if x.is_running()]
306  if pids_to_kill:
307  descendant_processes = [x for x in descendant_processes
308  if x.pid in pids_to_kill]
309 
310  if len(descendant_processes) > 0:
311  for descendant in descendant_processes:
312  logger.warn("Killing descendant process {} PID: {}"
313  .format(descendant.cmdline(), descendant.pid))
314  try:
315  descendant.kill()
316  descendant.wait()
317  except psutil.NoSuchProcess:
318  pass
319  logger.warn("Killed all descendant processes of {} PID: {}"
320  .format(this_process.cmdline(),
321  this_process.pid))

324 class AirflowImporter(object):
325  """
326  Importer that dynamically loads a class and module from its parent. This
327  allows Airflow to support ``from airflow.operators import BashOperator``
328  even though BashOperator is actually in
329  ``airflow.operators.bash_operator``.
330 
331  The importer also takes over for the parent_module by wrapping it. This is
332  required to support attribute-based usage:
333 
334  .. code:: python
335 
336  from airflow import operators
337  operators.BashOperator(...)
338  """
339 
340  def __init__(self, parent_module, module_attributes):
341  """
342  :param parent_module: The string package name of the parent module. For
343  example, 'airflow.operators'
344  :type parent_module: string
345  :param module_attributes: The file to class mappings for all importable
346  classes.
347  :type module_attributes: string
348  """
349  self._parent_module = parent_module
350  self._attribute_modules = self._build_attribute_modules(module_attributes)
351  self._loaded_modules = {}
352 
353  # Wrap the module so we can take over __getattr__.
354  sys.modules[parent_module.__name__] = self
355 
356  @staticmethod
357  def _build_attribute_modules(module_attributes):
358  """
359  Flips and flattens the module_attributes dictionary from:
360 
361  module => [Attribute, ...]
362 
363  To:
364 
365  Attribute => module
366 
367  This is useful so that we can find the module to use, given an
368  attribute.
369  """
370  attribute_modules = {}
371 
372  for module, attributes in list(module_attributes.items()):
373  for attribute in attributes:
374  attribute_modules[attribute] = module
375 
376  return attribute_modules
377 
378  def _load_attribute(self, attribute):
379  """
380  Load the class attribute if it hasn't been loaded yet, and return it.
381  """
382  module = self._attribute_modules.get(attribute, False)
383 
384  if not module:
385  # This shouldn't happen. The check happens in find_modules, too.
386  raise ImportError(attribute)
387  elif module not in self._loaded_modules:
388  # Note that it's very important to only load a given modules once.
389  # If they are loaded more than once, the memory reference to the
390  # class objects changes, and Python thinks that an object of type
391  # Foo that was declared before Foo's module was reloaded is no
392  # longer the same type as Foo after it's reloaded.
393  path = os.path.realpath(self._parent_module.__file__)
394  folder = os.path.dirname(path)
395  f, filename, description = imp.find_module(module, [folder])
396  self._loaded_modules[module] = imp.load_module(module, f, filename, description)
397 
398  # This functionality is deprecated, and AirflowImporter should be
399  # removed in 2.0.
400  warnings.warn(
401  "Importing {i} directly from {m} has been "
402  "deprecated. Please import from "
403  "'{m}.[operator_module]' instead. Support for direct "
404  "imports will be dropped entirely in Airflow 2.0.".format(
405  i=attribute, m=self._parent_module),
406  DeprecationWarning)
407 
408  loaded_module = self._loaded_modules[module]
409 
410  return getattr(loaded_module, attribute)
411 
412  def __getattr__(self, attribute):
413  """
414  Get an attribute from the wrapped module. If the attribute doesn't
415  exist, try and import it as a class from a submodule.
416 
417  This is a Python trick that allows the class to pretend it's a module,
418  so that attribute-based usage works:
419 
420  from airflow import operators
421  operators.BashOperator(...)
422 
423  It also allows normal from imports to work:
424 
425  from airflow.operators.bash_operator import BashOperator
426  """
427  if hasattr(self._parent_module, attribute):
428  # Always default to the parent module if the attribute exists.
429  return getattr(self._parent_module, attribute)
430  elif attribute in self._attribute_modules:
431  # Try and import the attribute if it's got a module defined.
432  loaded_attribute = self._load_attribute(attribute)
433  setattr(self, attribute, loaded_attribute)
434  return loaded_attribute
435 
436  raise AttributeError
