31 def renew_from_kt():
32  # The config is specified in seconds. But we ask for that same amount in
33  # minutes to give ourselves a large renewal buffer.
34  renewal_lifetime = "%sm" % configuration.getint('kerberos', 'reinit_frequency')
35  principal = configuration.get('kerberos', 'principal').replace("_HOST", socket.getfqdn())
36  cmdv = [configuration.get('kerberos', 'kinit_path'),
37  "-r", renewal_lifetime,
38  "-k", # host ticket
39  "-t", configuration.get('kerberos', 'keytab'), # specify keytab
40  "-c", configuration.get('kerberos', 'ccache'), # specify credentials cache
41  principal]
42  LOG.info("Reinitting kerberos from keytab: " +
43  " ".join(cmdv))
44 
45  subp = subprocess.Popen(cmdv,
46  stdout=subprocess.PIPE,
47  stderr=subprocess.PIPE,
48  close_fds=True,
49  bufsize=-1)
50  subp.wait()
51  if subp.returncode != 0:
52  LOG.error("Couldn't reinit from keytab! `kinit' exited with %s.\n%s\n%s" % (
53  subp.returncode,
54  "\n".join(subp.stdout.readlines()),
55  "\n".join(subp.stderr.readlines())))
56  sys.exit(subp.returncode)

58  global NEED_KRB181_WORKAROUND
59  if NEED_KRB181_WORKAROUND is None:
60  NEED_KRB181_WORKAROUND = detect_conf_var()
61  if NEED_KRB181_WORKAROUND:
62  # (From: HUE-640). Kerberos clock have seconds level granularity. Make sure we

68 def perform_krb181_workaround():
69  cmdv = [configuration.get('kerberos', 'kinit_path'),
70  "-R",
71  "-c", configuration.get('kerberos', 'ccache')]
72  LOG.info("Renewing kerberos ticket to work around kerberos 1.8.1: " +
73  " ".join(cmdv))
74  ret = subprocess.call(cmdv)
75  if ret != 0:
76  principal = "%s/%s" % (configuration.get('kerberos', 'principal'), socket.getfqdn())
77  fmt_dict = dict(princ=principal,
78  ccache=configuration.get('kerberos', 'principal'))
79  LOG.error("Couldn't renew kerberos ticket in order to work around "
80  "Kerberos 1.8.1 issue. Please check that the ticket for "
81  "'%(princ)s' is still renewable:\n"
82  " $ kinit -f -c %(ccache)s\n"
83  "If the 'renew until' date is the same as the 'valid starting' "
84  "date, the ticket cannot be renewed. Please check your KDC "
85  "configuration, and the ticket renewal policy (maxrenewlife) "
86  "for the '%(princ)s' and `krbtgt' principals." % fmt_dict)
87  sys.exit(ret)

90 def detect_conf_var():
91  """Return true if the ticket cache contains "conf" information as is found
92  in ticket caches of Kerboers 1.8.1 or later. This is incompatible with the
93  Sun Java Krb5LoginModule in Java6, so we need to take an action to work
94  around it.
95  """
96  with open(configuration.get('kerberos', 'ccache'), "rb") as f:
97  return "X-CACHECONF:" in f.read()
