20 class DatastoreHook(GoogleCloudBaseHook):
21  """
22  Interact with Google Cloud Datastore. This hook uses the Google Cloud Platform
23  connection.
24 
25  This object is not threads safe. If you want to make multiple requests
26  simultaniously, you will need to create a hook per thread.
27  """
28 
29  def __init__(self,
30  datastore_conn_id='google_cloud_datastore_default',
31  delegate_to=None):
32  super(DatastoreHook, self).__init__(datastore_conn_id, delegate_to)
33  # datasetId is the same as the project name
34  self.dataset_id = self._get_field('project')
35  self.connection = self.get_conn()

37  def get_conn(self):
38  """
39  Returns a Google Cloud Storage service object.

41  http_authorized = self._authorize()

44  def allocate_ids(self, partialKeys):
45  """
46  Allocate IDs for incomplete keys.
47  see https://cloud.google.com/datastore/docs/apis/v1beta2/datasets/allocateIds
48 
49  :param partialKeys: a list of partial keys
50  :return: a list of full keys.
51  """
52  resp = self.connection.datasets().allocateIds(datasetId=self.dataset_id, body={'keys': partialKeys}).execute()
53  return resp['keys']

55  def begin_transaction(self):

59 
60  :return: a transaction handle

65  def commit(self, body):
66  """

69 
70  :param body: the body of the commit request

76  def lookup(self, keys, read_consistency=None, transaction=None):
77  """
78  Lookup some entities by key
79  see https://cloud.google.com/datastore/docs/apis/v1beta2/datasets/lookup
80  :param keys: the keys to lookup
81  :param read_consistency: the read consistency to use. default, strong or eventual.
82  Cannot be used with a transaction.
83  :param transaction: the transaction to use, if any.
84  :return: the response body of the lookup request.
85  """
86  body = {'keys': keys}
87  if read_consistency:
88  body['readConsistency'] = read_consistency
89  if transaction:
90  body['transaction'] = transaction
91  return self.connection.datasets().lookup(datasetId=self.dataset_id, body=body).execute()
