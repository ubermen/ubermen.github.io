25 class DockerOperator(BaseOperator):
26  """
27  Execute a command inside a docker container.
28 
29  A temporary directory is created on the host and mounted into a container to allow storing files
30  that together exceed the default disk size of 10GB in a container. The path to the mounted
31  directory can be accessed via the environment variable ``AIRFLOW_TMP_DIR``.
32 
33  :param image: Docker image from which to create the container.
34  :type image: str
35  :param api_version: Remote API version.
36  :type api_version: str
37  :param command: Command to be run in the container.
38  :type command: str or list
39  :param cpus: Number of CPUs to assign to the container.
40  This value gets multiplied with 1024. See
41  https://docs.docker.com/engine/reference/run/#cpu-share-constraint
42  :type cpus: float
43  :param docker_url: URL of the host running the docker daemon.
44  :type docker_url: str
45  :param environment: Environment variables to set in the container.
46  :type environment: dict
47  :param force_pull: Pull the docker image on every run.
48  :type force_pull: bool
49  :param mem_limit: Maximum amount of memory the container can use. Either a float value, which
50  represents the limit in bytes, or a string like ``128m`` or ``1g``.
51  :type mem_limit: float or str
52  :param network_mode: Network mode for the container.
53  :type network_mode: str
54  :param tls_ca_cert: Path to a PEM-encoded certificate authority to secure the docker connection.
55  :type tls_ca_cert: str
56  :param tls_client_cert: Path to the PEM-encoded certificate used to authenticate docker client.
57  :type tls_client_cert: str
58  :param tls_client_key: Path to the PEM-encoded key used to authenticate docker client.
59  :type tls_client_key: str
60  :param tls_hostname: Hostname to match against the docker server certificate or False to
61  disable the check.
62  :type tls_hostname: str or bool
63  :param tls_ssl_version: Version of SSL to use when communicating with docker daemon.
64  :type tls_ssl_version: str
65  :param tmp_dir: Mount point inside the container to a temporary directory created on the host by
66  the operator. The path is also made available via the environment variable
67  ``AIRFLOW_TMP_DIR`` inside the container.
68  :type tmp_dir: str
69  :param user: Default user inside the docker container.
70  :type user: int or str
71  :param volumes: List of volumes to mount into the container, e.g.
72  ``['/host/path:/container/path', '/host/path2:/container/path2:ro']``.
73  :param xcom_push: Does the stdout will be pushed to the next step using XCom.
74  The default is False.
75  :type xcom_push: bool
76  :param xcom_all: Push all the stdout or just the last line. The default is False (last line).
77  :type xcom_all: bool
78  """

131  def execute(self, context):

133 
134  tls_config = None

136  tls_config = tls.TLSConfig(
137  ca_cert=self.tls_ca_cert,
138  client_cert=(self.tls_client_cert, self.tls_client_key),
139  verify=True,

142  )
143  self.docker_url = self.docker_url.replace('tcp://', 'https://')
144 
145  self.cli = Client(base_url=self.docker_url, version=self.api_version, tls=tls_config)
146 
147  if ':' not in self.image:
148  image = self.image + ':latest'
149  else:
150  image = self.image
151 
152  if self.force_pull or len(self.cli.images(name=image)) == 0:
153  logging.info('Pulling docker image ' + image)
154  for l in self.cli.pull(image, stream=True):
155  output = json.loads(l.decode('utf-8'))
156  logging.info("{}".format(output['status']))
157 
158  cpu_shares = int(round(self.cpus * 1024))
159 
160  with TemporaryDirectory(prefix='airflowtmp') as host_tmp_dir:
161  self.environment['AIRFLOW_TMP_DIR'] = self.tmp_dir
162  self.volumes.append('{0}:{1}'.format(host_tmp_dir, self.tmp_dir))
163 
164  self.container = self.cli.create_container(
165  command=self.get_command(),
166  cpu_shares=cpu_shares,
167  environment=self.environment,
168  host_config=self.cli.create_host_config(binds=self.volumes,
169  network_mode=self.network_mode),
170  image=image,
171  mem_limit=self.mem_limit,
172  user=self.user
173  )
174  self.cli.start(self.container['Id'])
175 
176  line = ''
177  for line in self.cli.logs(container=self.container['Id'], stream=True):
178  logging.info("{}".format(line.strip()))
179 
180  exit_code = self.cli.wait(self.container['Id'])
181  if exit_code != 0:
182  raise AirflowException('docker container failed')
183 
184  if self.xcom_push:
185  return self.cli.logs(container=self.container['Id']) if self.xcom_all else str(line.strip())
