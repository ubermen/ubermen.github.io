73 def group_contains_user(conn, search_base, group_filter, user_name_attr, username):
74  search_filter = '(&({0}))'.format(group_filter)
75  if not conn.search(native(search_base), native(search_filter),
76  attributes=[native(user_name_attr)]):
77  LOG.warning("Unable to find group for %s %s", search_base, search_filter)
78  else:
79  for resp in conn.response:
80  if (
81  'attributes' in resp and (
82  resp['attributes'].get(user_name_attr)[0] == username or
83  resp['attributes'].get(user_name_attr) == username
84  )
85  ):
86  return True
87  return False

90 def groups_user(conn, search_base, user_filter, user_name_att, username):
91  search_filter = "(&({0})({1}={2}))".format(user_filter, user_name_att, username)
92  res = conn.search(native(search_base), native(search_filter), attributes=[native("memberOf")])
93  if not res:
94  LOG.info("Cannot find user %s", username)
95  raise AuthenticationError("Invalid username or password")
96 
97  if conn.response and "memberOf" not in conn.response[0]["attributes"]:
98  LOG.warn("""Missing attribute "memberOf" when looked-up in Ldap database.
99  The user does not seem to be a member of a group and therefore won't see any dag
100  if the option filter_by_owner=True and owner_mode=ldapgroup are set""")
101  return []
102 
103  user_groups = conn.response[0]["attributes"]["memberOf"]
104 
105  regex = re.compile("cn=([^,]*).*", re.IGNORECASE)
106  groups_list = []
107  try:
108  groups_list = [regex.search(i).group(1) for i in user_groups]
109  except IndexError:
110  LOG.warning("Parsing error when retrieving the user's group(s)."
111  " Check if the user belongs to at least one group"
112  " or if the user's groups name do not contain special characters")
113 
114  return groups_list

179  native(search_filter),
180  search_scope=native(search_scope))
181 
182  # todo: use list or result?

184  LOG.info("Cannot find user %s", username)
185  raise AuthenticationError("Invalid username or password")
186 
187  entry = conn.response[0]
188 
189  conn.unbind()
190 
191  if 'dn' not in entry:
