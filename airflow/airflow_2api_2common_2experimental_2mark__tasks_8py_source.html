26 def _create_dagruns(dag, execution_dates, state, run_id_template):
27  """
28  Infers from the dates which dag runs need to be created and does so.
29  :param dag: the dag to create dag runs for
30  :param execution_dates: list of execution dates to evaluate
31  :param state: the state to set the dag run to
32  :param run_id_template:the template for run id to be with the execution date
33  :return: newly created and existing dag runs for the execution dates supplied
34  """
35  # find out if we need to create any dag runs
36  drs = DagRun.find(dag_id=dag.dag_id, execution_date=execution_dates)
37  dates_to_create = list(set(execution_dates) - set([dr.execution_date for dr in drs]))
38 
39  for date in dates_to_create:
40  dr = dag.create_dagrun(
41  run_id=run_id_template.format(date.isoformat()),
42  execution_date=date,
43  start_date=datetime.datetime.now(),
44  external_trigger=False,
45  state=state,
46  )
47  drs.append(dr)
48 
49  return drs

52 def set_state(task, execution_date, upstream=False, downstream=False,
53  future=False, past=False, state=State.SUCCESS, commit=False):
54  """
55  Set the state of a task instance and if needed its relatives. Can set state
56  for future tasks (calculated from execution_date) and retroactively
57  for past tasks. Will verify integrity of past dag runs in order to create
58  tasks that did not exist. It will not create dag runs that are missing
59  on the schedule (but it will as for subdag dag runs if needed).
60  :param task: the task from which to work. task.task.dag needs to be set
61  :param execution_date: the execution date from which to start looking
62  :param upstream: Mark all parents (upstream tasks)
63  :param downstream: Mark all siblings (downstream tasks) of task_id, including SubDags
64  :param future: Mark all future tasks on the interval of the dag up until
65  last execution date.
66  :param past: Retroactively mark all tasks starting from start_date of the DAG
67  :param state: State to which the tasks need to be set
68  :param commit: Commit tasks to be altered to the database
69  :return: list of tasks that have been created and updated
70  """
71  assert isinstance(execution_date, datetime.datetime)
72 
73  # microseconds are supported by the database, but is not handled
74  # correctly by airflow on e.g. the filesystem and in other places
75  execution_date = execution_date.replace(microsecond=0)
76 
77  assert task.dag is not None
78  dag = task.dag
79 
80  latest_execution_date = dag.latest_execution_date
81  assert latest_execution_date is not None
82 
83  # determine date range of dag runs and tasks to consider
84  end_date = latest_execution_date if future else execution_date
85 
86  if 'start_date' in dag.default_args:
87  start_date = dag.default_args['start_date']
88  elif dag.start_date:
89  start_date = dag.start_date
90  else:
91  start_date = execution_date
92 
93  start_date = execution_date if not past else start_date
94 
95  if dag.schedule_interval == '@once':
96  dates = [start_date]
97  else:
98  dates = dag.date_range(start_date=start_date, end_date=end_date)
99 
100  # find relatives (siblings = downstream, parents = upstream) if needed
101  task_ids = [task.task_id]
102  if downstream:
103  relatives = task.get_flat_relatives(upstream=False)
104  task_ids += [t.task_id for t in relatives]
105  if upstream:
106  relatives = task.get_flat_relatives(upstream=True)
107  task_ids += [t.task_id for t in relatives]
108 
109  # verify the integrity of the dag runs in case a task was added or removed
110  # set the confirmed execution dates as they might be different
111  # from what was provided
112  confirmed_dates = []
113  drs = DagRun.find(dag_id=dag.dag_id, execution_date=dates)
114  for dr in drs:
115  dr.dag = dag
116  dr.verify_integrity()
117  confirmed_dates.append(dr.execution_date)
118 
119  # go through subdagoperators and create dag runs. We will only work
120  # within the scope of the subdag. We wont propagate to the parent dag,
121  # but we will propagate from parent to subdag.
122  session = Session()
123  dags = [dag]
124  sub_dag_ids = []
125  while len(dags) > 0:
126  current_dag = dags.pop()
127  for task_id in task_ids:
128  if not current_dag.has_task(task_id):
129  continue
130 
131  current_task = current_dag.get_task(task_id)
132  if isinstance(current_task, SubDagOperator):
133  # this works as a kind of integrity check
134  # it creates missing dag runs for subdagoperators,
135  # maybe this should be moved to dagrun.verify_integrity
136  drs = _create_dagruns(current_task.subdag,
137  execution_dates=confirmed_dates,
138  state=State.RUNNING,
139  run_id_template=BackfillJob.ID_FORMAT_PREFIX)
140 
141  for dr in drs:
142  dr.dag = current_task.subdag
143  dr.verify_integrity()
144  if commit:
145  dr.state = state
146  session.merge(dr)
147 
148  dags.append(current_task.subdag)
149  sub_dag_ids.append(current_task.subdag.dag_id)
150 
151  # now look for the task instances that are affected
152  TI = TaskInstance
153 
154  # get all tasks of the main dag that will be affected by a state change
155  qry_dag = session.query(TI).filter(
156  TI.dag_id==dag.dag_id,
157  TI.execution_date.in_(confirmed_dates),
158  TI.task_id.in_(task_ids)).filter(
159  or_(TI.state.is_(None),
160  TI.state != state)
161  )
162 
163  # get *all* tasks of the sub dags
164  if len(sub_dag_ids) > 0:
165  qry_sub_dag = session.query(TI).filter(
166  TI.dag_id.in_(sub_dag_ids),
167  TI.execution_date.in_(confirmed_dates)).filter(
168  or_(TI.state.is_(None),
169  TI.state != state)
170  )
171 
172  if commit:
173  tis_altered = qry_dag.with_for_update().all()
174  if len(sub_dag_ids) > 0:
175  tis_altered += qry_sub_dag.with_for_update().all()
176  for ti in tis_altered:
177  ti.state = state
178  session.commit()
179  else:
180  tis_altered = qry_dag.all()
181  if len(sub_dag_ids) > 0:
182  tis_altered += qry_sub_dag.all()
183 
184  session.close()
185 
186  return tis_altered
