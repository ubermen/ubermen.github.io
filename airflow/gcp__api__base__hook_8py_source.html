26 class GoogleCloudBaseHook(BaseHook):
27  """
28  A base hook for Google cloud-related hooks. Google cloud has a shared REST
29  API client that is built in the same way no matter which service you use.
30  This class helps construct and authorize the credentials needed to then
31  call apiclient.discovery.build() to actually discover and build a client
32  for a Google cloud service.
33 
34  The class also contains some miscellaneous helper functions.
35 
36  All hook derived from this base hook use the 'Google Cloud Platform' connection
37  type. Two ways of authentication are supported:
38 
39  Default credentials: Only specify 'Project Id'. Then you need to have executed
40  ``gcloud auth`` on the Airflow worker machine.
41 
42  JSON key file: Specify 'Project Id', 'Key Path' and 'Scope'.
43 
44  Legacy P12 key files are not supported.
45  """
46 
47  def __init__(self, conn_id, delegate_to=None):
48  """
49  :param conn_id: The connection ID to use when fetching connection info.
50  :type conn_id: string
51  :param delegate_to: The account to impersonate, if any.
52  For this to work, the service account making the request must have
53  domain-wide delegation enabled.
54  :type delegate_to: string
55  """
56  self.conn_id = conn_id
57  self.delegate_to = delegate_to

60  def _authorize(self):
61  """
62  Returns an authorized HTTP object to be used to build a Google cloud
63  service hook connection.
64  """
65  key_path = self._get_field('key_path', False)
66  scope = self._get_field('scope', False)
67 
68  kwargs = {}
69  if self.delegate_to:
70  kwargs['sub'] = self.delegate_to
71 
72  if not key_path:
73  logging.info('Getting connection using `gcloud auth` user, since no key file '
74  'is defined for hook.')
75  credentials = GoogleCredentials.get_application_default()
76  else:
77  if not scope:
78  raise AirflowException('Scope should be defined when using a key file.')
79  scopes = [s.strip() for s in scope.split(',')]
80  if key_path.endswith('.json'):
81  logging.info('Getting connection using a JSON key file.')
82  credentials = ServiceAccountCredentials\
83  .from_json_keyfile_name(key_path, scopes)
84  elif key_path.endswith('.p12'):
85  raise AirflowException('Legacy P12 key file are not supported, '
86  'use a JSON key file.')
87  else:
88  raise AirflowException('Unrecognised extension for key file.')
89 
90  http = httplib2.Http()
91  return credentials.authorize(http)

95  Fetches a field from extras, and returns it. This is some Airflow
