80  def _prepare_command(self, cmd):
81  """
82  Construct the spark-sql command to execute. Verbose output is enabled
83  as default.
84  :param cmd: command to append to the spark-sql command
85  :type cmd: str
86  :return: full command to be executed
87  """
88  connection_cmd = ["spark-sql"]
89  if self._conf:
90  for conf_el in self._conf.split(","):
91  connection_cmd += ["--conf", conf_el]
92  if self._executor_cores:
93  connection_cmd += ["--executor-cores", str(self._executor_cores)]
94  if self._executor_memory:
95  connection_cmd += ["--executor-memory", self._executor_memory]
96  if self._keytab:
97  connection_cmd += ["--keytab", self._keytab]
98  if self._num_executors:
99  connection_cmd += ["--num-executors", str(self._num_executors)]
100  if self._sql:
101  if self._sql.endswith('.sql') or self._sql.endswith(".hql"):
102  connection_cmd += ["-f", self._sql]
103  else:
104  connection_cmd += ["-e", self._sql]
105  if self._master:
106  connection_cmd += ["--master", self._master]
107  if self._name:
108  connection_cmd += ["--name", self._name]
109  if self._verbose:
110  connection_cmd += ["--verbose"]
111  if self._yarn_queue:
112  connection_cmd += ["--queue", self._yarn_queue]
113 
114  connection_cmd += cmd
115  logging.debug("Spark-Sql cmd: {}".format(connection_cmd))
116 
117  return connection_cmd

119  def run_query(self, cmd="", **kwargs):
120  """
121  Remote Popen (actually execute the Spark-sql query)
122 
123  :param cmd: command to remotely execute
124  :param kwargs: extra arguments to Popen (see subprocess.Popen)
125  """
126  prefixed_cmd = self._prepare_command(cmd)
127  self._sp = subprocess.Popen(prefixed_cmd,
128  stdout=subprocess.PIPE,
129  stderr=subprocess.PIPE,
130  **kwargs)
131  # using two iterators here to support 'real-time' logging
132  for line in iter(self._sp.stdout.readline, b''):
133  line = line.decode('utf-8').strip()
134  logging.info(line)
135  for line in iter(self._sp.stderr.readline, b''):
136  line = line.decode('utf-8').strip()
137  logging.info(line)
138  output, stderr = self._sp.communicate()
139 
140  if self._sp.returncode:
141  raise AirflowException("Cannot execute {} on {}. Error code is: "
142  "{}. Output: {}, Stderr: {}"
143  .format(cmd, self._conn.host,
144  self._sp.returncode, output, stderr))
