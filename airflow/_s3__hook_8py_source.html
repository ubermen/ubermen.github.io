37 def _parse_s3_config(config_file_name, config_format='boto', profile=None):
38  """
39  Parses a config file for s3 credentials. Can currently
40  parse boto, s3cmd.conf and AWS SDK config formats
41 
42  :param config_file_name: path to the config file
43  :type config_file_name: str
44  :param config_format: config type. One of "boto", "s3cmd" or "aws".
45  Defaults to "boto"
46  :type config_format: str
47  :param profile: profile name in AWS type config file
48  :type profile: str
49  """
50  Config = configparser.ConfigParser()
51  if Config.read(config_file_name): # pragma: no cover
52  sections = Config.sections()
53  else:
54  raise AirflowException("Couldn't read {0}".format(config_file_name))
55  # Setting option names depending on file format
56  if config_format is None:
57  config_format = 'boto'
58  conf_format = config_format.lower()
59  if conf_format == 'boto': # pragma: no cover
60  if profile is not None and 'profile ' + profile in sections:
61  cred_section = 'profile ' + profile
62  else:
63  cred_section = 'Credentials'
64  elif conf_format == 'aws' and profile is not None:
65  cred_section = profile
66  else:
67  cred_section = 'default'
68  # Option names
69  if conf_format in ('boto', 'aws'): # pragma: no cover
70  key_id_option = 'aws_access_key_id'
71  secret_key_option = 'aws_secret_access_key'
72  # security_token_option = 'aws_security_token'
73  else:
74  key_id_option = 'access_key'
75  secret_key_option = 'secret_key'
76  # Actual Parsing
77  if cred_section not in sections:
78  raise AirflowException("This config file format is not recognized")
79  else:
80  try:
81  access_key = Config.get(cred_section, key_id_option)
82  secret_key = Config.get(cred_section, secret_key_option)
83  calling_format = None
84  if Config.has_option(cred_section, 'calling_format'):
85  calling_format = Config.get(cred_section, 'calling_format')
86  except:
87  logging.warning("Option Error in parsing s3 config file")
88  raise
89  return (access_key, secret_key, calling_format)

155  def get_conn(self):
156  """
157  Returns the boto S3Connection object.
158  """
159  if self._default_to_boto:
160  return S3Connection(profile_name=self.profile)
161  a_key = s_key = None
162  if self._creds_in_config_file:
163  a_key, s_key, calling_format = _parse_s3_config(self.s3_config_file,
164  self.s3_config_format,
165  self.profile)
166  elif self._creds_in_conn:
167  a_key = self._a_key
168  s_key = self._s_key
169  calling_format = self.calling_format
170 
171  if calling_format is None:
172  calling_format = 'boto.s3.connection.SubdomainCallingFormat'
173 
174  if self._sts_conn_required:
175  sts_connection = STSConnection(aws_access_key_id=a_key,
176  aws_secret_access_key=s_key,
177  profile_name=self.profile)
178  assumed_role_object = sts_connection.assume_role(
179  role_arn=self.role_arn,
180  role_session_name="Airflow_" + self.s3_conn_id
181  )
182  creds = assumed_role_object.credentials
183  connection = S3Connection(
184  aws_access_key_id=creds.access_key,
185  aws_secret_access_key=creds.secret_key,
186  calling_format=calling_format,
187  security_token=creds.session_token
188  )
189  else:
190  connection = S3Connection(aws_access_key_id=a_key,
191  aws_secret_access_key=s_key,
192  calling_format=calling_format,
193  profile_name=self.profile)
194  return connection

288  def get_wildcard_key(self, wildcard_key, bucket_name=None, delimiter=''):
289  """
290  Returns a boto.s3.key.Key object matching the regular expression
291 
292  :param regex_key: the path to the key
293  :type regex_key: str
294  :param bucket_name: the name of the bucket
295  :type bucket_name: str
296  """
297  if not bucket_name:
298  (bucket_name, wildcard_key) = self.parse_s3_url(wildcard_key)
299  bucket = self.get_bucket(bucket_name)
300  prefix = re.split(r'[*]', wildcard_key, 1)[0]
301  klist = self.list_keys(bucket_name, prefix=prefix, delimiter=delimiter)
302  if not klist:
303  return None
304  key_matches = [k for k in klist if fnmatch.fnmatch(k, wildcard_key)]
305  return bucket.get_key(key_matches[0]) if key_matches else None

307  def check_for_prefix(self, bucket_name, prefix, delimiter):
308  """
309  Checks that a prefix exists in a bucket
310  """
311  prefix = prefix + delimiter if prefix[-1] != delimiter else prefix
312  prefix_split = re.split(r'(\w+[{d}])$'.format(d=delimiter), prefix, 1)
313  previous_level = prefix_split[0]
314  plist = self.list_prefixes(bucket_name, previous_level, delimiter)

317  def load_file(
318  self,
319  filename,
320  key,
321  bucket_name=None,
322  replace=False,
323  multipart_bytes=5 * (1024 ** 3),
324  encrypt=False):
325  """
326  Loads a local file to S3
327 
328  :param filename: name of the file to load.
329  :type filename: str
330  :param key: S3 key that will point to the file
331  :type key: str
332  :param bucket_name: Name of the bucket in which to store the file
333  :type bucket_name: str
334  :param replace: A flag to decide whether or not to overwrite the key
335  if it already exists. If replace is False and the key exists, an
336  error will be raised.
337  :type replace: bool
338  :param multipart_bytes: If provided, the file is uploaded in parts of
339  this size (minimum 5242880). The default value is 5GB, since S3
340  cannot accept non-multipart uploads for files larger than 5GB. If
341  the file is smaller than the specified limit, the option will be
342  ignored.
343  :type multipart_bytes: int
344  :param encrypt: If True, the file will be encrypted on the server-side
345  by S3 and will be stored in an encrypted form while at rest in S3.
346  :type encrypt: bool
347  """
348  if not bucket_name:
349  (bucket_name, key) = self.parse_s3_url(key)
350  bucket = self.get_bucket(bucket_name)
351  key_obj = bucket.get_key(key)
352  if not replace and key_obj:
353  raise ValueError("The key {key} already exists.".format(
354  **locals()))
355 
356  key_size = os.path.getsize(filename)
357  if multipart_bytes and key_size >= multipart_bytes:
358  # multipart upload
359  from filechunkio import FileChunkIO
360  mp = bucket.initiate_multipart_upload(key_name=key,
361  encrypt_key=encrypt)
362  total_chunks = int(math.ceil(key_size / multipart_bytes))
363  sent_bytes = 0
364  try:
365  for chunk in range(total_chunks):
366  offset = chunk * multipart_bytes
367  bytes = min(multipart_bytes, key_size - offset)
368  with FileChunkIO(
369  filename, 'r', offset=offset, bytes=bytes) as fp:
370  logging.info('Sending chunk {c} of {tc}...'.format(
371  c=chunk + 1, tc=total_chunks))
372  mp.upload_part_from_file(fp, part_num=chunk + 1)
373  except:
374  mp.cancel_upload()
375  raise
376  mp.complete_upload()
377  else:
378  # regular upload
379  if not key_obj:
380  key_obj = bucket.new_key(key_name=key)
381  key_size = key_obj.set_contents_from_filename(filename,
382  replace=replace,
383  encrypt_key=encrypt)
384  logging.info("The key {key} now contains"
385  " {key_size} bytes".format(**locals()))
