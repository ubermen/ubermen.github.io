25 class SparkSubmitHook(BaseHook):
26  """
27  This hook is a wrapper around the spark-submit binary to kick off a spark-submit job.
28  It requires that the "spark-submit" binary is in the PATH.
29  :param conf: Arbitrary Spark configuration properties
30  :type conf: dict
31  :param conn_id: The connection id as configured in Airflow administration. When an
32  invalid connection_id is supplied, it will default to yarn.
33  :type conn_id: str
34  :param files: Upload additional files to the container running the job, separated by a
35  comma. For example hive-site.xml.
36  :type files: str
37  :param py_files: Additional python files used by the job, can be .zip, .egg or .py.
38  :type py_files: str
39  :param jars: Submit additional jars to upload and place them in executor classpath.
40  :type jars: str
41  :param executor_cores: Number of cores per executor (Default: 2)
42  :type executor_cores: int
43  :param executor_memory: Memory per executor (e.g. 1000M, 2G) (Default: 1G)
44  :type executor_memory: str
45  :param keytab: Full path to the file that contains the keytab
46  :type keytab: str
47  :param principal: The name of the kerberos principal used for keytab
48  :type principal: str
49  :param name: Name of the job (default airflow-spark)
50  :type name: str
51  :param num_executors: Number of executors to launch
52  :type num_executors: int
53  :param verbose: Whether to pass the verbose flag to spark-submit process for debugging
54  :type verbose: bool
55  """

88  def _resolve_connection(self):
89  # Build from connection master or default to yarn if not available
90  master = 'yarn'
91  queue = None
92  deploy_mode = None
93 
94  try:

96  conn = self.get_connection(self._conn_id)
97  if conn.port:

168  def submit(self, application="", **kwargs):
169  """
170  Remote Popen to execute the spark-submit job
171 
172  :param application: Submitted application, jar or py file
173  :type application: str
174  :param kwargs: extra arguments to Popen (see subprocess.Popen)
175  """
176  spark_submit_cmd = self._build_command(application)
177  self._sp = subprocess.Popen(spark_submit_cmd,
178  stdout=subprocess.PIPE,
179  stderr=subprocess.PIPE,
180  **kwargs)
181 
182  # Using two iterators here to support 'real-time' logging
183  sources = [self._sp.stdout, self._sp.stderr]
184 
185  for source in sources:
186  self._process_log(iter(source.readline, b''))
187 
188  output, stderr = self._sp.communicate()
189 
190  if self._sp.returncode:
191  raise AirflowException(
192  "Cannot execute: {}. Error code is: {}. Output: {}, Stderr: {}".format(
193  spark_submit_cmd, self._sp.returncode, output, stderr
194  )
195  )

197  def _process_log(self, itr):
198  """
199  Processes the log files and extracts useful information out of it
200 
201  :param itr: An iterator which iterates over the input of the subprocess
202  """
203  # Consume the iterator
204  for line in itr:
205  line = line.decode('utf-8').strip()
206  # If we run yarn cluster mode, we want to extract the application id from
207  # the logs so we can kill the application when we stop it unexpectedly
208  if self._is_yarn and self._deploy_mode == 'cluster':
209  match = re.search('(application[0-9_]+)', line)
210  if match:
211  self._yarn_application_id = match.groups()[0]
212 
213  # Pass to logging
214  logging.info(line)

216  def on_kill(self):
217  if self._sp and self._sp.poll() is None:
218  logging.info('Sending kill signal to spark-submit')
219  self.sp.kill()
220 
221  if self._yarn_application_id:

224  stdout=subprocess.PIPE,
