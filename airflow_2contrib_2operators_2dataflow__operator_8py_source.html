25 class DataFlowJavaOperator(BaseOperator):
26  """
27  Start a Java Cloud DataFlow batch job. The parameters of the operation
28  will be passed to the job.
29 
30  It's a good practice to define dataflow_* parameters in the default_args of the dag
31  like the project, zone and staging location.
32 
33  ```
34  default_args = {
35  'dataflow_default_options': {
36  'project': 'my-gcp-project',
37  'zone': 'europe-west1-d',
38  'stagingLocation': 'gs://my-staging-bucket/staging/'
39  }
40  }
41  ```
42 
43  You need to pass the path to your dataflow as a file reference with the ``jar``
44  parameter, the jar needs to be a self executing jar. Use ``options`` to pass on
45  options to your job.
46 
47  ```
48  t1 = DataFlowOperation(
49  task_id='datapflow_example',
50  jar='{{var.value.gcp_dataflow_base}}pipeline/build/libs/pipeline-example-1.0.jar',
51  options={
52  'autoscalingAlgorithm': 'BASIC',
53  'maxNumWorkers': '50',
54  'start': '{{ds}}',
55  'partitionType': 'DAY'
56  },
57  dag=my-dag)
58  ```
59 
60  Both ``jar`` and ``options`` are templated so you can use variables in them.
61  """
62  template_fields = ['options', 'jar']
63  ui_color = '#0273d4'

193 class GoogleCloudBucketHelper():
194  """GoogleCloudStorageHook helper class to download GCS object."""
195  GCS_PREFIX_LENGTH = 5
196 
197  def __init__(self,
198  gcp_conn_id='google_cloud_default',
199  delegate_to=None):
200  self._gcs_hook = GoogleCloudStorageHook(gcp_conn_id, delegate_to)

202  def google_cloud_to_local(self, file_name):
203  """
204  Checks whether the file specified by file_name is stored in Google Cloud
205  Storage (GCS), if so, downloads the file and saves it locally. The full
206  path of the saved file will be returned. Otherwise the local file_name
207  will be returned immediately.
208 
209  :param file_name: The full path of input file.
210  :type file_name: string
211  :return: The full path of local file.
212  :type: string
213  """
214  if not file_name.startswith('gs://'):
215  return file_name
216 
217  # Extracts bucket_id and object_id by first removing 'gs://' prefix and
218  # then split the remaining by path delimiter '/'.
219  path_components = file_name[self.GCS_PREFIX_LENGTH:].split('/')
220  if path_components < 2:
221  raise Exception(
222  'Invalid Google Cloud Storage (GCS) object path: {}.'
223  .format(file_name))
224 
225  bucket_id = path_components[0]
226  object_id = '/'.join(path_components[1:])
227  local_file = '/tmp/dataflow{}-{}'.format(str(uuid.uuid1())[:8],
228  path_components[-1])
229  file_size = self._gcs_hook.download(bucket_id, object_id, local_file)
230 
231  if file_size > 0:
232  return local_file
233  raise Exception(
234  'Failed to download Google Cloud Storage GCS object: {}'
235  .format(file_name))
