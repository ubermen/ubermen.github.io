98  def execute(self, context):
99  args = self.cls.parse(self.create_cmd_args(context))
100  self.cmd = self.cls.create(**args)
101  context['task_instance'].xcom_push(key='qbol_cmd_id', value=self.cmd.id)
102  logging.info("Qubole command created with Id: %s and Status: %s",
103  self.cmd.id, self.cmd.status)
104 
105  while not Command.is_done(self.cmd.status):
106  time.sleep(Qubole.poll_interval)
107  self.cmd = self.cls.find(self.cmd.id)
108  logging.info("Command Id: %s and Status: %s", self.cmd.id, self.cmd.status)
109 
110  if 'fetch_logs' in self.kwargs and self.kwargs['fetch_logs'] is True:
111  logging.info("Logs for Command Id: %s \n%s", self.cmd.id, self.cmd.get_log())
112 
113  if self.cmd.status != 'done':
114  raise AirflowException('Command Id: {0} failed with Status: {1}'.format(
115  self.cmd.id, self.cmd.status))

130  def get_results(self, ti=None, fp=None, inline=True, delim=None, fetch=True):
131  """
132  Get results (or just s3 locations) of a command from Qubole and save into a file
133  :param ti: Task Instance of the dag, used to determine the Quboles command id
134  :param fp: Optional file pointer, will create one and return if None passed
135  :param inline: True to download actual results, False to get s3 locations only
136  :param delim: Replaces the CTL-A chars with the given delim, defaults to ','
137  :param fetch: when inline is True, get results directly from s3 (if large)
138  :return: file location containing actual results or s3 locations of results

176  def create_cmd_args(self, context):
177  args = []
178  cmd_type = self.kwargs['command_type']
179  inplace_args = None
180  tags = set([self.dag_id, self.task_id, context['run_id']])
181 
182  for k,v in self.kwargs.items():
183  if k in COMMAND_ARGS[cmd_type]:
184  if k in HYPHEN_ARGS:
185  args.append("--{0}={1}".format(k.replace('_', '-'),v))
186  elif k in POSITIONAL_ARGS:
187  inplace_args = v
188  elif k == 'tags':
189  if isinstance(v, six.string_types):
190  tags.add(v)
191  elif isinstance(v, (list, tuple)):
192  for val in v:
193  tags.add(val)
194  else:
195  args.append("--{0}={1}".format(k,v))
196 
197  if k == 'notify' and v is True:
198  args.append("--notify")
199 
200  args.append("--tags={0}".format(','.join(filter(None,tags))))
201 
202  if inplace_args is not None:
203  if cmd_type == 'hadoopcmd':
204  args += inplace_args.split(' ', 1)
205  else:
206  args += inplace_args.split(' ')
207 
208  return args
