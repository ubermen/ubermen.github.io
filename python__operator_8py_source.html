25 class PythonOperator(BaseOperator):
26  """
27  Executes a Python callable
28 
29  :param python_callable: A reference to an object that is callable
30  :type python_callable: python callable
31  :param op_kwargs: a dictionary of keyword arguments that will get unpacked
32  in your function
33  :type op_kwargs: dict
34  :param op_args: a list of positional arguments that will get unpacked when
35  calling your callable
36  :type op_args: list
37  :param provide_context: if set to true, Airflow will pass a set of
38  keyword arguments that can be used in your function. This set of
39  kwargs correspond exactly to what you can use in your jinja
40  templates. For this to work, you need to define `**kwargs` in your
41  function header.
42  :type provide_context: bool
43  :param templates_dict: a dictionary where the values are templates that
44  will get templated by the Airflow engine sometime between
45  ``__init__`` and ``execute`` takes place and are made available
46  in your callable's context after the template has been applied
47  :type templates_dict: dict of str
48  :param templates_exts: a list of file extensions to resolve while
49  processing templated fields, for examples ``['.sql', '.hql']``
50  """

85 class BranchPythonOperator(PythonOperator):
86  """
87  Allows a workflow to "branch" or follow a single path following the
88  execution of this task.
89 
90  It derives the PythonOperator and expects a Python function that returns
91  the task_id to follow. The task_id returned should point to a task
92  directly downstream from {self}. All other "branches" or
93  directly downstream tasks are marked with a state of ``skipped`` so that
94  these paths can't move forward. The ``skipped`` states are propageted
95  downstream to allow for the DAG state to fill up and the DAG run's state
96  to be inferred.
97 
98  Note that using tasks with ``depends_on_past=True`` downstream from
99  ``BranchPythonOperator`` is logically unsound as ``skipped`` status
100  will invariably lead to block tasks that depend on their past successes.
101  ``skipped`` states propagates where all directly upstream tasks are
102  ``skipped``.
103  """
104  def execute(self, context):
105  branch = super(BranchPythonOperator, self).execute(context)
106  logging.info("Following branch " + branch)
107  logging.info("Marking other directly downstream tasks as skipped")
108  session = settings.Session()
109  for task in context['task'].downstream_list:
110  if task.task_id != branch:
111  ti = TaskInstance(
112  task, execution_date=context['ti'].execution_date)
113  ti.state = State.SKIPPED
114  ti.start_date = datetime.now()
115  ti.end_date = datetime.now()
116  session.merge(ti)
117  session.commit()
118  session.close()
119  logging.info("Done.")

122 class ShortCircuitOperator(PythonOperator):
123  """
124  Allows a workflow to continue only if a condition is met. Otherwise, the
125  workflow "short-circuits" and downstream tasks are skipped.
126 
127  The ShortCircuitOperator is derived from the PythonOperator. It evaluates a
128  condition and short-circuits the workflow if the condition is False. Any
129  downstream tasks are marked with a state of "skipped". If the condition is
130  True, downstream tasks proceed as normal.
131 
132  The condition is determined by the result of `python_callable`.
133  """
134  def execute(self, context):
135  condition = super(ShortCircuitOperator, self).execute(context)
136  logging.info("Condition result is {}".format(condition))
137  if condition:
138  logging.info('Proceeding with downstream tasks...')
139  return
140  else:
141  logging.info('Skipping downstream tasks...')
142  session = settings.Session()
143  for task in context['task'].downstream_list:
144  ti = TaskInstance(
145  task, execution_date=context['ti'].execution_date)
146  ti.state = State.SKIPPED
147  ti.start_date = datetime.now()
148  ti.end_date = datetime.now()
149  session.merge(ti)
150  session.commit()
151  session.close()
152  logging.info("Done.")
