116  def issue_field_checker(self, context, issue):
117  result = None
118  try:
119  if issue is not None \
120  and self.field is not None \
121  and self.expected_value is not None:
122 
123  field_val = getattr(issue.fields, self.field)
124  if field_val is not None:
125  if isinstance(field_val, list):
126  result = self.expected_value in field_val
127  elif isinstance(field_val, str):
128  result = self.expected_value.lower() == field_val.lower()
129  elif isinstance(field_val, Resource) and getattr(field_val, 'name'):
130  result = self.expected_value.lower() == field_val.name.lower()
131  else:
