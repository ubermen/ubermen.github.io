28 class PostgresHook(DbApiHook):

45  def get_conn(self):
46  conn = self.get_connection(self.postgres_conn_id)
47  conn_args = dict(
48  host=conn.host,
49  user=conn.login,
50  password=conn.password,
51  dbname=self.schema or conn.schema,
52  port=conn.port)
53  # check for ssl parameters in conn.extra
54  for arg_name, arg_val in conn.extra_dejson.items():
55  if arg_name in ['sslmode', 'sslcert', 'sslkey',
56  'sslrootcert', 'sslcrl', 'application_name',
57  'keepalives_idle']:
58  conn_args[arg_name] = arg_val
59 
60  self.conn = psycopg2.connect(**conn_args)
61  return self.conn

63  def copy_expert(self, sql, filename, open=open):
64  """
65  Executes SQL using psycopg2 copy_expert method.
66  Necessary to execute COPY command without access to a superuser.
67 
68  Note: if this method is called with a "COPY FROM" statement and
69  the specified input file does not exist, it creates an empty
70  file and no data is loaded, but the operation succeeds.
71  So if users want to be aware when the input file does not exist,
72  they have to check its existence by themselves.
73  """
74  if not os.path.isfile(filename):
75  with open(filename, 'w'):
76  pass
77 
78  with open(filename, 'r+') as f:
79  with closing(self.get_conn()) as conn:
80  with closing(conn.cursor()) as cur:
81  cur.copy_expert(sql, f)
82  f.truncate(f.tell())
83  conn.commit()
