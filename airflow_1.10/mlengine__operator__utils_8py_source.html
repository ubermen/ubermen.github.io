32 def create_evaluate_ops(task_prefix,
33  data_format,
34  input_paths,
35  prediction_path,
36  metric_fn_and_keys,
37  validate_fn,
38  batch_prediction_job_id=None,
39  project_id=None,
40  region=None,
41  dataflow_options=None,
42  model_uri=None,
43  model_name=None,
44  version_name=None,
45  dag=None):
46  """
47  Creates Operators needed for model evaluation and returns.
48 
49  It gets prediction over inputs via Cloud ML Engine BatchPrediction API by
50  calling MLEngineBatchPredictionOperator, then summarize and validate
51  the result via Cloud Dataflow using DataFlowPythonOperator.
52 
53  For details and pricing about Batch prediction, please refer to the website
54  https://cloud.google.com/ml-engine/docs/how-tos/batch-predict
55  and for Cloud Dataflow, https://cloud.google.com/dataflow/docs/
56 
57  It returns three chained operators for prediction, summary, and validation,
58  named as <prefix>-prediction, <prefix>-summary, and <prefix>-validation,
59  respectively.
60  (<prefix> should contain only alphanumeric characters or hyphen.)
61 
62  The upstream and downstream can be set accordingly like:
63  pred, _, val = create_evaluate_ops(...)
64  pred.set_upstream(upstream_op)
65  ...
66  downstream_op.set_upstream(val)
67 
68  Callers will provide two python callables, metric_fn and validate_fn, in
69  order to customize the evaluation behavior as they wish.
70  - metric_fn receives a dictionary per instance derived from json in the
71  batch prediction result. The keys might vary depending on the model.
72  It should return a tuple of metrics.
73  - validation_fn receives a dictionary of the averaged metrics that metric_fn
74  generated over all instances.
75  The key/value of the dictionary matches to what's given by
76  metric_fn_and_keys arg.
77  The dictionary contains an additional metric, 'count' to represent the
78  total number of instances received for evaluation.
79  The function would raise an exception to mark the task as failed, in a
80  case the validation result is not okay to proceed (i.e. to set the trained
81  version as default).
82 
83  Typical examples are like this:
84 
85  def get_metric_fn_and_keys():
86  import math # imports should be outside of the metric_fn below.
87  def error_and_squared_error(inst):
88  label = float(inst['input_label'])
89  classes = float(inst['classes']) # 0 or 1
90  err = abs(classes-label)
91  squared_err = math.pow(classes-label, 2)
92  return (err, squared_err) # returns a tuple.
93  return error_and_squared_error, ['err', 'mse'] # key order must match.
94 
95  def validate_err_and_count(summary):
96  if summary['err'] > 0.2:
97  raise ValueError('Too high err>0.2; summary=%s' % summary)
98  if summary['mse'] > 0.05:
99  raise ValueError('Too high mse>0.05; summary=%s' % summary)
100  if summary['count'] < 1000:
101  raise ValueError('Too few instances<1000; summary=%s' % summary)
102  return summary
103 
104  For the details on the other BatchPrediction-related arguments (project_id,
105  job_id, region, data_format, input_paths, prediction_path, model_uri),
106  please refer to MLEngineBatchPredictionOperator too.
107 
108  :param task_prefix: a prefix for the tasks. Only alphanumeric characters and
109  hyphen are allowed (no underscores), since this will be used as dataflow
110  job name, which doesn't allow other characters.
111  :type task_prefix: string
112 
113  :param data_format: either of 'TEXT', 'TF_RECORD', 'TF_RECORD_GZIP'
114  :type data_format: string
115 
116  :param input_paths: a list of input paths to be sent to BatchPrediction.
117  :type input_paths: list of strings
118 
119  :param prediction_path: GCS path to put the prediction results in.
120  :type prediction_path: string
121 
122  :param metric_fn_and_keys: a tuple of metric_fn and metric_keys:
123  - metric_fn is a function that accepts a dictionary (for an instance),
124  and returns a tuple of metric(s) that it calculates.
125  - metric_keys is a list of strings to denote the key of each metric.
126  :type metric_fn_and_keys: tuple of a function and a list of strings
127 
128  :param validate_fn: a function to validate whether the averaged metric(s) is
129  good enough to push the model.
130  :type validate_fn: function
131 
132  :param batch_prediction_job_id: the id to use for the Cloud ML Batch
133  prediction job. Passed directly to the MLEngineBatchPredictionOperator as
134  the job_id argument.
135  :type batch_prediction_job_id: string
136 
137  :param project_id: the Google Cloud Platform project id in which to execute
138  Cloud ML Batch Prediction and Dataflow jobs. If None, then the `dag`'s
139  `default_args['project_id']` will be used.
140  :type project_id: string
141 
142  :param region: the Google Cloud Platform region in which to execute Cloud ML
143  Batch Prediction and Dataflow jobs. If None, then the `dag`'s
144  `default_args['region']` will be used.
145  :type region: string
146 
147  :param dataflow_options: options to run Dataflow jobs. If None, then the
148  `dag`'s `default_args['dataflow_default_options']` will be used.
149  :type dataflow_options: dictionary
150 
151  :param model_uri: GCS path of the model exported by Tensorflow using
152  tensorflow.estimator.export_savedmodel(). It cannot be used with
153  model_name or version_name below. See MLEngineBatchPredictionOperator for
154  more detail.
155  :type model_uri: string
156 
157  :param model_name: Used to indicate a model to use for prediction. Can be
158  used in combination with version_name, but cannot be used together with
159  model_uri. See MLEngineBatchPredictionOperator for more detail. If None,
160  then the `dag`'s `default_args['model_name']` will be used.
161  :type model_name: string
162 
163  :param version_name: Used to indicate a model version to use for prediciton,
164  in combination with model_name. Cannot be used together with model_uri.
165  See MLEngineBatchPredictionOperator for more detail. If None, then the
166  `dag`'s `default_args['version_name']` will be used.
167  :type version_name: string
168 
169  :param dag: The `DAG` to use for all Operators.
170  :type dag: airflow.DAG
171 
172  :returns: a tuple of three operators, (prediction, summary, validation)
173  :rtype: tuple(DataFlowPythonOperator, DataFlowPythonOperator,
174  PythonOperator)
175  """
176 
177  # Verify that task_prefix doesn't have any special characters except hyphen
178  # '-', which is the only allowed non-alphanumeric character by Dataflow.
179  if not re.match(r"^[a-zA-Z][-A-Za-z0-9]*$", task_prefix):
180  raise AirflowException(
181  "Malformed task_id for DataFlowPythonOperator (only alphanumeric "
182  "and hyphens are allowed but got: " + task_prefix)
183 
184  metric_fn, metric_keys = metric_fn_and_keys
185  if not callable(metric_fn):
186  raise AirflowException("`metric_fn` param must be callable.")
187  if not callable(validate_fn):
188  raise AirflowException("`validate_fn` param must be callable.")
189 
190  if dag is not None and dag.default_args is not None:
191  default_args = dag.default_args
192  project_id = project_id or default_args.get('project_id')
193  region = region or default_args.get('region')
194  model_name = model_name or default_args.get('model_name')
195  version_name = version_name or default_args.get('version_name')
196  dataflow_options = dataflow_options or \
197  default_args.get('dataflow_default_options')
198 
199  evaluate_prediction = MLEngineBatchPredictionOperator(
200  task_id=(task_prefix + "-prediction"),
201  project_id=project_id,
202  job_id=batch_prediction_job_id,
203  region=region,
204  data_format=data_format,
205  input_paths=input_paths,
206  output_path=prediction_path,
207  uri=model_uri,
208  model_name=model_name,
209  version_name=version_name,
210  dag=dag)
211 
212  metric_fn_encoded = base64.b64encode(dill.dumps(metric_fn, recurse=True))
213  evaluate_summary = DataFlowPythonOperator(
214  task_id=(task_prefix + "-summary"),
215  py_options=["-m"],
216  py_file="airflow.contrib.operators.mlengine_prediction_summary",
217  dataflow_default_options=dataflow_options,
218  options={
219  "prediction_path": prediction_path,
220  "metric_fn_encoded": metric_fn_encoded,
221  "metric_keys": ','.join(metric_keys)
222  },
223  dag=dag)
224  evaluate_summary.set_upstream(evaluate_prediction)
225 
226  def apply_validate_fn(*args, **kwargs):
227  prediction_path = kwargs["templates_dict"]["prediction_path"]
228  scheme, bucket, obj, _, _ = urlsplit(prediction_path)
229  if scheme != "gs" or not bucket or not obj:
230  raise ValueError("Wrong format prediction_path: %s",
231  prediction_path)
232  summary = os.path.join(obj.strip("/"),
233  "prediction.summary.json")
234  gcs_hook = GoogleCloudStorageHook()
235  summary = json.loads(gcs_hook.download(bucket, summary))
236  return validate_fn(summary)
237 
238  evaluate_validation = PythonOperator(
239  task_id=(task_prefix + "-validation"),
240  python_callable=apply_validate_fn,
241  provide_context=True,
242  templates_dict={"prediction_path": prediction_path},
243  dag=dag)
244  evaluate_validation.set_upstream(evaluate_summary)
245 
246  return evaluate_prediction, evaluate_summary, evaluate_validation
