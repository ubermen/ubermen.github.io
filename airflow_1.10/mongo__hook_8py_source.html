40  def get_conn(self):

42  Fetches PyMongo Client
43  """

45  return self.client

48 
49  uri = 'mongodb://{creds}{host}{port}/{database}'.format(

53 
54  host=conn.host,

56  database='' if conn.schema is None else conn.schema

60  options = self.extras
