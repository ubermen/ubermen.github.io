25 class GenericTransfer(BaseOperator):
26  """
27  Moves data from a connection to another, assuming that they both
28  provide the required methods in their respective hooks. The source hook
29  needs to expose a `get_records` method, and the destination a
30  `insert_rows` method.
31 
32  This is meant to be used on small-ish datasets that fit in memory.
33 
34  :param sql: SQL query to execute against the source database. (templated)
35  :type sql: str
36  :param destination_table: target table. (templated)
37  :type destination_table: str
38  :param source_conn_id: source connection
39  :type source_conn_id: str
40  :param destination_conn_id: source connection
41  :type destination_conn_id: str
42  :param preoperator: sql statement or list of statements to be
43  executed prior to loading the data. (templated)
44  :type preoperator: str or list of str
45  """
46

48  template_ext = ('.sql', '.hql',)

68  source_hook = BaseHook.get_hook(self.source_conn_id)
69 
70  self.log.info("Extracting data from %s", self.source_conn_id)
71  self.log.info("Executing: \n %s", self.sql)
