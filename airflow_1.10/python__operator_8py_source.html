37 class PythonOperator(BaseOperator):
38  """
39  Executes a Python callable
40 
41  :param python_callable: A reference to an object that is callable
42  :type python_callable: python callable
43  :param op_kwargs: a dictionary of keyword arguments that will get unpacked
44  in your function
45  :type op_kwargs: dict
46  :param op_args: a list of positional arguments that will get unpacked when
47  calling your callable
48  :type op_args: list
49  :param provide_context: if set to true, Airflow will pass a set of
50  keyword arguments that can be used in your function. This set of
51  kwargs correspond exactly to what you can use in your jinja
52  templates. For this to work, you need to define `**kwargs` in your
53  function header.
54  :type provide_context: bool
55  :param templates_dict: a dictionary where the values are templates that
56  will get templated by the Airflow engine sometime between
57  ``__init__`` and ``execute`` takes place and are made available
58  in your callable's context after the template has been applied. (templated)
59  :type templates_dict: dict of str
60  :param templates_exts: a list of file extensions to resolve while
61  processing templated fields, for examples ``['.sql', '.hql']``
62  :type templates_exts: list(str)
63  """
64  template_fields = ('templates_dict',)
65  template_ext = tuple()
66  ui_color = '#ffefeb'

69  def __init__(
70  self,
71  python_callable,
72  op_args=None,
73  op_kwargs=None,
74  provide_context=False,
75  templates_dict=None,
76  templates_exts=None,
77  *args, **kwargs):
78  super(PythonOperator, self).__init__(*args, **kwargs)

103 class BranchPythonOperator(PythonOperator, SkipMixin):
104  """
105  Allows a workflow to "branch" or follow a single path following the
106  execution of this task.
107 
108  It derives the PythonOperator and expects a Python function that returns
109  the task_id to follow. The task_id returned should point to a task
110  directly downstream from {self}. All other "branches" or
111  directly downstream tasks are marked with a state of ``skipped`` so that
112  these paths can't move forward. The ``skipped`` states are propageted
113  downstream to allow for the DAG state to fill up and the DAG run's state
114  to be inferred.
115 
116  Note that using tasks with ``depends_on_past=True`` downstream from
117  ``BranchPythonOperator`` is logically unsound as ``skipped`` status
118  will invariably lead to block tasks that depend on their past successes.
119  ``skipped`` states propagates where all directly upstream tasks are
120  ``skipped``.
121  """
122  def execute(self, context):
123  branch = super(BranchPythonOperator, self).execute(context)
124  self.log.info("Following branch %s", branch)
125  self.log.info("Marking other directly downstream tasks as skipped")
126 
127  downstream_tasks = context['task'].downstream_list
128  self.log.debug("Downstream task_ids %s", downstream_tasks)
129 
130  skip_tasks = [t for t in downstream_tasks if t.task_id != branch]
131  if downstream_tasks:
132  self.skip(context['dag_run'], context['ti'].execution_date, skip_tasks)
133 
134  self.log.info("Done.")

137 class ShortCircuitOperator(PythonOperator, SkipMixin):
138  """
139  Allows a workflow to continue only if a condition is met. Otherwise, the
140  workflow "short-circuits" and downstream tasks are skipped.
141 
142  The ShortCircuitOperator is derived from the PythonOperator. It evaluates a
143  condition and short-circuits the workflow if the condition is False. Any
144  downstream tasks are marked with a state of "skipped". If the condition is
145  True, downstream tasks proceed as normal.
146 
147  The condition is determined by the result of `python_callable`.
148  """
149  def execute(self, context):
150  condition = super(ShortCircuitOperator, self).execute(context)
151  self.log.info("Condition result is %s", condition)
152 
153  if condition:
154  self.log.info('Proceeding with downstream tasks...')
155  return
156 
157  self.log.info('Skipping downstream tasks...')
158 
159  downstream_tasks = context['task'].get_flat_relatives(upstream=False)
160  self.log.debug("Downstream task_ids %s", downstream_tasks)
161 
162  if downstream_tasks:
163  self.skip(context['dag_run'], context['ti'].execution_date, downstream_tasks)
164 
165  self.log.info("Done.")

168 class PythonVirtualenvOperator(PythonOperator):
169  """
170  Allows one to run a function in a virtualenv that is created and destroyed
171  automatically (with certain caveats).
172 
173  The function must be defined using def, and not be
174  part of a class. All imports must happen inside the function
175  and no variables outside of the scope may be referenced. A global scope
176  variable named virtualenv_string_args will be available (populated by
177  string_args). In addition, one can pass stuff through op_args and op_kwargs, and one
178  can use a return value.
179 
180  Note that if your virtualenv runs in a different Python major version than Airflow,
181  you cannot use return values, op_args, or op_kwargs. You can use string_args though.
182 
183  :param python_callable: A python function with no references to outside variables,
184  defined with def, which will be run in a virtualenv
185  :type python_callable: function
186  :param requirements: A list of requirements as specified in a pip install command
187  :type requirements: list(str)
188  :param python_version: The Python version to run the virtualenv with. Note that
189  both 2 and 2.7 are acceptable forms.
190  :type python_version: str
191  :param use_dill: Whether to use dill to serialize
192  the args and result (pickle is default). This allow more complex types
193  but requires you to include dill in your requirements.
194  :type use_dill: bool
195  :param system_site_packages: Whether to include
196  system_site_packages in your virtualenv.
197  See virtualenv documentation for more information.
198  :type system_site_packages: bool
199  :param op_args: A list of positional arguments to pass to python_callable.
200  :type op_kwargs: list
201  :param op_kwargs: A dict of keyword arguments to pass to python_callable.
202  :type op_kwargs: dict
203  :param string_args: Strings that are present in the global var virtualenv_string_args,
204  available to python_callable at runtime as a list(str). Note that args are split
205  by newline.
206  :type string_args: list(str)
207  :param templates_dict: a dictionary where the values are templates that
208  will get templated by the Airflow engine sometime between
209  ``__init__`` and ``execute`` takes place and are made available
210  in your callable's context after the template has been applied
211  :type templates_dict: dict of str
212  :param templates_exts: a list of file extensions to resolve while
213  processing templated fields, for examples ``['.sql', '.hql']``
214  :type templates_exts: list(str)
215  """
216  def __init__(self, python_callable,
217  requirements=None,
218  python_version=None, use_dill=False,
219  system_site_packages=True,
220  op_args=None, op_kwargs=None, string_args=None,
221  templates_dict=None, templates_exts=None, *args, **kwargs):
222  super(PythonVirtualenvOperator, self).__init__(
223  python_callable=python_callable,
224  op_args=op_args,
225  op_kwargs=op_kwargs,
226  templates_dict=templates_dict,
227  templates_exts=templates_exts,
228  provide_context=False,
229  *args,
230  **kwargs)
231  self.requirements = requirements or []
232  self.string_args = string_args or []
233  self.python_version = python_version
234  self.use_dill = use_dill
235  self.system_site_packages = system_site_packages
236  # check that dill is present if needed
237  dill_in_requirements = map(lambda x: x.lower().startswith('dill'),
238  self.requirements)
239  if (not system_site_packages) and use_dill and not any(dill_in_requirements):
240  raise AirflowException('If using dill, dill must be in the environment ' +
241  'either via system_site_packages or requirements')
242  # check that a function is passed, and that it is not a lambda
243  if (not isinstance(self.python_callable,
244  types.FunctionType) or (self.python_callable.__name__ ==

247  self.__class__.__name__)

250  str(python_version)[0] != str(sys.version_info[0]) and
251  self._pass_op_args()):

253  "different Python major versions "
254  "for PythonVirtualenvOperator. "
255  "Please use string_args.")

360  def _generate_python_code(self):
361  if self.use_dill:
362  pickling_library = 'dill'
363  else:
364  pickling_library = 'pickle'
365  fn = self.python_callable
366  # dont try to read pickle if we didnt pass anything
367  if self._pass_op_args():
368  load_args_line = 'with open(sys.argv[1], "rb") as f: arg_dict = {}.load(f)'\
369  .format(pickling_library)
370  else:
371  load_args_line = 'arg_dict = {"args": [], "kwargs": {}}'
372 
373  # no indents in original code so we can accept
374  # any type of indents in the original function
375  # we deserialize args, call function, serialize result if necessary
376  return dedent("""\
377  import {pickling_library}
378  import sys
379  {load_args_code}
380  args = arg_dict["args"]
381  kwargs = arg_dict["kwargs"]
382  with open(sys.argv[3], 'r') as f:
383  virtualenv_string_args = list(map(lambda x: x.strip(), list(f)))
384  {python_callable_lines}
385  res = {python_callable_name}(*args, **kwargs)
386  with open(sys.argv[2], 'wb') as f:
387  res is not None and {pickling_library}.dump(res, f)
388  """).format(load_args_code=load_args_line,
389  python_callable_lines=dedent(inspect.getsource(fn)),
390  python_callable_name=fn.__name__,
391  pickling_library=pickling_library)
392 
393  self.log.info("Done.")
