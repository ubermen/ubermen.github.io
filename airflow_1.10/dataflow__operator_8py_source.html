31 class DataFlowJavaOperator(BaseOperator):
32  """
33  Start a Java Cloud DataFlow batch job. The parameters of the operation
34  will be passed to the job.
35 
36  It's a good practice to define dataflow_* parameters in the default_args of the dag
37  like the project, zone and staging location.
38 
39  .. code-block:: python
40 
41  default_args = {
42  'dataflow_default_options': {
43  'project': 'my-gcp-project',
44  'zone': 'europe-west1-d',
45  'stagingLocation': 'gs://my-staging-bucket/staging/'
46  }
47  }
48 
49  You need to pass the path to your dataflow as a file reference with the ``jar``
50  parameter, the jar needs to be a self executing jar (see documentation here:
51  https://beam.apache.org/documentation/runners/dataflow/#self-executing-jar).
52  Use ``options`` to pass on options to your job.
53 
54  .. code-block:: python
55 
56  t1 = DataFlowOperation(
57  task_id='datapflow_example',
58  jar='{{var.value.gcp_dataflow_base}}pipeline/build/libs/pipeline-example-1.0.jar',
59  options={
60  'autoscalingAlgorithm': 'BASIC',
61  'maxNumWorkers': '50',
62  'start': '{{ds}}',
63  'partitionType': 'DAY',
64  'labels': {'foo' : 'bar'}
65  },
66  gcp_conn_id='gcp-airflow-service-account',
67  dag=my-dag)
68 
69  Both ``jar`` and ``options`` are templated so you can use variables in them.
70  """

334 class GoogleCloudBucketHelper():
335  """GoogleCloudStorageHook helper class to download GCS object."""
336  GCS_PREFIX_LENGTH = 5
337 
338  def __init__(self,
339  gcp_conn_id='google_cloud_default',
340  delegate_to=None):
341  self._gcs_hook = GoogleCloudStorageHook(gcp_conn_id, delegate_to)
342 
343  def google_cloud_to_local(self, file_name):
344  """
345  Checks whether the file specified by file_name is stored in Google Cloud
346  Storage (GCS), if so, downloads the file and saves it locally. The full
347  path of the saved file will be returned. Otherwise the local file_name
348  will be returned immediately.
349 
350  :param file_name: The full path of input file.
351  :type file_name: string
352  :return: The full path of local file.
353  :type: string
354  """
355  if not file_name.startswith('gs://'):
356  return file_name
357 
358  # Extracts bucket_id and object_id by first removing 'gs://' prefix and
359  # then split the remaining by path delimiter '/'.
360  path_components = file_name[self.GCS_PREFIX_LENGTH:].split('/')
361  if path_components < 2:
362  raise Exception(
363  'Invalid Google Cloud Storage (GCS) object path: {}.'
364  .format(file_name))
365 
366  bucket_id = path_components[0]
367  object_id = '/'.join(path_components[1:])
368  local_file = '/tmp/dataflow{}-{}'.format(str(uuid.uuid1())[:8],
369  path_components[-1])
370  file_size = self._gcs_hook.download(bucket_id, object_id, local_file)
371 
372  if file_size > 0:
373  return local_file
374  raise Exception(
375  'Failed to download Google Cloud Storage GCS object: {}'
376  .format(file_name))
