34 class VerticaToMySqlTransfer(BaseOperator):
35  """
36  Moves data from Vertica to MySQL.
37 
38  :param sql: SQL query to execute against the Vertica database. (templated)
39  :type sql: str
40  :param vertica_conn_id: source Vertica connection
41  :type vertica_conn_id: str
42  :param mysql_table: target MySQL table, use dot notation to target a
43  specific database. (templated)
44  :type mysql_table: str
45  :param mysql_conn_id: source mysql connection
46  :type mysql_conn_id: str
47  :param mysql_preoperator: sql statement to run against MySQL prior to
48  import, typically use to truncate of delete in place of the data
49  coming in, allowing the task to be idempotent (running the task
50  twice won't double load data). (templated)
51  :type mysql_preoperator: str
52  :param mysql_postoperator: sql statement to run against MySQL after the
53  import, typically used to move data from staging to production
54  and issue cleanup commands. (templated)
55  :type mysql_postoperator: str
56  :param bulk_load: flag to use bulk_load option. This loads MySQL directly
57  from a tab-delimited text file using the LOAD DATA LOCAL INFILE command.
58  This option requires an extra connection parameter for the
59  destination MySQL connection: {'local_infile': true}.
60  :type bulk_load: bool
61  """
