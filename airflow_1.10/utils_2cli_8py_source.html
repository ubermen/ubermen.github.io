37 def action_logging(f):
38  """
39  Decorates function to execute function at the same time submitting action_logging
40  but in CLI context. It will call action logger callbacks twice,
41  one for pre-execution and the other one for post-execution.
42 
43  Action logger will be called with below keyword parameters:
44  sub_command : name of sub-command
45  start_datetime : start datetime instance by utc
46  end_datetime : end datetime instance by utc
47  full_command : full command line arguments
48  user : current user
49  log : airflow.models.Log ORM instance
50  dag_id : dag id (optional)
51  task_id : task_id (optional)
52  execution_date : execution date (optional)
53  error : exception instance if there's an exception
54 
55  :param f: function instance
56  :return: wrapped function
57  """
58  @functools.wraps(f)
59  def wrapper(*args, **kwargs):
60  """
61  An wrapper for cli functions. It assumes to have Namespace instance
62  at 1st positional argument
63  :param args: Positional argument. It assumes to have Namespace instance
64  at 1st positional argument
65  :param kwargs: A passthrough keyword argument
66  """
67  assert args
68  assert isinstance(args[0], Namespace), \
69  "1st positional argument should be argparse.Namespace instance, " \
70  "but {}".format(args[0])
71  metrics = _build_metrics(f.__name__, args[0])
72  cli_action_loggers.on_pre_execution(**metrics)
73  try:
74  return f(*args, **kwargs)
75  except Exception as e:
76  metrics['error'] = e
77  raise
78  finally:
79  metrics['end_datetime'] = datetime.utcnow()
80  cli_action_loggers.on_post_execution(**metrics)
81 
82  return wrapper

85 def _build_metrics(func_name, namespace):
86  """
87  Builds metrics dict from function args
88  It assumes that function arguments is from airflow.bin.cli module's function
89  and has Namespace instance where it optionally contains "dag_id", "task_id",
90  and "execution_date".
91 
92  :param func_name: name of function
93  :param namespace: Namespace instance from argparse
94  :return: dict with metrics
95  """
96 
97  metrics = {'sub_command': func_name}
98  metrics['start_datetime'] = datetime.utcnow()
99  metrics['full_command'] = '{}'.format(list(sys.argv))
100  metrics['user'] = getpass.getuser()
101 
102  assert isinstance(namespace, Namespace)
103  tmp_dic = vars(namespace)
104  metrics['dag_id'] = tmp_dic.get('dag_id')
105  metrics['task_id'] = tmp_dic.get('task_id')
106  metrics['execution_date'] = tmp_dic.get('execution_date')
107  metrics['host_name'] = socket.gethostname()
108 
109  extra = json.dumps(dict((k, metrics[k]) for k in ('host_name', 'full_command')))
110  log = airflow.models.Log(
111  event='cli_{}'.format(func_name),
112  task_instance=None,
113  owner=metrics['user'],
114  extra=extra,
115  task_id=metrics.get('task_id'),
116  dag_id=metrics.get('dag_id'),
117  execution_date=metrics.get('execution_date'))
118  metrics['log'] = log
119  return metrics
