29 class FileSensor(BaseSensorOperator):
30  """
31  Waits for a file or folder to land in a filesystem.
32 
33  If the path given is a directory then this sensor will only return true if
34  any files exist inside it (either directly, or within a subdirectory)
35 
36  :param fs_conn_id: reference to the File (path)
37  connection id
38  :type fs_conn_id: string
39  :param filepath: File or folder name (relative to
40  the base path set within the connection)
41  :type fs_conn_id: string
42  """
43  template_fields = ('filepath',)
44  ui_color = '#91818a'
45 
46  @apply_defaults
47  def __init__(self,
48  filepath,
49  fs_conn_id='fs_default2',
50  *args,
51  **kwargs):
52  super(FileSensor, self).__init__(*args, **kwargs)
53  self.filepath = filepath
54  self.fs_conn_id = fs_conn_id
55 
56  def poke(self, context):
57  hook = FSHook(self.fs_conn_id)
58  basepath = hook.get_path()
59  full_path = "/".join([basepath, self.filepath])
60  self.log.info('Poking for file {full_path}'.format(**locals()))
61  try:
62  if stat.S_ISDIR(os.stat(full_path).st_mode):
63  for root, dirs, files in os.walk(full_path):
64  if len(files):
65  return True
66  else:
67  # full_path was a file directly
68  return True
69  except OSError:
70  return False
71  return False
