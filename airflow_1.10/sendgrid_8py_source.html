37 def send_email(to, subject, html_content, files=None,
38  dryrun=False, cc=None, bcc=None,
39  mime_subtype='mixed', **kwargs):
40  """
41  Send an email with html content using sendgrid.
42 
43  To use this plugin:
44  0. include sendgrid subpackage as part of your Airflow installation, e.g.,
45  pip install airflow[sendgrid]
46  1. update [email] backend in airflow.cfg, i.e.,
47  [email]
48  email_backend = airflow.contrib.utils.sendgrid.send_email
49  2. configure Sendgrid specific environment variables at all Airflow instances:
50  SENDGRID_MAIL_FROM={your-mail-from}
51  SENDGRID_API_KEY={your-sendgrid-api-key}.
52  """
53  mail = Mail()
54  from_email = kwargs.get('from_email') or os.environ.get('SENDGRID_MAIL_FROM')
55  from_name = kwargs.get('from_name') or os.environ.get('SENDGRID_MAIL_SENDER')
56  mail.from_email = Email(from_email, from_name)
57  mail.subject = subject
58 
59  # Add the recipient list of to emails.
60  personalization = Personalization()
61  to = get_email_address_list(to)
62  for to_address in to:
63  personalization.add_to(Email(to_address))
64  if cc:
65  cc = get_email_address_list(cc)
66  for cc_address in cc:
67  personalization.add_cc(Email(cc_address))
68  if bcc:
69  bcc = get_email_address_list(bcc)
70  for bcc_address in bcc:
71  personalization.add_bcc(Email(bcc_address))
72 
73  # Add custom_args to personalization if present
74  pers_custom_args = kwargs.get('personalization_custom_args', None)
75  if isinstance(pers_custom_args, dict):
76  for key in pers_custom_args.keys():
77  personalization.add_custom_arg(CustomArg(key, pers_custom_args[key]))
78 
79  mail.add_personalization(personalization)
80  mail.add_content(Content('text/html', html_content))
81 
82  categories = kwargs.get('categories', [])
83  for cat in categories:
84  mail.add_category(Category(cat))
85 
86  # Add email attachment.
87  for fname in files or []:
88  basename = os.path.basename(fname)
89  attachment = Attachment()
90  with open(fname, "rb") as f:
91  attachment.content = base64.b64encode(f.read())
92  attachment.type = mimetypes.guess_type(basename)[0]
93  attachment.filename = basename
94  attachment.disposition = "attachment"
95  attachment.content_id = '<%s>' % basename
96  mail.add_attachment(attachment)
97  _post_sendgrid_mail(mail.get())

100 def _post_sendgrid_mail(mail_data):
101  log = LoggingMixin().log
102  sg = sendgrid.SendGridAPIClient(apikey=os.environ.get('SENDGRID_API_KEY'))
103  response = sg.client.mail.send.post(request_body=mail_data)
104  # 2xx status code.
