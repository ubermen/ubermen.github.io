25 class PubSubTopicCreateOperator(BaseOperator):
26  """Create a PubSub topic.
27 
28  By default, if the topic already exists, this operator will
29  not cause the DAG to fail. ::
30 
31  with DAG('successful DAG') as dag:
32  (
33  dag
34  >> PubSubTopicCreateOperator(project='my-project',
35  topic='my_new_topic')
36  >> PubSubTopicCreateOperator(project='my-project',
37  topic='my_new_topic')
38  )
39 
40  The operator can be configured to fail if the topic already exists. ::
41 
42  with DAG('failing DAG') as dag:
43  (
44  dag
45  >> PubSubTopicCreateOperator(project='my-project',
46  topic='my_new_topic')
47  >> PubSubTopicCreateOperator(project='my-project',
48  topic='my_new_topic',
49  fail_if_exists=True)
50  )
51 
52  Both ``project`` and ``topic`` are templated so you can use
53  variables in them.
54  """

100 class PubSubSubscriptionCreateOperator(BaseOperator):
101  """Create a PubSub subscription.
102 
103  By default, the subscription will be created in ``topic_project``. If
104  ``subscription_project`` is specified and the GCP credentials allow, the
105  Subscription can be created in a different project from its topic.
106 
107  By default, if the subscription already exists, this operator will
108  not cause the DAG to fail. However, the topic must exist in the project. ::
109 
110  with DAG('successful DAG') as dag:
111  (
112  dag
113  >> PubSubSubscriptionCreateOperator(
114  topic_project='my-project', topic='my-topic',
115  subscription='my-subscription')
116  >> PubSubSubscriptionCreateOperator(
117  topic_project='my-project', topic='my-topic',
118  subscription='my-subscription')
119  )
120 
121  The operator can be configured to fail if the subscription already exists.
122  ::
123 
124  with DAG('failing DAG') as dag:
125  (
126  dag
127  >> PubSubSubscriptionCreateOperator(
128  topic_project='my-project', topic='my-topic',
129  subscription='my-subscription')
130  >> PubSubSubscriptionCreateOperator(
131  topic_project='my-project', topic='my-topic',
132  subscription='my-subscription', fail_if_exists=True)
133  )
134 
135  Finally, subscription is not required. If not passed, the operator will
136  generated a universally unique identifier for the subscription's name. ::
137 
138  with DAG('DAG') as dag:
139  (
140  dag >> PubSubSubscriptionCreateOperator(
141  topic_project='my-project', topic='my-topic')
142  )
143 
144  ``topic_project``, ``topic``, ``subscription``, and
145  ``subscription`` are templated so you can use variables in them.
146  """
147  template_fields = ['topic_project', 'topic', 'subscription',
148  'subscription_project']
149  ui_color = '#0273d4'

163  **kwargs):
164  """
165  :param topic_project: the GCP project ID where the topic exists

169  ``projects/{project}/topics/{topic}``, provide only
170  ``{topic}``. (templated)
171  :type topic: string
172  :param subscription: the Pub/Sub subscription name. If empty, a random
173  name will be generated using the uuid module
174  :type subscription: string
175  :param subscription_project: the GCP project ID where the subscription
176  will be created. If empty, ``topic_project`` will be used.
177  :type subscription_project: string
178  :param ack_deadline_secs: Number of seconds that a subscriber has to
179  acknowledge each message pulled from the subscription
180  :type ack_deadline_secs: int
181  :param gcp_conn_id: The connection ID to use connecting to
182  Google Cloud Platform.
183  :type gcp_conn_id: string
184  :param delegate_to: The account to impersonate, if any.
185  For this to work, the service account making the request
186  must have domain-wide delegation enabled.
187  :type delegate_to: string
188  """
189  super(PubSubSubscriptionCreateOperator, self).__init__(*args, **kwargs)
190 
191  self.topic_project = topic_project
192  self.topic = topic
193  self.subscription = subscription
194  self.subscription_project = subscription_project
195  self.ack_deadline_secs = ack_deadline_secs
196  self.fail_if_exists = fail_if_exists
197  self.gcp_conn_id = gcp_conn_id
198  self.delegate_to = delegate_to
199 
200  def execute(self, context):
201  hook = PubSubHook(gcp_conn_id=self.gcp_conn_id,
202  delegate_to=self.delegate_to)
203 
204  return hook.create_subscription(
205  self.topic_project, self.topic, self.subscription,
206  self.subscription_project, self.ack_deadline_secs,
207  self.fail_if_exists)

360 class PubSubPublishOperator(BaseOperator):
361  """Publish messages to a PubSub topic.
362 
363  Each Task publishes all provided messages to the same topic
364  in a single GCP project. If the topic does not exist, this
365  task will fail. ::
366 
367  from base64 import b64encode as b64e
368 
369  m1 = {'data': b64e('Hello, World!'),
370  'attributes': {'type': 'greeting'}
371  }
372  m2 = {'data': b64e('Knock, knock')}
373  m3 = {'attributes': {'foo': ''}}
374 
375  t1 = PubSubPublishOperator(
376  project='my-project',topic='my_topic',
377  messages=[m1, m2, m3],
378  create_topic=True,
379  dag=dag)
380 
381  ``project`` , ``topic``, and ``messages`` are templated so you can use
382  variables in them.
383  """
384  template_fields = ['project', 'topic', 'messages']
385  ui_color = '#0273d4'
386 
387  @apply_defaults
388  def __init__(
389  self,
390  project,
391  topic,
392  messages,
393  gcp_conn_id='google_cloud_default',
394  delegate_to=None,
395  *args,
396  **kwargs):
397  """
398  :param project: the GCP project ID in which to work (templated)
399  :type project: string
400  :param topic: the topic to which to publish. Do not include the
401  full topic path. In other words, instead of
402  ``projects/{project}/topics/{topic}``, provide only
403  ``{topic}``. (templated)
404  :type topic: string
405  :param messages: a list of messages to be published to the
406  topic. Each message is a dict with one or more of the
407  following keys-value mappings:
408  * 'data': a base64-encoded string
409  * 'attributes': {'key1': 'value1', ...}
410  Each message must contain at least a non-empty 'data' value

412  https://cloud.google.com/pubsub/docs/reference/rest/v1/PubsubMessage
413  :type messages: list
