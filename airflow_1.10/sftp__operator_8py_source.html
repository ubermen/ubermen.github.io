30 class SFTPOperator(BaseOperator):
31  """
32  SFTPOperator for transferring files from remote host to local or vice a versa.
33  This operator uses ssh_hook to open sftp trasport channel that serve as basis
34  for file transfer.
35 
36  :param ssh_hook: predefined ssh_hook to use for remote execution
37  :type ssh_hook: :class:`SSHHook`
38  :param ssh_conn_id: connection id from airflow Connections
39  :type ssh_conn_id: str
40  :param remote_host: remote host to connect
41  :type remote_host: str
42  :param local_filepath: local file path to get or put. (templated)
43  :type local_filepath: str
44  :param remote_filepath: remote file path to get or put. (templated)
45  :type remote_filepath: str
46  :param operation: specify operation 'get' or 'put', defaults to get
47  :type get: bool
48  """
49  template_fields = ('local_filepath', 'remote_filepath')

91  sftp_client.get(self.remote_filepath, self.local_filepath)
92  else:

97 
98  except Exception as e:
