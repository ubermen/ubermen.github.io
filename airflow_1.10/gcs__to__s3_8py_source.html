26 class GoogleCloudStorageToS3Operator(GoogleCloudStorageListOperator):
27  """
28  Synchronizes a Google Cloud Storage bucket with an S3 bucket.
29 
30  :param bucket: The Google Cloud Storage bucket to find the objects. (templated)
31  :type bucket: string
32  :param prefix: Prefix string which filters objects whose name begin with
33  this prefix. (templated)
34  :type prefix: string
35  :param delimiter: The delimiter by which you want to filter the objects. (templated)
36  For e.g to lists the CSV files from in a directory in GCS you would use
37  delimiter='.csv'.
38  :type delimiter: string
39  :param google_cloud_storage_conn_id: The connection ID to use when
40  connecting to Google Cloud Storage.
41  :type google_cloud_storage_conn_id: string
42  :param delegate_to: The account to impersonate, if any.
43  For this to work, the service account making the request must have
44  domain-wide delegation enabled.
45  :type delegate_to: string
46  :param dest_aws_conn_id: The destination S3 connection
47  :type dest_aws_conn_id: str
48  :param dest_s3_key: The base S3 key to be used to store the files. (templated)
49  :type dest_s3_key: str
50  """

80  def execute(self, context):
81  # use the super to list all files in an Google Cloud Storage bucket
82  files = super(GoogleCloudStorageToS3Operator, self).execute(context)
83  s3_hook = S3Hook(aws_conn_id=self.dest_aws_conn_id)
84 
85  if not self.replace:
86  # if we are not replacing -> list all files in the S3 bucket
87  # and only keep those files which are present in
88  # Google Cloud Storage and not in S3
89  bucket_name, _ = S3Hook.parse_s3_url(self.dest_s3_key)
90  existing_files = s3_hook.list_keys(bucket_name)
91  files = set(files) - set(existing_files)
92 
93  if files:
94  hook = GoogleCloudStorageHook(
95  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
96  delegate_to=self.delegate_to
97  )
98 
99  for file in files:
100  file_bytes = hook.download(self.bucket, file)
101 
102  dest_key = self.dest_s3_key + file
103  self.log.info("Saving file to %s", dest_key)
104 
105  s3_hook.load_bytes(file_bytes,
106  key=dest_key,
107  replace=self.replace)
108 
109  self.log.info("All done, uploaded %d files to S3", len(files))
110  else:
111  self.log.info("In sync, no files needed to be uploaded to S3")
112 
113  return files
