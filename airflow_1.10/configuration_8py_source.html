67 def expand_env_var(env_var):
68  """
69  Expands (potentially nested) env vars by repeatedly applying
70  `expandvars` and `expanduser` until interpolation stops having
71  any effect.
72  """
73  if not env_var:
74  return env_var
75  while True:
76  interpolated = os.path.expanduser(os.path.expandvars(str(env_var)))
77  if interpolated == env_var:
78  return interpolated
79  else:
80  env_var = interpolated

83 def run_command(command):
84  """
85  Runs command and returns stdout
86  """
87  process = subprocess.Popen(
88  shlex.split(command),
89  stdout=subprocess.PIPE,
90  stderr=subprocess.PIPE,
91  close_fds=True)
92  output, stderr = [stream.decode(sys.getdefaultencoding(), 'ignore')
93  for stream in process.communicate()]
94 
95  if process.returncode != 0:
96  raise AirflowConfigException(
97  "Cannot execute {}. Error code is: {}. Output: {}, Stderr: {}"
98  .format(command, process.returncode, output, stderr)
99  )
100 
101  return output

189  def _get_env_var_option(self, section, key):
190  # must have format AIRFLOW__{SECTION}__{KEY} (note double underscore)
191  env_var = 'AIRFLOW__{S}__{K}'.format(S=section.upper(), K=key.upper())

195  def _get_cmd_option(self, section, key):
196  fallback_key = key + '_cmd'
197  # if this is a valid command key...
198  if (section, key) in self.as_command_stdout:
199  if super(AirflowConfigParser, self) \
200  .has_option(section, fallback_key):
201  command = super(AirflowConfigParser, self) \
202  .get(section, fallback_key)

205  def get(self, section, key, **kwargs):
206  section = str(section).lower()
207  key = str(key).lower()
208 
209  deprecated_name = self.deprecated_options.get(section, {}).get(key, None)
210 
211  # first check environment variables
212  option = self._get_env_var_option(section, key)
213  if option is not None:
214  return option
215  if deprecated_name:
216  option = self._get_env_var_option(section, deprecated_name)
217  if option is not None:
218  self._warn_deprecate(section, key, deprecated_name)
219  return option
220 
221  # ...then the config file
222  if super(AirflowConfigParser, self).has_option(section, key):
223  # Use the parent's methods to get the actual config here to be able to
224  # separate the config from default config.
225  return expand_env_var(
226  super(AirflowConfigParser, self).get(section, key, **kwargs))
227  if deprecated_name:
228  if super(AirflowConfigParser, self).has_option(section, deprecated_name):
229  self._warn_deprecate(section, key, deprecated_name)
230  return expand_env_var(super(AirflowConfigParser, self).get(
231  section,
232  deprecated_name,
233  **kwargs
234  ))
235 
236  # ...then commands
237  option = self._get_cmd_option(section, key)
238  if option:
239  return option
240  if deprecated_name:
241  option = self._get_cmd_option(section, deprecated_name)
242  if option:
243  self._warn_deprecate(section, key, deprecated_name)
244  return option
245 
246  # ...then the default config
247  if self.defaults.has_option(section, key):

304  def getsection(self, section):
305  """
306  Returns the section as a dict. Values are converted to int, float, bool
307  as required.
308  :param section: section from the config
309  :return: dict
310  """
311  if section not in self._sections and section not in self.defaults._sections:
312  return None
313 
314  _section = copy.deepcopy(self.defaults._sections[section])
315 
316  if section in self._sections:
317  _section.update(copy.deepcopy(self._sections[section]))
318 
319  for key, val in iteritems(_section):
320  try:
321  val = int(val)
322  except ValueError:
323  try:
324  val = float(val)
325  except ValueError:
326  if val.lower() in ('t', 'true'):
327  val = True
328  elif val.lower() in ('f', 'false'):
329  val = False
330  _section[key] = val
331  return _section

333  def as_dict(self, display_source=False, display_sensitive=False):
334  """
335  Returns the current configuration as an OrderedDict of OrderedDicts.
336  :param display_source: If False, the option value is returned. If True,
337  a tuple of (option_value, source) is returned. Source is either
338  'airflow.cfg' or 'default'.
339  :type display_source: bool
340  :param display_sensitive: If True, the values of options set by env
341  vars and bash commands will be displayed. If False, those options
342  are shown as '< hidden >'
343  :type display_sensitive: bool
344  """
345  cfg = copy.deepcopy(self.defaults._sections)
346  cfg.update(copy.deepcopy(self._sections))
347 
348  # remove __name__ (affects Python 2 only)
349  for options in cfg.values():
350  options.pop('__name__', None)
351 
352  # add source
353  if display_source:
354  for section in cfg:
355  for k, v in cfg[section].items():
356  cfg[section][k] = (v, 'airflow config')
357 
358  # add env vars and overwrite because they have priority
359  for ev in [ev for ev in os.environ if ev.startswith('AIRFLOW__')]:
360  try:
361  _, section, key = ev.split('__')
362  opt = self._get_env_var_option(section, key)
363  except ValueError:
364  opt = None
365  if opt:
366  if (
367  not display_sensitive and
368  ev != 'AIRFLOW__CORE__UNIT_TEST_MODE'):
369  opt = '< hidden >'
370  if display_source:
371  opt = (opt, 'env var')
372  cfg.setdefault(section.lower(), OrderedDict()).update(
373  {key.lower(): opt})
374 
375  # add bash commands
376  for (section, key) in self.as_command_stdout:
377  opt = self._get_cmd_option(section, key)
378  if opt:
379  if not display_sensitive:
380  opt = '< hidden >'
381  if display_source:
382  opt = (opt, 'bash cmd')
383  cfg.setdefault(section, OrderedDict()).update({key: opt})
384 
385  return cfg
