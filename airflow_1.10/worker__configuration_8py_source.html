38  """When using git to retrieve the DAGs, use the GitSync Init Container"""
39  # If we're using volume claims to mount the dags, no init container is needed

56  }, {
57  'name': 'GIT_SYNC_ONE_TIME',
58  'value': 'true'
59  }]

62  'name': 'GIT_SYNC_USERNAME',
63  'value': self.kube_config.git_user
64  })
65  if self.kube_config.git_password:
66  init_environment.append({
67  'name': 'GIT_SYNC_PASSWORD',
68  'value': self.kube_config.git_password
69  })
70 
71  volume_mounts[0]['readOnly'] = False
72  return [{
73  'name': self.kube_config.git_sync_init_container_name,
74  'image': self.kube_config.git_sync_container,
75  'securityContext': {'runAsUser': 0},
76  'env': init_environment,
