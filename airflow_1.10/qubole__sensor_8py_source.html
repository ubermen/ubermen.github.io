58  def poke(self, context):
59 
60  conn = BaseHook.get_connection(self.qubole_conn_id)
61  Qubole.configure(api_token=conn.password, api_url=conn.host)
62 
63  self.log.info('Poking: %s', self.data)
64 
65  status = False
66  try:
67  status = self.sensor_class.check(self.data)
68  except Exception as e:
69  self.log.exception(e)
70  status = False
71 
72  self.log.info('Status of this Poke: %s', status)
73 
74  return status
