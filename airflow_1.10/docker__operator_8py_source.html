31 class DockerOperator(BaseOperator):
32  """
33  Execute a command inside a docker container.
34 
35  A temporary directory is created on the host and
36  mounted into a container to allow storing files
37  that together exceed the default disk size of 10GB in a container.
38  The path to the mounted directory can be accessed
39  via the environment variable ``AIRFLOW_TMP_DIR``.
40 
41  If a login to a private registry is required prior to pulling the image, a
42  Docker connection needs to be configured in Airflow and the connection ID
43  be provided with the parameter ``docker_conn_id``.
44 
45  :param image: Docker image from which to create the container.
46  :type image: str
47  :param api_version: Remote API version. Set to ``auto`` to automatically
48  detect the server's version.
49  :type api_version: str
50  :param command: Command to be run in the container. (templated)
51  :type command: str or list
52  :param cpus: Number of CPUs to assign to the container.
53  This value gets multiplied with 1024. See
54  https://docs.docker.com/engine/reference/run/#cpu-share-constraint
55  :type cpus: float
56  :param docker_url: URL of the host running the docker daemon.
57  Default is unix://var/run/docker.sock
58  :type docker_url: str
59  :param environment: Environment variables to set in the container. (templated)
60  :type environment: dict
61  :param force_pull: Pull the docker image on every run. Default is false.
62  :type force_pull: bool
63  :param mem_limit: Maximum amount of memory the container can use.
64  Either a float value, which represents the limit in bytes,
65  or a string like ``128m`` or ``1g``.
66  :type mem_limit: float or str
67  :param network_mode: Network mode for the container.
68  :type network_mode: str
69  :param tls_ca_cert: Path to a PEM-encoded certificate authority
70  to secure the docker connection.
71  :type tls_ca_cert: str
72  :param tls_client_cert: Path to the PEM-encoded certificate
73  used to authenticate docker client.
74  :type tls_client_cert: str
75  :param tls_client_key: Path to the PEM-encoded key used to authenticate docker client.
76  :type tls_client_key: str
77  :param tls_hostname: Hostname to match against
78  the docker server certificate or False to disable the check.
79  :type tls_hostname: str or bool
80  :param tls_ssl_version: Version of SSL to use when communicating with docker daemon.
81  :type tls_ssl_version: str
82  :param tmp_dir: Mount point inside the container to
83  a temporary directory created on the host by the operator.
84  The path is also made available via the environment variable
85  ``AIRFLOW_TMP_DIR`` inside the container.
86  :type tmp_dir: str
87  :param user: Default user inside the docker container.
88  :type user: int or str
89  :param volumes: List of volumes to mount into the container, e.g.
90  ``['/host/path:/container/path', '/host/path2:/container/path2:ro']``.
91  :param working_dir: Working directory to
92  set on the container (equivalent to the -w switch the docker client)
93  :type working_dir: str
94  :param xcom_push: Does the stdout will be pushed to the next step using XCom.
95  The default is False.
96  :type xcom_push: bool
97  :param xcom_all: Push all the stdout or just the last line.
98  The default is False (last line).
99  :type xcom_all: bool
100  :param docker_conn_id: ID of the Airflow connection to use
101  :type docker_conn_id: str
102  """
