25 class GoogleCloudStorageListOperator(BaseOperator):
26  """
27  List all objects from the bucket with the give string prefix and delimiter in name.
28 
29  This operator returns a python list with the name of objects which can be used by
30  `xcom` in the downstream task.
31 
32  :param bucket: The Google cloud storage bucket to find the objects. (templated)
33  :type bucket: string
34  :param prefix: Prefix string which filters objects whose name begin with
35  this prefix. (templated)
36  :type prefix: string
37  :param delimiter: The delimiter by which you want to filter the objects. (templated)
38  For e.g to lists the CSV files from in a directory in GCS you would use
39  delimiter='.csv'.
40  :type delimiter: string
41  :param google_cloud_storage_conn_id: The connection ID to use when
42  connecting to Google cloud storage.
43  :type google_cloud_storage_conn_id: string
44  :param delegate_to: The account to impersonate, if any.
45  For this to work, the service account making the request must have
46  domain-wide delegation enabled.
47  :type delegate_to: string
48 
49  **Example**:
50  The following Operator would list all the Avro files from ``sales/sales-2017``
51  folder in ``data`` bucket. ::
52 
53  GCS_Files = GoogleCloudStorageListOperator(
54  task_id='GCS_Files',
55  bucket='data',
56  prefix='sales/sales-2017/',
57  delimiter='.avro',
58  google_cloud_storage_conn_id=google_cloud_conn_id
59  )
60  """
