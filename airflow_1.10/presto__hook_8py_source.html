93  def get_pandas_df(self, hql, parameters=None):
94  """
95  Get a pandas dataframe from a sql query.
96  """
97  import pandas
98  cursor = self.get_cursor()
99  try:
100  cursor.execute(self._strip_sql(hql), parameters)
101  data = cursor.fetchall()
102  except DatabaseError as e:
103  raise PrestoException(self._get_pretty_exception_message(e))
104  column_descriptions = cursor.description
105  if data:
106  df = pandas.DataFrame(data)
107  df.columns = [c[0] for c in column_descriptions]
108  else:
109  df = pandas.DataFrame()
110  return df
