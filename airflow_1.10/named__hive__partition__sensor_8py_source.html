26 class NamedHivePartitionSensor(BaseSensorOperator):
27  """
28  Waits for a set of partitions to show up in Hive.
29 
30  :param partition_names: List of fully qualified names of the
31  partitions to wait for. A fully qualified name is of the
32  form ``schema.table/pk1=pv1/pk2=pv2``, for example,
33  default.users/ds=2016-01-01. This is passed as is to the metastore
34  Thrift client ``get_partitions_by_name`` method. Note that
35  you cannot use logical or comparison operators as in
36  HivePartitionSensor.
37  :type partition_names: list of strings
38  :param metastore_conn_id: reference to the metastore thrift service
39  connection id
40  :type metastore_conn_id: str
41  """

71  if len(first_split) == 1:
72  schema = 'default'
