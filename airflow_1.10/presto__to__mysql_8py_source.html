26 class PrestoToMySqlTransfer(BaseOperator):
27  """
28  Moves data from Presto to MySQL, note that for now the data is loaded
29  into memory before being pushed to MySQL, so this operator should
30  be used for smallish amount of data.
31 
32  :param sql: SQL query to execute against Presto. (templated)
33  :type sql: str
34  :param mysql_table: target MySQL table, use dot notation to target a
35  specific database. (templated)
36  :type mysql_table: str
37  :param mysql_conn_id: source mysql connection
38  :type mysql_conn_id: str
39  :param presto_conn_id: source presto connection
40  :type presto_conn_id: str
41  :param mysql_preoperator: sql statement to run against mysql prior to
42  import, typically use to truncate of delete in place
43  of the data coming in, allowing the task to be idempotent (running
44  the task twice won't double load data). (templated)
45  :type mysql_preoperator: str
46  """
47

69  presto = PrestoHook(presto_conn_id=self.presto_conn_id)
70  self.log.info("Extracting data from Presto: %s", self.sql)
71  results = presto.get_records(self.sql)
72 
73  mysql = MySqlHook(mysql_conn_id=self.mysql_conn_id)
74  if self.mysql_preoperator:

77  mysql.run(self.mysql_preoperator)
78 
79  self.log.info("Inserting rows into MySQL")
