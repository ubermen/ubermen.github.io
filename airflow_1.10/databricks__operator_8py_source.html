33 class DatabricksSubmitRunOperator(BaseOperator):
34  """
35  Submits an Spark job run to Databricks using the
36  `api/2.0/jobs/runs/submit
37  <https://docs.databricks.com/api/latest/jobs.html#runs-submit>`_
38  API endpoint.
39 
40  There are two ways to instantiate this operator.
41 
42  In the first way, you can take the JSON payload that you typically use
43  to call the ``api/2.0/jobs/runs/submit`` endpoint and pass it directly
44  to our ``DatabricksSubmitRunOperator`` through the ``json`` parameter.
45  For example ::
46  json = {
47  'new_cluster': {
48  'spark_version': '2.1.0-db3-scala2.11',
49  'num_workers': 2
50  },
51  'notebook_task': {
52  'notebook_path': '/Users/airflow@example.com/PrepareData',
53  },
54  }
55  notebook_run = DatabricksSubmitRunOperator(task_id='notebook_run', json=json)
56 
57  Another way to accomplish the same thing is to use the named parameters
58  of the ``DatabricksSubmitRunOperator`` directly. Note that there is exactly
59  one named parameter for each top level parameter in the ``runs/submit``
60  endpoint. In this method, your code would look like this: ::
61  new_cluster = {
62  'spark_version': '2.1.0-db3-scala2.11',
63  'num_workers': 2
64  }
65  notebook_task = {
66  'notebook_path': '/Users/airflow@example.com/PrepareData',
67  }
68  notebook_run = DatabricksSubmitRunOperator(
69  task_id='notebook_run',
70  new_cluster=new_cluster,
71  notebook_task=notebook_task)
72 
73  In the case where both the json parameter **AND** the named parameters
74  are provided, they will be merged together. If there are conflicts during the merge,
75  the named parameters will take precedence and override the top level ``json`` keys.
76 
77  Currently the named parameters that ``DatabricksSubmitRunOperator`` supports are
78  - ``spark_jar_task``
79  - ``notebook_task``
80  - ``new_cluster``
81  - ``existing_cluster_id``
82  - ``libraries``
83  - ``run_name``
84  - ``timeout_seconds``
85 
86  :param json: A JSON object containing API parameters which will be passed
87  directly to the ``api/2.0/jobs/runs/submit`` endpoint. The other named parameters
88  (i.e. ``spark_jar_task``, ``notebook_task``..) to this operator will
89  be merged with this json dictionary if they are provided.
90  If there are conflicts during the merge, the named parameters will
91  take precedence and override the top level json keys. (templated)
92 
93  .. seealso::
94  For more information about templating see :ref:`jinja-templating`.
95  https://docs.databricks.com/api/latest/jobs.html#runs-submit
96  :type json: dict
97  :param spark_jar_task: The main class and parameters for the JAR task. Note that
98  the actual JAR is specified in the ``libraries``.
99  *EITHER* ``spark_jar_task`` *OR* ``notebook_task`` should be specified.
100  This field will be templated.
101 
102  .. seealso::
103  https://docs.databricks.com/api/latest/jobs.html#jobssparkjartask
104  :type spark_jar_task: dict
105  :param notebook_task: The notebook path and parameters for the notebook task.
106  *EITHER* ``spark_jar_task`` *OR* ``notebook_task`` should be specified.
107  This field will be templated.
108 
109  .. seealso::
110  https://docs.databricks.com/api/latest/jobs.html#jobsnotebooktask
111  :type notebook_task: dict
112  :param new_cluster: Specs for a new cluster on which this task will be run.
113  *EITHER* ``new_cluster`` *OR* ``existing_cluster_id`` should be specified.
114  This field will be templated.
115 
116  .. seealso::
117  https://docs.databricks.com/api/latest/jobs.html#jobsclusterspecnewcluster
118  :type new_cluster: dict
119  :param existing_cluster_id: ID for existing cluster on which to run this task.
120  *EITHER* ``new_cluster`` *OR* ``existing_cluster_id`` should be specified.
121  This field will be templated.
122  :type existing_cluster_id: string
123  :param libraries: Libraries which this run will use.
124  This field will be templated.
125 
126  .. seealso::
127  https://docs.databricks.com/api/latest/libraries.html#managedlibrarieslibrary
128  :type libraries: list of dicts
129  :param run_name: The run name used for this task.
130  By default this will be set to the Airflow ``task_id``. This ``task_id`` is a
131  required parameter of the superclass ``BaseOperator``.
132  This field will be templated.
133  :type run_name: string
134  :param timeout_seconds: The timeout for this run. By default a value of 0 is used
135  which means to have no timeout.
136  This field will be templated.
137  :type timeout_seconds: int32
138  :param databricks_conn_id: The name of the Airflow connection to use.
139  By default and in the common case this will be ``databricks_default``. To use
140  token based authentication, provide the key ``token`` in the extra field for the
141  connection.
142  :type databricks_conn_id: string
143  :param polling_period_seconds: Controls the rate which we poll for the result of
144  this run. By default the operator will poll every 30 seconds.
145  :type polling_period_seconds: int
146  :param databricks_retry_limit: Amount of times retry if the Databricks backend is
147  unreachable. Its value must be greater than or equal to 1.
148  :type databricks_retry_limit: int
149  :param do_xcom_push: Whether we should push run_id and run_page_url to xcom.
150  :type do_xcom_push: boolean
151  """

203  def _deep_string_coerce(self, content, json_path='json'):
204  """
205  Coerces content or all values of content if it is a dict to a string. The
206  function will throw if content contains non-string or non-numeric types.
207 
208  The reason why we have this function is because the ``self.json`` field must be a
209  dict with only string values. This is because ``render_template`` will fail
210  for numerical values.
211  """
212  c = self._deep_string_coerce
213  if isinstance(content, six.string_types):
214  return content
215  elif isinstance(content, six.integer_types + (float,)):
216  # Databricks can tolerate either numeric or string types in the API backend.
217  return str(content)
218  elif isinstance(content, (list, tuple)):
219  return [c(e, '{0}[{1}]'.format(json_path, i)) for i, e in enumerate(content)]
220  elif isinstance(content, dict):
221  return {k: c(v, '{0}[{1}]'.format(json_path, k))
222  for k, v in list(content.items())}
223  else:
224  param_type = type(content)
225  msg = 'Type {0} used for parameter {1} is not a number or a string'\
226  .format(param_type, json_path)
227  raise AirflowException(msg)

237  def execute(self, context):
238  hook = self.get_hook()
239  self.run_id = hook.submit_run(self.json)
240  if self.do_xcom_push:
241  context['ti'].xcom_push(key=XCOM_RUN_ID_KEY, value=self.run_id)
242  self.log.info('Run submitted with run_id: %s', self.run_id)
243  run_page_url = hook.get_run_page_url(self.run_id)
244  if self.do_xcom_push:
245  context['ti'].xcom_push(key=XCOM_RUN_PAGE_URL_KEY, value=run_page_url)
246  self._log_run_page_url(run_page_url)
247  while True:
248  run_state = hook.get_run_state(self.run_id)
249  if run_state.is_terminal:
250  if run_state.is_successful:
251  self.log.info('%s completed successfully.', self.task_id)
252  self._log_run_page_url(run_page_url)
253  return
254  else:
255  error_message = '{t} failed with terminal state: {s}'.format(
256  t=self.task_id,
257  s=run_state)
258  raise AirflowException(error_message)
259  else:
260  self.log.info('%s in run state: %s', self.task_id, run_state)
261  self._log_run_page_url(run_page_url)
262  self.log.info('Sleeping for %s seconds.', self.polling_period_seconds)
263  time.sleep(self.polling_period_seconds)
