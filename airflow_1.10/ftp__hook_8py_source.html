30 def mlsd(conn, path="", facts=None):
31  """
32  BACKPORT FROM PYTHON3 FTPLIB.
33 
34  List a directory in a standardized format by using MLSD
35  command (RFC-3659). If path is omitted the current directory
36  is assumed. "facts" is a list of strings representing the type
37  of information desired (e.g. ["type", "size", "perm"]).
38 
39  Return a generator object yielding a tuple of two elements
40  for every file found in path.
41  First element is the file name, the second one is a dictionary
42  including a variable number of "facts" depending on the server
43  and whether "facts" argument has been provided.
44  """
45  facts = facts or []
46  if facts:
47  conn.sendcmd("OPTS MLST " + ";".join(facts) + ";")
48  if path:
49  cmd = "MLSD %s" % path
50  else:
51  cmd = "MLSD"
52  lines = []
53  conn.retrlines(cmd, lines.append)
54  for line in lines:
55  facts_found, _, name = line.rstrip(ftplib.CRLF).partition(' ')
56  entry = {}
57  for fact in facts_found[:-1].split(";"):
58  key, _, value = fact.partition("=")
59  entry[key.lower()] = value
60  yield (name, entry)

151  def retrieve_file(self, remote_full_path, local_full_path_or_buffer):
152  """
153  Transfers the remote file to a local location.
154 
155  If local_full_path_or_buffer is a string path, the file will be put
156  at that location; if it is a file-like buffer, the file will
157  be written to the buffer but not closed.
158 
159  :param remote_full_path: full path to the remote file
160  :type remote_full_path: str
161  :param local_full_path_or_buffer: full path to the local file or a
162  file-like buffer
163  :type local_full_path_or_buffer: str or file-like buffer
164  """
165  conn = self.get_conn()
166 
167  is_path = isinstance(local_full_path_or_buffer, basestring)
168 
169  if is_path:
170  output_handle = open(local_full_path_or_buffer, 'wb')
171  else:
172  output_handle = local_full_path_or_buffer
173 
174  remote_path, remote_file_name = os.path.split(remote_full_path)
175  conn.cwd(remote_path)
176  self.log.info('Retrieving file from FTP: %s', remote_full_path)
177  conn.retrbinary('RETR %s' % remote_file_name, output_handle.write)
178  self.log.info('Finished retrieving file from FTP: %s', remote_full_path)
179 
180  if is_path:
181  output_handle.close()
