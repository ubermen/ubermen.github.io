149 class _DataProcOperation(LoggingMixin):
150  """Continuously polls Dataproc Operation until it completes."""
151  def __init__(self, dataproc_api, operation):
152  self.dataproc_api = dataproc_api
153  self.operation = operation
154  self.operation_name = self.operation['name']
155 
156  def wait_for_done(self):
157  if self._check_done():
158  return True
159 
160  self.log.info(
161  'Waiting for Dataproc Operation %s to finish', self.operation_name)
162  while True:
163  time.sleep(10)
164  self.operation = (
165  self.dataproc_api.projects()
166  .regions()
167  .operations()
168  .get(name=self.operation_name)
169  .execute(num_retries=5))
170 
171  if self._check_done():
172  return True
173 
174  def get(self):
175  return self.operation

177  def _check_done(self):
178  if 'done' in self.operation:
179  if 'error' in self.operation:
180  self.log.warning(
181  'Dataproc Operation %s failed with error: %s',
182  self.operation_name, self.operation['error']['message'])
183  self._raise_error()
184  else:
185  self.log.info(
186  'Dataproc Operation %s done', self.operation['name'])
187  return True
188  return False

190  def _raise_error(self):
191  raise Exception('Google Dataproc Operation %s failed: %s' %
