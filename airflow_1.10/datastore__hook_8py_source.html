156  def poll_operation_until_done(self, name, polling_interval_in_seconds):
157  """
158  Poll backup operation state until it's completed
159  """
160  while True:
161  result = self.get_operation(name)
162  state = result['metadata']['common']['state']
163  if state == 'PROCESSING':
164  self.log.info('Operation is processing. Re-polling state in {} seconds'
165  .format(polling_interval_in_seconds))
166  time.sleep(polling_interval_in_seconds)
167  else:
168  return result
