115  def _wait_for_task_ended(self):
116  """
117  Try to use a waiter from the below pull request
118 
119  * https://github.com/boto/botocore/pull/1307
120 
121  If the waiter is not available apply a exponential backoff
122 
123  * docs.aws.amazon.com/general/latest/gr/api-retries.html
124  """
125  try:
126  waiter = self.client.get_waiter('job_execution_complete')
127  waiter.config.max_attempts = sys.maxsize # timeout is managed by airflow
128  waiter.wait(jobs=[self.jobId])
129  except ValueError:
130  # If waiter not available use expo
131  retry = True
132  retries = 0
133 
134  while retries < self.max_retries and retry:
135  self.log.info('AWS Batch retry in the next %s seconds', retries)
136  response = self.client.describe_jobs(
137  jobs=[self.jobId]
138  )
139  if response['jobs'][-1]['status'] in ['SUCCEEDED', 'FAILED']:
140  retry = False
141 
142  sleep( 1 + pow(retries * 0.1, 2))
143  retries += 1

145  def _check_success_task(self):
146  response = self.client.describe_jobs(
147  jobs=[self.jobId],
148  )
149 
150  self.log.info('AWS Batch stopped, check status: %s', response)
151  if len(response.get('jobs')) < 1:
152  raise AirflowException('No job found for {}'.format(response))
153 
154  for job in response['jobs']:
155  if 'attempts' in job:
156  containers = job['attempts']
157  for container in containers:
158  if (job['status'] == 'FAILED' or
159  container['container']['exitCode'] != 0):
160  raise AirflowException(
161  'This containers encounter an error during '
162  'execution {}'.format(job))
163  elif job['status'] is not 'SUCCEEDED':
164  raise AirflowException(
165  'This task is still pending {}'.format(job['status']))
