86 def prepare_lineage(func):
87  """
88  Prepares the lineage inlets and outlets
89  inlets can be:
90  "auto" -> picks up any outlets from direct upstream tasks that have outlets
91  defined, as such that if A -> B -> C and B does not have outlets but A does,
92  these are provided as inlets.
93  "list of task_ids" -> picks up outlets from the upstream task_ids
94  "list of datasets" -> manually defined list of DataSet
95  """
96  @wraps(func)
97  def wrapper(self, context, *args, **kwargs):
98  self.log.debug("Preparing lineage inlets and outlets")
99 
100  task_ids = set(self._inlets['task_ids']).intersection(
101  self.get_flat_relative_ids(upstream=True)
102  )
103  if task_ids:
104  inlets = self.xcom_pull(context,
105  task_ids=task_ids,
106  dag_id=self.dag_id,
107  key=PIPELINE_OUTLETS)
108  inlets = [item for sublist in inlets if sublist for item in sublist]
109  inlets = [DataSet.map_type(i['typeName'])(data=i['attributes'])
110  for i in inlets]
111  self.inlets.extend(inlets)
112 
113  if self._inlets['auto']:
114  # dont append twice
115  task_ids = set(self._inlets['task_ids']).symmetric_difference(
116  self.upstream_task_ids
117  )
118  inlets = self.xcom_pull(context,
119  task_ids=task_ids,
120  dag_id=self.dag_id,
121  key=PIPELINE_OUTLETS)
122  inlets = [item for sublist in inlets if sublist for item in sublist]
123  inlets = [DataSet.map_type(i['typeName'])(data=i['attributes'])
124  for i in inlets]
125  self.inlets.extend(inlets)
126

128  self.inlets.extend(self._inlets['datasets'])

130  # outlets

132  self.outlets.extend(self._outlets['datasets'])
