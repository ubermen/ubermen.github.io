40  """
41  This runs a sub dag. By convention, a sub dag's dag_id
42  should be prefixed by its parent and a dot. As in `parent.child`.
43 
44  :param subdag: the DAG object to run as a subdag of the current DAG.
45  :type subdag: airflow.DAG.
46  :param dag: the parent DAG for the subdag.
47  :type dag: airflow.DAG.
48  :param executor: the executor for this subdag. Default to use SequentialExecutor.
49  Please find AIRFLOW-74 for more details.
50  :type executor: airflow.executors.
51  """
52  import airflow.models
53  dag = kwargs.get('dag') or airflow.models._CONTEXT_MANAGER_DAG
54  if not dag:
55  raise AirflowException('Please pass in the `dag` param or call '
56  'within a DAG context manager')
57  session = kwargs.pop('session')
58  super(SubDagOperator, self).__init__(*args, **kwargs)
59 
60  # validate subdag name
61  if dag.dag_id + '.' + kwargs['task_id'] != subdag.dag_id:
62  raise AirflowException(
63  "The subdag's dag_id should have the form "
64  "'{{parent_dag_id}}.{{this_task_id}}'. Expected "
65  "'{d}.{t}'; received '{rcvd}'.".format(
66  d=dag.dag_id, t=kwargs['task_id'], rcvd=subdag.dag_id))
67 
68  # validate that subdag operator and subdag tasks don't have a
69  # pool conflict
70  if self.pool:
71  conflicts = [t for t in subdag.tasks if t.pool == self.pool]
72  if conflicts:
73  # only query for pool conflicts if one may exist
74  pool = (
75  session
76  .query(Pool)
77  .filter(Pool.slots == 1)
78  .filter(Pool.pool == self.pool)
79  .first()
80  )
81  if pool and any(t.pool == self.pool for t in subdag.tasks):
82  raise AirflowException(
83  'SubDagOperator {sd} and subdag task{plural} {t} both '
84  'use pool {p}, but the pool only has 1 slot. The '
85  'subdag tasks will never run.'.format(
86  sd=self.task_id,
87  plural=len(conflicts) > 1,
88  t=', '.join(t.task_id for t in conflicts),
89  p=self.pool
90  )
91  )
92 
93  self.subdag = subdag
94  # Airflow pool is not honored by SubDagOperator.
95  # Hence resources could be consumed by SubdagOperators
96  # Use other executor with your own risk.
97  self.executor = executor
