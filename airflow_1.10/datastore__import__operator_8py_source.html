26 class DatastoreImportOperator(BaseOperator):
27  """
28  Import entities from Cloud Storage to Google Cloud Datastore
29 
30  :param bucket: container in Cloud Storage to store data
31  :type bucket: string
32  :param file: path of the backup metadata file in the specified Cloud Storage bucket.
33  It should have the extension .overall_export_metadata
34  :type file: string
35  :param namespace: optional namespace of the backup metadata file in
36  the specified Cloud Storage bucket.
37  :type namespace: str
38  :param entity_filter: description of what data from the project is included in
39  the export, refer to
40  https://cloud.google.com/datastore/docs/reference/rest/Shared.Types/EntityFilter
41  :type entity_filter: dict
42  :param labels: client-assigned labels for cloud storage
43  :type labels: dict
44  :param datastore_conn_id: the name of the connection id to use
45  :type datastore_conn_id: string
46  :param delegate_to: The account to impersonate, if any.
47  For this to work, the service account making the request must have domain-wide
48  delegation enabled.
49  :type delegate_to: string
50  :param polling_interval_in_seconds: number of seconds to wait before polling for
51  execution status again
52  :type polling_interval_in_seconds: int
53  :param xcom_push: push operation name to xcom for reference
54  :type xcom_push: bool
55  """
