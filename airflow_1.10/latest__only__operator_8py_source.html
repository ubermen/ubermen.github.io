24 class LatestOnlyOperator(BaseOperator, SkipMixin):
25  """
26  Allows a workflow to skip tasks that are not running during the most
27  recent schedule interval.
28 
29  If the task is run outside of the latest schedule interval, all
30  directly downstream tasks will be skipped.
31  """
32 
33  ui_color = '#e9ffdb' # nyanza
34 
35  def execute(self, context):
36  # If the DAG Run is externally triggered, then return without
37  # skipping downstream tasks
38  if context['dag_run'] and context['dag_run'].external_trigger:
39  self.log.info("Externally triggered DAG_Run: allowing execution to proceed.")
40  return
41 
42  now = timezone.utcnow()
43  left_window = context['dag'].following_schedule(
44  context['execution_date'])
45  right_window = context['dag'].following_schedule(left_window)
46  self.log.info(
47  'Checking latest only with left_window: %s right_window: %s now: %s',
48  left_window, right_window, now
49  )
50 
51  if not left_window < now <= right_window:
52  self.log.info('Not latest execution, skipping downstream.')
53 
54  downstream_tasks = context['task'].get_flat_relatives(upstream=False)
55  self.log.debug("Downstream task_ids %s", downstream_tasks)
56 
57  if downstream_tasks:
58  self.skip(context['dag_run'],
59  context['ti'].execution_date,
60  downstream_tasks)
61 
62  self.log.info('Done.')
63  else:
64  self.log.info('Latest, allowing execution to proceed.')
