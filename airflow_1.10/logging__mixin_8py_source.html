160 def set_context(logger, value):
161  """
162  Walks the tree of loggers and tries to set the context for each handler
163  :param logger: logger
164  :param value: value to set
165  """
166  _logger = logger
167  while _logger:
168  for handler in _logger.handlers:

174  pass

176  _logger = _logger.parent
177  else:
178  _logger = None
