30 class S3ToGoogleCloudStorageOperator(S3ListOperator):
31  """
32  Synchronizes an S3 key, possibly a prefix, with a Google Cloud Storage
33  destination path.
34 
35  :param bucket: The S3 bucket where to find the objects. (templated)
36  :type bucket: string
37  :param prefix: Prefix string which filters objects whose name begin with
38  such prefix. (templated)
39  :type prefix: string
40  :param delimiter: the delimiter marks key hierarchy. (templated)
41  :type delimiter: string
42  :param aws_conn_id: The source S3 connection
43  :type aws_conn_id: string
44  :param dest_gcs_conn_id: The destination connection ID to use
45  when connecting to Google Cloud Storage.
46  :type dest_gcs_conn_id: string
47  :param dest_gcs: The destination Google Cloud Storage bucket and prefix
48  where you want to store the files. (templated)
49  :type dest_gcs: string
50  :param delegate_to: The account to impersonate, if any.
51  For this to work, the service account making the request must have
52  domain-wide delegation enabled.
53  :type delegate_to: string
54  :param replace: Whether you want to replace existing destination files
55  or not.
56  :type replace: bool
57 
58 
59  **Example**:
60  .. code-block:: python
61  s3_to_gcs_op = S3ToGoogleCloudStorageOperator(
62  task_id='s3_to_gcs_example',
63  bucket='my-s3-bucket',
64  prefix='data/customers-201804',
65  dest_gcs_conn_id='google_cloud_default',
66  dest_gcs='gs://my.gcs.bucket/some/customers/',
67  replace=False,
68  dag=my-dag)
69 
70  Note that ``bucket``, ``prefix``, ``delimiter`` and ``dest_gcs`` are
71  templated, so you can use variables in them if you wish.
72  """

110  def execute(self, context):
111  # use the super method to list all the files in an S3 bucket/key
112  files = super(S3ToGoogleCloudStorageOperator, self).execute(context)
113 
114  gcs_hook = GoogleCloudStorageHook(
115  google_cloud_storage_conn_id=self.dest_gcs_conn_id,
116  delegate_to=self.delegate_to)
117 
118  if not self.replace:
119  # if we are not replacing -> list all files in the GCS bucket
120  # and only keep those files which are present in
121  # S3 and not in Google Cloud Storage
122  bucket_name, object_prefix = _parse_gcs_url(self.dest_gcs)
123  existing_files_prefixed = gcs_hook.list(
124  bucket_name, prefix=object_prefix)
125 
126  existing_files = []
127 
128  if existing_files_prefixed:
129  # Remove the object prefix itself, an empty directory was found
130  if object_prefix in existing_files_prefixed:
131  existing_files_prefixed.remove(object_prefix)
132 
133  # Remove the object prefix from all object string paths
134  for f in existing_files_prefixed:
135  if f.startswith(object_prefix):
136  existing_files.append(f[len(object_prefix):])
137  else:
138  existing_files.append(f)
139 
140  files = set(files) - set(existing_files)
141  if len(files) > 0:
142  self.log.info('{0} files are going to be synced: {1}.'.format(
143  len(files), files))
144  else:
145  self.log.info(
146  'There are no new files to sync. Have a nice day!')
147 
148  if files:
149  hook = S3Hook(aws_conn_id=self.aws_conn_id)
150 
151  for file in files:
152  # GCS hook builds its own in-memory file so we have to create
153  # and pass the path
154  file_object = hook.get_key(file, self.bucket)
155  with NamedTemporaryFile(mode='wb', delete=True) as f:
156  file_object.download_fileobj(f)
157  f.flush()
158 
159  dest_gcs_bucket, dest_gcs_object_prefix = _parse_gcs_url(
160  self.dest_gcs)
161  # There will always be a '/' before file because it is
162  # enforced at instantiation time
163  dest_gcs_object = dest_gcs_object_prefix + file
164 
165  # Sync is sequential and the hook already logs too much
166  # so skip this for now
167  # self.log.info(
168  # 'Saving file {0} from S3 bucket {1} in GCS bucket {2}'
169  # ' as object {3}'.format(file, self.bucket,
170  # dest_gcs_bucket,
171  # dest_gcs_object))
172 
173  gcs_hook.upload(dest_gcs_bucket, dest_gcs_object, f.name)
174 
175  self.log.info(
176  "All done, uploaded %d files to Google Cloud Storage",
177  len(files))
178  else:
179  self.log.info(
180  'In sync, no files needed to be uploaded to Google Cloud'
181  'Storage')
182 
183  return files
