29 class CassandraHook(BaseHook, LoggingMixin):
30  """
31  Hook used to interact with Cassandra
32 
33  Contact points can be specified as a comma-separated string in the 'hosts'
34  field of the connection.
35 
36  Port can be specified in the port field of the connection.
37 
38  If SSL is enabled in Cassandra, pass in a dict in the extra field as kwargs for
39  ``ssl.wrap_socket()``. For example:
40  {
41  'ssl_options' : {
42  'ca_certs' : PATH_TO_CA_CERTS
43  }
44  }
45 
46  Default load balancing policy is RoundRobinPolicy. To specify a different LB policy:
47  - DCAwareRoundRobinPolicy
48  {
49  'load_balancing_policy': 'DCAwareRoundRobinPolicy',
50  'load_balancing_policy_args': {
51  'local_dc': LOCAL_DC_NAME, // optional
52  'used_hosts_per_remote_dc': SOME_INT_VALUE, // optional
53  }
54  }
55  - WhiteListRoundRobinPolicy
56  {
57  'load_balancing_policy': 'WhiteListRoundRobinPolicy',
58  'load_balancing_policy_args': {
59  'hosts': ['HOST1', 'HOST2', 'HOST3']
60  }
61  }
62  - TokenAwarePolicy
63  {
64  'load_balancing_policy': 'TokenAwarePolicy',
65  'load_balancing_policy_args': {
66  'child_load_balancing_policy': CHILD_POLICY_NAME, // optional
67  'child_load_balancing_policy_args': { ... } // optional
68  }
69  }
70 
71  For details of the Cluster config, see cassandra.cluster.
72  """

124  @staticmethod
125  def get_lb_policy(policy_name, policy_args):
126  policies = {
127  'RoundRobinPolicy': RoundRobinPolicy,
128  'DCAwareRoundRobinPolicy': DCAwareRoundRobinPolicy,
129  'WhiteListRoundRobinPolicy': WhiteListRoundRobinPolicy,
130  'TokenAwarePolicy': TokenAwarePolicy,
131  }
132 
133  if not policies.get(policy_name) or policy_name == 'RoundRobinPolicy':
134  return RoundRobinPolicy()
135 
136  if policy_name == 'DCAwareRoundRobinPolicy':
137  local_dc = policy_args.get('local_dc', '')
138  used_hosts_per_remote_dc = int(policy_args.get('used_hosts_per_remote_dc', 0))
139  return DCAwareRoundRobinPolicy(local_dc, used_hosts_per_remote_dc)
140 
141  if policy_name == 'WhiteListRoundRobinPolicy':
142  hosts = policy_args.get('hosts')
143  if not hosts:
144  raise Exception('Hosts must be specified for WhiteListRoundRobinPolicy')
145  return WhiteListRoundRobinPolicy(hosts)
146 
147  if policy_name == 'TokenAwarePolicy':
148  allowed_child_policies = ('RoundRobinPolicy',
149  'DCAwareRoundRobinPolicy',
150  'WhiteListRoundRobinPolicy',)
151  child_policy_name = policy_args.get('child_load_balancing_policy',
152  'RoundRobinPolicy')
153  child_policy_args = policy_args.get('child_load_balancing_policy_args', {})
154  if child_policy_name not in allowed_child_policies:
155  return TokenAwarePolicy(RoundRobinPolicy())
156  else:
157  child_policy = CassandraHook.get_lb_policy(child_policy_name,
158  child_policy_args)
159  return TokenAwarePolicy(child_policy)

161  def record_exists(self, table, keys):
162  """
163  Checks if a record exists in Cassandra
164 
165  :param table: Target Cassandra table.
166  Use dot notation to target a specific keyspace.
167  :type table: string
168  :param keys: The keys and their values to check the existence.
169  :type keys: dict
170  """
171  keyspace = None
172  if '.' in table:
173  keyspace, table = table.split('.', 1)
174  ks = " AND ".join("{}=%({})s".format(key, key) for key in keys.keys())
175  cql = "SELECT * FROM {keyspace}.{table} WHERE {keys}".format(
176  keyspace=(keyspace or self.keyspace), table=table, keys=ks)
177 
178  try:
179  rs = self.get_conn().execute(cql, keys)
180  return rs.one() is not None
181  except Exception:
182  return False
