96  destination_object=None):
97  """
98  Has the same functionality as copy, except that will work on files
99  over 5 TB, as well as when copying between locations and/or storage
100  classes.
101 
102  destination_object can be omitted, in which case source_object is used.
103 
104  :param source_bucket: The bucket of the object to copy from.
105  :type source_bucket: string
106  :param source_object: The object to copy.
107  :type source_object: string
108  :param destination_bucket: The destination of the object to copied to.
109  :type destination_bucket: string
110  :param destination_object: The (renamed) path of the object if given.
111  Can be omitted; then the same name is used.
112  """
113  destination_object = destination_object or source_object
114  if (source_bucket == destination_bucket and
115  source_object == destination_object):
116  raise ValueError(
117  'Either source/destination bucket or source/destination object '
118  'must be different, not both the same: bucket=%s, object=%s' %
119  (source_bucket, source_object))
120  if not source_bucket or not source_object:
121  raise ValueError('source_bucket and source_object cannot be empty.')
122 
123  service = self.get_conn()
124  request_count = 1
125  try:
126  result = service.objects() \
127  .rewrite(sourceBucket=source_bucket, sourceObject=source_object,
128  destinationBucket=destination_bucket,
129  destinationObject=destination_object, body='') \
130  .execute()

288  def list(self, bucket, versions=None, maxResults=None, prefix=None, delimiter=None):
289  """
290  List all objects from the bucket with the give string prefix in name
291 
292  :param bucket: bucket name
293  :type bucket: string
294  :param versions: if true, list all versions of the objects
295  :type versions: boolean
296  :param maxResults: max count of items to return in a single page of responses
297  :type maxResults: integer
298  :param prefix: prefix string which filters objects whose name begin with
299  this prefix
300  :type prefix: string
301  :param delimiter: filters objects based on the delimiter (for e.g '.csv')
302  :type delimiter: string
303  :return: a stream of object names matching the filtering criteria
304  """
305  service = self.get_conn()
306 
307  ids = list()
308  pageToken = None
309  while(True):
310  response = service.objects().list(
311  bucket=bucket,
312  versions=versions,
313  maxResults=maxResults,
314  pageToken=pageToken,
315  prefix=prefix,
316  delimiter=delimiter
317  ).execute()
318 
319  if 'prefixes' not in response:
320  if 'items' not in response:
321  self.log.info("No items found for prefix: %s", prefix)
322  break
323 
324  for item in response['items']:
325  if item and 'name' in item:
326  ids.append(item['name'])
327  else:
328  for item in response['prefixes']:
329  ids.append(item)
330 
331  if 'nextPageToken' not in response:
332  # no further pages of results, so stop the loop
333  break
334 
335  pageToken = response['nextPageToken']
336  if not pageToken:
337  # empty next page token
338  break
339  return ids

426  def create_bucket(self,
427  bucket_name,
428  storage_class='MULTI_REGIONAL',
429  location='US',
430  project_id=None,
431  labels=None
432  ):
433  """
434  Creates a new bucket. Google Cloud Storage uses a flat namespace, so
435  you can't create a bucket with a name that is already in use.
436 
437  .. seealso::
438  For more information, see Bucket Naming Guidelines:
439  https://cloud.google.com/storage/docs/bucketnaming.html#requirements
440 
441  :param bucket_name: The name of the bucket.
442  :type bucket_name: string
443  :param storage_class: This defines how objects in the bucket are stored
444  and determines the SLA and the cost of storage. Values include
445 
446  - ``MULTI_REGIONAL``
447  - ``REGIONAL``
448  - ``STANDARD``
449  - ``NEARLINE``
450  - ``COLDLINE``.
451  If this value is not specified when the bucket is
452  created, it will default to STANDARD.
453  :type storage_class: string
454  :param location: The location of the bucket.
455  Object data for objects in the bucket resides in physical storage
456  within this region. Defaults to US.
457 
458  .. seealso::
459  https://developers.google.com/storage/docs/bucket-locations
460 
461  :type location: string
462  :param project_id: The ID of the GCP Project.
463  :type project_id: string
464  :param labels: User-provided labels, in key/value pairs.
465  :type labels: dict
466  :return: If successful, it returns the ``id`` of the bucket.
467  """
468 
469  project_id = project_id if project_id is not None else self.project_id
470  storage_classes = [
471  'MULTI_REGIONAL',
472  'REGIONAL',
473  'NEARLINE',
474  'COLDLINE',
475  'STANDARD', # alias for MULTI_REGIONAL/REGIONAL, based on location
476  ]
477 
478  self.log.info('Creating Bucket: %s; Location: %s; Storage Class: %s',
479  bucket_name, location, storage_class)
480  assert storage_class in storage_classes, \
481  'Invalid value ({}) passed to storage_class. Value should be ' \
482  'one of {}'.format(storage_class, storage_classes)
483 
484  assert re.match('[a-zA-Z0-9]+', bucket_name[0]), \
485  'Bucket names must start with a number or letter.'
486 
487  assert re.match('[a-zA-Z0-9]+', bucket_name[-1]), \
488  'Bucket names must end with a number or letter.'
489 
490  service = self.get_conn()
491  bucket_resource = {
492  'name': bucket_name,
493  'location': location,
494  'storageClass': storage_class
495  }
496 
497  self.log.info('The Default Project ID is %s', self.project_id)
498 
499  if labels is not None:
500  bucket_resource['labels'] = labels
501 
502  try:
503  response = service.buckets().insert(
504  project=project_id,
505  body=bucket_resource
506  ).execute()
507 
508  self.log.info('Bucket: %s created successfully.', bucket_name)

510  return response['id']

512  except errors.HttpError as ex:

518 def _parse_gcs_url(gsurl):
519  """
520  Given a Google Cloud Storage URL (gs://<bucket>/<blob>), returns a
521  tuple containing the corresponding bucket and blob.
522  """
523  # Python 3

526  # Python 2

529

531  if not parsed_url.netloc:
532  raise AirflowException('Please provide a bucket name')
533  else:
534  bucket = parsed_url.netloc
535  # Remove leading '/' but NOT trailing one
536  blob = parsed_url.path.lstrip('/')
537  return bucket, blob
