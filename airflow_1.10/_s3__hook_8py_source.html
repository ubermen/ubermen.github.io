254  def get_wildcard_key(self, wildcard_key, bucket_name=None, delimiter=''):
255  """
256  Returns a boto3.s3.Object object matching the wildcard expression
257 
258  :param wildcard_key: the path to the key
259  :type wildcard_key: str
260  :param bucket_name: the name of the bucket
261  :type bucket_name: str
262  """
263  if not bucket_name:
264  (bucket_name, wildcard_key) = self.parse_s3_url(wildcard_key)
265 
266  prefix = re.split(r'[*]', wildcard_key, 1)[0]
267  klist = self.list_keys(bucket_name, prefix=prefix, delimiter=delimiter)
268  if klist:
269  key_matches = [k for k in klist if fnmatch.fnmatch(k, wildcard_key)]
270  if key_matches:
271  return self.get_key(key_matches[0], bucket_name)

274  filename,
275  key,
276  bucket_name=None,
277  replace=False,
278  encrypt=False):
279  """
280  Loads a local file to S3
281 
282  :param filename: name of the file to load.
283  :type filename: str
284  :param key: S3 key that will point to the file
285  :type key: str
286  :param bucket_name: Name of the bucket in which to store the file
287  :type bucket_name: str
288  :param replace: A flag to decide whether or not to overwrite the key
289  if it already exists. If replace is False and the key exists, an
290  error will be raised.
291  :type replace: bool
292  :param encrypt: If True, the file will be encrypted on the server-side
293  by S3 and will be stored in an encrypted form while at rest in S3.
294  :type encrypt: bool
295  """
296  if not bucket_name:
297  (bucket_name, key) = self.parse_s3_url(key)
298 
299  if not replace and self.check_for_key(key, bucket_name):
300  raise ValueError("The key {key} already exists.".format(key=key))
301 
302  extra_args = {}
303  if encrypt:
304  extra_args['ServerSideEncryption'] = "AES256"
305 
306  client = self.get_conn()
307  client.upload_file(filename, bucket_name, key, ExtraArgs=extra_args)
