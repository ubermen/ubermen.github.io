25 class TimeDeltaSensor(BaseSensorOperator):
26  """
27  Waits for a timedelta after the task's execution_date + schedule_interval.
28  In Airflow, the daily task stamped with ``execution_date``
29  2016-01-01 can only start running on 2016-01-02. The timedelta here
30  represents the time after the execution period has closed.
31 
32  :param delta: time length to wait after execution_date before succeeding
33  :type delta: datetime.timedelta
34  """
35 
36  @apply_defaults
37  def __init__(self, delta, *args, **kwargs):
38  super(TimeDeltaSensor, self).__init__(*args, **kwargs)
39  self.delta = delta
40 
41  def poke(self, context):
42  dag = context['dag']
43  target_dttm = dag.following_schedule(context['execution_date'])
44  target_dttm += self.delta
45  self.log.info('Checking if the time (%s) has come', target_dttm)
46  return timezone.utcnow() > target_dttm
