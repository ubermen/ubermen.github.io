30 class HiveOperator(BaseOperator):
31  """
32  Executes hql code or hive script in a specific Hive database.
33 
34  :param hql: the hql to be executed. Note that you may also use
35  a relative path from the dag file of a (template) hive
36  script. (templated)
37  :type hql: string
38  :param hive_cli_conn_id: reference to the Hive database. (templated)
39  :type hive_cli_conn_id: string
40  :param hiveconfs: if defined, these key value pairs will be passed
41  to hive as ``-hiveconf "key"="value"``
42  :type hiveconfs: dict
43  :param hiveconf_jinja_translate: when True, hiveconf-type templating
44  ${var} gets translated into jinja-type templating {{ var }} and
45  ${hiveconf:var} gets translated into jinja-type templating {{ var }}.
46  Note that you may want to use this along with the
47  ``DAG(user_defined_macros=myargs)`` parameter. View the DAG
48  object documentation for more details.
49  :type hiveconf_jinja_translate: boolean
50  :param script_begin_tag: If defined, the operator will get rid of the
51  part of the script before the first occurrence of `script_begin_tag`
52  :type script_begin_tag: str
53  :param mapred_queue: queue used by the Hadoop CapacityScheduler. (templated)
54  :type mapred_queue: string
55  :param mapred_queue_priority: priority within CapacityScheduler queue.
56  Possible settings include: VERY_HIGH, HIGH, NORMAL, LOW, VERY_LOW
57  :type mapred_queue_priority: string
58  :param mapred_job_name: This name will appear in the jobtracker.
59  This can make monitoring easier.
60  :type mapred_job_name: string
61  """
