32 class DruidHook(BaseHook):
33  """
34  Connection to Druid overlord for ingestion
35 
36  :param druid_ingest_conn_id: The connection id to the Druid overlord machine
37  which accepts index jobs
38  :type druid_ingest_conn_id: string
39  :param timeout: The interval between polling
40  the Druid job for the status of the ingestion job
41  :type timeout: int
42  :param max_ingestion_time: The maximum ingestion time before assuming the job failed
43  :type max_ingestion_time: int
44  """
45  def __init__(
46  self,
47  druid_ingest_conn_id='druid_ingest_default',
48  timeout=1,
49  max_ingestion_time=None):
50 
51  self.druid_ingest_conn_id = druid_ingest_conn_id
52  self.timeout = timeout
53  self.max_ingestion_time = max_ingestion_time
54  self.header = {'content-type': 'application/json'}
55 
56  def get_conn_url(self):
57  conn = self.get_connection(self.druid_ingest_conn_id)
58  host = conn.host
59  port = conn.port
60  conn_type = 'http' if not conn.conn_type else conn.conn_type
61  endpoint = conn.extra_dejson.get('endpoint', '')
62  return "{conn_type}://{host}:{port}/{endpoint}".format(**locals())
63 
64  def submit_indexing_job(self, json_index_spec):
65  url = self.get_conn_url()
66 
67  req_index = requests.post(url, data=json_index_spec, headers=self.header)
68  if (req_index.status_code != 200):
69  raise AirflowException('Did not get 200 when '
70  'submitting the Druid job to {}'.format(url))
71 
72  req_json = req_index.json()
73  # Wait until the job is completed
74  druid_task_id = req_json['task']
75 
76  running = True
77 
78  sec = 0
79  while running:
80  req_status = requests.get("{0}/{1}/status".format(url, druid_task_id))
81 
82  self.log.info("Job still running for %s seconds...", sec)
83 
84  sec = sec + 1
85 
86  if self.max_ingestion_time and sec > self.max_ingestion_time:
87  # ensure that the job gets killed if the max ingestion time is exceeded
88  requests.post("{0}/{1}/shutdown".format(url, druid_task_id))
89  raise AirflowException('Druid ingestion took more than '
90  '%s seconds', self.max_ingestion_time)
91 
92  time.sleep(self.timeout)
93 
94  status = req_status.json()['status']['status']
95  if status == 'RUNNING':
96  running = True
97  elif status == 'SUCCESS':
98  running = False # Great success!
99  elif status == 'FAILED':
100  raise AirflowException('Druid indexing job failed, '
101  'check console for more info')
102  else:
103  raise AirflowException('Could not get status of the job, got %s', status)
104 
105  self.log.info('Successful index')
