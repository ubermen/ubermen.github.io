27 class ExternalTaskSensor(BaseSensorOperator):
28  """
29  Waits for a task to complete in a different DAG
30 
31  :param external_dag_id: The dag_id that contains the task you want to
32  wait for
33  :type external_dag_id: string
34  :param external_task_id: The task_id that contains the task you want to
35  wait for
36  :type external_task_id: string
37  :param allowed_states: list of allowed states, default is ``['success']``
38  :type allowed_states: list
39  :param execution_delta: time difference with the previous execution to
40  look at, the default is the same execution_date as the current task.
41  For yesterday, use [positive!] datetime.timedelta(days=1). Either
42  execution_delta or execution_date_fn can be passed to
43  ExternalTaskSensor, but not both.
44  :type execution_delta: datetime.timedelta
45  :param execution_date_fn: function that receives the current execution date

47  or execution_date_fn can be passed to ExternalTaskSensor, but not both.
48  :type execution_date_fn: callable
