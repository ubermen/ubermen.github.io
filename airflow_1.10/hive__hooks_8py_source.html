48 class HiveCliHook(BaseHook):
49  """Simple wrapper around the hive CLI.
50 
51  It also supports the ``beeline``
52  a lighter CLI that runs JDBC and is replacing the heavier
53  traditional CLI. To enable ``beeline``, set the use_beeline param in the
54  extra field of your connection as in ``{ "use_beeline": true }``
55 
56  Note that you can also set default hive CLI parameters using the
57  ``hive_cli_params`` to be used in your connection as in
58  ``{"hive_cli_params": "-hiveconf mapred.job.tracker=some.jobtracker:444"}``
59  Parameters passed here can be overridden by run_cli's hive_conf param
60 
61  The extra connection parameter ``auth`` gets passed as in the ``jdbc``
62  connection string as is.
63 
64  :param mapred_queue: queue used by the Hadoop Scheduler (Capacity or Fair)
65  :type mapred_queue: string
66  :param mapred_queue_priority: priority within the job queue.
67  Possible settings include: VERY_HIGH, HIGH, NORMAL, LOW, VERY_LOW
68  :type mapred_queue_priority: string
69  :param mapred_job_name: This name will appear in the jobtracker.
70  This can make monitoring easier.
71  :type mapred_job_name: string
72  """

162  def run_cli(self, hql, schema=None, verbose=True, hive_conf=None):
163  """
164  Run an hql statement using the hive cli. If hive_conf is specified
165  it should be a dict and the entries will be set as key/value pairs
166  in HiveConf
167 
168 
169  :param hive_conf: if specified these key value pairs will be passed
170  to hive as ``-hiveconf "key"="value"``. Note that they will be
171  passed after the ``hive_cli_params`` and thus will override
172  whatever values are specified in the database.
173  :type hive_conf: dict
174 
175  >>> hh = HiveCliHook()
176  >>> result = hh.run_cli("USE airflow;")
177  >>> ("OK" in result)
178  True
179  """
180  conn = self.conn
181  schema = schema or conn.schema
182  if schema:
183  hql = "USE {schema};\n{hql}".format(**locals())
184 
185  with TemporaryDirectory(prefix='airflow_hiveop_') as tmp_dir:
186  with NamedTemporaryFile(dir=tmp_dir) as f:
187  f.write(hql.encode('UTF-8'))
188  f.flush()
189  hive_cmd = self._prepare_cli_cmd()
190  hive_conf_params = self._prepare_hiveconf(hive_conf)
191  if self.mapred_queue:
192  hive_conf_params.extend(
193  ['-hiveconf',
194  'mapreduce.job.queuename={}'
195  .format(self.mapred_queue),
196  '-hiveconf',
197  'mapred.job.queue.name={}'
198  .format(self.mapred_queue),
199  '-hiveconf',
200  'tez.job.queue.name={}'
201  .format(self.mapred_queue)
202  ])
203 
204  if self.mapred_queue_priority:
205  hive_conf_params.extend(
206  ['-hiveconf',
207  'mapreduce.job.priority={}'
208  .format(self.mapred_queue_priority)])
209 
210  if self.mapred_job_name:
211  hive_conf_params.extend(
212  ['-hiveconf',
213  'mapred.job.name={}'
214  .format(self.mapred_job_name)])
215 
216  hive_cmd.extend(hive_conf_params)
217  hive_cmd.extend(['-f', f.name])
218 
219  if verbose:
220  self.log.info(" ".join(hive_cmd))
221  sp = subprocess.Popen(
222  hive_cmd,
223  stdout=subprocess.PIPE,
224  stderr=subprocess.STDOUT,
225  cwd=tmp_dir,
226  close_fds=True)
227  self.sp = sp
228  stdout = ''
229  while True:
230  line = sp.stdout.readline()
231  if not line:
232  break
233  stdout += line.decode('UTF-8')
234  if verbose:
235  self.log.info(line.decode('UTF-8').strip())
236  sp.wait()
237 
238  if sp.returncode:
239  raise AirflowException(stdout)
240 
241  return stdout

243  def test_hql(self, hql):
244  """
245  Test an hql statement using the hive cli and EXPLAIN
246 
247  """
248  create, insert, other = [], [], []
249  for query in hql.split(';'): # naive
250  query_original = query
251  query = query.lower().strip()
252 
253  if query.startswith('create table'):
254  create.append(query_original)
255  elif query.startswith(('set ',
256  'add jar ',
257  'create temporary function')):
258  other.append(query_original)
259  elif query.startswith('insert'):
260  insert.append(query_original)
261  other = ';'.join(other)
262  for query_set in [create, insert]:
263  for query in query_set:
264 
265  query_preview = ' '.join(query.split())[:50]
266  self.log.info("Testing HQL [%s (...)]", query_preview)
267  if query_set == insert:
268  query = other + '; explain ' + query
269  else:
270  query = 'explain ' + query
271  try:
272  self.run_cli(query, verbose=False)
273  except AirflowException as e:
274  message = e.args[0].split('\n')[-2]
275  self.log.info(message)
276  error_loc = re.search('(\d+):(\d+)', message)
277  if error_loc and error_loc.group(1).isdigit():
278  lst = int(error_loc.group(1))
279  begin = max(lst - 2, 0)
280  end = min(lst + 3, len(query.split('\n')))
281  context = '\n'.join(query.split('\n')[begin:end])
282  self.log.info("Context :\n %s", context)
283  else:
284  self.log.info("SUCCESS")

317  def _infer_field_types_from_df(df):
318  DTYPE_KIND_HIVE_TYPE = {
319  'b': 'BOOLEAN', # boolean
320  'i': 'BIGINT', # signed integer
321  'u': 'BIGINT', # unsigned integer
322  'f': 'DOUBLE', # floating-point
323  'c': 'STRING', # complex floating-point
324  'M': 'TIMESTAMP', # datetime
325  'O': 'STRING', # object
326  'S': 'STRING', # (byte-)string
327  'U': 'STRING', # Unicode
328  'V': 'STRING' # void
329  }
330 
331  d = OrderedDict()
332  for col, dtype in df.dtypes.iteritems():
333  d[col] = DTYPE_KIND_HIVE_TYPE[dtype.kind]
334  return d
335 
336  if pandas_kwargs is None:
337  pandas_kwargs = {}
338 
339  with TemporaryDirectory(prefix='airflow_hiveop_') as tmp_dir:
340  with NamedTemporaryFile(dir=tmp_dir, mode="w") as f:
341 
342  if field_dict is None:
343  field_dict = _infer_field_types_from_df(df)
344 
345  df.to_csv(path_or_buf=f,
346  sep=(delimiter.encode(encoding)
347  if six.PY2 and isinstance(delimiter, unicode)
348  else delimiter),

350  index=False,
351  encoding=encoding,
352  date_format="%Y-%m-%d %H:%M:%S",
353  **pandas_kwargs)

362  def load_file(
363  self,
364  filepath,
365  table,
366  delimiter=",",
367  field_dict=None,
368  create=True,
369  overwrite=True,
370  partition=None,
371  recreate=False,
372  tblproperties=None):
373  """
374  Loads a local file into Hive
375 
376  Note that the table generated in Hive uses ``STORED AS textfile``
377  which isn't the most efficient serialization format. If a
378  large amount of data is loaded and/or if the tables gets
379  queried considerably, you may want to use this operator only to
380  stage the data into a temporary table before loading it into its
381  final destination using a ``HiveOperator``.
382 
383  :param filepath: local filepath of the file to load
384  :type filepath: str
385  :param table: target Hive table, use dot notation to target a
386  specific database
387  :type table: str
388  :param delimiter: field delimiter in the file
389  :type delimiter: str
390  :param field_dict: A dictionary of the fields name in the file
391  as keys and their Hive types as values.
392  Note that it must be OrderedDict so as to keep columns' order.
393  :type field_dict: OrderedDict
394  :param create: whether to create the table if it doesn't exist
395  :type create: bool
396  :param overwrite: whether to overwrite the data in table or partition
397  :type overwrite: bool
398  :param partition: target partition as a dict of partition columns
399  and values
400  :type partition: dict
401  :param recreate: whether to drop and recreate the table at every
402  execution
403  :type recreate: bool
404  :param tblproperties: TBLPROPERTIES of the hive table being created
405  :type tblproperties: dict
406  """
407  hql = ''
408  if recreate:
409  hql += "DROP TABLE IF EXISTS {table};\n"
410  if create or recreate:
411  if field_dict is None:
412  raise ValueError("Must provide a field dict when creating a table")
413  fields = ",\n ".join(
414  [k + ' ' + v for k, v in field_dict.items()])
415  hql += "CREATE TABLE IF NOT EXISTS {table} (\n{fields})\n"
416  if partition:
417  pfields = ",\n ".join(
418  [p + " STRING" for p in partition])
419  hql += "PARTITIONED BY ({pfields})\n"
420  hql += "ROW FORMAT DELIMITED\n"
421  hql += "FIELDS TERMINATED BY '{delimiter}'\n"
422  hql += "STORED AS textfile\n"
423  if tblproperties is not None:
424  tprops = ", ".join(
425  ["'{0}'='{1}'".format(k, v) for k, v in tblproperties.items()])
426  hql += "TBLPROPERTIES({tprops})\n"
427  hql += ";"
428  hql = hql.format(**locals())
429  self.log.info(hql)
430  self.run_cli(hql)
431  hql = "LOAD DATA LOCAL INPATH '{filepath}' "
432  if overwrite:
433  hql += "OVERWRITE "
434  hql += "INTO TABLE {table} "
435  if partition:
436  pvals = ", ".join(
437  ["{0}='{1}'".format(k, v) for k, v in partition.items()])
438  hql += "PARTITION ({pvals});"
439 
440  # As a workaround for HIVE-10541, add a newline character
441  # at the end of hql (AIRFLOW-2412).
442  hql += '\n'
443 
444  hql = hql.format(**locals())
445  self.log.info(hql)
446  self.run_cli(hql)

596  def get_partitions(
597  self, schema, table_name, filter=None):
598  """
599  Returns a list of all partitions in a table. Works only
600  for tables with less than 32767 (java short max val).
601  For subpartitioned table, the number might easily exceed this.
602 
603  >>> hh = HiveMetastoreHook()
604  >>> t = 'static_babynames_partitioned'
605  >>> parts = hh.get_partitions(schema='airflow', table_name=t)
606  >>> len(parts)
607  1
608  >>> parts
609  [{'ds': '2015-01-01'}]
610  """
611  with self.metastore as client:
612  table = client.get_table(dbname=schema, tbl_name=table_name)
613  if len(table.partitionKeys) == 0:
614  raise AirflowException("The table isn't partitioned")
615  else:
616  if filter:
617  parts = client.get_partitions_by_filter(
618  db_name=schema, tbl_name=table_name,
619  filter=filter, max_parts=HiveMetastoreHook.MAX_PART_COUNT)
620  else:
621  parts = client.get_partitions(
622  db_name=schema, tbl_name=table_name,
623  max_parts=HiveMetastoreHook.MAX_PART_COUNT)
624 
625  pnames = [p.name for p in table.partitionKeys]
626  return [dict(zip(pnames, p.values)) for p in parts]

628  @staticmethod
629  def _get_max_partition_from_part_specs(part_specs, partition_key, filter_map):
630  """
631  Helper method to get max partition of partitions with partition_key
632  from part specs. key:value pair in filter_map will be used to
633  filter out partitions.
634 
635  :param part_specs: list of partition specs.
636  :type part_specs: list
637  :param partition_key: partition key name.
638  :type partition_key: string
639  :param filter_map: partition_key:partition_value map used for partition filtering,
640  e.g. {'key1': 'value1', 'key2': 'value2'}.
641  Only partitions matching all partition_key:partition_value
642  pairs will be considered as candidates of max partition.
643  :type filter_map: map
644  :return: Max partition or None if part_specs is empty.
645  """
646  if not part_specs:
647  return None
648 
649  # Assuming all specs have the same keys.
650  if partition_key not in part_specs[0].keys():
651  raise AirflowException("Provided partition_key {} "
652  "is not in part_specs.".format(partition_key))
653  if filter_map:
654  is_subset = set(filter_map.keys()).issubset(set(part_specs[0].keys()))
655  if filter_map and not is_subset:
656  raise AirflowException("Keys in provided filter_map {} "
657  "are not subset of part_spec keys: {}"
658  .format(', '.join(filter_map.keys()),
659  ', '.join(part_specs[0].keys())))
660 
661  candidates = [p_dict[partition_key] for p_dict in part_specs
662  if filter_map is None or
663  all(item in p_dict.items() for item in filter_map.items())]
664 
665  if not candidates:
666  return None
667  else:
668  return max(candidates).encode('utf-8')

670  def max_partition(self, schema, table_name, field=None, filter_map=None):
671  """
672  Returns the maximum value for all partitions with given field in a table.
673  If only one partition key exist in the table, the key will be used as field.
674  filter_map should be a partition_key:partition_value map and will be used to
675  filter out partitions.
676 
677  :param schema: schema name.
678  :type schema: string
679  :param table_name: table name.
680  :type table_name: string
681  :param field: partition key to get max partition from.
682  :type field: string
683  :param filter_map: partition_key:partition_value map used for partition filtering.
684  :type filter_map: map
685 
686  >>> hh = HiveMetastoreHook()
687  >>> filter_map = {'ds': '2015-01-01', 'ds': '2014-01-01'}
688  >>> t = 'static_babynames_partitioned'
689  >>> hh.max_partition(schema='airflow',\
690  ... table_name=t, field='ds', filter_map=filter_map)
691  '2015-01-01'
692  """
693  with self.metastore as client:
694  table = client.get_table(dbname=schema, tbl_name=table_name)
695  key_name_set = set(key.name for key in table.partitionKeys)
696  if len(table.partitionKeys) == 1:
697  field = table.partitionKeys[0].name
698  elif not field:
699  raise AirflowException("Please specify the field you want the max "
700  "value for.")
701  elif field not in key_name_set:
702  raise AirflowException("Provided field is not a partition key.")
703 
704  if filter_map and not set(filter_map.keys()).issubset(key_name_set):
705  raise AirflowException("Provided filter_map contains keys "
706  "that are not partition key.")
707 
708  part_names = \
709  client.get_partition_names(schema,
710  table_name,
711  max_parts=HiveMetastoreHook.MAX_PART_COUNT)
712  part_specs = [client.partition_name_to_spec(part_name)
713  for part_name in part_names]
714 
715  return HiveMetastoreHook._get_max_partition_from_part_specs(part_specs,
716  field,
717  filter_map)

780  with contextlib.closing(self.get_conn(schema)) as conn, \
