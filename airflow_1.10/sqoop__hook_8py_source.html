32 class SqoopHook(BaseHook, LoggingMixin):
33  """
34  This hook is a wrapper around the sqoop 1 binary. To be able to use the hook
35  it is required that "sqoop" is in the PATH.
36 
37  Additional arguments that can be passed via the 'extra' JSON field of the
38  sqoop connection:
39  * job_tracker: Job tracker local|jobtracker:port.
40  * namenode: Namenode.
41  * lib_jars: Comma separated jar files to include in the classpath.
42  * files: Comma separated files to be copied to the map reduce cluster.
43  * archives: Comma separated archives to be unarchived on the compute
44  machines.
45  * password_file: Path to file containing the password.
46 
47  :param conn_id: Reference to the sqoop connection.
48  :type conn_id: str
49  :param verbose: Set sqoop to verbose.
50  :type verbose: bool
51  :param num_mappers: Number of map tasks to import in parallel.
52  :type num_mappers: int
53  :param properties: Properties to set via the -D argument
54  :type properties: dict
55  """

83  def cmd_mask_password(self, cmd_orig):
84  cmd = deepcopy(cmd_orig)
85  try:
86  password_index = cmd.index('--password')
87  cmd[password_index + 1] = 'MASKED'
88  except ValueError:
89  self.log.debug("No password in sqoop cmd")
90  return cmd

92  def Popen(self, cmd, **kwargs):
93  """
94  Remote Popen
95 
96  :param cmd: command to remotely execute
97  :param kwargs: extra arguments to Popen (see subprocess.Popen)
98  :return: handle to subprocess
99  """
100  masked_cmd = ' '.join(self.cmd_mask_password(cmd))
101  self.log.info("Executing command: {}".format(masked_cmd))
102  self.sp = subprocess.Popen(
103  cmd,
104  stdout=subprocess.PIPE,
105  stderr=subprocess.STDOUT,
106  **kwargs)
107 
108  for line in iter(self.sp.stdout):
109  self.log.info(line.strip())
110 
111  self.sp.wait()
112 
113  self.log.info("Command exited with return code %s", self.sp.returncode)
114 
115  if self.sp.returncode:
116  raise AirflowException("Sqoop command failed: {}".format(masked_cmd))

118  def _prepare_command(self, export=False):
119  sqoop_cmd_type = "export" if export else "import"
120  connection_cmd = ["sqoop", sqoop_cmd_type]
121 
122  for key, value in self.properties.items():
123  connection_cmd += ["-D", "{}={}".format(key, value)]
124 
125  if self.namenode:
126  connection_cmd += ["-fs", self.namenode]
127  if self.job_tracker:
128  connection_cmd += ["-jt", self.job_tracker]
129  if self.libjars:
130  connection_cmd += ["-libjars", self.libjars]
131  if self.files:
132  connection_cmd += ["-files", self.files]
133  if self.archives:
134  connection_cmd += ["-archives", self.archives]
135  if self.conn.login:
136  connection_cmd += ["--username", self.conn.login]
137  if self.conn.password:
138  connection_cmd += ["--password", self.conn.password]
139  if self.password_file:
140  connection_cmd += ["--password-file", self.password_file]
141  if self.verbose:
142  connection_cmd += ["--verbose"]
143  if self.num_mappers:
144  connection_cmd += ["--num-mappers", str(self.num_mappers)]
145  if self.hcatalog_database:
146  connection_cmd += ["--hcatalog-database", self.hcatalog_database]
147  if self.hcatalog_table:
148  connection_cmd += ["--hcatalog-table", self.hcatalog_table]
149 
150  connection_cmd += ["--connect", "{}:{}/{}".format(
151  self.conn.host,
152  self.conn.port,
153  self.conn.schema
154  )]
155 
156  return connection_cmd
