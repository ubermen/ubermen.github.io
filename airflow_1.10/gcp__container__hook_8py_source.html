61  def wait_for_operation(self, operation):
62  """
63  Given an operation, continuously fetches the status from Google Cloud until either
64  completion or an error occurring
65  :param operation: The Operation to wait for
66  :type operation: A google.cloud.container_V1.gapic.enums.Operator
67  :return: A new, updated operation fetched from Google Cloud
68  """
69  self.log.info("Waiting for OPERATION_NAME %s" % operation.name)
70  time.sleep(OPERATIONAL_POLL_INTERVAL)
71  while operation.status != Operation.Status.DONE:
72  if operation.status == Operation.Status.RUNNING or operation.status == \
73  Operation.Status.PENDING:
74  time.sleep(OPERATIONAL_POLL_INTERVAL)
75  else:
76  raise exceptions.GoogleCloudError(
77  "Operation has failed with status: %s" % operation.status)
78  # To update status of operation
79  operation = self.get_operation(operation.name)
80  return operation

93  def _append_label(self, cluster_proto, key, val):
94  """
95  Append labels to provided Cluster Protobuf
96 
97  Labels must fit the regex [a-z]([-a-z0-9]*[a-z0-9])? (current airflow version
98  string follows semantic versioning spec: x.y.z).
99  :param cluster_proto: The proto to append resource_label airflow version to
100  :type cluster_proto: google.cloud.container_v1.types.Cluster
101  :param key: The key label
102  :type key: str
103  :param val:
104  :type val: str
105  :return: The cluster proto updated with new label
106  """
107  val = val.replace('.', '-').replace('+', '-')
108  cluster_proto.resource_labels.update({key: val})
