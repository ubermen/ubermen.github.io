28 class DatadogHook(BaseHook, LoggingMixin):
29  """
30  Uses datadog API to send metrics of practically anything measurable,
31  so it's possible to track # of db records inserted/deleted, records read
32  from file and many other useful metrics.
33 
34  Depends on the datadog API, which has to be deployed on the same server where
35  Airflow runs.
36 
37  :param datadog_conn_id: The connection to datadog, containing metadata for api keys.
38  :param datadog_conn_id: string
39  """
40  def __init__(self, datadog_conn_id='datadog_default'):
41  conn = self.get_connection(datadog_conn_id)
42  self.api_key = conn.extra_dejson.get('api_key', None)
43  self.app_key = conn.extra_dejson.get('app_key', None)
44  self.source_type_name = conn.extra_dejson.get('source_type_name', None)
45 
46  # If the host is populated, it will use that hostname instead.
47  # for all metric submissions.
48  self.host = conn.host
49 
50  if self.api_key is None:
51  raise AirflowException("api_key must be specified in the "
52  "Datadog connection details")
53  if self.app_key is None:
54  raise AirflowException("app_key must be specified in the "
55  "Datadog connection details")
56 
57  self.log.info("Setting up api keys for Datadog")
58  options = {
59  'api_key': self.api_key,
60  'app_key': self.app_key
61  }
62  initialize(**options)
63 
64  def validate_response(self, response):
65  if response['status'] != 'ok':
66  self.log.error("Datadog returned: %s", response)
67  raise AirflowException("Error status received from Datadog")

69  def send_metric(self, metric_name, datapoint, tags=None):
70  """
71  Sends a single datapoint metric to DataDog
72 
73  :param metric_name: The name of the metric
74  :type metric_name: string
75  :param datapoint: A single integer or float related to the metric
76  :type datapoint: integer or float
77  :param tags: A list of tags associated with the metric
78  :type tags: list
79  """
80  response = api.Metric.send(
81  metric=metric_name,
82  points=datapoint,
83  host=self.host,
84  tags=tags)
85 
86  self.validate_response(response)
87  return response

89  def query_metric(self,
90  query,
91  from_seconds_ago,
92  to_seconds_ago):
93  """
94  Queries datadog for a specific metric, potentially with some
95  function applied to it and returns the results.
96 
97  :param query: The datadog query to execute (see datadog docs)
98  :type query: string
99  :param from_seconds_ago: How many seconds ago to start querying for.
100  :type from_seconds_ago: int
101  :param to_seconds_ago: Up to how many seconds ago to query for.
102  :type to_seconds_ago: int
103  """
104  now = int(time.time())
105 
106  response = api.Metric.query(
107  start=now - from_seconds_ago,
108  end=now - to_seconds_ago,
109  query=query)
110 
111  self.validate_response(response)
112  return response

114  def post_event(self, title, text, tags=None, alert_type=None, aggregation_key=None):
115  """
116  Posts an event to datadog (processing finished, potentially alerts, other issues)
117  Think about this as a means to maintain persistence of alerts, rather than
118  alerting itself.
119 
120  :param title: The title of the event
121  :type title: string
122  :param text: The body of the event (more information)
123  :type text: string
124  :param tags: List of string tags to apply to the event
125  :type tags: list
126  :param alert_type: The alert type for the event, one of
127  ["error", "warning", "info", "success"]
128  :type alert_type: string
129  :param aggregation_key: Key that can be used to aggregate this event in a stream
130  :type aggregation_key: string
131  """
132  response = api.Event.create(
133  title=title,
134  text=text,
135  host=self.host,
136  tags=tags,
137  alert_type=alert_type,
138  aggregation_key=aggregation_key,
139  source_type_name=self.source_type_name)
140 
141  self.validate_response(response)
142  return response
