43  """
44  Interact with HDFS. This class is a wrapper around the hdfscli library.
45  """
46  def __init__(self, webhdfs_conn_id='webhdfs_default', proxy_user=None):
47  self.webhdfs_conn_id = webhdfs_conn_id
48  self.proxy_user = proxy_user

50  def get_conn(self):
51  """
52  Returns a hdfscli InsecureClient object.
53  """
54  nn_connections = self.get_connections(self.webhdfs_conn_id)
55  for nn in nn_connections:
56  try:
57  self.log.debug('Trying namenode %s', nn.host)
58  connection_str = 'http://{nn.host}:{nn.port}'.format(nn=nn)
59  if _kerberos_security_mode:
60  client = KerberosClient(connection_str)
61  else:
62  proxy_user = self.proxy_user or nn.login
63  client = InsecureClient(connection_str, user=proxy_user)
64  client.status('/')
65  self.log.debug('Using namenode %s for hook', nn.host)
66  return client
67  except HdfsError as e:
68  self.log.debug(
69  "Read operation on namenode {nn.host} "
70  "failed with error: {e}".format(**locals())
71  )
72  nn_hosts = [c.host for c in nn_connections]
73  no_nn_error = "Read operations failed " \
74  "on the namenodes below:\n{}".format("\n".join(nn_hosts))
75  raise AirflowWebHDFSHookException(no_nn_error)

86  """
87  Uploads a file to HDFS

91  of files will not be created remotely).
92  :type source: str

95  :type destination: str
96  :param overwrite: Overwrite any existing file or directory.
97  :type overwrite: bool
98  :param parallelism: Number of threads to use for parallelization. A value of
99  `0` (or negative) uses as many threads as there are files.
100  :type parallelism: int
101  :param \*\*kwargs: Keyword arguments forwarded to :meth:`upload`.
102 
103
