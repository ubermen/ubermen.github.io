138 def init_role(sm, role_name, role_vms, role_perms):
139  sm_session = sm.get_session
140  pvms = sm_session.query(sqla_models.PermissionView).all()
141  pvms = [p for p in pvms if p.permission and p.view_menu]
142 
143  valid_perms = [p.permission.name for p in pvms]
144  valid_vms = [p.view_menu.name for p in pvms]
145  invalid_perms = [p for p in role_perms if p not in valid_perms]
146  if invalid_perms:
147  raise Exception('The following permissions are not valid: {}'
148  .format(invalid_perms))
149  invalid_vms = [v for v in role_vms if v not in valid_vms]
150  if invalid_vms:
151  raise Exception('The following view menus are not valid: {}'
152  .format(invalid_vms))
153 
154  role = sm.add_role(role_name)
155  role_pvms = []
156  for pvm in pvms:
157  if pvm.view_menu.name in role_vms and pvm.permission.name in role_perms:
158  role_pvms.append(pvm)
159  role_pvms = list(set(role_pvms))
160  role.permissions = role_pvms
161  sm_session.merge(role)
162  sm_session.commit()
