56  @staticmethod

58  """
59  Will test the filepath result and test if its size is at least self.filesize
60 
61  :param result: a list of dicts returned by Snakebite ls
62  :param size: the file size in MB a file should be at least to trigger True
63  :return: (bool) depending on the matching criteria
64  """
65  if size:
66  log = LoggingMixin().log
67  log.debug(
68  'Filtering for file size >= %s in files: %s',
69  size, map(lambda x: x['path'], result)
70  )
71  size *= settings.MEGABYTE

76  @staticmethod

78  """
79  Will filter if instructed to do so the result to remove matching criteria
80 
81  :param result: (list) of dicts returned by Snakebite ls
82  :param ignored_ext: (list) of ignored extensions
83  :param ignore_copying: (bool) shall we ignore ?
84  :return: (list) of dicts which were not removed

94  result = [x for x in result if not ignored_extentions_regex.match(x['path'])]
95  log.debug('HdfsSensor.poke: after ext filter result is %s', result)
