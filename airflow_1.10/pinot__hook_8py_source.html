53  def get_uri(self):
54  """
55  Get the connection uri for pinot broker.
56 
57  e.g: http://localhost:9000/pql
58  """
59  conn = self.get_connection(getattr(self, self.conn_name_attr))
60  host = conn.host
61  if conn.port is not None:
62  host += ':{port}'.format(port=conn.port)
63  conn_type = 'http' if not conn.conn_type else conn.conn_type
64  endpoint = conn.extra_dejson.get('endpoint', 'pql')
65  return '{conn_type}://{host}/{endpoint}'.format(
66  conn_type=conn_type, host=host, endpoint=endpoint)
