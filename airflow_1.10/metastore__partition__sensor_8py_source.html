24 class MetastorePartitionSensor(SqlSensor):
25  """
26  An alternative to the HivePartitionSensor that talk directly to the
27  MySQL db. This was created as a result of observing sub optimal
28  queries generated by the Metastore thrift service when hitting
29  subpartitioned tables. The Thrift service's queries were written in a
30  way that wouldn't leverage the indexes.
31 
32  :param schema: the schema
33  :type schema: str
34  :param table: the table
35  :type table: str
36  :param partition_name: the partition name, as defined in the PARTITIONS
37  table of the Metastore. Order of the fields does matter.
38  Examples: ``ds=2016-01-01`` or
39  ``ds=2016-01-01/sub=foo`` for a sub partitioned table
40  :type partition_name: str
41  :param mysql_conn_id: a reference to the MySQL conn_id for the metastore
42  :type mysql_conn_id: str
43  """
44  template_fields = ('partition_name', 'table', 'schema')
45  ui_color = '#8da7be'
46

53  *args,

61  # TODO(aoen): We shouldn't be using SqlSensor here but MetastorePartitionSensor.
62  # The problem is the way apply_defaults works isn't compatible with inheritance.
63  # The inheritance model needs to be reworked in order to support overriding args/
64  # kwargs with arguments here, then 'conn_id' and 'sql' can be passed into the
65  # constructor below and apply_defaults will no longer throw an exception.

68  def poke(self, context):
69  if self.first_poke:
70  self.first_poke = False
71  if '.' in self.table:
72  self.schema, self.table = self.table.split('.')
73  self.sql = """
74  SELECT 'X'
75  FROM PARTITIONS A0
76  LEFT OUTER JOIN TBLS B0 ON A0.TBL_ID = B0.TBL_ID
77  LEFT OUTER JOIN DBS C0 ON B0.DB_ID = C0.DB_ID
78  WHERE
79  B0.TBL_NAME = '{self.table}' AND
80  C0.NAME = '{self.schema}' AND
81  A0.PART_NAME = '{self.partition_name}';
82  """.format(self=self)
83  return super(MetastorePartitionSensor, self).poke(context)
