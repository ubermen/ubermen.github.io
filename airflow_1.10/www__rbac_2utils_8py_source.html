75 def generate_pages(current_page, num_of_pages,
76  search=None, showPaused=None, window=7):
77  """
78  Generates the HTML for a paging component using a similar logic to the paging
79  auto-generated by Flask managed views. The paging component defines a number of
80  pages visible in the pager (window) and once the user goes to a page beyond the
81  largest visible, it would scroll to the right the page numbers and keeps the
82  current one in the middle of the pager component. When in the last pages,
83  the pages won't scroll and just keep moving until the last page. Pager also contains
84  <first, previous, ..., next, last> pages.
85  This component takes into account custom parameters such as search and showPaused,
86  which could be added to the pages link in order to maintain the state between
87  client and server. It also allows to make a bookmark on a specific paging state.
88  :param current_page:
89  the current page number, 0-indexed
90  :param num_of_pages:
91  the total number of pages
92  :param search:
93  the search query string, if any
94  :param showPaused:
95  false if paused dags will be hidden, otherwise true to show them
96  :param window:
97  the number of pages to be shown in the paging component (7 default)
98  :return:
99  the HTML string of the paging component
100  """
101 
102  void_link = 'javascript:void(0)'
103  first_node = """<li class="paginate_button {disabled}" id="dags_first">
104  <a href="{href_link}" aria-controls="dags" data-dt-idx="0" tabindex="0">&laquo;</a>
105 </li>"""
106 
107  previous_node = """<li class="paginate_button previous {disabled}" id="dags_previous">
108  <a href="{href_link}" aria-controls="dags" data-dt-idx="0" tabindex="0">&lt;</a>
109 </li>"""
110 
111  next_node = """<li class="paginate_button next {disabled}" id="dags_next">
112  <a href="{href_link}" aria-controls="dags" data-dt-idx="3" tabindex="0">&gt;</a>
113 </li>"""
114 
115  last_node = """<li class="paginate_button {disabled}" id="dags_last">
116  <a href="{href_link}" aria-controls="dags" data-dt-idx="3" tabindex="0">&raquo;</a>
117 </li>"""
118 
119  page_node = """<li class="paginate_button {is_active}">
120  <a href="{href_link}" aria-controls="dags" data-dt-idx="2" tabindex="0">{page_num}</a>
121 </li>"""
122 
123  output = ['<ul class="pagination" style="margin-top:0px;">']
124 
125  is_disabled = 'disabled' if current_page <= 0 else ''
126  output.append(first_node.format(href_link="?{}"
127  .format(get_params(page=0,
128  search=search,
129  showPaused=showPaused)),
130  disabled=is_disabled))
131 
132  page_link = void_link
133  if current_page > 0:
134  page_link = '?{}'.format(get_params(page=(current_page - 1),
135  search=search,
136  showPaused=showPaused))
137 
138  output.append(previous_node.format(href_link=page_link,
139  disabled=is_disabled))
140 
141  mid = int(window / 2)
142  last_page = num_of_pages - 1
143 
144  if current_page <= mid or num_of_pages < window:
145  pages = [i for i in range(0, min(num_of_pages, window))]
146  elif mid < current_page < last_page - mid:
147  pages = [i for i in range(current_page - mid, current_page + mid + 1)]
148  else:
149  pages = [i for i in range(num_of_pages - window, last_page + 1)]
150 
151  def is_current(current, page):
152  return page == current
153 
154  for page in pages:
155  vals = {
156  'is_active': 'active' if is_current(current_page, page) else '',
157  'href_link': void_link if is_current(current_page, page)
158  else '?{}'.format(get_params(page=page,
159  search=search,
160  showPaused=showPaused)),
161  'page_num': page + 1
162  }
163  output.append(page_node.format(**vals))
164 
165  is_disabled = 'disabled' if current_page >= num_of_pages - 1 else ''
166 
167  page_link = (void_link if current_page >= num_of_pages - 1
168  else '?{}'.format(get_params(page=current_page + 1,
169  search=search,
170  showPaused=showPaused)))
171 
172  output.append(next_node.format(href_link=page_link, disabled=is_disabled))
173  output.append(last_node.format(href_link="?{}"
174  .format(get_params(page=last_page,
175  search=search,
176  showPaused=showPaused)),
177  disabled=is_disabled))
178 
179  output.append('</ul>')
180 
181  return wtforms.widgets.core.HTMLString('\n'.join(output))

297  out = ""
