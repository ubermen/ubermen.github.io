83  """
84  Template method that contains custom logic of reading
85  logs given the try_number.
86  :param ti: task instance record
87  :param try_number: current try_number to read log from
88  :param metadata: log metadata,
89  can be used for steaming log reading and auto-tailing.
90  :return: log message as a string and metadata.
91  """
92  # Task instance here might be different from task instance when
93  # initializing the handler. Thus explicitly getting log location
94  # is needed to get correct log path.
95  log_relative_path = self._render_filename(ti, try_number)
96  location = os.path.join(self.local_base, log_relative_path)
97 
98  log = ""
99 
100  if os.path.exists(location):
101  try:
102  with open(location) as f:
103  log += "*** Reading local file: {}\n".format(location)
104  log += "".join(f.readlines())
105  except Exception as e:
106  log = "*** Failed to load local log file: {}\n".format(location)
107  log += "*** {}\n".format(str(e))
108  else:
109  url = os.path.join(
110  "http://{ti.hostname}:{worker_log_server_port}/log", log_relative_path
111  ).format(
112  ti=ti,
113  worker_log_server_port=conf.get('celery', 'WORKER_LOG_SERVER_PORT')
114  )
115  log += "*** Log file does not exist: {}\n".format(location)
116  log += "*** Fetching from: {}\n".format(url)
117  try:

121  except (AirflowConfigException, ValueError):

135  def read(self, task_instance, try_number=None, metadata=None):
136  """
137  Read logs of given task instance from local machine.
138  :param task_instance: task instance object
139  :param try_number: task instance try_number to read logs from. If None
140  it returns all logs separated by try_number
141  :param metadata: log metadata,
142  can be used for steaming log reading and auto-tailing.
143  :return: a list of logs
144  """
145  # Task instance increments its try number when it starts to run.
146  # So the log for a particular task try will only show up when
147  # try number gets incremented in DB, i.e logs produced the time
148  # after cli run and before try_number + 1 in DB will not be displayed.
149 
150  if try_number is None:
151  next_try = task_instance.next_try_number
152  try_numbers = list(range(1, next_try))
153  elif try_number < 1:
154  logs = [
155  'Error fetching the logs. Try number {} is invalid.'.format(try_number),
156  ]
157  return logs
158  else:
159  try_numbers = [try_number]
160 
161  logs = [''] * len(try_numbers)
162  metadatas = [{}] * len(try_numbers)
163  for i, try_number in enumerate(try_numbers):
164  log, metadata = self._read(task_instance, try_number, metadata)
165  logs[i] += log
166  metadatas[i] = metadata
167 
168  return logs, metadatas

170  def _init_file(self, ti):
171  """
172  Create log directory and give it correct permissions.
173  :param ti: task instance object
174  :return relative log path of the given task instance
175  """
176  # To handle log writing when tasks are impersonated, the log files need to
177  # be writable by the user that runs the Airflow command and the user
178  # that is impersonated. This is mainly to handle corner cases with the
179  # SubDagOperator. When the SubDagOperator is run, all of the operators
180  # run under the impersonated user and create appropriate log files
181  # as the impersonated user. However, if the user manually runs tasks
182  # of the SubDagOperator through the UI, then the log files are created
183  # by the user that runs the Airflow command. For example, the Airflow
184  # run command may be run by the `airflow_sudoable` user, but the Airflow
185  # tasks may be run by the `airflow` user. If the log files are not
186  # writable by both users, then it's possible that re-running a task
187  # via the UI (or vice versa) results in a permission error as the task
188  # tries to write to a log file created by the other user.
189  relative_path = self._render_filename(ti, ti.try_number)
190  full_path = os.path.join(self.local_base, relative_path)
191  directory = os.path.dirname(full_path)
192  # Create the log file and give it group writable permissions
193  # TODO(aoen): Make log dirs and logs globally readable for now since the SubDag
194  # operator is not compatible with impersonation (e.g. if a Celery executor is used
195  # for a SubDag operator and the SubDag operator has a different owner than the
196  # parent DAG)
197  if not os.path.exists(directory):
198  # Create the directory as globally writable using custom mkdirs
199  # as os.makedirs doesn't set mode properly.
200  mkdirs(directory, 0o777)
201 
202  if not os.path.exists(full_path):
203  open(full_path, "a").close()
204  # TODO: Investigate using 444 instead of 666.
205  os.chmod(full_path, 0o666)
206 
207  return full_path
