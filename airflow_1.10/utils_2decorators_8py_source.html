37 def apply_defaults(func):
38  """
39  Function decorator that Looks for an argument named "default_args", and
40  fills the unspecified arguments from it.
41 
42  Since python2.* isn't clear about which arguments are missing when
43  calling a function, and that this can be quite confusing with multi-level
44  inheritance and argument defaults, this decorator also alerts with
45  specific information about the missing arguments.
46  """
47 
48  import airflow.models
49  # Cache inspect.signature for the wrapper closure to avoid calling it
50  # at every decorated invocation. This is separate sig_cache created
51  # per decoration, i.e. each function decorated using apply_defaults will
52  # have a different sig_cache.
53  sig_cache = signature(func)
54  non_optional_args = {
55  name for (name, param) in sig_cache.parameters.items()
56  if param.default == param.empty and
57  param.name != 'self' and
58  param.kind not in (param.VAR_POSITIONAL, param.VAR_KEYWORD)}
59 
60  @wraps(func)
61  def wrapper(*args, **kwargs):
62  if len(args) > 1:
63  raise AirflowException(
64  "Use keyword arguments when initializing operators")
65  dag_args = {}
66  dag_params = {}
67 
68  dag = kwargs.get('dag', None) or airflow.models._CONTEXT_MANAGER_DAG
69  if dag:
70  dag_args = copy(dag.default_args) or {}
71  dag_params = copy(dag.params) or {}
72 
73  params = {}
74  if 'params' in kwargs:
75  params = kwargs['params']
76  dag_params.update(params)
77 
78  default_args = {}
79  if 'default_args' in kwargs:
80  default_args = kwargs['default_args']
81  if 'params' in default_args:
82  dag_params.update(default_args['params'])
83  del default_args['params']
84 
85  dag_args.update(default_args)
86  default_args = dag_args
87 
88  for arg in sig_cache.parameters:
89  if arg not in kwargs and arg in default_args:
90  kwargs[arg] = default_args[arg]
91  missing_args = list(non_optional_args - set(kwargs))
92  if missing_args:
93  msg = "Argument {0} is required".format(missing_args)
94  raise AirflowException(msg)
95 
96  kwargs['params'] = dag_params
97 
98  result = func(*args, **kwargs)
99  return result
100  return wrapper
