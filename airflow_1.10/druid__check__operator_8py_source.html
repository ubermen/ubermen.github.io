26 class DruidCheckOperator(CheckOperator):
27  """
28  Performs checks against Druid. The ``DruidCheckOperator`` expects
29  a sql query that will return a single row. Each value on that
30  first row is evaluated using python ``bool`` casting. If any of the
31  values return ``False`` the check is failed and errors out.
32 
33  Note that Python bool casting evals the following as ``False``:
34 
35  * ``False``
36  * ``0``
37  * Empty string (``""``)
38  * Empty list (``[]``)
39  * Empty dictionary or set (``{}``)
40 
41  Given a query like ``SELECT COUNT(*) FROM foo``, it will fail only if
42  the count ``== 0``. You can craft much more complex query that could,
43  for instance, check that the table has the same number of rows as
44  the source table upstream, or that the count of today's partition is
45  greater than yesterday's partition, or that a set of metrics are less
46  than 3 standard deviation for the 7 day average.
47  This operator can be used as a data quality check in your pipeline, and
48  depending on where you put it in your DAG, you have the choice to
49  stop the critical path, preventing from
50  publishing dubious data, or on the side and receive email alterts
51  without stopping the progress of the DAG.
52 
53  :param sql: the sql to be executed
54  :type sql: string
55  :param druid_broker_conn_id: reference to the druid broker
56  :type druid_broker_conn_id: string
57  """
58 
59  @apply_defaults
60  def __init__(
61  self, sql,
62  druid_broker_conn_id='druid_broker_default',
63  *args, **kwargs):
64  super(DruidCheckOperator, self).__init__(sql=sql, *args, **kwargs)
65  self.druid_broker_conn_id = druid_broker_conn_id
66  self.sql = sql
67 
68  def get_db_hook(self):
69  """
70  Return the druid db api hook.
71  """
72  return DruidDbApiHook(druid_broker_conn_id=self.druid_broker_conn_id)
73 
74  def get_first(self, sql):
75  """
76  Executes the druid sql to druid broker and returns the first resulting row.
77 
78  :param sql: the sql statement to be executed (str)
79  :type sql: str
80  """
81  with self.get_db_hook().get_conn() as cur:
82  cur.execute(sql)
83  return cur.fetchone()
84 
85  def execute(self, context=None):
86  self.log.info('Executing SQL check: {}'.format(self.sql))
87  record = self.get_first(self.sql)
88  self.log.info("Record: {}".format(str(record)))
89  if not record:
90  raise AirflowException("The query returned None")
91  self.log.info("Success.")
