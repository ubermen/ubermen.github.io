66  def execute(self, context):
67  started_at = timezone.utcnow()
68  while not self.poke(context):
69  if (timezone.utcnow() - started_at).total_seconds() > self.timeout:
70  # If sensor is in soft fail mode but will be retried then
71  # give it a chance and fail with timeout.
72  # This gives the ability to set up non-blocking AND soft-fail sensors.
73  if self.soft_fail and not context['ti'].is_eligible_to_retry():
74  self._do_skip_downstream_tasks(context)
75  raise AirflowSkipException('Snap. Time is OUT.')
76  else:
77  raise AirflowSensorTimeout('Snap. Time is OUT.')
78  sleep(self.poke_interval)
79  self.log.info("Success criteria met. Exiting.")
