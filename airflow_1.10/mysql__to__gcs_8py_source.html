132  def _write_local_data_files(self, cursor):
133  """
134  Takes a cursor, and writes results to a local file.
135 
136  :return: A dictionary where keys are filenames to be used as object
137  names in GCS, and values are file handles to local files that
138  contain the data for the GCS objects.
139  """
140  schema = list(map(lambda schema_tuple: schema_tuple[0], cursor.description))
141  col_type_dict = self._get_col_type_dict()
142  file_no = 0
143  tmp_file_handle = NamedTemporaryFile(delete=True)
144  tmp_file_handles = {self.filename.format(file_no): tmp_file_handle}
145 
146  for row in cursor:
147  # Convert datetime objects to utc seconds, and decimals to floats.
148  # Convert binary type object to string encoded with base64.
149  row = self._convert_types(schema, col_type_dict, row)
150  row_dict = dict(zip(schema, row))
151 
152  # TODO validate that row isn't > 2MB. BQ enforces a hard row size of 2MB.
153  s = json.dumps(row_dict)
154  if PY3:
155  s = s.encode('utf-8')
156  tmp_file_handle.write(s)
157 
158  # Append newline to make dumps BigQuery compatible.
159  tmp_file_handle.write(b'\n')
160 
161  # Stop if the file exceeds the file size limit.
162  if tmp_file_handle.tell() >= self.approx_max_file_size_bytes:
163  file_no += 1
164  tmp_file_handle = NamedTemporaryFile(delete=True)
165  tmp_file_handles[self.filename.format(file_no)] = tmp_file_handle
166 
167  return tmp_file_handles

169  def _write_local_schema_file(self, cursor):
170  """
171  Takes a cursor, and writes the BigQuery schema for the results to a
172  local file system.
173 
174  :return: A dictionary where key is a filename to be used as an object
175  name in GCS, and values are file handles to local files that
176  contains the BigQuery schema fields in .json format.
177  """
178  schema_str = None
179  tmp_schema_file_handle = NamedTemporaryFile(delete=True)
180  if self.schema is not None and isinstance(self.schema, string_types):
181  schema_str = self.schema
182  elif self.schema is not None and isinstance(self.schema, list):
183  schema_str = json.dumps(self.schema)
184  else:
185  schema = []
186  for field in cursor.description:
187  # See PEP 249 for details about the description tuple.
188  field_name = field[0]
189  field_type = self.type_map(field[1])
190  # Always allow TIMESTAMP to be nullable. MySQLdb returns None types
191  # for required fields because some MySQL timestamps can't be
192  # represented by Python's datetime (e.g. 0000-00-00 00:00:00).
193  if field[6] or field_type == 'TIMESTAMP':
194  field_mode = 'NULLABLE'
195  else:
196  field_mode = 'REQUIRED'
197  schema.append({
198  'name': field_name,
199  'type': field_type,
200  'mode': field_mode,
201  })
202  schema_str = json.dumps(schema)
203  if PY3:
204  schema_str = schema_str.encode('utf-8')
205  tmp_schema_file_handle.write(schema_str)
206 
207  self.log.info('Using schema for %s: %s', self.schema_filename, schema_str)
208  return {self.schema_filename: tmp_schema_file_handle}

210  def _upload_to_gcs(self, files_to_upload):
211  """
212  Upload all of the file splits (and optionally the schema .json file) to
213  Google cloud storage.
214  """
215  hook = GoogleCloudStorageHook(
216  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
217  delegate_to=self.delegate_to)
218  for object, tmp_file_handle in files_to_upload.items():
219  hook.upload(self.bucket, object, tmp_file_handle.name, 'application/json')

221  def _convert_types(self, schema, col_type_dict, row):
222  """
223  Takes a value from MySQLdb, and converts it to a value that's safe for
224  JSON/Google cloud storage/BigQuery. Dates are converted to UTC seconds.
225  Decimals are converted to floats. Binary type fields are encoded with base64,
226  as imported BYTES data must be base64-encoded according to Bigquery SQL
227  date type documentation: https://cloud.google.com/bigquery/data-types
228  """
229  converted_row = []
230  for col_name, col_val in zip(schema, row):
231  if type(col_val) in (datetime, date):
232  col_val = time.mktime(col_val.timetuple())
233  elif isinstance(col_val, Decimal):
234  col_val = float(col_val)
235  elif col_type_dict.get(col_name) == "BYTES":
236  col_val = base64.standard_b64encode(col_val)
237  if PY3:
238  col_val = col_val.decode('ascii')
239  else:
240  col_val = col_val
241  converted_row.append(col_val)
242  return converted_row

244  def _get_col_type_dict(self):
245  """
246  Return a dict of column name and column type based on self.schema if not None.
247  """
248  schema = []
249  if isinstance(self.schema, string_types):
250  schema = json.loads(self.schema)
251  elif isinstance(self.schema, list):
252  schema = self.schema
253  elif self.schema is not None:
254  self.log.warn('Using default schema due to unexpected type.'
255  'Should be a string or list.')
256 
257  col_type_dict = {}
258  try:
259  col_type_dict = {col['name']: col['type'] for col in schema}
260  except KeyError:
261  self.log.warn('Using default schema due to missing name or type. Please '
262  'refer to: https://cloud.google.com/bigquery/docs/schemas'
263  '#specifying_a_json_schema_file')
264  return col_type_dict
