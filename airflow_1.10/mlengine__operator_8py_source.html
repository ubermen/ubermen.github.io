29 def _normalize_mlengine_job_id(job_id):
30  """
31  Replaces invalid MLEngine job_id characters with '_'.
32 
33  This also adds a leading 'z' in case job_id starts with an invalid
34  character.
35 
36  Args:
37  job_id: A job_id str that may have invalid characters.
38 
39  Returns:
40  A valid job_id representation.
41  """
42 
43  # Add a prefix when a job_id starts with a digit or a template
44  match = re.search(r'\d|\{{2}', job_id)
45  if match and match.start() is 0:
46  job = 'z_{}'.format(job_id)
47  else:
48  job = job_id
49 
50  # Clean up 'bad' characters except templates
51  tracker = 0
52  cleansed_job_id = ''
53  for m in re.finditer(r'\{{2}.+?\}{2}', job):
54  cleansed_job_id += re.sub(r'[^0-9a-zA-Z]+', '_',
55  job[tracker:m.start()])
56  cleansed_job_id += job[m.start():m.end()]
57  tracker = m.end()
58 
59  # Clean up last substring or the full string if no templates
60  cleansed_job_id += re.sub(r'[^0-9a-zA-Z]+', '_', job[tracker:])
61 
62  return cleansed_job_id

65 class MLEngineBatchPredictionOperator(BaseOperator):
66  """
67  Start a Google Cloud ML Engine prediction job.
68 
69  NOTE: For model origin, users should consider exactly one from the
70  three options below:
71  1. Populate 'uri' field only, which should be a GCS location that
72  points to a tensorflow savedModel directory.
73  2. Populate 'model_name' field only, which refers to an existing
74  model, and the default version of the model will be used.
75  3. Populate both 'model_name' and 'version_name' fields, which
76  refers to a specific version of a specific model.
77 
78  In options 2 and 3, both model and version name should contain the
79  minimal identifier. For instance, call
80 
81  ::
82 
83  MLEngineBatchPredictionOperator(
84  ...,
85  model_name='my_model',
86  version_name='my_version',
87  ...)
88 
89  if the desired model version is
90  "projects/my_project/models/my_model/versions/my_version".
91 
92  See https://cloud.google.com/ml-engine/reference/rest/v1/projects.jobs
93  for further documentation on the parameters.
94 
95  :param project_id: The Google Cloud project name where the
96  prediction job is submitted. (templated)
97  :type project_id: string
98 
99  :param job_id: A unique id for the prediction job on Google Cloud
100  ML Engine. (templated)
101  :type job_id: string
102 
103  :param data_format: The format of the input data.
104  It will default to 'DATA_FORMAT_UNSPECIFIED' if is not provided
105  or is not one of ["TEXT", "TF_RECORD", "TF_RECORD_GZIP"].
106  :type data_format: string
107 
108  :param input_paths: A list of GCS paths of input data for batch
109  prediction. Accepting wildcard operator *, but only at the end. (templated)
110  :type input_paths: list of string
111 
112  :param output_path: The GCS path where the prediction results are
113  written to. (templated)
114  :type output_path: string
115 
116  :param region: The Google Compute Engine region to run the
117  prediction job in. (templated)
118  :type region: string
119 
120  :param model_name: The Google Cloud ML Engine model to use for prediction.
121  If version_name is not provided, the default version of this
122  model will be used.
123  Should not be None if version_name is provided.
124  Should be None if uri is provided. (templated)
125  :type model_name: string
126 
127  :param version_name: The Google Cloud ML Engine model version to use for
128  prediction.
129  Should be None if uri is provided. (templated)
130  :type version_name: string

133  Should be None if model_name is provided.
134  It should be a GCS path pointing to a tensorflow SavedModel. (templated)
135  :type uri: string

138  for parallel processing. Defaults to 10 if not specified.
139  :type max_worker_count: int

146  Cloud Platform.

249  prediction_request['predictionInput'][
250  'maxWorkerCount'] = self._max_worker_count
251 
252  if self._runtime_version:
253  prediction_request['predictionInput'][
254  'runtimeVersion'] = self._runtime_version
255 
256  hook = MLEngineHook(self._gcp_conn_id, self._delegate_to)
257 
258  # Helper method to check if the existing job's prediction input is the
259  # same as the request we get here.
260  def check_existing_job(existing_job):
261  return existing_job.get('predictionInput', None) == \
262  prediction_request['predictionInput']
263 
264  try:
265  finished_prediction_job = hook.create_job(
266  self._project_id, prediction_request, check_existing_job)
267  except errors.HttpError:
268  raise
269 
270  if finished_prediction_job['state'] != 'SUCCEEDED':
271  self.log.error('MLEngine batch prediction job failed: {}'.format(
272  str(finished_prediction_job)))
273  raise RuntimeError(finished_prediction_job['errorMessage'])
274 
275  return finished_prediction_job['predictionOutput']

339 class MLEngineVersionOperator(BaseOperator):
340  """
341  Operator for managing a Google Cloud ML Engine version.
342 
343  :param project_id: The Google Cloud project name to which MLEngine
344  model belongs.
345  :type project_id: string
346 
347  :param model_name: The name of the Google Cloud ML Engine model that the version
348  belongs to. (templated)
349  :type model_name: string
350 
351  :param version_name: A name to use for the version being operated upon.
352  If not None and the `version` argument is None or does not have a value for
353  the `name` key, then this will be populated in the payload for the
354  `name` key. (templated)
355  :type version_name: string
356 
357  :param version: A dictionary containing the information about the version.
358  If the `operation` is `create`, `version` should contain all the
359  information about this version such as name, and deploymentUrl.
360  If the `operation` is `get` or `delete`, the `version` parameter
361  should contain the `name` of the version.
362  If it is None, the only `operation` possible would be `list`. (templated)
363  :type version: dict
364 
365  :param operation: The operation to perform. Available operations are:
366 
367  * ``create``: Creates a new version in the model specified by `model_name`,
368  in which case the `version` parameter should contain all the
369  information to create that version
370  (e.g. `name`, `deploymentUrl`).
371 
372  * ``get``: Gets full information of a particular version in the model
373  specified by `model_name`.
374  The name of the version should be specified in the `version`
375  parameter.
376 
377  * ``list``: Lists all available versions of the model specified
378  by `model_name`.
379 
380  * ``delete``: Deletes the version specified in `version` parameter from the
381  model specified by `model_name`).
382  The name of the version should be specified in the `version`
383  parameter.
384  :type operation: string
385 
386  :param gcp_conn_id: The connection ID to use when fetching connection info.
387  :type gcp_conn_id: string
388 
389  :param delegate_to: The account to impersonate, if any.
390  For this to work, the service account making the request must have
391  domain-wide delegation enabled.
392  :type delegate_to: string
393  """
