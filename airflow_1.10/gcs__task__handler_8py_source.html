27 class GCSTaskHandler(FileTaskHandler, LoggingMixin):
28  """
29  GCSTaskHandler is a python log handler that handles and reads
30  task instance logs. It extends airflow FileTaskHandler and

32  failure, it reads from host machine's local disk.
33  """

70  def close(self):
71  """
72  Close and upload local log file to remote storage S3.
73  """
74  # When application exit, system shuts down all handlers by
75  # calling close method. Here we check if logger is already
76  # closed to prevent uploading the log to remote storage multiple
77  # times when `logging.shutdown` is called.
78  if self.closed:
79  return
80 
81  super(GCSTaskHandler, self).close()
82 
83  if not self.upload_on_close:
84  return
85 
86  local_loc = os.path.join(self.local_base, self.log_relative_path)
87  remote_loc = os.path.join(self.remote_base, self.log_relative_path)
88  if os.path.exists(local_loc):
89  # read log and remove old logs to get just the latest additions
90  with open(local_loc, 'r') as logfile:
91  log = logfile.read()

93

95  self.closed = True

97  def _read(self, ti, try_number, metadata=None):
98  """
99  Read logs of given task instance and try_number from GCS.
100  If failed, read the log from task instance host machine.
101  :param ti: task instance object
102  :param try_number: task instance try_number to read logs from
103  :param metadata: log metadata,
104  can be used for steaming log reading and auto-tailing.
105  """
106  # Explicitly getting log relative path is necessary as the given
107  # task instance might be different than task instance passed in
108  # in set_context method.
109  log_relative_path = self._render_filename(ti, try_number)
110  remote_loc = os.path.join(self.remote_base, log_relative_path)
111 
112  try:
113  remote_log = self.gcs_read(remote_loc)
114  log = '*** Reading remote log from {}.\n{}\n'.format(
115  remote_loc, remote_log)
116  return log, {'end_of_log': True}
117  except Exception as e:
118  log = '*** Unable to read remote log from {}\n*** {}\n\n'.format(
119  remote_loc, str(e))
120  self.log.error(log)
121  local_log, metadata = super(GCSTaskHandler, self)._read(ti, try_number)
122  log += local_log
123  return log, metadata

134  def gcs_write(self, log, remote_log_location, append=True):
135  """
136  Writes the log to the remote_log_location. Fails silently if no hook
137  was created.
138  :param log: the log to write to the remote_log_location
139  :type log: string
140  :param remote_log_location: the log's location in remote storage
141  :type remote_log_location: string (path)
142  :param append: if False, any existing log file is overwritten. If True,
143  the new log is appended to any existing logs.
144  :type append: bool
145  """
146  if append:
147  try:
148  old_log = self.gcs_read(remote_log_location)
149  log = '\n'.join([old_log, log]) if old_log else log
150  except Exception as e:
151  if not hasattr(e, 'resp') or e.resp.get('status') != '404':
152  log = '*** Previous log discarded: {}\n\n'.format(str(e)) + log
153 
154  try:
155  bkt, blob = self.parse_gcs_url(remote_log_location)
156  from tempfile import NamedTemporaryFile
157  with NamedTemporaryFile(mode='w+') as tmpfile:
158  tmpfile.write(log)
159  # Force the file to be flushed, since we're doing the
160  # upload from within the file context (it hasn't been
161  # closed).
162  tmpfile.flush()
163  self.hook.upload(bkt, blob, tmpfile.name)
164  except Exception as e:
165  self.log.error('Could not write logs to %s: %s', remote_log_location, e)

167  def parse_gcs_url(self, gsurl):
168  """
169  Given a Google Cloud Storage URL (gs://<bucket>/<blob>), returns a
170  tuple containing the corresponding bucket and blob.
171  """
172  # Python 3
173  try:
174  from urllib.parse import urlparse
175  # Python 2
176  except ImportError:
177  from urlparse import urlparse
178 
179  parsed_url = urlparse(gsurl)
180  if not parsed_url.netloc:
181  raise AirflowException('Please provide a bucket name')
182  else:
183  bucket = parsed_url.netloc
184  blob = parsed_url.path.strip('/')
185  return bucket, blob
