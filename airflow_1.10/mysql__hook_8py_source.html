59  def get_conn(self):
60  """
61  Returns a mysql connection object
62  """
63  conn = self.get_connection(self.mysql_conn_id)
64  conn_config = {
65  "user": conn.login,
66  "passwd": conn.password or '',
67  "host": conn.host or 'localhost',
68  "db": self.schema or conn.schema or ''
69  }
70 
71  if not conn.port:
72  conn_config["port"] = 3306
73  else:
74  conn_config["port"] = int(conn.port)
75 
76  if conn.extra_dejson.get('charset', False):
77  conn_config["charset"] = conn.extra_dejson["charset"]
78  if (conn_config["charset"]).lower() == 'utf8' or\
79  (conn_config["charset"]).lower() == 'utf-8':
80  conn_config["use_unicode"] = True
81  if conn.extra_dejson.get('cursor', False):
82  if (conn.extra_dejson["cursor"]).lower() == 'sscursor':
83  conn_config["cursorclass"] = MySQLdb.cursors.SSCursor
84  elif (conn.extra_dejson["cursor"]).lower() == 'dictcursor':
85  conn_config["cursorclass"] = MySQLdb.cursors.DictCursor
86  elif (conn.extra_dejson["cursor"]).lower() == 'ssdictcursor':
87  conn_config["cursorclass"] = MySQLdb.cursors.SSDictCursor
88  local_infile = conn.extra_dejson.get('local_infile', False)
89  if conn.extra_dejson.get('ssl', False):
90  conn_config['ssl'] = conn.extra_dejson['ssl']
91  if local_infile:
92  conn_config["local_infile"] = 1
93  conn = MySQLdb.connect(**conn_config)
