29 class WasbTaskHandler(FileTaskHandler, LoggingMixin):

31  WasbTaskHandler is a python log handler that handles and reads

65  super(WasbTaskHandler, self).set_context(ti)
66  # Local location and remote location is needed to open and

71  def close(self):
72  """
73  Close and upload local log file to remote storage Wasb.
74  """
75  # When application exit, system shuts down all handlers by
76  # calling close method. Here we check if logger is already
77  # closed to prevent uploading the log to remote storage multiple
78  # times when `logging.shutdown` is called.
79  if self.closed:
80  return
81 
82  super(WasbTaskHandler, self).close()
83 
84  if not self.upload_on_close:
85  return
86 
87  local_loc = os.path.join(self.local_base, self.log_relative_path)
88  remote_loc = os.path.join(self.remote_base, self.log_relative_path)
89  if os.path.exists(local_loc):
90  # read log and remove old logs to get just the latest additions
91  with open(local_loc, 'r') as logfile:
92  log = logfile.read()
93  self.wasb_write(log, remote_loc, append=True)
94 
95  if self.delete_local_copy:
96  shutil.rmtree(os.path.dirname(local_loc))
97  # Mark closed so we don't double write if close is called twice
98  self.closed = True

100  def _read(self, ti, try_number, metadata=None):
101  """
102  Read logs of given task instance and try_number from Wasb remote storage.
103  If failed, read the log from task instance host machine.
104  :param ti: task instance object
105  :param try_number: task instance try_number to read logs from
106  :param metadata: log metadata,
107  can be used for steaming log reading and auto-tailing.
108  """
109  # Explicitly getting log relative path is necessary as the given
110  # task instance might be different than task instance passed in
111  # in set_context method.
112  log_relative_path = self._render_filename(ti, try_number)
113  remote_loc = os.path.join(self.remote_base, log_relative_path)
114 
115  if self.wasb_log_exists(remote_loc):
116  # If Wasb remote file exists, we do not fetch logs from task instance
117  # local machine even if there are errors reading remote logs, as
118  # returned remote_log will contain error messages.
119  remote_log = self.wasb_read(remote_loc, return_error=True)
120  log = '*** Reading remote log from {}.\n{}\n'.format(
121  remote_loc, remote_log)
122  return log, {'end_of_log': True}
123  else:
124  return super(WasbTaskHandler, self)._read(ti, try_number)

133  return self.hook.check_for_blob(self.wasb_container, remote_log_location)

138  def wasb_read(self, remote_log_location, return_error=False):
139  """
140  Returns the log found at the remote_log_location. Returns '' if no
141  logs are found or there is an error.
142  :param remote_log_location: the log's location in remote storage
143  :type remote_log_location: string (path)
144  :param return_error: if True, returns a string error message if an
145  error occurs. Otherwise returns '' when an error occurs.
146  :type return_error: bool
147  """
148  try:
149  return self.hook.read_file(self.wasb_container, remote_log_location)
150  except AzureHttpError:
151  msg = 'Could not read logs from {}'.format(remote_log_location)
152  self.log.exception(msg)
153  # return error if needed
154  if return_error:
155  return msg

157  def wasb_write(self, log, remote_log_location, append=True):
158  """
159  Writes the log to the remote_log_location. Fails silently if no hook
160  was created.
161  :param log: the log to write to the remote_log_location
162  :type log: string
163  :param remote_log_location: the log's location in remote storage
164  :type remote_log_location: string (path)
165  :param append: if False, any existing log file is overwritten. If True,
166  the new log is appended to any existing logs.
167  :type append: bool
168  """
169  if append and self.wasb_log_exists(remote_log_location):
170  old_log = self.wasb_read(remote_log_location)
171  log = '\n'.join([old_log, log]) if old_log else log
172 
173  try:
174  self.hook.load_string(
175  log,
176  self.wasb_container,
177  remote_log_location,
178  )
179  except AzureHttpError:
180  self.log.exception('Could not write logs to %s',
181  remote_log_location)
