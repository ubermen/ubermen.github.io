129  def execute(self, context):
130  cursor = self._query_cassandra()
131  files_to_upload = self._write_local_data_files(cursor)
132 
133  # If a schema is set, create a BQ schema JSON file.
134  if self.schema_filename:
135  files_to_upload.update(self._write_local_schema_file(cursor))
136 
137  # Flush all files before uploading
138  for file_handle in files_to_upload.values():
139  file_handle.flush()
140 
141  self._upload_to_gcs(files_to_upload)
142 
143  # Close all temp file handles.
144  for file_handle in files_to_upload.values():
145  file_handle.close()
146 
147  # Close all sessions and connection associated with this Cassandra cluster
148  self.hook.shutdown_cluster()

150  def _query_cassandra(self):
151  """
152  Queries cassandra and returns a cursor to the results.
153  """
154  self.hook = CassandraHook(cassandra_conn_id=self.cassandra_conn_id)
155  session = self.hook.get_conn()
156  cursor = session.execute(self.cql)
157  return cursor

159  def _write_local_data_files(self, cursor):
160  """
161  Takes a cursor, and writes results to a local file.
162 
163  :return: A dictionary where keys are filenames to be used as object
164  names in GCS, and values are file handles to local files that
165  contain the data for the GCS objects.
166  """
167  file_no = 0
168  tmp_file_handle = NamedTemporaryFile(delete=True)
169  tmp_file_handles = {self.filename.format(file_no): tmp_file_handle}
170  for row in cursor:
171  row_dict = self.generate_data_dict(row._fields, row)
172  s = json.dumps(row_dict)
173  if PY3:
174  s = s.encode('utf-8')
175  tmp_file_handle.write(s)
176 
177  # Append newline to make dumps BigQuery compatible.
178  tmp_file_handle.write(b'\n')
179 
180  if tmp_file_handle.tell() >= self.approx_max_file_size_bytes:
181  file_no += 1
182  tmp_file_handle = NamedTemporaryFile(delete=True)
183  tmp_file_handles[self.filename.format(file_no)] = tmp_file_handle
184 
185  return tmp_file_handles

187  def _write_local_schema_file(self, cursor):
188  """
189  Takes a cursor, and writes the BigQuery schema for the results to a
190  local file system.
191 
192  :return: A dictionary where key is a filename to be used as an object
193  name in GCS, and values are file handles to local files that
194  contains the BigQuery schema fields in .json format.
195  """
196  schema = []
197  tmp_schema_file_handle = NamedTemporaryFile(delete=True)
198 
199  for name, type in zip(cursor.column_names, cursor.column_types):
200  schema.append(self.generate_schema_dict(name, type))
201  json_serialized_schema = json.dumps(schema)
202  if PY3:
203  json_serialized_schema = json_serialized_schema.encode('utf-8')
204 
205  tmp_schema_file_handle.write(json_serialized_schema)
206  return {self.schema_filename: tmp_schema_file_handle}

208  def _upload_to_gcs(self, files_to_upload):
209  hook = GoogleCloudStorageHook(
210  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
211  delegate_to=self.delegate_to)
212  for object, tmp_file_handle in files_to_upload.items():
213  hook.upload(self.bucket, object, tmp_file_handle.name, 'application/json')

222  @classmethod
223  def convert_value(cls, name, value):
224  if not value:
225  return value
226  elif isinstance(value, (text_type, int, float, bool, dict)):
227  return value
228  elif isinstance(value, binary_type):
229  return b64encode(value).decode('ascii')
230  elif isinstance(value, UUID):
231  return b64encode(value.bytes).decode('ascii')
232  elif isinstance(value, (datetime, Date)):
233  return str(value)
234  elif isinstance(value, Decimal):
235  return float(value)
236  elif isinstance(value, Time):
237  return str(value).split('.')[0]
238  elif isinstance(value, (list, SortedSet)):
239  return cls.convert_array_types(name, value)
240  elif hasattr(value, '_fields'):
241  return cls.convert_user_type(name, value)
242  elif isinstance(value, tuple):
243  return cls.convert_tuple_type(name, value)
244  elif isinstance(value, OrderedMapSerializedKey):
245  return cls.convert_map_type(name, value)
246  else:
247  raise AirflowException('unexpected value: ' + str(value))

250  def convert_array_types(cls, name, value):

253  @classmethod
254  def convert_user_type(cls, name, value):
255  """
256  Converts a user type to RECORD that contains n fields, where n is the
257  number of attributes. Each element in the user type class will be converted to its
258  corresponding data type in BQ.
259  """
260  names = value._fields
261  values = [cls.convert_value(name, getattr(value, name)) for name in names]
262  return cls.generate_data_dict(names, values)
263 
264  @classmethod
265  def convert_tuple_type(cls, name, value):
266  """
267  Converts a tuple to RECORD that contains n fields, each will be converted
268  to its corresponding data type in bq and will be named 'field_<index>', where
269  index is determined by the order of the tuple elments defined in cassandra.
270  """
271  names = ['field_' + str(i) for i in range(len(value))]
272  values = [cls.convert_value(name, value) for name, value in zip(names, value)]
273  return cls.generate_data_dict(names, values)

275  @classmethod
276  def convert_map_type(cls, name, value):
277  """
278  Converts a map to a repeated RECORD that contains two fields: 'key' and 'value',
279  each will be converted to its corresopnding data type in BQ.
280  """
281  converted_map = []
282  for k, v in zip(value.keys(), value.values()):
283  converted_map.append({
284  'key': cls.convert_value('key', k),
285  'value': cls.convert_value('value', v)
286  })
287  return converted_map

300  @classmethod
301  def get_bq_fields(cls, name, type):
302  fields = []
303 
304  if not cls.is_simple_type(type):
305  names, types = [], []
306 
307  if cls.is_array_type(type) and cls.is_record_type(type.subtypes[0]):
308  names = type.subtypes[0].fieldnames
309  types = type.subtypes[0].subtypes
310  elif cls.is_record_type(type):
311  names = type.fieldnames
312  types = type.subtypes
313 
314  if types and not names and type.cassname == 'TupleType':
315  names = ['field_' + str(i) for i in range(len(types))]
316  elif types and not names and type.cassname == 'MapType':
317  names = ['key', 'value']
318 
319  for name, type in zip(names, types):
320  field = cls.generate_schema_dict(name, type)
321  fields.append(field)
322 
323  return fields
