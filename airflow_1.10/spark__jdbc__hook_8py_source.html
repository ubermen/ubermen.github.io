175  def _resolve_jdbc_connection(self):
176  conn_data = {'url': '',
177  'schema': '',
178  'conn_prefix': '',
179  'user': '',
180  'password': ''
181  }
182  try:
183  conn = self.get_connection(self._jdbc_conn_id)
184  if conn.port:
185  conn_data['url'] = "{}:{}".format(conn.host, conn.port)
186  else:
187  conn_data['url'] = conn.host
188  conn_data['schema'] = conn.schema
189  conn_data['user'] = conn.login
190  conn_data['password'] = conn.password
191  extra = conn.extra_dejson
192  conn_data['conn_prefix'] = extra.get('conn_prefix', '')
193  except AirflowException:
194  self.log.debug(
195  "Could not load jdbc connection string %s, defaulting to %s",
196  self._jdbc_conn_id, ""
197  )
198  return conn_data

200  def _build_jdbc_application_arguments(self, jdbc_conn):
201  arguments = []
202  arguments += ["-cmdType", self._cmd_type]
203  if self._jdbc_connection['url']:
204  arguments += ['-url', "{0}{1}/{2}".format(
205  jdbc_conn['conn_prefix'], jdbc_conn['url'], jdbc_conn['schema']
206  )]
207  if self._jdbc_connection['user']:
208  arguments += ['-user', self._jdbc_connection['user']]
209  if self._jdbc_connection['password']:
210  arguments += ['-password', self._jdbc_connection['password']]
211  if self._metastore_table:
212  arguments += ['-metastoreTable', self._metastore_table]
213  if self._jdbc_table:
214  arguments += ['-jdbcTable', self._jdbc_table]
215  if self._jdbc_truncate:
216  arguments += ['-jdbcTruncate', str(self._jdbc_truncate)]
217  if self._jdbc_driver:
218  arguments += ['-jdbcDriver', self._jdbc_driver]
219  if self._batch_size:
220  arguments += ['-batchsize', str(self._batch_size)]
221  if self._fetch_size:
222  arguments += ['-fetchsize', str(self._fetch_size)]
223  if self._num_partitions:
224  arguments += ['-numPartitions', str(self._num_partitions)]
225  if (self._partition_column and self._lower_bound and
226  self._upper_bound and self._num_partitions):
227  # these 3 parameters need to be used all together to take effect.
228  arguments += ['-partitionColumn', self._partition_column,
229  '-lowerBound', self._lower_bound,
230  '-upperBound', self._upper_bound]
231  if self._save_mode:
232  arguments += ['-saveMode', self._save_mode]
233  if self._save_format:
234  arguments += ['-saveFormat', self._save_format]
235  if self._create_table_column_types:
236  arguments += ['-createTableColumnTypes', self._create_table_column_types]
237  return arguments
