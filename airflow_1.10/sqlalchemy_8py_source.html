42  @event.listens_for(engine, "engine_connect")
43  def ping_connection(connection, branch):
44  """
45  Pessimistic SQLAlchemy disconnect handling. Ensures that each
46  connection returned from the pool is properly connected to the database.
47 
48  http://docs.sqlalchemy.org/en/rel_1_1/core/pooling.html#disconnect-handling-pessimistic
49  """
50  if branch:
51  # "branch" refers to a sub-connection of a connection,
52  # we don't want to bother pinging on these.
53  return
54 
55  start = time.time()
56  backoff = initial_backoff_seconds
57 
58  # turn off "close with result". This flag is only used with
59  # "connectionless" execution, otherwise will be False in any case
60  save_should_close_with_result = connection.should_close_with_result
61 
62  while True:
63  connection.should_close_with_result = False
64 
65  try:
66  connection.scalar(select([1]))
67  # If we made it here then the connection appears to be healty
68  break
69  except exc.DBAPIError as err:
70  if time.time() - start >= reconnect_timeout_seconds:
71  log.error(
72  "Failed to re-establish DB connection within %s secs: %s",
73  reconnect_timeout_seconds,
74  err)
75  raise
76  if err.connection_invalidated:
77  log.warning("DB connection invalidated. Reconnecting...")
78 
79  # Use a truncated binary exponential backoff. Also includes
80  # a jitter to prevent the thundering herd problem of
81  # simultaneous client reconnects
82  backoff += backoff * random.random()
83  time.sleep(min(backoff, max_backoff_seconds))
84 
85  # run the same SELECT again - the connection will re-validate
86  # itself and establish a new connection. The disconnect detection
87  # here also causes the whole connection pool to be invalidated
88  # so that all stale connections are discarded.
89  continue
90  else:
91  log.error(
92  "Unknown database connection error. Not retrying: %s",
93  err)
94  raise
95  finally:
96  # restore "close with result"
97  connection.should_close_with_result = save_should_close_with_result
