31 class ElasticsearchTaskHandler(FileTaskHandler, LoggingMixin):
32  PAGE = 0

35  """
36  ElasticsearchTaskHandler is a python log handler that
37  reads logs from Elasticsearch. Note logs are not directly
38  indexed into Elasticsearch. Instead, it flushes logs
39  into local files. Additional software setup is required
40  to index the log into Elasticsearch, such as using
41  Filebeat and Logstash.
42  To efficiently query and sort Elasticsearch results, we assume each
43  log message has a field `log_id` consists of ti primary keys:
44  `log_id = {dag_id}-{task_id}-{execution_date}-{try_number}`
45  Log messages with specific log_id are sorted based on `offset`,
46  which is a unique integer indicates log message's order.
47  Timestamp here are unreliable because multiple log messages
48  might have the same timestamp.
49  """

83  def _read(self, ti, try_number, metadata=None):
84  """
85  Endpoint for streaming log.
86  :param ti: task instance object
87  :param try_number: try_number of the task instance
88  :param metadata: log metadata,
89  can be used for steaming log reading and auto-tailing.
90  :return a list of log documents and metadata.
91  """
92  if not metadata:
93  metadata = {'offset': 0}
94  if 'offset' not in metadata:
95  metadata['offset'] = 0
96 
97  offset = metadata['offset']
98  log_id = self._render_log_id(ti, try_number)
99 
100  logs = self.es_read(log_id, offset)
101 
102  next_offset = offset if not logs else logs[-1].offset
103 
104  metadata['offset'] = next_offset
105  # end_of_log_mark may contain characters like '\n' which is needed to
106  # have the log uploaded but will not be stored in elasticsearch.
107  metadata['end_of_log'] = False if not logs \
108  else logs[-1].message == self.end_of_log_mark.strip()
109 
110  cur_ts = pendulum.now()
111  # Assume end of log after not receiving new log for 5 min,
112  # as executor heartbeat is 1 min and there might be some
113  # delay before Elasticsearch makes the log available.
114  if 'last_log_timestamp' in metadata:
115  last_log_ts = timezone.parse(metadata['last_log_timestamp'])
116  if cur_ts.diff(last_log_ts).in_minutes() >= 5:
117  metadata['end_of_log'] = True
118 
119  if offset != next_offset or 'last_log_timestamp' not in metadata:
120  metadata['last_log_timestamp'] = str(cur_ts)
121 
122  message = '\n'.join([log.message for log in logs])
123 
124  return message, metadata

126  def es_read(self, log_id, offset):
127  """
128  Returns the logs matching log_id in Elasticsearch and next offset.
129  Returns '' if no log is found or there was an error.
130  :param log_id: the log_id of the log to read.
131  :type log_id: str
132  :param offset: the offset start to read log from.
133  :type offset: str
134  """
135 
136  # Offset is the unique key for sorting logs given log_id.
137  s = Search(using=self.client) \
138  .query('match', log_id=log_id) \
139  .sort('offset')
140 
141  s = s.filter('range', offset={'gt': offset})
142 
143  logs = []
144  if s.count() != 0:
145  try:
146 
147  logs = s[self.MAX_LINE_PER_PAGE * self.PAGE:self.MAX_LINE_PER_PAGE] \
148  .execute()
149  except Exception as e:
150  msg = 'Could not read log with log_id: {}, ' \
151  'error: {}'.format(log_id, str(e))
152  self.log.exception(msg)
153 
154  return logs
