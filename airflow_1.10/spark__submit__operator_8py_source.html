26 class SparkSubmitOperator(BaseOperator):
27  """
28  This hook is a wrapper around the spark-submit binary to kick off a spark-submit job.
29  It requires that the "spark-submit" binary is in the PATH or the spark-home is set
30  in the extra on the connection.
31 
32  :param application: The application that submitted as a job, either jar or
33  py file. (templated)
34  :type application: str
35  :param conf: Arbitrary Spark configuration properties
36  :type conf: dict
37  :param conn_id: The connection id as configured in Airflow administration. When an
38  invalid connection_id is supplied, it will default to yarn.
39  :type conn_id: str
40  :param files: Upload additional files to the executor running the job, separated by a
41  comma. Files will be placed in the working directory of each executor.
42  For example, serialized objects.
43  :type files: str
44  :param py_files: Additional python files used by the job, can be .zip, .egg or .py.
45  :type py_files: str
46  :param jars: Submit additional jars to upload and place them in executor classpath.
47  :param driver_classpath: Additional, driver-specific, classpath settings.
48  :type driver_classpath: str
49  :type jars: str
50  :param java_class: the main class of the Java application
51  :type java_class: str
52  :param packages: Comma-separated list of maven coordinates of jars to include on the
53  driver and executor classpaths. (templated)
54  :type packages: str
55  :param exclude_packages: Comma-separated list of maven coordinates of jars to exclude
56  while resolving the dependencies provided in 'packages'
57  :type exclude_packages: str
58  :param repositories: Comma-separated list of additional remote repositories to search
59  for the maven coordinates given with 'packages'
60  :type repositories: str
61  :param total_executor_cores: (Standalone & Mesos only) Total cores for all executors
62  (Default: all the available cores on the worker)
63  :type total_executor_cores: int
64  :param executor_cores: (Standalone & YARN only) Number of cores per executor
65  (Default: 2)
66  :type executor_cores: int
67  :param executor_memory: Memory per executor (e.g. 1000M, 2G) (Default: 1G)
68  :type executor_memory: str
69  :param driver_memory: Memory allocated to the driver (e.g. 1000M, 2G) (Default: 1G)
70  :type driver_memory: str
71  :param keytab: Full path to the file that contains the keytab
72  :type keytab: str
73  :param principal: The name of the kerberos principal used for keytab
74  :type principal: str
75  :param name: Name of the job (default airflow-spark). (templated)
76  :type name: str
77  :param num_executors: Number of executors to launch
78  :type num_executors: int
79  :param application_args: Arguments for the application being submitted
80  :type application_args: list
81  :param env_vars: Environment variables for spark-submit. It
82  supports yarn and k8s mode too.
83  :type env_vars: dict
84  :param verbose: Whether to pass the verbose flag to spark-submit process for debugging
85  :type verbose: bool
86  """
