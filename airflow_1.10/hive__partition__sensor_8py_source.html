24 class HivePartitionSensor(BaseSensorOperator):
25  """
26  Waits for a partition to show up in Hive.
27 
28  Note: Because ``partition`` supports general logical operators, it
29  can be inefficient. Consider using NamedHivePartitionSensor instead if
30  you don't need the full flexibility of HivePartitionSensor.
31 
32  :param table: The name of the table to wait for, supports the dot
33  notation (my_database.my_table)
34  :type table: string
35  :param partition: The partition clause to wait for. This is passed as
36  is to the metastore Thrift client ``get_partitions_by_filter`` method,
37  and apparently supports SQL like notation as in ``ds='2015-01-01'
38  AND type='value'`` and comparison operators as in ``"ds>=2015-01-01"``
39  :type partition: string
40  :param metastore_conn_id: reference to the metastore thrift service
41  connection id
42  :type metastore_conn_id: str
43  """
44  template_fields = ('schema', 'table', 'partition',)
45  ui_color = '#C5CAE9'

48  def __init__(self,
49  table, partition="ds='{{ ds }}'",
50  metastore_conn_id='metastore_default',
51  schema='default',
52  poke_interval=60 * 3,
53  *args,
54  **kwargs):
55  super(HivePartitionSensor, self).__init__(
56  poke_interval=poke_interval, *args, **kwargs)
57  if not partition:
58  partition = "ds='{{ ds }}'"
59  self.metastore_conn_id = metastore_conn_id
60  self.table = table

63 
64  def poke(self, context):
65  if '.' in self.table:
66  self.schema, self.table = self.table.split('.')
67  self.log.info(
68  'Poking for table {self.schema}.{self.table}, '
69  'partition {self.partition}'.format(**locals()))
70  if not hasattr(self, 'hook'):
71  from airflow.hooks.hive_hooks import HiveMetastoreHook
72  self.hook = HiveMetastoreHook(
73  metastore_conn_id=self.metastore_conn_id)
74  return self.hook.check_for_partition(
75  self.schema, self.table, self.partition)
