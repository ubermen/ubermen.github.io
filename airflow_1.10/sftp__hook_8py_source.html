27 class SFTPHook(BaseHook):
28  """
29  Interact with SFTP. Aims to be interchangeable with FTPHook.
30 
31  Pitfalls: - In contrast with FTPHook describe_directory only returns size, type and
32  modify. It doesn't return unix.owner, unix.mode, perm, unix.group and
33  unique.
34  - retrieve_file and store_file only take a local full path and not a
35  buffer.
36  - If no mode is passed to create_directory it will be created with 777
37  permissions.
38 
39  Errors that may occur throughout but should be handled downstream.
40  """
41 
42  def __init__(self, ftp_conn_id='sftp_default'):
43  self.ftp_conn_id = ftp_conn_id
44  self.conn = None

46  def get_conn(self):
47  """
48  Returns an SFTP connection object
49  """
50  if self.conn is None:
51  params = self.get_connection(self.ftp_conn_id)
52  cnopts = pysftp.CnOpts()
53  if ('ignore_hostkey_verification' in params.extra_dejson and
54  params.extra_dejson['ignore_hostkey_verification']):
55  cnopts.hostkeys = None
56  conn_params = {
57  'host': params.host,
58  'port': params.port,
59  'username': params.login,
60  'cnopts': cnopts
61  }
62  if params.password is not None:
63  conn_params['password'] = params.password
64  if 'private_key' in params.extra_dejson:
65  conn_params['private_key'] = params.extra_dejson['private_key']
66  if 'private_key_pass' in params.extra_dejson:
67  conn_params['private_key_pass'] = params.extra_dejson['private_key_pass']
68  self.conn = pysftp.Connection(**conn_params)
69  return self.conn
70 
71  def close_conn(self):
72  """
73  Closes the connection. An error will occur if the
74  connection wasnt ever opened.
75  """
76  conn = self.conn
77  conn.close()
78  self.conn = None

80  def describe_directory(self, path):
81  """
82  Returns a dictionary of {filename: {attributes}} for all files
83  on the remote system (where the MLSD command is supported).
84  :param path: full path to the remote directory
85  :type path: str
86  """
87  conn = self.get_conn()
88  flist = conn.listdir_attr(path)
89  files = {}
90  for f in flist:
91  modify = datetime.datetime.fromtimestamp(
92  f.st_mtime).strftime('%Y%m%d%H%M%S')
93  files[f.filename] = {
94  'size': f.st_size,
95  'type': 'dir' if stat.S_ISDIR(f.st_mode) else 'file',
96  'modify': modify}
97  return files
