35 class GoogleCloudBaseHook(BaseHook, LoggingMixin):
36  """
37  A base hook for Google cloud-related hooks. Google cloud has a shared REST
38  API client that is built in the same way no matter which service you use.
39  This class helps construct and authorize the credentials needed to then
40  call apiclient.discovery.build() to actually discover and build a client
41  for a Google cloud service.
42 
43  The class also contains some miscellaneous helper functions.
44 
45  All hook derived from this base hook use the 'Google Cloud Platform' connection
46  type. Two ways of authentication are supported:
47 
48  Default credentials: Only the 'Project Id' is required. You'll need to
49  have set up default credentials, such as by the
50  ``GOOGLE_APPLICATION_DEFAULT`` environment variable or from the metadata
51  server on Google Compute Engine.
52 
53  JSON key file: Specify 'Project Id', 'Key Path' and 'Scope'.
54 
55  Legacy P12 key files are not supported.
56  """
57

71  def _get_credentials(self):
72  """
73  Returns the Credentials object for Google API
74  """
75  key_path = self._get_field('key_path', False)
76  keyfile_dict = self._get_field('keyfile_dict', False)
77  scope = self._get_field('scope', None)
78  if scope is not None:
79  scopes = [s.strip() for s in scope.split(',')]
80  else:
81  scopes = _DEFAULT_SCOPES
82 
83  if not key_path and not keyfile_dict:
84  self.log.info('Getting connection using `google.auth.default()` '
85  'since no key file is defined for hook.')
86  credentials, _ = google.auth.default(scopes=scopes)
87  elif key_path:
88  # Get credentials from a JSON file.
89  if key_path.endswith('.json'):
90  self.log.info('Getting connection using a JSON key file.')
91  credentials = (
92  google.oauth2.service_account.Credentials.from_service_account_file(
93  key_path, scopes=scopes)
94  )
95  elif key_path.endswith('.p12'):
96  raise AirflowException('Legacy P12 key file are not supported, '
97  'use a JSON key file.')
98  else:
99  raise AirflowException('Unrecognised extension for key file.')
100  else:
101  # Get credentials from JSON data provided in the UI.
102  try:
103  keyfile_dict = json.loads(keyfile_dict)
104 
105  # Depending on how the JSON was formatted, it may contain
106  # escaped newlines. Convert those to actual newlines.
107  keyfile_dict['private_key'] = keyfile_dict['private_key'].replace(
108  '\\n', '\n')
109 
110  credentials = (
111  google.oauth2.service_account.Credentials.from_service_account_info(
112  keyfile_dict, scopes=scopes)
113  )
114  except json.decoder.JSONDecodeError:
115  raise AirflowException('Invalid key JSON.')
116 
117  return credentials.with_subject(self.delegate_to) \
118  if self.delegate_to else credentials
