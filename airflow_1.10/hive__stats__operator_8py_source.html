32 class HiveStatsCollectionOperator(BaseOperator):
33  """
34  Gathers partition statistics using a dynamically generated Presto
35  query, inserts the stats into a MySql table with this format. Stats
36  overwrite themselves if you rerun the same date/partition. ::
37 
38  CREATE TABLE hive_stats (
39  ds VARCHAR(16),
40  table_name VARCHAR(500),
41  metric VARCHAR(200),
42  value BIGINT
43  );
44 
45  :param table: the source table, in the format ``database.table_name``. (templated)
46  :type table: str
47  :param partition: the source partition. (templated)
48  :type partition: dict of {col:value}
49  :param extra_exprs: dict of expression to run against the table where
50  keys are metric names and values are Presto compatible expressions
51  :type extra_exprs: dict
52  :param col_blacklist: list of columns to blacklist, consider
53  blacklisting blobs, large json columns, ...
54  :type col_blacklist: list
55  :param assignment_func: a function that receives a column name and
56  a type, and returns a dict of metric names and an Presto expressions.
57  If None is returned, the global defaults are applied. If an
58  empty dictionary is returned, no stats are computed for that
59  column.
60  :type assignment_func: function
61  """

91  def get_default_exprs(self, col, col_type):
92  if col in self.col_blacklist:
93  return {}
94  d = {}
95  d[(col, 'non_null')] = "COUNT({col})"
96  if col_type in ['double', 'int', 'bigint', 'float', 'double']:
97  d[(col, 'sum')] = 'SUM({col})'
98  d[(col, 'min')] = 'MIN({col})'
99  d[(col, 'max')] = 'MAX({col})'
100  d[(col, 'avg')] = 'AVG({col})'
101  elif col_type == 'boolean':
102  d[(col, 'true')] = 'SUM(CASE WHEN {col} THEN 1 ELSE 0 END)'
103  d[(col, 'false')] = 'SUM(CASE WHEN NOT {col} THEN 1 ELSE 0 END)'
104  elif col_type in ['string']:
105  d[(col, 'len')] = 'SUM(CAST(LENGTH({col}) AS BIGINT))'
106  d[(col, 'approx_distinct')] = 'APPROX_DISTINCT({col})'
107 
108  return {k: v.format(col=col) for k, v in d.items()}

110  def execute(self, context=None):
111  metastore = HiveMetastoreHook(metastore_conn_id=self.metastore_conn_id)
112  table = metastore.get_table(table_name=self.table)
113  field_types = {col.name: col.type for col in table.sd.cols}
114 
115  exprs = {
116  ('', 'count'): 'COUNT(*)'
117  }
118  for col, col_type in list(field_types.items()):
119  d = {}
120  if self.assignment_func:
121  d = self.assignment_func(col, col_type)
122  if d is None:
123  d = self.get_default_exprs(col, col_type)
124  else:
125  d = self.get_default_exprs(col, col_type)
126  exprs.update(d)
127  exprs.update(self.extra_exprs)
128  exprs = OrderedDict(exprs)
129  exprs_str = ",\n ".join([
130  v + " AS " + k[0] + '__' + k[1]
131  for k, v in exprs.items()])
132 
133  where_clause = [
134  "{0} = '{1}'".format(k, v) for k, v in self.partition.items()]
135  where_clause = " AND\n ".join(where_clause)
136  sql = """
137  SELECT
138  {exprs_str}
139  FROM {self.table}
140  WHERE
141  {where_clause};

143

145  self.log.info('Executing SQL check: %s', sql)
146  row = hook.get_first(hql=sql)

150

157  WHERE
158  table_name='{self.table}' AND

177  for r in zip(exprs, row)]
178  mysql.insert_rows(
