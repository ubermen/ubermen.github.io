62  def run_pod(self, pod, startup_timeout=120, get_logs=True):
63  # type: (Pod) -> (State, result)
64  """
65  Launches the pod synchronously and waits for completion.
66  Args:
67  pod (Pod):
68  startup_timeout (int): Timeout for startup of the pod (if pod is pending for
69  too long, considers task a failure
70  """
71  resp = self.run_pod_async(pod)
72  curr_time = dt.now()
73  if resp.status.start_time is None:
74  while self.pod_not_started(pod):
75  delta = dt.now() - curr_time
76  if delta.seconds >= startup_timeout:
77  raise AirflowException("Pod took too long to start")
78  time.sleep(1)
79  self.log.debug('Pod not yet started')
80 
81  return self._monitor_pod(pod, get_logs)

83  def _monitor_pod(self, pod, get_logs):
84  # type: (Pod) -> (State, content)
85 
86  if get_logs:
87  logs = self._client.read_namespaced_pod_log(
88  name=pod.name,
89  namespace=pod.namespace,
90  container='base',
91  follow=True,
92  tail_lines=10,
93  _preload_content=False)
94  for line in logs:
95  self.log.info(line)
96  result = None
97  if self.extract_xcom:
98  while self.base_container_is_running(pod):
99  self.log.info('Container %s has state %s', pod.name, State.RUNNING)
100  time.sleep(2)
101  result = self._extract_xcom(pod)
102  self.log.info(result)
103  result = json.loads(result)
104  while self.pod_is_running(pod):
105  self.log.info('Pod %s has state %s', pod.name, State.RUNNING)
106  time.sleep(2)
107  return (self._task_status(self.read_pod(pod)), result)

138  def _extract_xcom(self, pod):
139  resp = kubernetes_stream(self._client.connect_get_namespaced_pod_exec,
140  pod.name, pod.namespace,
141  container=self.kube_req_factory.SIDECAR_CONTAINER_NAME,
142  command=['/bin/sh'], stdin=True, stdout=True,
143  stderr=True, tty=False,
144  _preload_content=False)

147  resp, 'cat {}/return.json'.format(self.kube_req_factory.XCOM_MOUNT_PATH))
148  self._exec_pod_command(resp, 'kill -s SIGINT 1')

155  def _exec_pod_command(self, resp, command):
156  if resp.is_open():
157  self.log.info('Running command... %s\n' % command)
158  resp.write_stdin(command + '\n')
159  while resp.is_open():
160  resp.update(timeout=1)
161  if resp.peek_stdout():
162  return resp.read_stdout()
163  if resp.peek_stderr():
164  self.log.info(resp.read_stderr())
165  break

167  def process_status(self, job_id, status):
168  status = status.lower()
169  if status == PodStatus.PENDING:
170  return State.QUEUED
171  elif status == PodStatus.FAILED:
172  self.log.info('Event with job id %s Failed', job_id)
173  return State.FAILED
174  elif status == PodStatus.SUCCEEDED:
175  self.log.info('Event with job id %s Succeeded', job_id)
176  return State.SUCCESS
177  elif status == PodStatus.RUNNING:
178  return State.RUNNING
179  else:
180  self.log.info('Event: Invalid state %s on job %s', status, job_id)
181  return State.FAILED
