59  def publish(self, project, topic, messages):
60  """Publishes messages to a Pub/Sub topic.
61 
62  :param project: the GCP project ID in which to publish
63  :type project: string
64  :param topic: the Pub/Sub topic to which to publish; do not
65  include the ``projects/{project}/topics/`` prefix.
66  :type topic: string
67  :param messages: messages to publish; if the data field in a
68  message is set, it should already be base64 encoded.
69  :type messages: list of PubSub messages; see
70  http://cloud.google.com/pubsub/docs/reference/rest/v1/PubsubMessage
71  """
72  body = {'messages': messages}
73  full_topic = _format_topic(project, topic)
74  request = self.get_conn().projects().topics().publish(
75  topic=full_topic, body=body)
76  try:
77  request.execute()
78  except errors.HttpError as e:
79  raise PubSubException(
80  'Error publishing to topic {}'.format(full_topic), e)

82  def create_topic(self, project, topic, fail_if_exists=False):
83  """Creates a Pub/Sub topic, if it does not already exist.
84 
85  :param project: the GCP project ID in which to create
86  the topic
87  :type project: string
88  :param topic: the Pub/Sub topic name to create; do not
89  include the ``projects/{project}/topics/`` prefix.
90  :type topic: string
91  :param fail_if_exists: if set, raise an exception if the topic
92  already exists
93  :type fail_if_exists: bool
94  """
95  service = self.get_conn()
96  full_topic = _format_topic(project, topic)
97  try:
98  service.projects().topics().create(
99  name=full_topic, body={}).execute()
100  except errors.HttpError as e:
101  # Status code 409 indicates that the topic already exists.
102  if str(e.resp['status']) == '409':
103  message = 'Topic already exists: {}'.format(full_topic)
104  self.log.warning(message)
105  if fail_if_exists:
106  raise PubSubException(message)
107  else:
108  raise PubSubException(
109  'Error creating topic {}'.format(full_topic), e)

138  def create_subscription(self, topic_project, topic, subscription=None,
139  subscription_project=None, ack_deadline_secs=10,
140  fail_if_exists=False):
141  """Creates a Pub/Sub subscription, if it does not already exist.
142 
143  :param topic_project: the GCP project ID of the topic that the
144  subscription will be bound to.
145  :type topic_project: string
146  :param topic: the Pub/Sub topic name that the subscription will be bound
147  to create; do not include the ``projects/{project}/subscriptions/``
148  prefix.
149  :type topic: string
150  :param subscription: the Pub/Sub subscription name. If empty, a random
151  name will be generated using the uuid module
152  :type subscription: string
153  :param subscription_project: the GCP project ID where the subscription
154  will be created. If unspecified, ``topic_project`` will be used.
155  :type subscription_project: string
156  :param ack_deadline_secs: Number of seconds that a subscriber has to
157  acknowledge each message pulled from the subscription
158  :type ack_deadline_secs: int
159  :param fail_if_exists: if set, raise an exception if the topic
160  already exists
161  :type fail_if_exists: bool
162  :return: subscription name which will be the system-generated value if
163  the ``subscription`` parameter is not supplied
164  :rtype: string
165  """
166  service = self.get_conn()
167  full_topic = _format_topic(topic_project, topic)
168  if not subscription:
169  subscription = 'sub-{}'.format(uuid4())
170  if not subscription_project:
171  subscription_project = topic_project
172  full_subscription = _format_subscription(subscription_project,
173  subscription)
174  body = {
175  'topic': full_topic,
176  'ackDeadlineSeconds': ack_deadline_secs
177  }
178  try:
179  service.projects().subscriptions().create(
180  name=full_subscription, body=body).execute()
181  except errors.HttpError as e:
182  # Status code 409 indicates that the subscription already exists.
183  if str(e.resp['status']) == '409':
184  message = 'Subscription already exists: {}'.format(
185  full_subscription)
186  self.log.warning(message)
187  if fail_if_exists:
188  raise PubSubException(message)
189  else:
190  raise PubSubException(
191  'Error creating subscription {}'.format(full_subscription),
192  e)
193  return subscription

226  def pull(self, project, subscription, max_messages,
227  return_immediately=False):
228  """Pulls up to ``max_messages`` messages from Pub/Sub subscription.
229 
230  :param project: the GCP project ID where the subscription exists
231  :type project: string
232  :param subscription: the Pub/Sub subscription name to pull from; do not
233  include the 'projects/{project}/topics/' prefix.
234  :type subscription: string
235  :param max_messages: The maximum number of messages to return from
236  the Pub/Sub API.
237  :type max_messages: int
238  :param return_immediately: If set, the Pub/Sub API will immediately
239  return if no messages are available. Otherwise, the request will
240  block for an undisclosed, but bounded period of time
241  :type return_immediately: bool
242  :return A list of Pub/Sub ReceivedMessage objects each containing
243  an ``ackId`` property and a ``message`` property, which includes
244  the base64-encoded message content. See
245  https://cloud.google.com/pubsub/docs/reference/rest/v1/\
246  projects.subscriptions/pull#ReceivedMessage
247  """
248  service = self.get_conn()
249  full_subscription = _format_subscription(project, subscription)
250  body = {
251  'maxMessages': max_messages,
252  'returnImmediately': return_immediately
253  }
254  try:
255  response = service.projects().subscriptions().pull(
256  subscription=full_subscription, body=body).execute()
257  return response.get('receivedMessages', [])
258  except errors.HttpError as e:
259  raise PubSubException(
260  'Error pulling messages from subscription {}'.format(
261  full_subscription), e)
