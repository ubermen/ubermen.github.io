25 class PrevDagrunDep(BaseTIDep):
26  """
27  Is the past dagrun in a state that allows this task instance to run, e.g. did this
28  task instance's task in the previous dagrun complete if we are depending on past.
29  """
30  NAME = "Previous Dagrun State"
31  IGNOREABLE = True
32  IS_TASK_DEP = True
33 
34  @provide_session
35  def _get_dep_statuses(self, ti, session, dep_context):
36  if dep_context.ignore_depends_on_past:
37  yield self._passing_status(
38  reason="The context specified that the state of past DAGs could be "
39  "ignored.")
40  return
41 
42  if not ti.task.depends_on_past:
43  yield self._passing_status(
44  reason="The task did not have depends_on_past set.")
45  return
46 
47  # Don't depend on the previous task instance if we are the first task
48  dag = ti.task.dag
49  if dag.catchup:
50  if dag.previous_schedule(ti.execution_date) is None:
51  yield self._passing_status(
52  reason="This task does not have a schedule or is @once"
53  )
54  return
55  if dag.previous_schedule(ti.execution_date) < ti.task.start_date:
56  yield self._passing_status(
57  reason="This task instance was the first task instance for its task.")
58  return
59  else:
60  dr = ti.get_dagrun()
61  last_dagrun = dr.get_previous_dagrun() if dr else None
62 
63  if not last_dagrun:
64  yield self._passing_status(
65  reason="This task instance was the first task instance for its task.")
66  return
67 
68  previous_ti = ti.previous_ti
69  if not previous_ti:
70  yield self._failing_status(
71  reason="depends_on_past is true for this task's DAG, but the previous "
72  "task instance has not run yet.")
73  return
74 
75  if previous_ti.state not in {State.SKIPPED, State.SUCCESS}:
76  yield self._failing_status(
77  reason="depends_on_past is true for this task, but the previous task "
78  "instance {0} is in the state '{1}' which is not a successful "
79  "state.".format(previous_ti, previous_ti.state))
80 
81  previous_ti.task = ti.task
82  if (ti.task.wait_for_downstream and
83  not previous_ti.are_dependents_done(session=session)):
84  yield self._failing_status(
85  reason="The tasks downstream of the previous task instance {0} haven't "
86  "completed.".format(previous_ti))
