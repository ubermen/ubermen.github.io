144  def heartbeat(self):
145  """
146  Heartbeats update the job's entry in the database with a timestamp
147  for the latest_heartbeat and allows for the job to be killed
148  externally. This allows at the system level to monitor what is
149  actually active.
150 
151  For instance, an old heartbeat for SchedulerJob would mean something
152  is wrong.
153 
154  This also allows for any job to be killed externally, regardless
155  of who is running it or on which machine it is running.
156 
157  Note that if your heartbeat is set to 60 seconds and you call this
158  method after 10 seconds of processing since the last heartbeat, it
159  will sleep 50 seconds to complete the 60 seconds and keep a steady
160  heart rate. If you go over 60 seconds before calling it, it won't
161  sleep at all.
162  """
163  with create_session() as session:
164  job = session.query(BaseJob).filter_by(id=self.id).one()
165  make_transient(job)
166  session.commit()
167 
168  if job.state == State.SHUTDOWN:
169  self.kill()
170 
171  # Figure out how long to sleep for
172  sleep_for = 0
173  if job.latest_heartbeat:
174  sleep_for = max(
175  0,
176  self.heartrate - (timezone.utcnow() - job.latest_heartbeat).total_seconds())
177 
178  sleep(sleep_for)
179 
180  # Update last heartbeat time
181  with create_session() as session:
182  job = session.query(BaseJob).filter(BaseJob.id == self.id).first()
183  job.latest_heartbeat = timezone.utcnow()
184  session.merge(job)
185  session.commit()
186

188  self.log.debug('[heartbeat]')
