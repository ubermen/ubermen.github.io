32 class MsSqlToHiveTransfer(BaseOperator):
33  """
34  Moves data from Microsoft SQL Server to Hive. The operator runs
35  your query against Microsoft SQL Server, stores the file locally
36  before loading it into a Hive table. If the ``create`` or
37  ``recreate`` arguments are set to ``True``,
38  a ``CREATE TABLE`` and ``DROP TABLE`` statements are generated.
39  Hive data types are inferred from the cursor's metadata.
40  Note that the table generated in Hive uses ``STORED AS textfile``
41  which isn't the most efficient serialization format. If a
42  large amount of data is loaded and/or if the table gets
43  queried considerably, you may want to use this operator only to
44  stage the data into a temporary table before loading it into its
45  final destination using a ``HiveOperator``.
46 
47  :param sql: SQL query to execute against the Microsoft SQL Server
48  database. (templated)
49  :type sql: str
50  :param hive_table: target Hive table, use dot notation to target a specific
51  database. (templated)
52  :type hive_table: str
53  :param create: whether to create the table if it doesn't exist
54  :type create: bool
55  :param recreate: whether to drop and recreate the table at every execution
56  :type recreate: bool
57  :param partition: target partition as a dict of partition columns and
58  values. (templated)
59  :type partition: dict
60  :param delimiter: field delimiter in the file
61  :type delimiter: str
62  :param mssql_conn_id: source Microsoft SQL Server connection
63  :type mssql_conn_id: str
64  :param hive_conn_id: destination hive connection
65  :type hive_conn_id: str
66  :param tblproperties: TBLPROPERTIES of the hive table being created
67  :type tblproperties: dict
68  """

109  def execute(self, context):
110  hive = HiveCliHook(hive_cli_conn_id=self.hive_cli_conn_id)
111  mssql = MsSqlHook(mssql_conn_id=self.mssql_conn_id)
112 
113  self.log.info("Dumping Microsoft SQL Server query results to local file")
114  conn = mssql.get_conn()
115  cursor = conn.cursor()
116  cursor.execute(self.sql)
117  with NamedTemporaryFile("w") as f:
118  csv_writer = csv.writer(f, delimiter=self.delimiter, encoding='utf-8')
119  field_dict = OrderedDict()
120  col_count = 0
121  for field in cursor.description:
122  col_count += 1
123  col_position = "Column{position}".format(position=col_count)
124  field_dict[col_position if field[0] == '' else field[0]] \
125  = self.type_map(field[1])
126  csv_writer.writerows(cursor)
