121  def resourceOffers(self, driver, offers):
122  for offer in offers:
123  tasks = []
124  offerCpus = 0
125  offerMem = 0
126  for resource in offer.resources:
127  if resource.name == "cpus":
128  offerCpus += resource.scalar.value
129  elif resource.name == "mem":
130  offerMem += resource.scalar.value
131 
132  self.log.info("Received offer %s with cpus: %s and mem: %s",
133  offer.id.value, offerCpus, offerMem)
134 
135  remainingCpus = offerCpus
136  remainingMem = offerMem
137 
138  while (not self.task_queue.empty()) and \
139  remainingCpus >= self.task_cpu and \
140  remainingMem >= self.task_mem:
141  key, cmd = self.task_queue.get()
142  tid = self.task_counter
143  self.task_counter += 1
144  self.task_key_map[str(tid)] = key
145 
146  self.log.info("Launching task %d using offer %s", tid, offer.id.value)
147 
148  task = mesos_pb2.TaskInfo()
149  task.task_id.value = str(tid)
150  task.slave_id.value = offer.slave_id.value
151  task.name = "AirflowTask %d" % tid
152 
153  cpus = task.resources.add()
154  cpus.name = "cpus"
155  cpus.type = mesos_pb2.Value.SCALAR
156  cpus.scalar.value = self.task_cpu
157 
158  mem = task.resources.add()
159  mem.name = "mem"
160  mem.type = mesos_pb2.Value.SCALAR
161  mem.scalar.value = self.task_mem
162 
163  command = mesos_pb2.CommandInfo()
164  command.shell = True
165  command.value = cmd
166  task.command.MergeFrom(command)
167 
168  # If docker image for airflow is specified in config then pull that
169  # image before running the above airflow command
170  if self.mesos_slave_docker_image:
171  network = mesos_pb2.ContainerInfo.DockerInfo.Network.Value('BRIDGE')
172  docker = mesos_pb2.ContainerInfo.DockerInfo(
173  image=self.mesos_slave_docker_image,
174  force_pull_image=False,
175  network=network
176  )
177  container = mesos_pb2.ContainerInfo(
178  type=mesos_pb2.ContainerInfo.DOCKER,
179  docker=docker
180  )
181  task.container.MergeFrom(container)
182 
183  tasks.append(task)
184 
185  remainingCpus -= self.task_cpu
186  remainingMem -= self.task_mem
187 
188  driver.launchTasks(offer.id, tasks)

190  def statusUpdate(self, driver, update):
191  self.log.info(
192  "Task %s is in state %s, data %s",
193  update.task_id.value, mesos_pb2.TaskState.Name(update.state), str(update.data)
194  )
195 
196  try:
197  key = self.task_key_map[update.task_id.value]
198  except KeyError:
199  # The map may not contain an item if the framework re-registered
200  # after a failover.
201  # Discard these tasks.
202  self.log.warning("Unrecognised task key %s", update.task_id.value)
203  return
204 
205  if update.state == mesos_pb2.TASK_FINISHED:
206  self.result_queue.put((key, State.SUCCESS))
207  self.task_queue.task_done()
208 
209  if update.state == mesos_pb2.TASK_LOST or \
210  update.state == mesos_pb2.TASK_KILLED or \
211  update.state == mesos_pb2.TASK_FAILED:
212  self.result_queue.put((key, State.FAILED))
213  self.task_queue.task_done()

216 class MesosExecutor(BaseExecutor, LoginMixin):
217  """
218  MesosExecutor allows distributing the execution of task
219  instances to multiple mesos workers.
220 
221  Apache Mesos is a distributed systems kernel which abstracts
222  CPU, memory, storage, and other compute resources away from
223  machines (physical or virtual), enabling fault-tolerant and
224  elastic distributed systems to easily be built and run effectively.
225  See http://mesos.apache.org/
226  """
