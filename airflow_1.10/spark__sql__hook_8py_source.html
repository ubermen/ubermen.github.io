33  :type conf: str (format: PROP=VALUE)
34  :param conn_id: connection_id string

38  :type total_executor_cores: int
39  :param executor_cores: (Standalone & YARN only) Number of cores per
40  executor (Default: 2)
41  :type executor_cores: int
42  :param executor_memory: Memory per executor (e.g. 1000M, 2G) (Default: 1G)
43  :type executor_memory: str
44  :param keytab: Full path to the file that contains the keytab
45  :type keytab: str

48  :param name: Name of the job.

90  def _prepare_command(self, cmd):
91  """
92  Construct the spark-sql command to execute. Verbose output is enabled
93  as default.
94  :param cmd: command to append to the spark-sql command
95  :type cmd: str
96  :return: full command to be executed
97  """
98  connection_cmd = ["spark-sql"]
99  if self._conf:
100  for conf_el in self._conf.split(","):
101  connection_cmd += ["--conf", conf_el]
102  if self._total_executor_cores:
103  connection_cmd += ["--total-executor-cores", str(self._total_executor_cores)]
104  if self._executor_cores:
105  connection_cmd += ["--executor-cores", str(self._executor_cores)]
106  if self._executor_memory:
107  connection_cmd += ["--executor-memory", self._executor_memory]
108  if self._keytab:
109  connection_cmd += ["--keytab", self._keytab]
110  if self._principal:
111  connection_cmd += ["--principal", self._principal]
112  if self._num_executors:
113  connection_cmd += ["--num-executors", str(self._num_executors)]
114  if self._sql:
115  sql = self._sql.strip()
116  if sql.endswith(".sql") or sql.endswith(".hql"):
117  connection_cmd += ["-f", sql]
118  else:
119  connection_cmd += ["-e", sql]
120  if self._master:
121  connection_cmd += ["--master", self._master]
122  if self._name:
123  connection_cmd += ["--name", self._name]
124  if self._verbose:
125  connection_cmd += ["--verbose"]
126  if self._yarn_queue:
127  connection_cmd += ["--queue", self._yarn_queue]
128 
129  connection_cmd += cmd
130  self.log.debug("Spark-Sql cmd: %s", connection_cmd)
131 
132  return connection_cmd

134  def run_query(self, cmd="", **kwargs):
135  """
136  Remote Popen (actually execute the Spark-sql query)
137 
138  :param cmd: command to remotely execute
139  :param kwargs: extra arguments to Popen (see subprocess.Popen)
140  """
141  spark_sql_cmd = self._prepare_command(cmd)
142  self._sp = subprocess.Popen(spark_sql_cmd,
143  stdout=subprocess.PIPE,
