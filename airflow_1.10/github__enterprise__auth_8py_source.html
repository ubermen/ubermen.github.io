145  def ghe_team_check(self, username, ghe_token):
146  try:
147  # the response from ghe returns the id of the team as an integer
148  try:
149  allowed_teams = [int(team.strip())
150  for team in
151  get_config_param('allowed_teams').split(',')]
152  except ValueError:
153  # this is to deprecate using the string name for a team
154  raise ValueError(
155  'it appears that you are using the string name for a team, '
156  'please use the id number instead')
157 
158  except AirflowConfigException:
159  # No allowed teams defined, let anyone in GHE in.
160  return True
161 
162  # https://developer.github.com/v3/orgs/teams/#list-user-teams
163  resp = self.ghe_oauth.get(self.ghe_api_route('/user/teams'),
164  token=(ghe_token, ''))
165 
166  if not resp or resp.status != 200:
167  raise AuthenticationError(
168  'Bad response from GHE ({0})'.format(
169  resp.status if resp else 'None'))
170 
171  for team in resp.data:
172  # mylons: previously this line used to be if team['slug'] in teams
173  # however, teams are part of organizations. organizations are unique,
174  # but teams are not therefore 'slug' for a team is not necessarily unique.
175  # use id instead
176  if team['id'] in allowed_teams:
177  return True
178 
179  log.debug('Denying access for user "%s", not a member of "%s"',
180  username,
181  str(allowed_teams))
182 
183  return False

187  if not userid or userid == 'None':

194  @provide_session
195  def oauth_callback(self, session=None):
196  log.debug('GHE OAuth callback called')
197 
198  next_url = request.args.get('state') or url_for('admin.index')
199 
200  resp = self.ghe_oauth.authorized_response()
201 
202  try:
203  if resp is None:
204  raise AuthenticationError(
205  'Null response from GHE, denying access.'
206  )
207 
208  ghe_token = resp['access_token']
209 
210  username, email = self.get_ghe_user_profile_info(ghe_token)
211 
212  if not self.ghe_team_check(username, ghe_token):
213  return redirect(url_for('airflow.noaccess'))
214 
215  except AuthenticationError:
216  log.exception('')
217  return redirect(url_for('airflow.noaccess'))
218 
219  user = session.query(models.User).filter(
220  models.User.username == username).first()
221 
222  if not user:
223  user = models.User(
224  username=username,
225  email=email,
226  is_superuser=False)
227 
228  session.merge(user)
229  session.commit()
230  login_user(GHEUser(user))
231  session.commit()
232
