93 def groups_user(conn, search_base, user_filter, user_name_att, username):
94  search_filter = "(&({0})({1}={2}))".format(user_filter, user_name_att, username)
95  try:
96  memberof_attr = configuration.conf.get("ldap", "group_member_attr")
97  except:
98  memberof_attr = "memberOf"
99  res = conn.search(native(search_base), native(search_filter),
100  attributes=[native(memberof_attr)])
101  if not res:
102  log.info("Cannot find user %s", username)
103  raise AuthenticationError("Invalid username or password")
104 
105  if conn.response and memberof_attr not in conn.response[0]["attributes"]:
106  log.warning("""Missing attribute "%s" when looked-up in Ldap database.
107  The user does not seem to be a member of a group and therefore won't see any dag
108  if the option filter_by_owner=True and owner_mode=ldapgroup are set""",
109  memberof_attr)
110  return []
111 
112  user_groups = conn.response[0]["attributes"][memberof_attr]
113 
114  regex = re.compile("cn=([^,]*).*", re.IGNORECASE)
115  groups_list = []
116  try:
117  groups_list = [regex.search(i).group(1) for i in user_groups]
118  except IndexError:
119  log.warning("Parsing error when retrieving the user's group(s)."
120  " Check if the user belongs to at least one group"
121  " or if the user's groups name do not contain special characters")
122 
123  return groups_list
