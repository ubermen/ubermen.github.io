26 class DiscordWebhookOperator(SimpleHttpOperator):
27  """
28  This operator allows you to post messages to Discord using incoming webhooks.
29  Takes a Discord connection ID with a default relative webhook endpoint. The
30  default endpoint can be overridden using the webhook_endpoint parameter
31  (https://discordapp.com/developers/docs/resources/webhook).
32 
33  Each Discord webhook can be pre-configured to use a specific username and
34  avatar_url. You can override these defaults in this operator.
35 
36  :param http_conn_id: Http connection ID with host as "https://discord.com/api/" and
37  default webhook endpoint in the extra field in the form of
38  {"webhook_endpoint": "webhooks/{webhook.id}/{webhook.token}"}
39  :type http_conn_id: str
40  :param webhook_endpoint: Discord webhook endpoint in the form of
41  "webhooks/{webhook.id}/{webhook.token}"
42  :type webhook_endpoint: str
43  :param message: The message you want to send to your Discord channel
44  (max 2000 characters). (templated)
45  :type message: str
46  :param username: Override the default username of the webhook. (templated)
47  :type username: str
48  :param avatar_url: Override the default avatar of the webhook
49  :type avatar_url: str
50  :param tts: Is a text-to-speech message
51  :type tts: bool
52  :param proxy: Proxy to use to make the Discord webhook call
53  :type proxy: str
54  """
