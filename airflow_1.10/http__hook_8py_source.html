54  Returns http session for use with requests
55  :param headers: additional headers to be passed through as a dictionary
56  :type headers: dict
57  """
58  conn = self.get_connection(self.http_conn_id)

79  def run(self, endpoint, data=None, headers=None, extra_options=None):
80  """
81  Performs the request
82  :param endpoint: the endpoint to be called i.e. resource/v1/query?
83  :type endpoint: str
84  :param data: payload to be uploaded or request parameters
85  :type data: dict
86  :param headers: additional headers to be passed through as a dictionary
87  :type headers: dict
88  :param extra_options: additional options to be used when executing the request
89  i.e. {'check_response': False} to avoid checking raising exceptions on non
90  2XX or 3XX status codes
91  :type extra_options: dict
92  """
93  extra_options = extra_options or {}
94 
95  session = self.get_conn(headers)
96 
97  url = self.base_url + endpoint
98  req = None
99  if self.method == 'GET':
100  # GET uses params
101  req = requests.Request(self.method,
102  url,
103  params=data,
104  headers=headers)
105  elif self.method == 'HEAD':
106  # HEAD doesn't use params
107  req = requests.Request(self.method,
108  url,
109  headers=headers)
110  else:
111  # Others use data
112  req = requests.Request(self.method,
113  url,
114  data=data,
115  headers=headers)
116 
117  prepped_request = session.prepare_request(req)
118  self.log.info("Sending '%s' to url: %s", self.method, url)
119  return self.run_and_check(session, prepped_request, extra_options)

121  def check_response(self, response):
122  """
123  Checks the status code and raise an AirflowException exception on non 2XX or 3XX
124  status codes
125  :param response: A requests response object
126  :type response: requests.response
127  """
128  try:
129  response.raise_for_status()
130  except requests.exceptions.HTTPError:
131  self.log.error("HTTP error: %s", response.reason)
132  if self.method not in ['GET', 'HEAD']:
133  self.log.error(response.text)
134  raise AirflowException(str(response.status_code) + ":" + response.reason)

136  def run_and_check(self, session, prepped_request, extra_options):
137  """
138  Grabs extra options like timeout and actually runs the request,
139  checking for the result
140  :param session: the session to be used to execute the request
141  :type session: requests.Session
142  :param prepped_request: the prepared request generated in run()
143  :type prepped_request: session.prepare_request
144  :param extra_options: additional options to be used when executing the request
145  i.e. {'check_response': False} to avoid checking raising exceptions on non 2XX
146  or 3XX status codes
147  :type extra_options: dict
148  """
149  extra_options = extra_options or {}
150 
151  try:
152  response = session.send(
153  prepped_request,
154  stream=extra_options.get("stream", False),
155  verify=extra_options.get("verify", False),
156  proxies=extra_options.get("proxies", {}),
157  cert=extra_options.get("cert"),
158  timeout=extra_options.get("timeout"),
159  allow_redirects=extra_options.get("allow_redirects", True))
160 
161  if extra_options.get('check_response', True):
162  self.check_response(response)
163  return response
164 
165  except requests.exceptions.ConnectionError as ex:
166  self.log.warn(str(ex) + ' Tenacity will retry to execute the operation')
167  raise ex
