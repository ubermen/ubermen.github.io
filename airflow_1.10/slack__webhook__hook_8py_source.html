26 class SlackWebhookHook(HttpHook):
27  """
28  This hook allows you to post messages to Slack using incoming webhooks.
29  Takes both Slack webhook token directly and connection that has Slack webhook token.
30  If both supplied, Slack webhook token will be used.
31 
32  Each Slack webhook token can be pre-configured to use a specific channel, username and
33  icon. You can override these defaults in this hook.
34 
35  :param http_conn_id: connection that has Slack webhook token in the extra field
36  :type http_conn_id: str
37  :param webhook_token: Slack webhook token
38  :type webhook_token: str
39  :param message: The message you want to send on Slack
40  :type message: str
41  :param channel: The channel the message should be posted to
42  :type channel: str
43  :param username: The username to post to slack with
44  :type username: str
45  :param icon_emoji: The emoji to use as icon for the user posting to Slack
46  :type icon_emoji: str
47  :param link_names: Whether or not to find and link channel and usernames in your
48  message
49  :type link_names: bool
50  :param proxy: Proxy to use to make the Slack webhook call
51  :type proxy: str
52  """

75  def _get_token(self, token, http_conn_id):
76  """
77  Given either a manually set token or a conn_id, return the webhook_token to use
78  :param token: The manually provided token
79  :param conn_id: The conn_id provided
80  :return: webhook_token (str) to use
81  """
82  if token:

84  elif http_conn_id:

92  def _build_slack_message(self):
93  """
94  Construct the Slack message. All relevant parameters are combined here to a valid
95  Slack json message
96  :return: Slack message (str) to send
97  """
98  cmd = {}
99 
100  if self.channel:
101  cmd['channel'] = self.channel
102  if self.username:
103  cmd['username'] = self.username
104  if self.icon_emoji:
105  cmd['icon_emoji'] = self.icon_emoji
106  if self.link_names:
107  cmd['link_names'] = 1
108 
109  # there should always be a message to post ;-)
110  cmd['text'] = self.message
111  return json.dumps(cmd)
