29 class BashSensor(BaseSensorOperator):

31  Executes a bash command/script and returns True if and only if the

33

45  """

60  def poke(self, context):
61  """
62  Execute the bash command in a temporary directory
63  which will be cleaned afterwards
64  """
65  bash_command = self.bash_command
66  self.log.info("Tmp dir root location: \n %s", gettempdir())
67  with TemporaryDirectory(prefix='airflowtmp') as tmp_dir:
68  with NamedTemporaryFile(dir=tmp_dir, prefix=self.task_id) as f:
69  f.write(bytes(bash_command, 'utf_8'))
70  f.flush()
71  fname = f.name
72  script_location = tmp_dir + "/" + fname
73  self.log.info(
74  "Temporary script location: %s",
75  script_location
76  )
77  self.log.info("Running command: %s", bash_command)
78  sp = Popen(
79  ['bash', fname],
80  stdout=PIPE, stderr=STDOUT,
81  close_fds=True, cwd=tmp_dir,
82  env=self.env, preexec_fn=os.setsid)
83 
84  self.sp = sp
85 
86  self.log.info("Output:")
87  line = ''
88  for line in iter(sp.stdout.readline, b''):
89  line = line.decode(self.output_encoding).strip()
90  self.log.info(line)
91  sp.wait()
92  self.log.info("Command exited with "
93  "return code {0}".format(sp.returncode))
94 
95  return not sp.returncode
