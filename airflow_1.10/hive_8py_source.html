27  Gets the max partition for a table.
28 
29  :param schema: The hive schema the table lives in
30  :type schema: string
31  :param table: The hive table you are interested in, supports the dot
32  notation as in "my_database.my_table", if a dot is found,

58 def _closest_date(target_dt, date_list, before_target=None):
59  """
60  This function finds the date in a list closest to the target date.
61  An optional parameter can be given to get the closest before or after.
62 
63  :param target_dt: The target date
64  :type target_dt: datetime.date
65  :param date_list: The list of dates to search
66  :type date_list: datetime.date list
67  :param before_target: closest before or after the target
68  :type before_target: bool or None
69  :returns: The closest date
70  :rtype: datetime.date or None
71  """
72  fb = lambda d: target_dt - d if d <= target_dt else datetime.timedelta.max
73  fa = lambda d: d - target_dt if d >= target_dt else datetime.timedelta.max
74  fnone = lambda d: target_dt - d if d < target_dt else d - target_dt
75  if before_target is None:
76  return min(date_list, key=fnone).date()
77  if before_target:
78  return min(date_list, key=fb).date()
79  else:
80  return min(date_list, key=fa).date()

93  :type ds: datetime.date list
94  :param before: closest before (True), after (False) or either side of ds
95  :type before: bool or None
96  :returns: The closest date
97  :rtype: str or None
98 
99  >>> tbl = 'airflow.static_babynames_partitioned'
100  >>> closest_ds_partition(tbl, '2015-01-02')
101  '2015-01-01'
102  """
103  from airflow.hooks.hive_hooks import HiveMetastoreHook
104  if '.' in table:
105  schema, table = table.split('.')
106  hh = HiveMetastoreHook(metastore_conn_id=metastore_conn_id)
107  partitions = hh.get_partitions(schema=schema, table_name=table)
108  if not partitions:
109  return None
110  part_vals = [list(p.values())[0] for p in partitions]
111  if ds in part_vals:
112  return ds
113  else:
114  parts = [datetime.datetime.strptime(pv, '%Y-%m-%d')
115  for pv in part_vals]
116  target_dt = datetime.datetime.strptime(ds, '%Y-%m-%d')
117  closest_ds = _closest_date(target_dt, parts, before_target=before)
118  return closest_ds.isoformat()
