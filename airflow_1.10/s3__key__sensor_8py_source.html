28 class S3KeySensor(BaseSensorOperator):
29  """
30  Waits for a key (a file-like instance on S3) to be present in a S3 bucket.
31  S3 being a key/value it does not support folders. The path is just a key
32  a resource.
33 
34  :param bucket_key: The key being waited on. Supports full s3:// style url
35  or relative path from root level.
36  :type bucket_key: str
37  :param bucket_name: Name of the S3 bucket
38  :type bucket_name: str
39  :param wildcard_match: whether the bucket_key should be interpreted as a
40  Unix wildcard pattern
41  :type wildcard_match: bool
42  :param aws_conn_id: a reference to the s3 connection
43  :type aws_conn_id: str
44  """
