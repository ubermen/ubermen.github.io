37 class HDFSHook(BaseHook):
38  """
39  Interact with HDFS. This class is a wrapper around the snakebite library.
40 
41  :param hdfs_conn_id: Connection id to fetch connection info
42  :type conn_id: string
43  :param proxy_user: effective user for HDFS operations
44  :type proxy_user: string
45  :param autoconfig: use snakebite's automatically configured client
46  :type autoconfig: bool
47  """
48  def __init__(self, hdfs_conn_id='hdfs_default', proxy_user=None,
49  autoconfig=False):
50  if not snakebite_imported:
51  raise ImportError(
52  'This HDFSHook implementation requires snakebite, but '
53  'snakebite is not compatible with Python 3 '
54  '(as of August 2015). Please use Python 2 if you require '
55  'this hook -- or help by submitting a PR!')
56  self.hdfs_conn_id = hdfs_conn_id
57  self.proxy_user = proxy_user
58  self.autoconfig = autoconfig
59 
60  def get_conn(self):
61  """
62  Returns a snakebite HDFSClient object.
63  """
64  # When using HAClient, proxy_user must be the same, so is ok to always
65  # take the first.
66  effective_user = self.proxy_user
67  autoconfig = self.autoconfig
68  use_sasl = configuration.conf.get('core', 'security') == 'kerberos'
69 
70  try:
71  connections = self.get_connections(self.hdfs_conn_id)
72 
73  if not effective_user:
74  effective_user = connections[0].login
75  if not autoconfig:
76  autoconfig = connections[0].extra_dejson.get('autoconfig',
77  False)
78  hdfs_namenode_principal = connections[0].extra_dejson.get(
79  'hdfs_namenode_principal')
80  except AirflowException:
81  if not autoconfig:
82  raise
83 
84  if autoconfig:
85  # will read config info from $HADOOP_HOME conf files
86  client = AutoConfigClient(effective_user=effective_user,
87  use_sasl=use_sasl)
88  elif len(connections) == 1:
89  client = Client(connections[0].host, connections[0].port,
90  effective_user=effective_user, use_sasl=use_sasl,
91  hdfs_namenode_principal=hdfs_namenode_principal)
92  elif len(connections) > 1:
93  nn = [Namenode(conn.host, conn.port) for conn in connections]
94  client = HAClient(nn, effective_user=effective_user,
95  use_sasl=use_sasl,
96  hdfs_namenode_principal=hdfs_namenode_principal)
97  else:
98  raise HDFSHookException("conn_id doesn't exist in the repository "
99  "and autoconfig is not specified")
100 
101  return client
