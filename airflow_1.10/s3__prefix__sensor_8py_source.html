25 class S3PrefixSensor(BaseSensorOperator):
26  """
27  Waits for a prefix to exist. A prefix is the first part of a key,
28  thus enabling checking of constructs similar to glob airfl* or
29  SQL LIKE 'airfl%'. There is the possibility to precise a delimiter to
30  indicate the hierarchy or keys, meaning that the match will stop at that
31  delimiter. Current code accepts sane delimiters, i.e. characters that
32  are NOT special characters in the Python regex engine.
33 
34  :param bucket_name: Name of the S3 bucket
35  :type bucket_name: str
36  :param prefix: The prefix being waited on. Relative path from bucket root level.
37  :type prefix: str
38  :param delimiter: The delimiter intended to show hierarchy.
39  Defaults to '/'.
40  :type delimiter: str
41  """
42  template_fields = ('prefix', 'bucket_name')
