99  def execute(self, context):
100  hive = HiveCliHook(hive_cli_conn_id=self.hive_cli_conn_id)
101  self.log.info("Extracting data from Hive")
102  hive_table = 'druid.' + context['task_instance_key_str'].replace('.', '_')
103  sql = self.sql.strip().strip(';')
104  tblproperties = ''.join([", '{}' = '{}'"
105  .format(k, v)
106  for k, v in self.hive_tblproperties.items()])
107  hql = """\
108  SET mapred.output.compress=false;
109  SET hive.exec.compress.output=false;
110  DROP TABLE IF EXISTS {hive_table};
111  CREATE TABLE {hive_table}
112  ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t'
113  STORED AS TEXTFILE
114  TBLPROPERTIES ('serialization.null.format' = ''{tblproperties})
115  AS
116  {sql}
117  """.format(**locals())
118  self.log.info("Running command:\n %s", hql)
119  hive.run_cli(hql)
120 
121  m = HiveMetastoreHook(self.metastore_conn_id)
122
