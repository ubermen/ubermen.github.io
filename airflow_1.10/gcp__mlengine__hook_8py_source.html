25 def _poll_with_exponential_delay(request, max_n, is_done_func, is_error_func):
26  log = LoggingMixin().log
27 
28  for i in range(0, max_n):
29  try:
30  response = request.execute()
31  if is_error_func(response):
32  raise ValueError(
33  'The response contained an error: {}'.format(response)
34  )
35  elif is_done_func(response):
36  log.info('Operation is done: %s', response)
37  return response
38  else:
39  time.sleep((2**i) + (random.randint(0, 1000) / 1000))
40  except errors.HttpError as e:
41  if e.resp.status != 429:
42  log.info('Something went wrong. Not retrying: %s', format(e))
43  raise
44  else:
45  time.sleep((2**i) + (random.randint(0, 1000) / 1000))

60  def create_job(self, project_id, job, use_existing_job_fn=None):
61  """
62  Launches a MLEngine job and wait for it to reach a terminal state.
63 
64  :param project_id: The Google Cloud project id within which MLEngine
65  job will be launched.
66  :type project_id: string
67 
68  :param job: MLEngine Job object that should be provided to the MLEngine
69  API, such as: ::
70  {
71  'jobId': 'my_job_id',
72  'trainingInput': {
73  'scaleTier': 'STANDARD_1',
74  ...
75  }
76  }
77  :type job: dict
78 
79  :param use_existing_job_fn: In case that a MLEngine job with the same
80  job_id already exist, this method (if provided) will decide whether
81  we should use this existing job, continue waiting for it to finish
82  and returning the job object. It should accepts a MLEngine job
83  object, and returns a boolean value indicating whether it is OK to
84  reuse the existing job. If 'use_existing_job_fn' is not provided,
85  we by default reuse the existing MLEngine job.
86  :type use_existing_job_fn: function
87 
88  :return: The MLEngine job object if the job successfully reach a
89  terminal state (which might be FAILED or CANCELLED state).
90  :rtype: dict
91  """
92  request = self._mlengine.projects().jobs().create(
93  parent='projects/{}'.format(project_id),
94  body=job)
95  job_id = job['jobId']
96 
97  try:
98  request.execute()
99  except errors.HttpError as e:
100  # 409 means there is an existing job with the same job ID.
101  if e.resp.status == 409:
102  if use_existing_job_fn is not None:
103  existing_job = self._get_job(project_id, job_id)
104  if not use_existing_job_fn(existing_job):
105  self.log.error(
106  'Job with job_id %s already exist, but it does '
107  'not match our expectation: %s',
108  job_id, existing_job
109  )
110  raise
111  self.log.info(
112  'Job with job_id %s already exist. Will waiting for it to finish',
113  job_id
114  )
115  else:
116  self.log.error('Failed to create MLEngine job: {}'.format(e))
117  raise
118 
119  return self._wait_for_job_done(project_id, job_id)

121  def _get_job(self, project_id, job_id):
122  """
123  Gets a MLEngine job based on the job name.
124 
125  :return: MLEngine job object if succeed.
126  :rtype: dict
127 
128  Raises:
129  apiclient.errors.HttpError: if HTTP error is returned from server
130  """
131  job_name = 'projects/{}/jobs/{}'.format(project_id, job_id)
132  request = self._mlengine.projects().jobs().get(name=job_name)
133  while True:
134  try:
135  return request.execute()
136  except errors.HttpError as e:
137  if e.resp.status == 429:
138  # polling after 30 seconds when quota failure occurs
139  time.sleep(30)
140  else:
141  self.log.error('Failed to get MLEngine job: {}'.format(e))
142  raise

144  def _wait_for_job_done(self, project_id, job_id, interval=30):
145  """
146  Waits for the Job to reach a terminal state.
147 
148  This method will periodically check the job state until the job reach
149  a terminal state.
150 
151  Raises:
152  apiclient.errors.HttpError: if HTTP error is returned when getting
153  the job
154  """
155  assert interval > 0
156  while True:
157  job = self._get_job(project_id, job_id)
158  if job['state'] in ['SUCCEEDED', 'FAILED', 'CANCELLED']:
159  return job
160  time.sleep(interval)

162  def create_version(self, project_id, model_name, version_spec):
163  """
164  Creates the Version on Google Cloud ML Engine.
165

200  """
201  Lists all available versions of a model. Blocks until finished.
202  """
203  result = []
204  full_parent_name = 'projects/{}/models/{}'.format(
205  project_id, model_name)
206  request = self._mlengine.projects().models().versions().list(

208 
209  response = request.execute()

217  response = next_request.execute()
218  next_page_token = response.get('nextPageToken', None)
219  result.extend(response.get('versions', []))
