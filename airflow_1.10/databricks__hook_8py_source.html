67  def _parse_host(self, host):
68  """
69  The purpose of this function is to be robust to improper connections
70  settings provided by users, specifically in the host field.
71 
72 
73  For example -- when users supply ``https://xx.cloud.databricks.com`` as the
74  host, we must strip out the protocol to get the host.
75  >>> h = DatabricksHook()
76  >>> assert h._parse_host('https://xx.cloud.databricks.com') == \
77  'xx.cloud.databricks.com'
78 
79  In the case where users supply the correct ``xx.cloud.databricks.com`` as the
80  host, this function is a no-op.
81  >>> assert h._parse_host('xx.cloud.databricks.com') == 'xx.cloud.databricks.com'
82  """
83  urlparse_host = urlparse.urlparse(host).hostname
84  if urlparse_host:
85  # In this case, host = https://xx.cloud.databricks.com
86  return urlparse_host
87  else:
88  # In this case, host = xx.cloud.databricks.com
89  return host

91  def _do_api_call(self, endpoint_info, json):
92  """
93  Utility function to perform an API call with retries
94  :param endpoint_info: Tuple of method and endpoint
95  :type endpoint_info: (string, string)
96  :param json: Parameters for this API call.
97  :type json: dict
98  :return: If the api call returns a OK status code,
99  this function returns the response in JSON. Otherwise,
100  we throw an AirflowException.
101  :rtype: dict
102  """
103  method, endpoint = endpoint_info
104  url = 'https://{host}/{endpoint}'.format(
105  host=self._parse_host(self.databricks_conn.host),
106  endpoint=endpoint)
107  if 'token' in self.databricks_conn.extra_dejson:
108  self.log.info('Using token auth.')
109  auth = _TokenAuth(self.databricks_conn.extra_dejson['token'])
110  else:
111  self.log.info('Using basic auth.')
112  auth = (self.databricks_conn.login, self.databricks_conn.password)
113  if method == 'GET':
114  request_func = requests.get
115  elif method == 'POST':
116  request_func = requests.post
117  else:
118  raise AirflowException('Unexpected HTTP Method: ' + method)
119 
120  for attempt_num in range(1, self.retry_limit + 1):
121  try:
122  response = request_func(
123  url,
124  json=json,
125  auth=auth,
126  headers=USER_AGENT_HEADER,
127  timeout=self.timeout_seconds)
128  if response.status_code == requests.codes.ok:
129  return response.json()
130  else:
131  # In this case, the user probably made a mistake.
132  # Don't retry.
133  raise AirflowException('Response: {0}, Status Code: {1}'.format(
134  response.content, response.status_code))
135  except (requests_exceptions.ConnectionError,
136  requests_exceptions.Timeout) as e:
137  self.log.error(
138  'Attempt %s API Request to Databricks failed with reason: %s',
139  attempt_num, e
140  )
141  raise AirflowException(('API requests to Databricks failed {} times. ' +
142  'Giving up.').format(self.retry_limit))

186 class RunState:
187  """
188  Utility class for the run state concept of Databricks runs.
189  """
190  def __init__(self, life_cycle_state, result_state, state_message):
191  self.life_cycle_state = life_cycle_state
192  self.result_state = result_state
193  self.state_message = state_message
194 
195  @property
196  def is_terminal(self):
197  if self.life_cycle_state not in RUN_LIFE_CYCLE_STATES:
198  raise AirflowException(
199  ('Unexpected life cycle state: {}: If the state has '
200  'been introduced recently, please check the Databricks user '
201  'guide for troubleshooting information').format(
202  self.life_cycle_state))
203  return self.life_cycle_state in ('TERMINATED', 'SKIPPED', 'INTERNAL_ERROR')
