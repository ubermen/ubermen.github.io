109 def policy(task_instance):
110  """
111  This policy setting allows altering task instances right before they
112  are executed. It allows administrator to rewire some task parameters.
113 
114  Note that the ``TaskInstance`` object has an attribute ``task`` pointing
115  to its related task object, that in turns has a reference to the DAG
116  object. So you can use the attributes of all of these to define your
117  policy.
118 
119  To define policy, add a ``airflow_local_settings`` module
120  to your PYTHONPATH that defines this ``policy`` function. It receives
121  a ``TaskInstance`` object and can alter it where needed.
122 
123  Here are a few examples of how this can be useful:
124 
125  * You could enforce a specific queue (say the ``spark`` queue)
126  for tasks using the ``SparkOperator`` to make sure that these
127  task instances get wired to the right workers
128  * You could force all task instances running on an
129  ``execution_date`` older than a week old to run in a ``backfill``
130  pool.
131  * ...
132  """
133  pass

145 def configure_orm(disable_connection_pool=False):
146  log.debug("Setting up DB connection pool (PID %s)" % os.getpid())
147  global engine
148  global Session
149  engine_args = {}
150 
151  pool_connections = conf.getboolean('core', 'SQL_ALCHEMY_POOL_ENABLED')
152  if disable_connection_pool or not pool_connections:
153  engine_args['poolclass'] = NullPool
154  log.debug("settings.configure_orm(): Using NullPool")
155  elif 'sqlite' not in SQL_ALCHEMY_CONN:
156  # Engine args not supported by sqlite.
157  # If no config value is defined for the pool size, select a reasonable value.
158  # 0 means no limit, which could lead to exceeding the Database connection limit.
159  try:
160  pool_size = conf.getint('core', 'SQL_ALCHEMY_POOL_SIZE')
161  except conf.AirflowConfigException:
162  pool_size = 5
163 
164  # The DB server already has a value for wait_timeout (number of seconds after
165  # which an idle sleeping connection should be killed). Since other DBs may
166  # co-exist on the same server, SQLAlchemy should set its
167  # pool_recycle to an equal or smaller value.
168  try:
169  pool_recycle = conf.getint('core', 'SQL_ALCHEMY_POOL_RECYCLE')
170  except conf.AirflowConfigException:
171  pool_recycle = 1800
172 
173  log.info("setting.configure_orm(): Using pool settings. pool_size={}, "
174  "pool_recycle={}".format(pool_size, pool_recycle))
175  engine_args['pool_size'] = pool_size
176  engine_args['pool_recycle'] = pool_recycle
177 
178  engine = create_engine(SQL_ALCHEMY_CONN, **engine_args)
179  reconnect_timeout = conf.getint('core', 'SQL_ALCHEMY_RECONNECT_TIMEOUT')
180  setup_event_handlers(engine, reconnect_timeout)
181 
182  Session = scoped_session(
183  sessionmaker(autocommit=False, autoflush=False, bind=engine))

197  engine = None
