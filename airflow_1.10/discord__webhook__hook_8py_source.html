27 class DiscordWebhookHook(HttpHook):
28  """
29  This hook allows you to post messages to Discord using incoming webhooks.
30  Takes a Discord connection ID with a default relative webhook endpoint. The
31  default endpoint can be overridden using the webhook_endpoint parameter
32  (https://discordapp.com/developers/docs/resources/webhook).
33 
34  Each Discord webhook can be pre-configured to use a specific username and
35  avatar_url. You can override these defaults in this hook.
36 
37  :param http_conn_id: Http connection ID with host as "https://discord.com/api/" and
38  default webhook endpoint in the extra field in the form of
39  {"webhook_endpoint": "webhooks/{webhook.id}/{webhook.token}"}
40  :type http_conn_id: str
41  :param webhook_endpoint: Discord webhook endpoint in the form of
42  "webhooks/{webhook.id}/{webhook.token}"
43  :type webhook_endpoint: str
44  :param message: The message you want to send to your Discord channel
45  (max 2000 characters)
46  :type message: str
47  :param username: Override the default username of the webhook
48  :type username: str
49  :param avatar_url: Override the default avatar of the webhook
50  :type avatar_url: str
51  :param tts: Is a text-to-speech message
52  :type tts: bool
53  :param proxy: Proxy to use to make the Discord webhook call
54  :type proxy: str
55  """

76  def _get_webhook_endpoint(self, http_conn_id, webhook_endpoint):
77  """
78  Given a Discord http_conn_id, return the default webhook endpoint or override if a
79  webhook_endpoint is manually supplied.
80 
81  :param http_conn_id: The provided connection ID
82  :param webhook_endpoint: The manually provided webhook endpoint
83  :return: Webhook endpoint (str) to use
84  """
85  if webhook_endpoint:
86  endpoint = webhook_endpoint
87  elif http_conn_id:
88  conn = self.get_connection(http_conn_id)
89  extra = conn.extra_dejson
90  endpoint = extra.get('webhook_endpoint', '')
91  else:
92  raise AirflowException('Cannot get webhook endpoint: No valid Discord '
93  'webhook endpoint or http_conn_id supplied.')
94 
95  # make sure endpoint matches the expected Discord webhook format
96  if not re.match('^webhooks/[0-9]+/[a-zA-Z0-9_-]+$', endpoint):
97  raise AirflowException('Expected Discord webhook endpoint in the form '
98  'of "webhooks/{webhook.id}/{webhook.token}".')
99 
100  return endpoint
