84  unload_options = '\n\t\t\t'.join(self.unload_options)

86  if self.include_header:

88  self.schema, self.table)
89 
90  columns_query = """SELECT column_name
91  FROM information_schema.columns
92  WHERE table_schema = '{schema}'
93  AND table_name = '{table}'
94  ORDER BY ordinal_position
95  """.format(schema=self.schema,
96  table=self.table)
97 
98  cursor = self.hook.get_conn().cursor()
99  cursor.execute(columns_query)
100  rows = cursor.fetchall()
101  columns = [row[0] for row in rows]
102  column_names = ', '.join("{0}".format(c) for c in columns)
103  column_headers = ', '.join("\\'{0}\\'".format(c) for c in columns)
104  column_castings = ', '.join("CAST({0} AS text) AS {0}".format(c)
105  for c in columns)
106 
107  select_query = """SELECT {column_names} FROM
108  (SELECT 2 sort_order, {column_castings}
109  FROM {schema}.{table}
110  UNION ALL
111  SELECT 1 sort_order, {column_headers})
112  ORDER BY sort_order"""\
113  .format(column_names=column_names,
114  column_castings=column_castings,
115  column_headers=column_headers,
116  schema=self.schema,
117  table=self.table)
118  else:

120  .format(schema=self.schema,
121  table=self.table)

123  unload_query = """
