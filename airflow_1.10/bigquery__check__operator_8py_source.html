26 class BigQueryCheckOperator(CheckOperator):
27  """
28  Performs checks against BigQuery. The ``BigQueryCheckOperator`` expects
29  a sql query that will return a single row. Each value on that
30  first row is evaluated using python ``bool`` casting. If any of the
31  values return ``False`` the check is failed and errors out.
32 
33  Note that Python bool casting evals the following as ``False``:
34 
35  * ``False``
36  * ``0``
37  * Empty string (``""``)
38  * Empty list (``[]``)
39  * Empty dictionary or set (``{}``)
40 
41  Given a query like ``SELECT COUNT(*) FROM foo``, it will fail only if
42  the count ``== 0``. You can craft much more complex query that could,
43  for instance, check that the table has the same number of rows as
44  the source table upstream, or that the count of today's partition is
45  greater than yesterday's partition, or that a set of metrics are less
46  than 3 standard deviation for the 7 day average.
47 
48  This operator can be used as a data quality check in your pipeline, and
49  depending on where you put it in your DAG, you have the choice to
50  stop the critical path, preventing from
51  publishing dubious data, or on the side and receive email alterts
52  without stopping the progress of the DAG.
53 
54  :param sql: the sql to be executed
55  :type sql: string
56  :param bigquery_conn_id: reference to the BigQuery database
57  :type bigquery_conn_id: string
58  """
59

61  def __init__(
62  self,
63  sql,
64  bigquery_conn_id='bigquery_default',
65  *args,
66  **kwargs):
67  super(BigQueryCheckOperator, self).__init__(sql=sql, *args, **kwargs)

69  self.sql = sql

72  return BigQueryHook(bigquery_conn_id=self.bigquery_conn_id)

97 class BigQueryIntervalCheckOperator(IntervalCheckOperator):
98  """
99  Checks that the values of metrics given as SQL expressions are within
100  a certain tolerance of the ones from days_back before.
101 
102  This method constructs a query like so ::
103 
104  SELECT {metrics_threshold_dict_key} FROM {table}
105  WHERE {date_filter_column}=<date>
106 
107  :param table: the table name
108  :type table: str
109  :param days_back: number of days between ds and the ds we want to check
110  against. Defaults to 7 days
111  :type days_back: int
112  :param metrics_threshold: a dictionary of ratios indexed by metrics, for
113  example 'COUNT(*)': 1.5 would require a 50 percent or less difference
