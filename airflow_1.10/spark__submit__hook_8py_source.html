40  invalid connection_id is supplied, it will default to yarn.
41  :type conn_id: str

43  comma. Files will be placed in the working directory of each executor.

45  :type files: str
46  :param py_files: Additional python files used by the job, can be .zip, .egg or .py.
47  :type py_files: str

307  def submit(self, application="", **kwargs):
308  """
309  Remote Popen to execute the spark-submit job
310 
311  :param application: Submitted application, jar or py file
312  :type application: str
313  :param kwargs: extra arguments to Popen (see subprocess.Popen)
314  """
315  spark_submit_cmd = self._build_spark_submit_command(application)
316 
317  if hasattr(self, '_env'):
318  env = os.environ.copy()
319  env.update(self._env)
320  kwargs["env"] = env
321 
322  self._submit_sp = subprocess.Popen(spark_submit_cmd,
323  stdout=subprocess.PIPE,
324  stderr=subprocess.STDOUT,
325  bufsize=-1,
326  universal_newlines=True,
327  **kwargs)
328 
329  self._process_spark_submit_log(iter(self._submit_sp.stdout.readline, ''))
330  returncode = self._submit_sp.wait()
331 
332  # Check spark-submit return code. In Kubernetes mode, also check the value
333  # of exit code in the log, as it may differ.
334  if returncode or (self._is_kubernetes and self._spark_exit_code != 0):
335  raise AirflowException(
336  "Cannot execute: {}. Error code is: {}.".format(
337  spark_submit_cmd, returncode
338  )
339  )
340 
341  self.log.debug("Should track driver: {}".format(self._should_track_driver_status))
342 
343  # We want the Airflow job to wait until the Spark driver is finished
344  if self._should_track_driver_status:
345  if self._driver_id is None:
346  raise AirflowException(
347  "No driver id is known: something went wrong when executing " +
348  "the spark submit command"
349  )
350 
351  # We start with the SUBMITTED status as initial status
352  self._driver_status = "SUBMITTED"
353 
354  # Start tracking the driver status (blocking function)
355  self._start_driver_status_tracking()
356 
357  if self._driver_status != "FINISHED":
358  raise AirflowException(
359  "ERROR : Driver {} badly exited with status {}"
360  .format(self._driver_id, self._driver_status)
361  )

363  def _process_spark_submit_log(self, itr):
364  """
365  Processes the log files and extracts useful information out of it.
366 
367  If the deploy-mode is 'client', log the output of the submit command as those
368  are the output logs of the Spark worker directly.
369 
370  Remark: If the driver needs to be tracked for its status, the log-level of the
371  spark deploy needs to be at least INFO (log4j.logger.org.apache.spark.deploy=INFO)
372 
373  :param itr: An iterator which iterates over the input of the subprocess
374  """
375  # Consume the iterator
376  for line in itr:
377  line = line.strip()
378  # If we run yarn cluster mode, we want to extract the application id from
379  # the logs so we can kill the application when we stop it unexpectedly
380  if self._is_yarn and self._connection['deploy_mode'] == 'cluster':
381  match = re.search('(application[0-9_]+)', line)
382  if match:
383  self._yarn_application_id = match.groups()[0]
384  self.log.info("Identified spark driver id: %s",
385  self._yarn_application_id)
386 
387  # If we run Kubernetes cluster mode, we want to extract the driver pod id
388  # from the logs so we can kill the application when we stop it unexpectedly
389  elif self._is_kubernetes:
390  match = re.search('\s*pod name: ((.+?)-([a-z0-9]+)-driver)', line)
391  if match:
392  self._kubernetes_driver_pod = match.groups()[0]
393  self.log.info("Identified spark driver pod: %s",
394  self._kubernetes_driver_pod)
395 
396  # Store the Spark Exit code
397  match_exit_code = re.search('\s*Exit code: (\d+)', line)
398  if match_exit_code:
399  self._spark_exit_code = int(match_exit_code.groups()[0])
400 
401  # if we run in standalone cluster mode and we want to track the driver status
402  # we need to extract the driver id from the logs. This allows us to poll for
403  # the status using the driver id. Also, we can kill the driver when needed.
404  elif self._should_track_driver_status and not self._driver_id:
405  match_driver_id = re.search('(driver-[0-9\-]+)', line)
406  if match_driver_id:
407  self._driver_id = match_driver_id.groups()[0]
408  self.log.info("identified spark driver id: {}"
409  .format(self._driver_id))
410 
411  else:
412  self.log.info(line)
413 
414  self.log.debug("spark submit log: {}".format(line))

416  def _process_spark_status_log(self, itr):

433  def _start_driver_status_tracking(self):
434  """
435  Polls the driver based on self._driver_id to get the status.
436  Finish successfully when the status is FINISHED.
437  Finish failed when the status is ERROR/UNKNOWN/KILLED/FAILED.
438 
439  Possible status:
440  SUBMITTED: Submitted but not yet scheduled on a worker
441  RUNNING: Has been allocated to a worker to run
442  FINISHED: Previously ran and exited cleanly
443  RELAUNCHING: Exited non-zero or due to worker failure, but has not yet
444  started running again
445  UNKNOWN: The status of the driver is temporarily not known due to
446  master failure recovery
447  KILLED: A user manually killed this driver
448  FAILED: The driver exited non-zero and was not supervised
449  ERROR: Unable to run or restart due to an unrecoverable error
450  (e.g. missing jar file)
451  """
452 
453  # When your Spark Standalone cluster is not performing well
454  # due to misconfiguration or heavy loads.
455  # it is possible that the polling request will timeout.
456  # Therefore we use a simple retry mechanism.
457  missed_job_status_reports = 0
458  max_missed_job_status_reports = 10
459 
460  # Keep polling as long as the driver is processing
461  while self._driver_status not in ["FINISHED", "UNKNOWN",
462  "KILLED", "FAILED", "ERROR"]:
463 
464  # Sleep for 1 second as we do not want to spam the cluster
465  time.sleep(1)
466 
467  self.log.debug("polling status of spark driver with id {}"
468  .format(self._driver_id))
469 
470  poll_drive_status_cmd = self._build_track_driver_status_command()
471  status_process = subprocess.Popen(poll_drive_status_cmd,
472  stdout=subprocess.PIPE,
473  stderr=subprocess.STDOUT,
474  bufsize=-1,
475  universal_newlines=True)
476 
477  self._process_spark_status_log(iter(status_process.stdout.readline, ''))
478  returncode = status_process.wait()
479 
480  if returncode:
481  if missed_job_status_reports < max_missed_job_status_reports:
482  missed_job_status_reports = missed_job_status_reports + 1
483  else:
484  raise AirflowException(
485  "Failed to poll for the driver status {} times: returncode = {}"
486  .format(max_missed_job_status_reports, returncode)
487  )
