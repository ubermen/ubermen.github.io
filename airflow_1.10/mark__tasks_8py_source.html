30 def _create_dagruns(dag, execution_dates, state, run_id_template):
31  """
32  Infers from the dates which dag runs need to be created and does so.
33  :param dag: the dag to create dag runs for
34  :param execution_dates: list of execution dates to evaluate
35  :param state: the state to set the dag run to
36  :param run_id_template:the template for run id to be with the execution date
37  :return: newly created and existing dag runs for the execution dates supplied
38  """
39  # find out if we need to create any dag runs
40  drs = DagRun.find(dag_id=dag.dag_id, execution_date=execution_dates)
41  dates_to_create = list(set(execution_dates) - set([dr.execution_date for dr in drs]))
42 
43  for date in dates_to_create:
44  dr = dag.create_dagrun(
45  run_id=run_id_template.format(date.isoformat()),
46  execution_date=date,
47  start_date=timezone.utcnow(),
48  external_trigger=False,
49  state=state,
50  )
51  drs.append(dr)
52 
53  return drs

56 def set_state(task, execution_date, upstream=False, downstream=False,
57  future=False, past=False, state=State.SUCCESS, commit=False):
58  """
59  Set the state of a task instance and if needed its relatives. Can set state
60  for future tasks (calculated from execution_date) and retroactively
61  for past tasks. Will verify integrity of past dag runs in order to create
62  tasks that did not exist. It will not create dag runs that are missing
63  on the schedule (but it will as for subdag dag runs if needed).
64  :param task: the task from which to work. task.task.dag needs to be set
65  :param execution_date: the execution date from which to start looking
66  :param upstream: Mark all parents (upstream tasks)
67  :param downstream: Mark all siblings (downstream tasks) of task_id, including SubDags
68  :param future: Mark all future tasks on the interval of the dag up until
69  last execution date.
70  :param past: Retroactively mark all tasks starting from start_date of the DAG
71  :param state: State to which the tasks need to be set
72  :param commit: Commit tasks to be altered to the database
73  :return: list of tasks that have been created and updated
74  """
75  assert timezone.is_localized(execution_date)
76 
77  # microseconds are supported by the database, but is not handled
78  # correctly by airflow on e.g. the filesystem and in other places
79  execution_date = execution_date.replace(microsecond=0)
80 
81  assert task.dag is not None
82  dag = task.dag
83 
84  latest_execution_date = dag.latest_execution_date
85  assert latest_execution_date is not None
86 
87  # determine date range of dag runs and tasks to consider
88  end_date = latest_execution_date if future else execution_date
89 
90  if 'start_date' in dag.default_args:
91  start_date = dag.default_args['start_date']
92  elif dag.start_date:
93  start_date = dag.start_date
94  else:
95  start_date = execution_date
96 
97  start_date = execution_date if not past else start_date
98 
99  if dag.schedule_interval == '@once':
100  dates = [start_date]
101  else:
102  dates = dag.date_range(start_date=start_date, end_date=end_date)
103 
104  # find relatives (siblings = downstream, parents = upstream) if needed
105  task_ids = [task.task_id]
106  if downstream:
107  relatives = task.get_flat_relatives(upstream=False)
108  task_ids += [t.task_id for t in relatives]
109  if upstream:
110  relatives = task.get_flat_relatives(upstream=True)
111  task_ids += [t.task_id for t in relatives]
112 
113  # verify the integrity of the dag runs in case a task was added or removed
114  # set the confirmed execution dates as they might be different
115  # from what was provided
116  confirmed_dates = []
117  drs = DagRun.find(dag_id=dag.dag_id, execution_date=dates)
118  for dr in drs:
119  dr.dag = dag
120  dr.verify_integrity()
121  confirmed_dates.append(dr.execution_date)
122 
123  # go through subdagoperators and create dag runs. We will only work
124  # within the scope of the subdag. We wont propagate to the parent dag,
125  # but we will propagate from parent to subdag.
126  session = Session()
127  dags = [dag]
128  sub_dag_ids = []
129  while len(dags) > 0:
130  current_dag = dags.pop()
131  for task_id in task_ids:
132  if not current_dag.has_task(task_id):
133  continue
134 
135  current_task = current_dag.get_task(task_id)
136  if isinstance(current_task, SubDagOperator):
137  # this works as a kind of integrity check
138  # it creates missing dag runs for subdagoperators,
139  # maybe this should be moved to dagrun.verify_integrity
140  drs = _create_dagruns(current_task.subdag,
141  execution_dates=confirmed_dates,
142  state=State.RUNNING,
143  run_id_template=BackfillJob.ID_FORMAT_PREFIX)
144 
145  for dr in drs:
146  dr.dag = current_task.subdag
147  dr.verify_integrity()
148  if commit:
149  dr.state = state
150  session.merge(dr)
151 
152  dags.append(current_task.subdag)
153  sub_dag_ids.append(current_task.subdag.dag_id)
154 
155  # now look for the task instances that are affected
156  TI = TaskInstance
157 
158  # get all tasks of the main dag that will be affected by a state change
159  qry_dag = session.query(TI).filter(
160  TI.dag_id == dag.dag_id,
161  TI.execution_date.in_(confirmed_dates),
162  TI.task_id.in_(task_ids)).filter(
163  or_(TI.state.is_(None),
164  TI.state != state)
165  )
166 
167  # get *all* tasks of the sub dags
168  if len(sub_dag_ids) > 0:
169  qry_sub_dag = session.query(TI).filter(
170  TI.dag_id.in_(sub_dag_ids),
171  TI.execution_date.in_(confirmed_dates)).filter(
172  or_(TI.state.is_(None),
173  TI.state != state)
174  )
175 
176  if commit:
177  tis_altered = qry_dag.with_for_update().all()
178  if len(sub_dag_ids) > 0:
179  tis_altered += qry_sub_dag.with_for_update().all()
180  for ti in tis_altered:
181  ti.state = state
182  session.commit()
183  else:
184  tis_altered = qry_dag.all()
185  if len(sub_dag_ids) > 0:
186  tis_altered += qry_sub_dag.all()
187 
188  session.expunge_all()
189  session.close()
190 
191  return tis_altered

194 def set_dag_run_state(dag, execution_date, state=State.SUCCESS, commit=False):
195  """
196  Set the state of a dag run and all task instances associated with the dag
197  run for a specific execution date.
198  :param dag: the DAG of which to alter state
199  :param execution_date: the execution date from which to start looking
200  :param state: the state to which the DAG need to be set
201  :param commit: commit DAG and tasks to be altered to the database
202  :return: list of tasks that have been created and updated
203  :raises: AssertionError if dag or execution_date is invalid
204  """
205  res = []
206 
207  if not dag or not execution_date:
208  return res
209 
210  # Mark all task instances in the dag run
211  for task in dag.tasks:
212  task.dag = dag
213  new_state = set_state(task=task, execution_date=execution_date,
214  state=state, commit=commit)
215  res.extend(new_state)
216 
217  # Mark the dag run
218  if commit:
219  drs = DagRun.find(dag.dag_id, execution_date=execution_date)
220  for dr in drs:
221  dr.dag = dag
222  dr.update_state()
223 
224  return res
