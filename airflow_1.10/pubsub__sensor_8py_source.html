25 class PubSubPullSensor(BaseSensorOperator):
26  """Pulls messages from a PubSub subscription and passes them through XCom.
27 
28  This sensor operator will pull up to ``max_messages`` messages from the
29  specified PubSub subscription. When the subscription returns messages,
30  the poke method's criteria will be fulfilled and the messages will be
31  returned from the operator and passed through XCom for downstream tasks.
32 
33  If ``ack_messages`` is set to True, messages will be immediately
34  acknowledged before being returned, otherwise, downstream tasks will be
35  responsible for acknowledging them.
36 
37  ``project`` and ``subscription`` are templated so you can use
38  variables in them.
39  """
40  template_fields = ['project', 'subscription']
41  ui_color = '#ff7f50'
