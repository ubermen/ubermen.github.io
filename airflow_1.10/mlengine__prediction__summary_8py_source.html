17 """A template called by DataFlowPythonOperator to summarize BatchPrediction.
18 
19 It accepts a user function to calculate the metric(s) per instance in
20 the prediction results, then aggregates to output as a summary.
21 
22 Args:
23  --prediction_path:
24  The GCS folder that contains BatchPrediction results, containing
25  prediction.results-NNNNN-of-NNNNN files in the json format.
26  Output will be also stored in this folder, as 'prediction.summary.json'.
27 
28  --metric_fn_encoded:
29  An encoded function that calculates and returns a tuple of metric(s)
30  for a given instance (as a dictionary). It should be encoded
31  via base64.b64encode(dill.dumps(fn, recurse=True)).
32 
33  --metric_keys:
34  A comma-separated key(s) of the aggregated metric(s) in the summary
35  output. The order and the size of the keys must match to the output
36  of metric_fn.
37  The summary will have an additional key, 'count', to represent the
38  total number of instances, so the keys shouldn't include 'count'.
39 
40 # Usage example:
41 def get_metric_fn():
42  import math # all imports must be outside of the function to be passed.
43  def metric_fn(inst):
44  label = float(inst["input_label"])
45  classes = float(inst["classes"])
46  prediction = float(inst["scores"][1])
47  log_loss = math.log(1 + math.exp(
48  -(label * 2 - 1) * math.log(prediction / (1 - prediction))))
49  squared_err = (classes-label)**2
50  return (log_loss, squared_err)
51  return metric_fn
52 metric_fn_encoded = base64.b64encode(dill.dumps(get_metric_fn(), recurse=True))

112 @beam.ptransform_fn
113 def MakeSummary(pcoll, metric_fn, metric_keys): # pylint: disable=invalid-name
114  return (
115  pcoll
116  | "ApplyMetricFnPerInstance" >> beam.Map(metric_fn)
117  | "PairWith1" >> beam.Map(lambda tup: tup + (1,))
118  | "SumTuple" >> beam.CombineGlobally(beam.combiners.TupleCombineFn(
119  *([sum] * (len(metric_keys) + 1))))
120  | "AverageAndMakeDict" >> beam.Map(
121  lambda tup: dict(
122  [(name, tup[i]/tup[-1]) for i, name in enumerate(metric_keys)] +
123  [("count", tup[-1])])))
