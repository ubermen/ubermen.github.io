82  def wait_for_done(self):
83  while True:
84  if self._job and 'currentState' in self._job:
85  if 'JOB_STATE_DONE' == self._job['currentState']:
86  return True
87  elif 'JOB_STATE_RUNNING' == self._job['currentState'] and \
88  'JOB_TYPE_STREAMING' == self._job['type']:
89  return True
90  elif 'JOB_STATE_FAILED' == self._job['currentState']:
91  raise Exception("Google Cloud Dataflow job {} has failed.".format(
92  self._job['name']))
93  elif 'JOB_STATE_CANCELLED' == self._job['currentState']:
94  raise Exception("Google Cloud Dataflow job {} was cancelled.".format(
95  self._job['name']))
96  elif 'JOB_STATE_RUNNING' == self._job['currentState']:
97  time.sleep(self._poll_sleep)
98  elif 'JOB_STATE_PENDING' == self._job['currentState']:
99  time.sleep(15)
100  else:
101  self.log.debug(str(self._job))
102  raise Exception(
103  "Google Cloud Dataflow job {} was unknown state: {}".format(
104  self._job['name'], self._job['currentState']))
105  else:
106  time.sleep(15)

114 class _Dataflow(LoggingMixin):
115  def __init__(self, cmd):
116  self.log.info("Running command: %s", ' '.join(cmd))
117  self._proc = subprocess.Popen(

124  def _line(self, fd):
125  if fd == self._proc.stderr.fileno():
126  lines = self._proc.stderr.readlines()
127  for line in lines:
128  self.log.warning(line[:-1])
129  if lines:
130  return lines[-1]
131  if fd == self._proc.stdout.fileno():
132  line = self._proc.stdout.readline()
133  return line

141  def wait_for_done(self):
142  reads = [self._proc.stderr.fileno(), self._proc.stdout.fileno()]
143  self.log.info("Start waiting for DataFlow process to complete.")
144  while self._proc.poll() is None:
145  ret = select.select(reads, [], [], 5)
146  if ret is not None:
147  for fd in ret[0]:
148  line = self._line(fd)
149  if line:
150  self.log.debug(line[:-1])
151  else:
152  self.log.info("Waiting for DataFlow process to complete.")
153  if self._proc.returncode is not 0:
154  raise Exception("DataFlow failed with return code {}".format(
155  self._proc.returncode))
