28 class HiveToDynamoDBTransferOperator(BaseOperator):
29  """
30  Moves data from Hive to DynamoDB, note that for now the data is loaded
31  into memory before being pushed to DynamoDB, so this operator should
32  be used for smallish amount of data.
33 
34  :param sql: SQL query to execute against the hive database. (templated)
35  :type sql: str
36  :param table_name: target DynamoDB table
37  :type table_name: str
38  :param table_keys: partition key and sort key
39  :type table_keys: list
40  :param pre_process: implement pre-processing of source data
41  :type pre_process: function
42  :param pre_process_args: list of pre_process function arguments
43  :type pre_process_args: list
44  :param pre_process_kwargs: dict of pre_process function arguments
45  :type pre_process_kwargs: dict
46  :param region_name: aws region name (example: us-east-1)
47  :type region_name: str
48  :param schema: hive database schema
49  :type schema: str
50  :param hiveserver2_conn_id: source hive connection
51  :type hiveserver2_conn_id: str
52  :param aws_conn_id: aws connection
53  :type aws_conn_id: str
54  """

88

90  self.log.info(self.sql)
91 
92  data = hive.get_pandas_df(self.sql, schema=self.schema)
93  dynamodb = AwsDynamoDBHook(aws_conn_id=self.aws_conn_id,
94  table_name=self.table_name,
95  table_keys=self.table_keys,
96  region_name=self.region_name)
97 
98  self.log.info('Inserting rows into dynamodb')
99 
100  if self.pre_process is None:
101  dynamodb.write_batch_data(
102  json.loads(data.to_json(orient='records')))
103  else:
104  dynamodb.write_batch_data(
105  self.pre_process(data=data,
106  args=self.pre_process_args,
107  kwargs=self.pre_process_kwargs))
108 
109  self.log.info('Done.')
