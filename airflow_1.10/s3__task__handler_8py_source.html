26 class S3TaskHandler(FileTaskHandler, LoggingMixin):
27  """
28  S3TaskHandler is a python log handler that handles and reads
29  task instance logs. It extends airflow FileTaskHandler and
30  uploads to and reads from S3 remote storage.
31  """
32  def __init__(self, base_log_folder, s3_log_folder, filename_template):
33  super(S3TaskHandler, self).__init__(base_log_folder, filename_template)
34  self.remote_base = s3_log_folder

36  self._hook = None
37  self.closed = False

40  def _build_hook(self):
41  remote_conn_id = configuration.conf.get('core', 'REMOTE_LOG_CONN_ID')
42  try:
43  from airflow.hooks.S3_hook import S3Hook
44  return S3Hook(remote_conn_id)
45  except:
46  self.log.error(
47  'Could not create an S3Hook with connection id "%s". '
48  'Please make sure that airflow[s3] is installed and '
49  'the S3 connection exists.', remote_conn_id
50  )

53  def hook(self):
54  if self._hook is None:
55  self._hook = self._build_hook()
56  return self._hook

58  def set_context(self, ti):
59  super(S3TaskHandler, self).set_context(ti)
60  # Local location and remote location is needed to open and

62  self.log_relative_path = self._render_filename(ti, ti.try_number)

65  def close(self):
66  """
67  Close and upload local log file to remote storage S3.
68  """
69  # When application exit, system shuts down all handlers by
70  # calling close method. Here we check if logger is already
71  # closed to prevent uploading the log to remote storage multiple
72  # times when `logging.shutdown` is called.
73  if self.closed:
74  return
75 
76  super(S3TaskHandler, self).close()
77 
78  if not self.upload_on_close:
79  return
80 
81  local_loc = os.path.join(self.local_base, self.log_relative_path)
82  remote_loc = os.path.join(self.remote_base, self.log_relative_path)
83  if os.path.exists(local_loc):
84  # read log and remove old logs to get just the latest additions
85  with open(local_loc, 'r') as logfile:
86  log = logfile.read()
87  self.s3_write(log, remote_loc)
88 
89  # Mark closed so we don't double write if close is called twice
90  self.closed = True

92  def _read(self, ti, try_number, metadata=None):
93  """
94  Read logs of given task instance and try_number from S3 remote storage.
95  If failed, read the log from task instance host machine.
96  :param ti: task instance object
97  :param try_number: task instance try_number to read logs from
98  :param metadata: log metadata,
99  can be used for steaming log reading and auto-tailing.
100  """
101  # Explicitly getting log relative path is necessary as the given
102  # task instance might be different than task instance passed in
103  # in set_context method.
104  log_relative_path = self._render_filename(ti, try_number)
105  remote_loc = os.path.join(self.remote_base, log_relative_path)
106 
107  if self.s3_log_exists(remote_loc):
108  # If S3 remote file exists, we do not fetch logs from task instance
109  # local machine even if there are errors reading remote logs, as
110  # returned remote_log will contain error messages.
111  remote_log = self.s3_read(remote_loc, return_error=True)
112  log = '*** Reading remote log from {}.\n{}\n'.format(
113  remote_loc, remote_log)
114  return log, {'end_of_log': True}
115  else:
116  return super(S3TaskHandler, self)._read(ti, try_number)

118  def s3_log_exists(self, remote_log_location):
119  """
120  Check if remote_log_location exists in remote storage
121  :param remote_log_location: log's location in remote storage
122  :return: True if location exists else False
123  """
124  try:
125  return self.hook.get_key(remote_log_location) is not None
126  except Exception:
127  pass
128  return False

130  def s3_read(self, remote_log_location, return_error=False):
131  """
132  Returns the log found at the remote_log_location. Returns '' if no
133  logs are found or there is an error.
134  :param remote_log_location: the log's location in remote storage
135  :type remote_log_location: string (path)
136  :param return_error: if True, returns a string error message if an
137  error occurs. Otherwise returns '' when an error occurs.
138  :type return_error: bool
139  """
140  try:
141  return self.hook.read_key(remote_log_location)
142  except:
143  msg = 'Could not read logs from {}'.format(remote_log_location)
144  self.log.exception(msg)
145  # return error if needed
146  if return_error:
147  return msg

149  def s3_write(self, log, remote_log_location, append=True):
150  """
151  Writes the log to the remote_log_location. Fails silently if no hook
152  was created.
153  :param log: the log to write to the remote_log_location
154  :type log: string
155  :param remote_log_location: the log's location in remote storage
156  :type remote_log_location: string (path)
157  :param append: if False, any existing log file is overwritten. If True,
158  the new log is appended to any existing logs.
159  :type append: bool
160  """
161  if append and self.s3_log_exists(remote_log_location):
162  old_log = self.s3_read(remote_log_location)
163  log = '\n'.join([old_log, log]) if old_log else log
164 
165  try:
166  self.hook.load_string(
167  log,
168  key=remote_log_location,
169  replace=True,
170  encrypt=configuration.conf.getboolean('core', 'ENCRYPT_S3_LOGS'),
171  )
172  except:
173  self.log.exception('Could not write logs to %s', remote_log_location)
