68 class CeleryExecutor(BaseExecutor):
69  """
70  CeleryExecutor is recommended for production use of Airflow. It allows
71  distributing the execution of task instances to multiple worker nodes.
72 
73  Celery is a simple, flexible and reliable distributed system to process
74  vast amounts of messages, while providing operations with the tools
75  required to maintain such a system.
76  """
77  def start(self):
78  self.tasks = {}
79  self.last_state = {}
80 
81  def execute_async(self, key, command,
82  queue=DEFAULT_CELERY_CONFIG['task_default_queue'],
83  executor_config=None):
84  self.log.info("[celery] queuing {key} through celery, "
85  "queue={queue}".format(**locals()))
86  self.tasks[key] = execute_command.apply_async(
87  args=[command], queue=queue)
88  self.last_state[key] = celery_states.PENDING
89 
90  def sync(self):
91  self.log.debug("Inquiring about %s celery task(s)", len(self.tasks))
92  for key, async in list(self.tasks.items()):
93  try:
94  state = async.state
95  if self.last_state[key] != state:
96  if state == celery_states.SUCCESS:
97  self.success(key)
98  del self.tasks[key]
99  del self.last_state[key]
100  elif state == celery_states.FAILURE:
101  self.fail(key)
102  del self.tasks[key]
103  del self.last_state[key]
104  elif state == celery_states.REVOKED:
105  self.fail(key)
106  del self.tasks[key]
107  del self.last_state[key]
108  else:
109  self.log.info("Unexpected state: %s", async.state)
110  self.last_state[key] = async.state
111  except Exception as e:
112  self.log.error("Error syncing the celery executor, ignoring it:")
113  self.log.exception(e)
114 
115  def end(self, synchronous=False):
116  if synchronous:
117  while any([
118  async.state not in celery_states.READY_STATES
119  for async in self.tasks.values()]):
120  time.sleep(5)
121  self.sync()
