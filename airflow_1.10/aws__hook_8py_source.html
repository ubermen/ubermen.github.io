29 def _parse_s3_config(config_file_name, config_format='boto', profile=None):
30  """
31  Parses a config file for s3 credentials. Can currently
32  parse boto, s3cmd.conf and AWS SDK config formats
33 
34  :param config_file_name: path to the config file
35  :type config_file_name: str
36  :param config_format: config type. One of "boto", "s3cmd" or "aws".
37  Defaults to "boto"
38  :type config_format: str
39  :param profile: profile name in AWS type config file
40  :type profile: str
41  """
42  config = configparser.ConfigParser()
43  if config.read(config_file_name): # pragma: no cover
44  sections = config.sections()
45  else:
46  raise AirflowException("Couldn't read {0}".format(config_file_name))
47  # Setting option names depending on file format
48  if config_format is None:
49  config_format = 'boto'
50  conf_format = config_format.lower()
51  if conf_format == 'boto': # pragma: no cover
52  if profile is not None and 'profile ' + profile in sections:
53  cred_section = 'profile ' + profile
54  else:
55  cred_section = 'Credentials'
56  elif conf_format == 'aws' and profile is not None:
57  cred_section = profile
58  else:
59  cred_section = 'default'
60  # Option names
61  if conf_format in ('boto', 'aws'): # pragma: no cover
62  key_id_option = 'aws_access_key_id'
63  secret_key_option = 'aws_secret_access_key'
64  # security_token_option = 'aws_security_token'
65  else:
66  key_id_option = 'access_key'
67  secret_key_option = 'secret_key'
68  # Actual Parsing
69  if cred_section not in sections:
70  raise AirflowException("This config file format is not recognized")
71  else:
72  try:
73  access_key = config.get(cred_section, key_id_option)
74  secret_key = config.get(cred_section, secret_key_option)
75  except:
76  logging.warning("Option Error in parsing s3 config file")
77  raise
78  return access_key, secret_key
