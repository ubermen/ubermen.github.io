109  def run_command(self, run_with, join_args=False):
110  """
111  Run the task command
112 
113  :param run_with: list of tokens to run the task command with
114  E.g. ['bash', '-c']
115  :type run_with: list
116  :param join_args: whether to concatenate the list of command tokens
117  E.g. ['airflow', 'run'] vs ['airflow run']
118  :param join_args: bool
119  :return: the process that was run
120  :rtype: subprocess.Popen
121  """
122  cmd = [" ".join(self._command)] if join_args else self._command
123  full_cmd = run_with + cmd
124  self.log.info('Running: %s', full_cmd)
125  proc = subprocess.Popen(
126  full_cmd,
127  stdout=subprocess.PIPE,
128  stderr=subprocess.STDOUT,
129  universal_newlines=True,
130  close_fds=True,
131  env=os.environ.copy(),
132  preexec_fn=os.setsid
133  )
134 
135  # Start daemon thread to read subprocess logging output
136  log_reader = threading.Thread(

138  args=(proc.stdout,),
139  )
140  log_reader.daemon = True
141  log_reader.start()
