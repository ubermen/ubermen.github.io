51  def call(self, path, query=None, get_all_pages=True, side_loading=False):
52  """
53  Call Zendesk API and return results
54 
55  :param path: The Zendesk API to call
56  :param query: Query parameters
57  :param get_all_pages: Accumulate results over all pages before
58  returning. Due to strict rate limiting, this can often timeout.
59  Waits for recommended period between tries after a timeout.
60  :param side_loading: Retrieve related records as part of a single
61  request. In order to enable side-loading, add an 'include'
62  query parameter containing a comma-separated list of resources
63  to load. For more information on side-loading see
64  https://developer.zendesk.com/rest_api/docs/core/side_loading
65  """
66  zendesk = self.get_conn()
67  first_request_successful = False
68 
69  while not first_request_successful:
70  try:
71  results = zendesk.call(path, query)
72  first_request_successful = True
73  except RateLimitError as rle:
74  self.__handle_rate_limit_exception(rle)
75 
76  # Find the key with the results
77  keys = [path.split("/")[-1].split(".json")[0]]
78  next_page = results['next_page']
79  if side_loading:
80  keys += query['include'].split(',')
81  results = {key: results[key] for key in keys}
82 
83  if get_all_pages:
84  while next_page is not None:
85  try:
86  # Need to split because the next page URL has
87  # `github.zendesk...`
88  # in it, but the call function needs it removed.
89  next_url = next_page.split(self.__url)[1]
90  self.log.info("Calling %s", next_url)
91  more_res = zendesk.call(next_url)
92  for key in results:
93  results[key].extend(more_res[key])
94  if next_page == more_res['next_page']:
95  # Unfortunately zdesk doesn't always throw ZendeskError
96  # when we are done getting all the data. Sometimes the
97  # next just refers to the current set of results.
98  # Hence, need to deal with this special case
99  break
100  else:
101  next_page = more_res['next_page']
102  except RateLimitError as rle:
103  self.__handle_rate_limit_exception(rle)
104  except ZendeskError as ze:
105  if b"Use a start_time older than 5 minutes" in ze.msg:
106  # We have pretty up to date data
107  break
108  else:
109  raise ze
110 
111  return results
