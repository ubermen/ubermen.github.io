127  def _write_local_data_files(self, cursor):
128  """
129  Takes a cursor, and writes results to a local file.
130 
131  :return: A dictionary where keys are filenames to be used as object
132  names in GCS, and values are file handles to local files that
133  contain the data for the GCS objects.
134  """
135  schema = list(map(lambda schema_tuple: schema_tuple[0], cursor.description))
136  file_no = 0
137  tmp_file_handle = NamedTemporaryFile(delete=True)
138  tmp_file_handles = {self.filename.format(file_no): tmp_file_handle}
139 
140  for row in cursor:
141  # Convert datetime objects to utc seconds, and decimals to floats
142  row = map(self.convert_types, row)
143  row_dict = dict(zip(schema, row))
144 
145  s = json.dumps(row_dict, sort_keys=True)
146  if PY3:
147  s = s.encode('utf-8')
148  tmp_file_handle.write(s)
149 
150  # Append newline to make dumps BigQuery compatible.
151  tmp_file_handle.write(b'\n')
152 
153  # Stop if the file exceeds the file size limit.
154  if tmp_file_handle.tell() >= self.approx_max_file_size_bytes:
155  file_no += 1
156  tmp_file_handle = NamedTemporaryFile(delete=True)
157  tmp_file_handles[self.filename.format(file_no)] = tmp_file_handle
158 
159  return tmp_file_handles

161  def _write_local_schema_file(self, cursor):
162  """
163  Takes a cursor, and writes the BigQuery schema for the results to a
164  local file system.
165 
166  :return: A dictionary where key is a filename to be used as an object
167  name in GCS, and values are file handles to local files that
168  contains the BigQuery schema fields in .json format.
169  """
170  schema = []
171  for field in cursor.description:
172  # See PEP 249 for details about the description tuple.
173  field_name = field[0]
174  field_type = self.type_map(field[1])
175  field_mode = 'REPEATED' if field[1] in (1009, 1005, 1007,
176  1016) else 'NULLABLE'
177  schema.append({
178  'name': field_name,
179  'type': field_type,
180  'mode': field_mode,
181  })
182 
183  self.log.info('Using schema for %s: %s', self.schema_filename, schema)
184  tmp_schema_file_handle = NamedTemporaryFile(delete=True)
185  s = json.dumps(schema, sort_keys=True)
186  if PY3:
187  s = s.encode('utf-8')
188  tmp_schema_file_handle.write(s)
189  return {self.schema_filename: tmp_schema_file_handle}

191  def _upload_to_gcs(self, files_to_upload):
192  """
193  Upload all of the file splits (and optionally the schema .json file) to
194  Google Cloud Storage.
195  """
196  hook = GoogleCloudStorageHook(
197  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
198  delegate_to=self.delegate_to)
199  for object, tmp_file_handle in files_to_upload.items():
200  hook.upload(self.bucket, object, tmp_file_handle.name,
201  'application/json')

203  @classmethod
204  def convert_types(cls, value):
205  """
206  Takes a value from Postgres, and converts it to a value that's safe for
207  JSON/Google Cloud Storage/BigQuery. Dates are converted to UTC seconds.
208  Decimals are converted to floats. Times are converted to seconds.
209  """
210  if type(value) in (datetime.datetime, datetime.date):
211  return time.mktime(value.timetuple())
212  elif type(value) == datetime.time:
213  formated_time = time.strptime(str(value), "%H:%M:%S")
214  return datetime.timedelta(
215  hours=formated_time.tm_hour,
216  minutes=formated_time.tm_min,
217  seconds=formated_time.tm_sec).seconds
218  elif isinstance(value, Decimal):
219  return float(value)
220  else:
221  return value
