111  def heartbeat(self):
112  # Triggering new jobs
113  if not self.parallelism:
114  open_slots = len(self.queued_tasks)
115  else:
116  open_slots = self.parallelism - len(self.running)
117 
118  self.log.debug("%s running task instances", len(self.running))
119  self.log.debug("%s in queue", len(self.queued_tasks))
120  self.log.debug("%s open slots", open_slots)
121 
122  sorted_queue = sorted(
123  [(k, v) for k, v in self.queued_tasks.items()],
124  key=lambda x: x[1][1],
125  reverse=True)
126  for i in range(min((open_slots, len(self.queued_tasks)))):
127  key, (command, _, queue, ti) = sorted_queue.pop(0)
128  # TODO(jlowin) without a way to know what Job ran which tasks,
129  # there is a danger that another Job started running a task
130  # that was also queued to this executor. This is the last chance
131  # to check if that happened. The most probable way is that a
132  # Scheduler tried to run a task that was originally queued by a
133  # Backfill. This fix reduces the probability of a collision but
134  # does NOT eliminate it.
135  self.queued_tasks.pop(key)
136  ti.refresh_from_db()
137  if ti.state != State.RUNNING:
138  self.running[key] = command
139  self.execute_async(key=key,
140  command=command,
141  queue=queue,
142  executor_config=ti.executor_config)
143  else:
144  self.logger.info(
145  'Task is already running, not sending to '
146  'executor: {}'.format(key))
147 
148  # Calling child class sync method
149  self.log.debug("Calling the %s sync method", self.__class__)
150  self.sync()

163  def get_event_buffer(self, dag_ids=None):
164  """
165  Returns and flush the event buffer. In case dag_ids is specified
166  it will only return and flush events for the given dag_ids. Otherwise
167  it returns and flushes all
168 
169  :param dag_ids: to dag_ids to return events for, if None returns all
170  :return: a dict of events

172  cleared_events = dict()

176  else:

178  dag_id, _, _ = key
