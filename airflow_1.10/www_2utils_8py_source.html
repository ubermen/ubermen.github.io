102 def generate_pages(current_page, num_of_pages,
103  search=None, showPaused=None, window=7):
104  """
105  Generates the HTML for a paging component using a similar logic to the paging
106  auto-generated by Flask managed views. The paging component defines a number of
107  pages visible in the pager (window) and once the user goes to a page beyond the
108  largest visible, it would scroll to the right the page numbers and keeps the
109  current one in the middle of the pager component. When in the last pages,
110  the pages won't scroll and just keep moving until the last page. Pager also contains
111  <first, previous, ..., next, last> pages.
112  This component takes into account custom parameters such as search and showPaused,
113  which could be added to the pages link in order to maintain the state between
114  client and server. It also allows to make a bookmark on a specific paging state.
115  :param current_page:
116  the current page number, 0-indexed
117  :param num_of_pages:
118  the total number of pages
119  :param search:
120  the search query string, if any
121  :param showPaused:
122  false if paused dags will be hidden, otherwise true to show them
123  :param window:
124  the number of pages to be shown in the paging component (7 default)
125  :return:
126  the HTML string of the paging component
127  """
128 
129  void_link = 'javascript:void(0)'
130  first_node = """<li class="paginate_button {disabled}" id="dags_first">
131  <a href="{href_link}" aria-controls="dags" data-dt-idx="0" tabindex="0">&laquo;</a>
132 </li>"""
133 
134  previous_node = """<li class="paginate_button previous {disabled}" id="dags_previous">
135  <a href="{href_link}" aria-controls="dags" data-dt-idx="0" tabindex="0">&lt;</a>
136 </li>"""
137 
138  next_node = """<li class="paginate_button next {disabled}" id="dags_next">
139  <a href="{href_link}" aria-controls="dags" data-dt-idx="3" tabindex="0">&gt;</a>
140 </li>"""
141 
142  last_node = """<li class="paginate_button {disabled}" id="dags_last">
143  <a href="{href_link}" aria-controls="dags" data-dt-idx="3" tabindex="0">&raquo;</a>
144 </li>"""
145 
146  page_node = """<li class="paginate_button {is_active}">
147  <a href="{href_link}" aria-controls="dags" data-dt-idx="2" tabindex="0">{page_num}</a>
148 </li>"""
149 
150  output = ['<ul class="pagination" style="margin-top:0px;">']
151 
152  is_disabled = 'disabled' if current_page <= 0 else ''
153  output.append(first_node.format(href_link="?{}"
154  .format(get_params(page=0,
155  search=search,
156  showPaused=showPaused)),
157  disabled=is_disabled))
158 
159  page_link = void_link
160  if current_page > 0:
161  page_link = '?{}'.format(get_params(page=(current_page - 1),
162  search=search,
163  showPaused=showPaused))
164 
165  output.append(previous_node.format(href_link=page_link,
166  disabled=is_disabled))
167 
168  mid = int(window / 2)
169  last_page = num_of_pages - 1
170 
171  if current_page <= mid or num_of_pages < window:
172  pages = [i for i in range(0, min(num_of_pages, window))]
173  elif mid < current_page < last_page - mid:
174  pages = [i for i in range(current_page - mid, current_page + mid + 1)]
175  else:
176  pages = [i for i in range(num_of_pages - window, last_page + 1)]
177 
178  def is_current(current, page):
179  return page == current
180 
181  for page in pages:
182  vals = {
183  'is_active': 'active' if is_current(current_page, page) else '',
184  'href_link': void_link if is_current(current_page, page)
185  else '?{}'.format(get_params(page=page,
186  search=search,
187  showPaused=showPaused)),
188  'page_num': page + 1
189  }
190  output.append(page_node.format(**vals))
191 
192  is_disabled = 'disabled' if current_page >= num_of_pages - 1 else ''
193 
194  page_link = (void_link if current_page >= num_of_pages - 1
195  else '?{}'.format(get_params(page=current_page + 1,
196  search=search,
197  showPaused=showPaused)))
198 
199  output.append(next_node.format(href_link=page_link, disabled=is_disabled))
200  output.append(last_node.format(href_link="?{}"
201  .format(get_params(page=last_page,
202  search=search,
203  showPaused=showPaused)),
204  disabled=is_disabled))
205 
206  output.append('</ul>')
207 
208  return wtforms.widgets.core.HTMLString('\n'.join(output))

332 def gzipped(f):
333  """
334  Decorator to make a view compressed
335  """
336  @functools.wraps(f)
337  def view_func(*args, **kwargs):
338  @after_this_request
339  def zipper(response):
340  accept_encoding = request.headers.get('Accept-Encoding', '')
341 
342  if 'gzip' not in accept_encoding.lower():
343  return response
344 
345  response.direct_passthrough = False
346 
347  if (response.status_code < 200 or
348  response.status_code >= 300 or
349  'Content-Encoding' in response.headers):
350  return response
351  gzip_buffer = IO()
352  gzip_file = gzip.GzipFile(mode='wb',
353  fileobj=gzip_buffer)
354  gzip_file.write(response.data)
355  gzip_file.close()
356 
357  response.data = gzip_buffer.getvalue()
358  response.headers['Content-Encoding'] = 'gzip'
359  response.headers['Vary'] = 'Accept-Encoding'
360  response.headers['Content-Length'] = len(response.data)
361 
362  return response
363 
364  return f(*args, **kwargs)
365 
366  return view_func
