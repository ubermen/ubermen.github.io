40 def jenkins_request_with_headers(jenkins_server, req, add_crumb=True):
41  """
42  We need to get the headers in addition to the body answer
43  to get the location from them
44  This function is just a copy of the one present in python-jenkins library
45  with just the return call changed
46  :param jenkins_server: The server to query
47  :param req: The request to execute
48  :param add_crumb: Boolean to indicate if it should add crumb to the request
49  :return:
50  """
51  try:
52  if jenkins_server.auth:
53  req.add_header('Authorization', jenkins_server.auth)
54  if add_crumb:
55  jenkins_server.maybe_add_crumb(req)
56  response = urlopen(req, timeout=jenkins_server.timeout)
57  response_body = response.read()
58  response_headers = response.info()
59  if response_body is None:
60  raise jenkins.EmptyResponseException(
61  "Error communicating with server[%s]: "
62  "empty response" % jenkins_server.server)
63  return {'body': response_body.decode('utf-8'), 'headers': response_headers}
64  except HTTPError as e:
65  # Jenkins's funky authentication means its nigh impossible to
66  # distinguish errors.
67  if e.code in [401, 403, 500]:
68  # six.moves.urllib.error.HTTPError provides a 'reason'
69  # attribute for all python version except for ver 2.6
70  # Falling back to HTTPError.msg since it contains the
71  # same info as reason
72  raise JenkinsException(
73  'Error in request. ' +
74  'Possibly authentication failed [%s]: %s' % (
75  e.code, e.msg)
76  )
77  elif e.code == 404:
78  raise jenkins.NotFoundException('Requested item could not be found')
79  else:
80  raise
81  except socket.timeout as e:
82  raise jenkins.TimeoutException('Error in request: %s' % e)
83  except URLError as e:
84  # python 2.6 compatibility to ensure same exception raised
85  # since URLError wraps a socket timeout on python 2.6.
86  if str(e.reason) == "timed out":
87  raise jenkins.TimeoutException('Error in request: %s' % e.reason)
88  raise JenkinsException('Error in request: %s' % e.reason)

91 class JenkinsJobTriggerOperator(BaseOperator):
92  """
93  Trigger a Jenkins Job and monitor it's execution.
94  This operator depend on python-jenkins library,
95  version >= 0.4.15 to communicate with jenkins server.
96  You'll also need to configure a Jenkins connection in the connections screen.
97  :param jenkins_connection_id: The jenkins connection to use for this job
98  :type jenkins_connection_id: string
99  :param job_name: The name of the job to trigger
100  :type job_name: string
101  :param parameters: The parameters block to provide to jenkins. (templated)
102  :type parameters: string
103  :param sleep_time: How long will the operator sleep between each status
104  request for the job (min 1, default 10)
105  :type sleep_time: int
106  :param max_try_before_job_appears: The maximum number of requests to make
107  while waiting for the job to appears on jenkins server (default 10)
108  :type max_try_before_job_appears: int
109  """

132  def build_job(self, jenkins_server):
133  """
134  This function makes an API call to Jenkins to trigger a build for 'job_name'
135  It returned a dict with 2 keys : body and headers.
136  headers contains also a dict-like object which can be queried to get
137  the location to poll in the queue.
138  :param jenkins_server: The jenkins server where the job should be triggered
139  :return: Dict containing the response body (key body)
140  and the headers coming along (headers)
141  """
142  # Warning if the parameter is too long, the URL can be longer than
143  # the maximum allowed size
144  if self.parameters and isinstance(self.parameters, basestring):
145  import ast
146  self.parameters = ast.literal_eval(self.parameters)
147 
148  if not self.parameters:
149  # We need a None to call the non parametrized jenkins api end point
150  self.parameters = None
151 
152  request = Request(jenkins_server.build_job_url(self.job_name,
153  self.parameters, None), b'')
154  return jenkins_request_with_headers(jenkins_server, request)

156  def poll_job_in_queue(self, location, jenkins_server):
157  """
158  This method poll the jenkins queue until the job is executed.
159  When we trigger a job through an API call,
160  the job is first put in the queue without having a build number assigned.
161  Thus we have to wait the job exit the queue to know its build number.
162  To do so, we have to add /api/json (or /api/xml) to the location
163  returned by the build_job call and poll this file.
164  When a 'executable' block appears in the json, it means the job execution started
165  and the field 'number' then contains the build number.
166  :param location: Location to poll, returned in the header of the build_job call
167  :param jenkins_server: The jenkins server to poll
168  :return: The build_number corresponding to the triggered job
169  """
170  try_count = 0
171  location = location + '/api/json'
172  # TODO Use get_queue_info instead
173  # once it will be available in python-jenkins (v > 0.4.15)
174  self.log.info('Polling jenkins queue at the url %s', location)
175  while try_count < self.max_try_before_job_appears:
176  location_answer = jenkins_request_with_headers(jenkins_server,
177  Request(location))
178  if location_answer is not None:
179  json_response = json.loads(location_answer['body'])
180  if 'executable' in json_response:
181  build_number = json_response['executable']['number']
182  self.log.info('Job executed on Jenkins side with the build number %s',
183  build_number)
184  return build_number
185  try_count += 1
186  time.sleep(self.sleep_time)
187  raise AirflowException("The job hasn't been executed"
188  " after polling the queue %d times",
189  self.max_try_before_job_appears)

205  raise AirflowException('The job_name parameter is missing,'
206  'impossible to trigger the job')
207 
208  self.log.info(
209  'Triggering the job %s on the jenkins : %s with the parameters : %s',
210  self.job_name, self.jenkins_connection_id, self.parameters)
211  jenkins_server = self.get_hook().get_jenkins_server()
212  jenkins_response = self.build_job(jenkins_server)
213  build_number = self.poll_job_in_queue(
214  jenkins_response['headers']['Location'], jenkins_server)
215 
216  time.sleep(self.sleep_time)
217  keep_polling_job = True
218  build_info = None
219  while keep_polling_job:
220  try:
221  build_info = jenkins_server.get_build_info(name=self.job_name,
222  number=build_number)
223  if build_info['result'] is not None:
224  keep_polling_job = False
225  # Check if job had errors.
226  if build_info['result'] != 'SUCCESS':
227  raise AirflowException(
228  'Jenkins job failed, final state : %s.'
229  'Find more information on job url : %s'
230  % (build_info['result'], build_info['url']))
231  else:
232  self.log.info('Waiting for job to complete : %s , build %s',
233  self.job_name, build_number)
234  time.sleep(self.sleep_time)
235  except jenkins.NotFoundException as err:
236  raise AirflowException(
237  'Jenkins job status check failed. Final error was: %s'
238  % err.resp.status)
239  except jenkins.JenkinsException as err:
240  raise AirflowException(
241  'Jenkins call failed with error : %s, if you have parameters '
242  'double check them, jenkins sends back '
243  'this exception for unknown parameters'
244  'You can also check logs for more details on this exception '
245  '(jenkins_url/log/rss)', str(err))
246  if build_info:
247  # If we can we return the url of the job
248  # for later use (like retrieving an artifact)
249  return build_info['url']
