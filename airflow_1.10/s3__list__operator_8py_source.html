25 class S3ListOperator(BaseOperator):
26  """
27  List all objects from the bucket with the given string prefix in name.
28 
29  This operator returns a python list with the name of objects which can be
30  used by `xcom` in the downstream task.
31 
32  :param bucket: The S3 bucket where to find the objects. (templated)
33  :type bucket: string
34  :param prefix: Prefix string to filters the objects whose name begin with
35  such prefix. (templated)
36  :type prefix: string
37  :param delimiter: the delimiter marks key hierarchy. (templated)
38  :type delimiter: string
39  :param aws_conn_id: The connection ID to use when connecting to S3 storage.
40  :type aws_conn_id: string
41 
42  **Example**:
43  The following operator would list all the files
44  (excluding subfolders) from the S3
45  ``customers/2018/04/`` key in the ``data`` bucket. ::
46 
47  s3_file = S3ListOperator(
48  task_id='list_3s_files',
49  bucket='data',
50  prefix='customers/2018/04/',
51  delimiter='/',
52  aws_conn_id='aws_customers_conn'
53  )
54  """
55  template_fields = ('bucket', 'prefix', 'delimiter')
