33 class KubernetesPodOperator(BaseOperator):
34  """
35  Execute a task in a Kubernetes Pod
36 
37  :param image: Docker image you wish to launch. Defaults to dockerhub.io,
38  but fully qualified URLS will point to custom repositories
39  :type image: str
40  :param: namespace: the namespace to run within kubernetes
41  :type: namespace: str
42  :param cmds: entrypoint of the container. (templated)
43  The docker images's entrypoint is used if this is not provide.
44  :type cmds: list of str
45  :param arguments: arguments of to the entrypoint. (templated)
46  The docker image's CMD is used if this is not provided.
47  :type arguments: list of str
48  :param volume_mounts: volumeMounts for launched pod
49  :type volume_mounts: list of VolumeMount
50  :param volumes: volumes for launched pod. Includes ConfigMaps and PersistentVolumes
51  :type volumes: list of Volume
52  :param labels: labels to apply to the Pod
53  :type labels: dict
54  :param startup_timeout_seconds: timeout in seconds to startup the pod
55  :type startup_timeout_seconds: int
56  :param name: name of the task you want to run,
57  will be used to generate a pod id
58  :type name: str
59  :param env_vars: Environment variables initialized in the container. (templated)
60  :type env_vars: dict
61  :param secrets: Kubernetes secrets to inject in the container,
62  They can be exposed as environment vars or files in a volume.
63  :type secrets: list of Secret
64  :param in_cluster: run kubernetes client with in_cluster configuration
65  :type in_cluster: bool
66  :param cluster_context: context that points to kubernetes cluster.
67  Ignored when in_cluster is True. If None, current-context is used.
68  :type cluster_context: string
69  :param get_logs: get the stdout of the container as logs of the tasks
70  :type get_logs: bool
71  :param affinity: A dict containing a group of affinity scheduling rules
72  :type affinity: dict
73  :param config_file: The path to the Kubernetes config file
74  :type config_file: str
75  :param xcom_push: If xcom_push is True, the content of the file
76  /airflow/xcom/return.json in the container will also be pushed to an
77  XCom when the container completes.
78  :type xcom_push: bool
79  """

82  def execute(self, context):
83  try:
84  client = kube_client.get_kube_client(in_cluster=self.in_cluster,
85  cluster_context=self.cluster_context,
86  config_file=self.config_file)
87  gen = pod_generator.PodGenerator()
88 
89  for mount in self.volume_mounts:
90  gen.add_mount(mount)
91  for volume in self.volumes:
92  gen.add_volume(volume)
93 
94  pod = gen.make_pod(
95  namespace=self.namespace,
96  image=self.image,
97  pod_id=self.name,
98  cmds=self.cmds,
99  arguments=self.arguments,
100  labels=self.labels,
101  )
102 
103  pod.secrets = self.secrets
104  pod.envs = self.env_vars
105  pod.image_pull_policy = self.image_pull_policy
106  pod.annotations = self.annotations
107  pod.resources = self.resources
108  pod.affinity = self.affinity
109 
110  launcher = pod_launcher.PodLauncher(kube_client=client,
111  extract_xcom=self.xcom_push)
112  (final_state, result) = launcher.run_pod(
113  pod,
114  startup_timeout=self.startup_timeout_seconds,
115  get_logs=self.get_logs)
116  if final_state != State.SUCCESS:
117  raise AirflowException(
118  'Pod returned a failure: {state}'.format(state=final_state)
119  )
120  if self.xcom_push:
121  return result
122  except AirflowException as ex:
123  raise AirflowException('Pod Launching failed: {error}'.format(error=ex))
