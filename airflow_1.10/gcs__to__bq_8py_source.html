28 class GoogleCloudStorageToBigQueryOperator(BaseOperator):
29  """
30  Loads files from Google cloud storage into BigQuery.
31 
32  The schema to be used for the BigQuery table may be specified in one of
33  two ways. You may either directly pass the schema fields in, or you may
34  point the operator to a Google cloud storage object name. The object in
35  Google cloud storage must be a JSON file with the schema fields in it.
36 
37  :param bucket: The bucket to load from. (templated)
38  :type bucket: string
39  :param source_objects: List of Google cloud storage URIs to load from. (templated)
40  If source_format is 'DATASTORE_BACKUP', the list must only contain a single URI.
41  :type object: list
42  :param destination_project_dataset_table: The dotted (<project>.)<dataset>.<table>
43  BigQuery table to load data into. If <project> is not included,
44  project will be the project defined in the connection json. (templated)
45  :type destination_project_dataset_table: string
46  :param schema_fields: If set, the schema field list as defined here:
47  https://cloud.google.com/bigquery/docs/reference/v2/jobs#configuration.load
48  Should not be set when source_format is 'DATASTORE_BACKUP'.
49  :type schema_fields: list
50  :param schema_object: If set, a GCS object path pointing to a .json file that
51  contains the schema for the table. (templated)
52  :param schema_object: string
53  :param source_format: File format to export.
54  :type source_format: string
55  :param compression: [Optional] The compression type of the data source.
56  Possible values include GZIP and NONE.
57  The default value is NONE.
58  This setting is ignored for Google Cloud Bigtable,
59  Google Cloud Datastore backups and Avro formats.
60  :type compression: string
61  :param create_disposition: The create disposition if the table doesn't exist.
62  :type create_disposition: string
63  :param skip_leading_rows: Number of rows to skip when loading from a CSV.
64  :type skip_leading_rows: int
65  :param write_disposition: The write disposition if the table already exists.
66  :type write_disposition: string
67  :param field_delimiter: The delimiter to use when loading from a CSV.
68  :type field_delimiter: string
69  :param max_bad_records: The maximum number of bad records that BigQuery can
70  ignore when running the job.
71  :type max_bad_records: int
72  :param quote_character: The value that is used to quote data sections in a CSV file.
73  :type quote_character: string
74  :param ignore_unknown_values: [Optional] Indicates if BigQuery should allow
75  extra values that are not represented in the table schema.
76  If true, the extra values are ignored. If false, records with extra columns
77  are treated as bad records, and if there are too many bad records, an
78  invalid error is returned in the job result.
79  :type ignore_unknown_values: bool
80  :param allow_quoted_newlines: Whether to allow quoted newlines (true) or not (false).
81  :type allow_quoted_newlines: boolean
82  :param allow_jagged_rows: Accept rows that are missing trailing optional columns.
83  The missing values are treated as nulls. If false, records with missing trailing
84  columns are treated as bad records, and if there are too many bad records, an
85  invalid error is returned in the job result. Only applicable to CSV, ignored
86  for other formats.
87  :type allow_jagged_rows: bool
88  :param max_id_key: If set, the name of a column in the BigQuery table
89  that's to be loaded. Thsi will be used to select the MAX value from
90  BigQuery after the load occurs. The results will be returned by the
91  execute() command, which in turn gets stored in XCom for future
92  operators to use. This can be helpful with incremental loads--during
93  future executions, you can pick up from the max ID.
94  :type max_id_key: string
95  :param bigquery_conn_id: Reference to a specific BigQuery hook.
96  :type bigquery_conn_id: string
97  :param google_cloud_storage_conn_id: Reference to a specific Google
98  cloud storage hook.
99  :type google_cloud_storage_conn_id: string
100  :param delegate_to: The account to impersonate, if any. For this to
101  work, the service account making the request must have domain-wide
102  delegation enabled.
103  :type delegate_to: string
104  :param schema_update_options: Allows the schema of the destination
105  table to be updated as a side effect of the load job.
106  :type schema_update_options: list
107  :param src_fmt_configs: configure optional fields specific to the source format
108  :type src_fmt_configs: dict
109  :param external_table: Flag to specify if the destination table should be
110  a BigQuery external table. Default Value is False.
111  :type external_table: bool
112  :param time_partitioning: configure optional time partitioning fields i.e.
113  partition by field, type and expiration as per API specifications.
114  Note that 'field' is not available in concurrency with
115  dataset.table$partition.
116  :type time_partitioning: dict
117  """

184  bq_hook = BigQueryHook(bigquery_conn_id=self.bigquery_conn_id,
185  delegate_to=self.delegate_to)
186 
187  if not self.schema_fields and \
188  self.schema_object and \
189  self.source_format != 'DATASTORE_BACKUP':
190  gcs_hook = GoogleCloudStorageHook(
191  google_cloud_storage_conn_id=self.google_cloud_storage_conn_id,
192  delegate_to=self.delegate_to)
193  schema_fields = json.loads(gcs_hook.download(
194  self.bucket,

196  else:
197  schema_fields = self.schema_fields
198 
199  source_uris = ['gs://{}/{}'.format(self.bucket, source_object)
200  for source_object in self.source_objects]
201  conn = bq_hook.get_conn()
202  cursor = conn.cursor()
203 
204  if self.external_table:
205  cursor.create_external_table(

207  schema_fields=schema_fields,
