37  def get_conn(self):
38  """
39  Returns a oracle connection object
40  Optional parameters for using a custom DSN connection
41  (instead of using a server alias from tnsnames.ora)
42  The dsn (data source name) is the TNS entry
43  (from the Oracle names server or tnsnames.ora file)
44  or is a string like the one returned from makedsn().
45 
46  :param dsn: the host address for the Oracle server
47  :param service_name: the db_unique_name of the database
48  that you are connecting to (CONNECT_DATA part of TNS)
49  You can set these parameters in the extra fields of your connection
50  as in ``{ "dsn":"some.host.address" , "service_name":"some.service.name" }``
51  """
52  conn = self.get_connection(self.oracle_conn_id)
53  dsn = conn.extra_dejson.get('dsn', None)
54  sid = conn.extra_dejson.get('sid', None)
55  mod = conn.extra_dejson.get('module', None)
56 
57  service_name = conn.extra_dejson.get('service_name', None)
58  if dsn and sid and not service_name:
59  dsn = cx_Oracle.makedsn(dsn, conn.port, sid)
60  conn = cx_Oracle.connect(conn.login, conn.password, dsn=dsn)
61  elif dsn and service_name and not sid:
62  dsn = cx_Oracle.makedsn(dsn, conn.port, service_name=service_name)
63  conn = cx_Oracle.connect(conn.login, conn.password, dsn=dsn)
64  else:
65  conn = cx_Oracle.connect(conn.login, conn.password, conn.host)
66 
67  if mod is not None:
68  conn.module = mod
69 
70  return conn

72  def insert_rows(self, table, rows, target_fields=None, commit_every=1000):
73  """
74  A generic way to insert a set of tuples into a table,
75  the whole set of inserts is treated as one transaction
76  Changes from standard DbApiHook implementation:
77  - Oracle SQL queries in cx_Oracle can not be terminated with a semicolon (';')
78  - Replace NaN values with NULL using numpy.nan_to_num (not using is_nan()
79  because of input types error for strings)
80  - Coerce datetime cells to Oracle DATETIME format during insert
81  """
82  if target_fields:
83  target_fields = ', '.join(target_fields)
84  target_fields = '({})'.format(target_fields)
85  else:
86  target_fields = ''
87  conn = self.get_conn()
88  cur = conn.cursor()
89  if self.supports_autocommit:
90  cur.execute('SET autocommit = 0')
91  conn.commit()
92  i = 0
93  for row in rows:
94  i += 1
95  lst = []
96  for cell in row:
97  if isinstance(cell, basestring):
98  lst.append("'" + str(cell).replace("'", "''") + "'")
99  elif cell is None:
100  lst.append('NULL')
101  elif type(cell) == float and \
102  numpy.isnan(cell): # coerce numpy NaN to NULL
103  lst.append('NULL')
104  elif isinstance(cell, numpy.datetime64):
105  lst.append("'" + str(cell) + "'")
106  elif isinstance(cell, datetime):
107  lst.append("to_date('" +
108  cell.strftime('%Y-%m-%d %H:%M:%S') +
109  "','YYYY-MM-DD HH24:MI:SS')")
110  else:
111  lst.append(str(cell))
112  values = tuple(lst)
113  sql = 'INSERT /*+ APPEND */ ' \
114  'INTO {0} {1} VALUES ({2})'.format(table,
115  target_fields,
116  ','.join(values))
117  cur.execute(sql)
118  if i % commit_every == 0:
119  conn.commit()
120  self.log.info('Loaded {i} into {table} rows so far'.format(**locals()))
121  conn.commit()
122  cur.close()
123  conn.close()
124  self.log.info('Done loading. Loaded a total of {i} rows'.format(**locals()))

126  def bulk_insert_rows(self, table, rows, target_fields=None, commit_every=5000):
127  """
128  A performant bulk insert for cx_Oracle
129  that uses prepared statements via `executemany()`.
130  For best performance, pass in `rows` as an iterator.
131  """
132  conn = self.get_conn()
133  cursor = conn.cursor()
134  values = ', '.join(':%s' % i for i in range(1, len(target_fields) + 1))
135  prepared_stm = 'insert into {tablename} ({columns}) values ({values})'.format(
136  tablename=table,
137  columns=', '.join(target_fields),
138  values=values,
139  )
140  row_count = 0
141  # Chunk the rows
142  row_chunk = []
143  for row in rows:
144  row_chunk.append(row)
145  row_count += 1
146  if row_count % commit_every == 0:
147  cursor.prepare(prepared_stm)
148  cursor.executemany(None, row_chunk)
149  conn.commit()
150  self.log.info('[%s] inserted %s rows', table, row_count)
151  # Empty chunk
152  row_chunk = []
153  # Commit the leftover chunk
154  cursor.prepare(prepared_stm)
155  cursor.executemany(None, row_chunk)
156  conn.commit()
157  self.log.info('[%s] inserted %s rows', table, row_count)
158  cursor.close()
159  conn.close()
