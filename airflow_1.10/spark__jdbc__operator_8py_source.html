25 class SparkJDBCOperator(SparkSubmitOperator):
26  """
27  This operator extends the SparkSubmitOperator specifically for performing data
28  transfers to/from JDBC-based databases with Apache Spark. As with the
29  SparkSubmitOperator, it assumes that the "spark-submit" binary is available on the
30  PATH.
31 
32  :param spark_app_name: Name of the job (default airflow-spark-jdbc)
33  :type spark_app_name: str
34  :param spark_conn_id: Connection id as configured in Airflow administration
35  :type spark_conn_id: str
36  :param spark_conf: Any additional Spark configuration properties
37  :type spark_conf: dict
38  :param spark_py_files: Additional python files used (.zip, .egg, or .py)
39  :type spark_py_files: str
40  :param spark_files: Additional files to upload to the container running the job
41  :type spark_files: str
42  :param spark_jars: Additional jars to upload and add to the driver and
43  executor classpath
44  :type spark_jars: str
45  :param num_executors: number of executor to run. This should be set so as to manage
46  the number of connections made with the JDBC database
47  :type num_executors: int
48  :param executor_cores: Number of cores per executor
49  :type executor_cores: int
50  :param executor_memory: Memory per executor (e.g. 1000M, 2G)
51  :type executor_memory: str
52  :param driver_memory: Memory allocated to the driver (e.g. 1000M, 2G)
53  :type driver_memory: str
54  :param verbose: Whether to pass the verbose flag to spark-submit for debugging
55  :type verbose: bool
56  :param keytab: Full path to the file that contains the keytab
57  :type keytab: str
58  :param principal: The name of the kerberos principal used for keytab
59  :type principal: str
60  :param cmd_type: Which way the data should flow. 2 possible values:
61  spark_to_jdbc: data written by spark from metastore to jdbc
62  jdbc_to_spark: data written by spark from jdbc to metastore
63  :type cmd_type: str
64  :param jdbc_table: The name of the JDBC table
65  :type jdbc_table: str
66  :param jdbc_conn_id: Connection id used for connection to JDBC database
67  :type: jdbc_conn_id: str
68  :param jdbc_driver: Name of the JDBC driver to use for the JDBC connection. This
69  driver (usually a jar) should be passed in the 'jars' parameter
70  :type jdbc_driver: str
71  :param metastore_table: The name of the metastore table,
72  :type metastore_table: str
73  :param jdbc_truncate: (spark_to_jdbc only) Whether or not Spark should truncate or
74  drop and recreate the JDBC table. This only takes effect if
75  'save_mode' is set to Overwrite. Also, if the schema is
76  different, Spark cannot truncate, and will drop and recreate
77  :type jdbc_truncate: bool
78  :param save_mode: The Spark save-mode to use (e.g. overwrite, append, etc.)
79  :type save_mode: str
80  :param save_format: (jdbc_to_spark-only) The Spark save-format to use (e.g. parquet)
81  :type save_format: str
82  :param batch_size: (spark_to_jdbc only) The size of the batch to insert per round
83  trip to the JDBC database. Defaults to 1000
84  :type batch_size: int
85  :param fetch_size: (jdbc_to_spark only) The size of the batch to fetch per round trip
86  from the JDBC database. Default depends on the JDBC driver
87  :type fetch_size: int
88  :param num_partitions: The maximum number of partitions that can be used by Spark
89  simultaneously, both for spark_to_jdbc and jdbc_to_spark
90  operations. This will also cap the number of JDBC connections
91  that can be opened
92  :type num_partitions: int
93  :param partition_column: (jdbc_to_spark-only) A numeric column to be used to
94  partition the metastore table by. If specified, you must
95  also specify:
96  num_partitions, lower_bound, upper_bound
97  :type partition_column: str
98  :param lower_bound: (jdbc_to_spark-only) Lower bound of the range of the numeric
99  partition column to fetch. If specified, you must also specify:
100  num_partitions, partition_column, upper_bound
101  :type lower_bound: int
102  :param upper_bound: (jdbc_to_spark-only) Upper bound of the range of the numeric
103  partition column to fetch. If specified, you must also specify:
104  num_partitions, partition_column, lower_bound
105  :type upper_bound: int
106  :param create_table_column_types: (spark_to_jdbc-only) The database column data types
107  to use instead of the defaults, when creating the
108  table. Data type information should be specified in
109  the same format as CREATE TABLE columns syntax
110  (e.g: "name CHAR(64), comments VARCHAR(1024)").
111  The specified types should be valid spark sql data
112  types.
113  """
