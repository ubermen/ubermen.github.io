138  def heartbeat(self):
139  '''
140  Heartbeats update the job's entry in the database with a timestamp
141  for the latest_heartbeat and allows for the job to be killed
142  externally. This allows at the system level to monitor what is
143  actually active.
144 
145  For instance, an old heartbeat for SchedulerJob would mean something
146  is wrong.
147 
148  This also allows for any job to be killed externally, regardless
149  of who is running it or on which machine it is running.
150 
151  Note that if your heartbeat is set to 60 seconds and you call this
152  method after 10 seconds of processing since the last heartbeat, it
153  will sleep 50 seconds to complete the 60 seconds and keep a steady
154  heart rate. If you go over 60 seconds before calling it, it won't
155  sleep at all.
156  '''
157  session = settings.Session()
158  job = session.query(BaseJob).filter_by(id=self.id).one()
159  make_transient(job)
160  session.commit()
161  session.close()
162 
163  if job.state == State.SHUTDOWN:
164  self.kill()
165 
166  # Figure out how long to sleep for
167  sleep_for = 0
168  if job.latest_heartbeat:
169  sleep_for = max(
170  0,
171  self.heartrate - (datetime.now() - job.latest_heartbeat).total_seconds())
172 
173  # Don't keep session open while sleeping as it leaves a connection open
174  session.close()
175  sleep(sleep_for)
176 
177  # Update last heartbeat time
178  session = settings.Session()
179  job = session.query(BaseJob).filter(BaseJob.id == self.id).first()
180  job.latest_heartbeat = datetime.now()
181  session.merge(job)
182  session.commit()
183 
184  self.heartbeat_callback(session=session)
185  session.close()
186  self.logger.debug('[heart] Boom.')

711  state=State.RUNNING,
712  external_trigger=False,
713  session=session
714  )

716  if len(active_runs) >= dag.max_active_runs and not dag.dagrun_timeout:
717  return
718  timedout_runs = 0
719  for dr in active_runs:
720  if (
721  dr.start_date and dag.dagrun_timeout and
722  dr.start_date < datetime.now() - dag.dagrun_timeout):
723  dr.state = State.FAILED
724  dr.end_date = datetime.now()
725  timedout_runs += 1
726  session.commit()
727  if len(active_runs) - timedout_runs >= dag.max_active_runs:
728  return
729 
730  # this query should be replaced by find dagrun
731  qry = (
732  session.query(func.max(DagRun.execution_date))
733  .filter_by(dag_id=dag.dag_id)
734  .filter(or_(
735  DagRun.external_trigger == False,
736  # add % as a wildcard for the like query
737  DagRun.run_id.like(DagRun.ID_PREFIX + '%')
738  ))
739  )
740  last_scheduled_run = qry.scalar()

742  # don't schedule @once again
743  if dag.schedule_interval == '@once' and last_scheduled_run:
744  return None
745 
746  # don't do scheduler catchup for dag's that don't have dag.catchup = True
747  if not dag.catchup:
748  # The logic is that we move start_date up until
749  # one period before, so that datetime.now() is AFTER
750  # the period end, and the job can be created...
751  now = datetime.now()
752  next_start = dag.following_schedule(now)
753  last_start = dag.previous_schedule(now)
754  if next_start <= now:
755  new_start = last_start
756  else:
757  new_start = dag.previous_schedule(last_start)
758 
759  if dag.start_date:
760  if new_start >= dag.start_date:
761  dag.start_date = new_start
762  else:
763  dag.start_date = new_start

766  if not last_scheduled_run:
767  # First run
768  task_start_dates = [t.start_date for t in dag.tasks]
769  if task_start_dates:
770  next_run_date = dag.normalize_schedule(min(task_start_dates))
771  self.logger.debug("Next run date based on tasks {}"
772  .format(next_run_date))
773  else:
774  next_run_date = dag.following_schedule(last_scheduled_run)
775

777  last_run = dag.get_last_dagrun(session=session)

780  next_run_date = dag.following_schedule(next_run_date)
781 
782  # don't ever schedule prior to the dag's start_date
783  if dag.start_date:
784  next_run_date = (dag.start_date if not next_run_date
785  else max(next_run_date, dag.start_date))
786  if next_run_date == dag.start_date:
787  next_run_date = dag.normalize_schedule(dag.start_date)

799  period_end = next_run_date
800  elif next_run_date:
801  period_end = dag.following_schedule(next_run_date)

814  return

817  next_run = dag.create_dagrun(
818  run_id='scheduled__' + next_run_date.isoformat(),
819  execution_date=next_run_date,
820  start_date=datetime.now(),
821  state=State.RUNNING,
822  external_trigger=False
823  )

971  session
972  .query(TI)
973  .filter(TI.dag_id.in_(simple_dag_bag.dag_ids))
974  .filter(TI.state.in_(states))
975  .all()
976  )

979  if len(task_instances_to_examine) == 0:

981  return
