451  public synchronized void newPixels(int x, int y, int w, int h,
452  boolean framenotify) {
453  if (animating) {
454  if (fullbuffers) {
455  x = y = 0;
456  w = width;
457  h = height;
458  } else {
459  if (x < 0) {
