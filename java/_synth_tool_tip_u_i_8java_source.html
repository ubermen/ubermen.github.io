203  Insets insets = c.getInsets();
204  Dimension prefSize = new Dimension(insets.left+insets.right,
205  insets.top+insets.bottom);
206  String text = ((JToolTip)c).getTipText();
207 
208  if (text != null) {
209  View v = (c != null) ? (View) c.getClientProperty("html") : null;
210  if (v != null) {
211  prefSize.width += (int) v.getPreferredSpan(View.X_AXIS);
212  prefSize.height += (int) v.getPreferredSpan(View.Y_AXIS);
213  } else {
214  Font font = context.getStyle().getFont(context);
215  FontMetrics fm = c.getFontMetrics(font);
216  prefSize.width += context.getStyle().getGraphicsUtils(context).
217  computeStringWidth(context, font, fm, text);
218  prefSize.height += fm.getHeight();
219  }
220  }
221  context.dispose();
222  return prefSize;
