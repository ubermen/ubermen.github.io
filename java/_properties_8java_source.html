538  private String loadConvert (char[] in, int off, int len, char[] convtBuf) {

562  case '5': case '6': case '7': case '8': case '9':
563  value = (value << 4) + aChar - '0';
564  break;

566  case 'd': case 'e': case 'f':
567  value = (value << 4) + 10 + aChar - 'a';
568  break;
569  case 'A': case 'B': case 'C':
570  case 'D': case 'E': case 'F':
571  value = (value << 4) + 10 + aChar - 'A';
572  break;

576  }
577  }

579  } else {

582  else if (aChar == 'n') aChar = '\n';

594  * Converts unicodes to encoded &#92;uxxxx and escapes
595  * special characters with a preceding slash

597  private String saveConvert(String theString,
598  boolean escapeSpace,
599  boolean escapeUnicode) {

604  }

610  // avoids the specials below

612  if (aChar == '\\') {
613  outBuffer.append('\\'); outBuffer.append('\\');
614  continue;
615  }

617  continue;
618  }

620  case ' ':

622  outBuffer.append('\\');
623  outBuffer.append(' ');
624  break;
625  case '\t':outBuffer.append('\\'); outBuffer.append('t');
626  break;
627  case '\n':outBuffer.append('\\'); outBuffer.append('n');
628  break;
629  case '\r':outBuffer.append('\\'); outBuffer.append('r');
630  break;
631  case '\f':outBuffer.append('\\'); outBuffer.append('f');
632  break;
633  case '=': // Fall through
634  case ':': // Fall through
635  case '#': // Fall through
636  case '!':
637  outBuffer.append('\\'); outBuffer.append(aChar);
638  break;
639  default:
640  if (((aChar < 0x0020) || (aChar > 0x007e)) & escapeUnicode ) {
641  outBuffer.append('\\');

643  outBuffer.append(toHex((aChar >> 12) & 0xF));
644  outBuffer.append(toHex((aChar >> 8) & 0xF));

646  outBuffer.append(toHex( aChar & 0xF));
647  } else {
