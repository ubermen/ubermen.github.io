401  protected Class<?> loadClass(String name, boolean resolve)
402  throws ClassNotFoundException
403  {
404  synchronized (getClassLoadingLock(name)) {
405  // First, check if the class has already been loaded
406  Class<?> c = findLoadedClass(name);
407  if (c == null) {
408  long t0 = System.nanoTime();
409  try {
410  if (parent != null) {
411  c = parent.loadClass(name, false);
412  } else {
413  c = findBootstrapClassOrNull(name);
414  }
415  } catch (ClassNotFoundException e) {
416  // ClassNotFoundException thrown if class not found
417  // from the non-null parent class loader
418  }
419

421  // If still not found, then invoke findClass in order
422  // to find the class.
423  long t1 = System.nanoTime();
424  c = findClass(name);

877  private void checkCerts(String name, CodeSource cs) {
878  int i = name.lastIndexOf('.');
879  String pname = (i == -1) ? "" : name.substring(0, i);
880 
881  Certificate[] certs = null;
882  if (cs != null) {

884  }

886  if (parallelLockMap == null) {
887  synchronized (this) {

890  package2certs.put(pname, (certs == null? nocerts:certs));

895  putIfAbsent(pname, (certs == null? nocerts:certs));

907  private boolean compareCerts(Certificate[] pcerts,
908  Certificate[] certs)
909  {
910  // certs can be null, indicating no certs.
911  if ((certs == null) || (certs.length == 0)) {
912  return pcerts.length == 0;
913  }
914 
915  // the length must be the same at this point
916  if (certs.length != pcerts.length)
917  return false;
918 
919  // go through and make sure all the certs in one array
920  // are in the other and vice-versa.
921  boolean match;
922  for (int i = 0; i < certs.length; i++) {
923  match = false;
924  for (int j = 0; j < pcerts.length; j++) {
925  if (certs[i].equals(pcerts[j])) {
926  match = true;
927  break;
928  }
929  }
930  if (!match) return false;
931  }
932 
933  // now do the same for pcerts
934  for (int i = 0; i < pcerts.length; i++) {
935  match = false;
936  for (int j = 0; j < certs.length; j++) {
937  if (pcerts[i].equals(certs[j])) {
938  match = true;
939  break;
940  }
941  }
942  if (!match) return false;
943  }
944 
945  return true;
946  }
