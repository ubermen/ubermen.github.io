383  Enumeration<Permission> e;
384  List<Permission> pdVector = new ArrayList<>(vcap);
385  List<Permission> plVector = new ArrayList<>(swag);

387  //

403  while (e.hasMoreElements()) {
404  plVector.add(e.nextElement());

406  }
407  }
408  }
409 
410  if (perms != null && permissions != null) {
411  //
412  // Weed out the duplicates from the policy. Unless a refresh
413  // has occurred since the pd was consed this should result in
414  // an empty vector.
415  synchronized (permissions) {
416  e = permissions.elements(); // domain vs policy
417  while (e.hasMoreElements()) {

419  Class<?> pdpClass = pdp.getClass();
420  String pdpActions = pdp.getActions();

422  for (int i = 0; i < plVector.size(); i++) {

424  if (pdpClass.isInstance(pp)) {

426  // have some side effects so this manual
427  // comparison is sufficient.
428  if (pdpName.equals(pp.getName()) &&
429  pdpActions.equals(pp.getActions())) {
430  plVector.remove(i);
431  break;
432  }
433  }
434  }
435  }
436  }
437  }
438 
439  if (perms !=null) {

441  // needs to preserve the bugfix 4301064
442 
443  for (int i = plVector.size()-1; i >= 0; i--) {
444  mergedPerms.add(plVector.get(i));
445  }
446  }
