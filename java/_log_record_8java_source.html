533  // Private method to infer the caller's class and method names
534  private void inferCaller() {
535  needToInferCaller = false;
536  JavaLangAccess access = SharedSecrets.getJavaLangAccess();
537  Throwable throwable = new Throwable();
538  int depth = access.getStackTraceDepth(throwable);
539 
540  boolean lookingForLogger = true;
541  for (int ix = 0; ix < depth; ix++) {
542  // Calling getStackTraceElement directly prevents the VM
543  // from paying the cost of building the entire stack frame.
544  StackTraceElement frame =
545  access.getStackTraceElement(throwable, ix);
546  String cname = frame.getClassName();
547  boolean isLoggerImpl = isLoggerImplFrame(cname);
548  if (lookingForLogger) {
549  // Skip all frames until we have found the first logger frame.
550  if (isLoggerImpl) {
551  lookingForLogger = false;
552  }
553  } else {
554  if (!isLoggerImpl) {
555  // skip reflection call
556  if (!cname.startsWith("java.lang.reflect.") && !cname.startsWith("sun.reflect.")) {
557  // We've found the relevant frame.
558  setSourceClassName(cname);
559  setSourceMethodName(frame.getMethodName());
560  return;
561  }
562  }
563  }
564  }
565  // We haven't found a suitable frame, so just punt. This is
566  // OK as we are only committed to making a "best effort" here.
567  }
