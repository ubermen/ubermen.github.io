234  /*
235  * Checks if a new focus cycle takes place and returns a Component to traverse focus to.
236  * @param comp a possible focus cycle root or policy provider
237  * @param traversalDirection the direction of the traversal
238  * @return a Component to traverse focus to if {@code comp} is a root or provider
239  * and implicit down-cycle is set, otherwise {@code null}
240  */
241  private Component getComponentDownCycle(Component comp, int traversalDirection) {
242  Component retComp = null;
243 
244  if (comp instanceof Container) {
245  Container cont = (Container)comp;
246 
247  if (cont.isFocusCycleRoot()) {
248  if (getImplicitDownCycleTraversal()) {
249  retComp = cont.getFocusTraversalPolicy().getDefaultComponent(cont);
250 
251  if (retComp != null && log.isLoggable(PlatformLogger.Level.FINE)) {
252  log.fine("### Transfered focus down-cycle to " + retComp +
253  " in the focus cycle root " + cont);
254  }
255  } else {
256  return null;
257  }
258  } else if (cont.isFocusTraversalPolicyProvider()) {
259  retComp = (traversalDirection == FORWARD_TRAVERSAL ?
260  cont.getFocusTraversalPolicy().getDefaultComponent(cont) :
261  cont.getFocusTraversalPolicy().getLastComponent(cont));
262 
263  if (retComp != null && log.isLoggable(PlatformLogger.Level.FINE)) {
264  log.fine("### Transfered focus to " + retComp + " in the FTP provider " + cont);
265  }
266  }
267  }
268  return retComp;

293  public Component getComponentAfter(Container aContainer, Component aComponent) {
294  if (log.isLoggable(PlatformLogger.Level.FINE)) {
295  log.fine("### Searching in " + aContainer + " for component after " + aComponent);
296  }
297 
298  if (aContainer == null || aComponent == null) {
299  throw new IllegalArgumentException("aContainer and aComponent cannot be null");
300  }
301  if (!aContainer.isFocusTraversalPolicyProvider() && !aContainer.isFocusCycleRoot()) {
302  throw new IllegalArgumentException("aContainer should be focus cycle root or focus traversal policy provider");
303 
304  } else if (aContainer.isFocusCycleRoot() && !aComponent.isFocusCycleRoot(aContainer)) {
305  throw new IllegalArgumentException("aContainer is not a focus cycle root of aComponent");
306  }
307 
308  // Before all the ckecks below we first see if it's an FTP provider or a focus cycle root.
309  // If it's the case just go down cycle (if it's set to "implicit").
310  Component comp = getComponentDownCycle(aComponent, FORWARD_TRAVERSAL);
311  if (comp != null) {
312  return comp;
313  }
314 
315  // See if the component is inside of policy provider.
316  Container provider = getTopmostProvider(aContainer, aComponent);
317  if (provider != null) {
318  if (log.isLoggable(PlatformLogger.Level.FINE)) {
319  log.fine("### Asking FTP " + provider + " for component after " + aComponent);
320  }
321 
322  // FTP knows how to find component after the given. We don't.
323  FocusTraversalPolicy policy = provider.getFocusTraversalPolicy();
324  Component afterComp = policy.getComponentAfter(provider, aComponent);
325 
326  // Null result means that we overstepped the limit of the FTP's cycle.
327  // In that case we must quit the cycle, otherwise return the component found.
328  if (afterComp != null) {
329  if (log.isLoggable(PlatformLogger.Level.FINE)) {
330  log.fine("### FTP returned " + afterComp);
331  }
332  return afterComp;
333  }
334  aComponent = provider;
335  }
336 
337  List<Component> cycle = getFocusTraversalCycle(aContainer);
338 
339  if (log.isLoggable(PlatformLogger.Level.FINE)) {
340  log.fine("### Cycle is " + cycle + ", component is " + aComponent);
341  }
342 
343  int index = getComponentIndex(cycle, aComponent);
344 
345  if (index < 0) {
346  if (log.isLoggable(PlatformLogger.Level.FINE)) {
347  log.fine("### Didn't find component " + aComponent + " in a cycle " + aContainer);
348  }
349  return getFirstComponent(aContainer);
350  }
351 
352  for (index++; index < cycle.size(); index++) {
353  comp = cycle.get(index);
354  if (accept(comp)) {
355  return comp;
356  } else if ((comp = getComponentDownCycle(comp, FORWARD_TRAVERSAL)) != null) {
357  return comp;

359  }

362  this.cachedRoot = aContainer;

366
