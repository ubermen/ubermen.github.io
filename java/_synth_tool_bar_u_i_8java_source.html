358 
359  if (tb.getOrientation() == JToolBar.HORIZONTAL) {
360  dim.width = tb.isFloatable() ?
361  SynthIcon.getIconWidth(handleIcon, context) : 0;

364  Component component = tb.getComponent(i);

367  dim.width += compDim.width;

429  public void layoutContainer(Container parent) {
430  JToolBar tb = (JToolBar)parent;
431  Insets insets = tb.getInsets();
432  boolean ltr = tb.getComponentOrientation().isLeftToRight();
433  SynthContext context = getContext(tb);
434 
435  Component c;
436  Dimension d;
437 
438  // JToolBar by default uses a somewhat modified BoxLayout as
439  // its layout manager. For compatibility reasons, we want to
440  // support Box "glue" as a way to move things around on the
441  // toolbar. "glue" is represented in BoxLayout as a Box.Filler
442  // with a minimum and preferred size of (0,0).
443  // So what we do here is find the number of such glue fillers
444  // and figure out how much space should be allocated to them.
445  int glueCount = 0;
446  for (int i=0; i<tb.getComponentCount(); i++) {
447  if (isGlue(tb.getComponent(i))) glueCount++;
448  }
449 
450  if (tb.getOrientation() == JToolBar.HORIZONTAL) {
451  int handleWidth = tb.isFloatable() ?
452  SynthIcon.getIconWidth(handleIcon, context) : 0;
453 
454  // Note: contentRect does not take insets into account
455  // since it is used for determining the bounds that are
456  // passed to paintToolBarContentBackground().
457  contentRect.x = ltr ? handleWidth : 0;
458  contentRect.y = 0;
459  contentRect.width = tb.getWidth() - handleWidth;
460  contentRect.height = tb.getHeight();
461 
462  // However, we do take the insets into account here for
463  // the purposes of laying out the toolbar child components.
464  int x = ltr ?
465  handleWidth + insets.left :
466  tb.getWidth() - handleWidth - insets.right;
467  int baseY = insets.top;
468  int baseH = tb.getHeight() - insets.top - insets.bottom;
469 
470  // we need to get the minimum width for laying things out
471  // so that we can calculate how much empty space needs to
472  // be distributed among the "glue", if any
473  int extraSpacePerGlue = 0;
474  if (glueCount > 0) {
475  int minWidth = minimumLayoutSize(parent).width;
476  extraSpacePerGlue = (tb.getWidth() - minWidth) / glueCount;
477  if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
478  }
479 
480  for (int i = 0; i < tb.getComponentCount(); i++) {
481  c = tb.getComponent(i);
482  if (c.isVisible()) {
483  d = c.getPreferredSize();
484  int y, h;
485  if (d.height >= baseH || c instanceof JSeparator) {
486  // Fill available height
487  y = baseY;
488  h = baseH;
489  } else {
490  // Center component vertically in the available space
491  y = baseY + (baseH / 2) - (d.height / 2);
492  h = d.height;
493  }
494  //if the component is a "glue" component then add to its
495  //width the extraSpacePerGlue it is due
496  if (isGlue(c)) d.width += extraSpacePerGlue;
497  c.setBounds(ltr ? x : x - d.width, y, d.width, h);
498  x = ltr ? x + d.width : x - d.width;
499  }
500  }
501  } else {
502  int handleHeight = tb.isFloatable() ?
503  SynthIcon.getIconHeight(handleIcon, context) : 0;
504 
505  // See notes above regarding the use of insets
506  contentRect.x = 0;
507  contentRect.y = handleHeight;
508  contentRect.width = tb.getWidth();
509  contentRect.height = tb.getHeight() - handleHeight;
510 
511  int baseX = insets.left;
512  int baseW = tb.getWidth() - insets.left - insets.right;
513  int y = handleHeight + insets.top;
514 
515  // we need to get the minimum height for laying things out
516  // so that we can calculate how much empty space needs to
517  // be distributed among the "glue", if any
518  int extraSpacePerGlue = 0;
519  if (glueCount > 0) {
520  int minHeight = minimumLayoutSize(parent).height;
521  extraSpacePerGlue = (tb.getHeight() - minHeight) / glueCount;
522  if (extraSpacePerGlue < 0) extraSpacePerGlue = 0;
523  }
524 
525  for (int i = 0; i < tb.getComponentCount(); i++) {
526  c = tb.getComponent(i);
527  if (c.isVisible()) {
528  d = c.getPreferredSize();
529  int x, w;
530  if (d.width >= baseW || c instanceof JSeparator) {
531  // Fill available width
532  x = baseX;
533  w = baseW;
534  } else {
535  // Center component horizontally in the available space
536  x = baseX + (baseW / 2) - (d.width / 2);
537  w = d.width;
538  }
539  //if the component is a "glue" component then add to its
540  //height the extraSpacePerGlue it is due
541  if (isGlue(c)) d.height += extraSpacePerGlue;
542  c.setBounds(x, y, w, d.height);
543  y += d.height;
544  }
545  }
546  }
547  context.dispose();
548  }

550  private boolean isGlue(Component c) {
551  if (c.isVisible() && c instanceof Box.Filler) {
552  Box.Filler f = (Box.Filler)c;
553  Dimension min = f.getMinimumSize();
554  Dimension pref = f.getPreferredSize();
555  return min.width == 0 && min.height == 0 &&
556  pref.width == 0 && pref.height == 0;
557  }
558  return false;
559  }
