363  public void read(DataInputStream in) throws IOException {
364  if (in.readInt() != FILE_VERSION) {
365  }
366 
367  //
368  // Read the list of names
369  //
370  String[] names = new String[in.readShort()];
371  for (int i = 0; i < names.length; i++) {
372  names[i] = in.readUTF();
373  }
374 
375 
376  //
377  // Read the entities
378  //
379  int num = in.readShort();
380  for (int i = 0; i < num; i++) {
381  short nameId = in.readShort();
382  int type = in.readByte();
383  String name = in.readUTF();
384  defEntity(names[nameId], type | GENERAL, name);
385  }
386 
387  // Read the elements
388  //
389  num = in.readShort();
390  for (int i = 0; i < num; i++) {
391  short nameId = in.readShort();
392  int type = in.readByte();
393  byte flags = in.readByte();
394  ContentModel m = readContentModel(in, names);
395  String[] exclusions = readNameArray(in, names);
396  String[] inclusions = readNameArray(in, names);
397  AttributeList atts = readAttributeList(in, names);
398  defElement(names[nameId], type,
399  ((flags & 0x01) != 0), ((flags & 0x02) != 0),
400  m, exclusions, inclusions, atts);
401  }
402  }

404  private ContentModel readContentModel(DataInputStream in, String[] names)
405  throws IOException {
406  byte flag = in.readByte();
407  switch(flag) {
408  case 0: // null
409  return null;
410  case 1: { // content_c
411  int type = in.readByte();
412  ContentModel m = readContentModel(in, names);
413  ContentModel next = readContentModel(in, names);
414  return defContentModel(type, m, next);
415  }
416  case 2: { // content_e
417  int type = in.readByte();
418  Element el = getElement(names[in.readShort()]);
419  ContentModel next = readContentModel(in, names);
420  return defContentModel(type, el, next);
421  }
422  default:
423  throw new IOException("bad bdtd");
424  }

441  private AttributeList readAttributeList(DataInputStream in, String[] names)
442  throws IOException {
443  AttributeList result = null;
444  for (int num = in.readByte(); num > 0; --num) {
445  short nameId = in.readShort();
446  int type = in.readByte();
447  int modifier = in.readByte();
448  short valueId = in.readShort();
449  String value = (valueId == -1) ? null : names[valueId];
450  Vector<String> values = null;
451  short numValues = in.readShort();
452  if (numValues > 0) {
453  values = new Vector<String>(numValues);
454  for (int i = 0; i < numValues; i++) {
455  values.addElement(names[in.readShort()]);
456  }
457  }
458 result = new AttributeList(names[nameId], type, modifier, value,
459  values, result);
460  // We reverse the order of the linked list by doing this, but
461  // that order isn't important.
462  }
