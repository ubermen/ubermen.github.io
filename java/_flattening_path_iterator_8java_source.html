197  private void next(boolean doNext) {

200  if (holdIndex >= holdEnd) {
201  if (doNext) {
202  src.next();
203  }

209  levelIndex = 0;
210  levels[0] = 0;
211  }

214  case SEG_MOVETO:
215  case SEG_LINETO:
216  curx = hold[0];
217  cury = hold[1];
218  if (holdType == SEG_MOVETO) {
219  movx = curx;

224  break;
