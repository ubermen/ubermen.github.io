470  private static synchronized String makeConnectionId(String protocol,
471  Subject subject) {
472  connectionIdNumber++;
473 
474  String clientHost = "";
475  try {
476  clientHost = RemoteServer.getClientHost();
477  /*
478  * According to the rules specified in the javax.management.remote
479  * package description, a numeric IPv6 address (detected by the
480  * presence of otherwise forbidden ":" character) forming a part
481  * of the connection id must be enclosed in square brackets.
482  */
483  if (clientHost.contains(":")) {
484  clientHost = "[" + clientHost + "]";
485  }
486  } catch (ServerNotActiveException e) {
487  logger.trace("makeConnectionId", "getClientHost", e);
488  }
489 
490  final StringBuilder buf = new StringBuilder();
491  buf.append(protocol).append(":");
492  if (clientHost.length() > 0)
493  buf.append("//").append(clientHost);
494  buf.append(" ");
495  if (subject != null) {
496  Set<Principal> principals = subject.getPrincipals();
497  String sep = "";
498  for (Iterator<Principal> it = principals.iterator(); it.hasNext(); ) {
499  Principal p = it.next();
500  String name = p.getName().replace(' ', '_').replace(';', ':');
501  buf.append(sep).append(name);
502  sep = ";";
503  }
504  }
505  buf.append(" ").append(connectionIdNumber);
506  if (logger.traceOn())
507  logger.trace("newConnectionId","connectionId="+buf);
508  return buf.toString();
509  }
