611  protected void processFocusEvent(FocusEvent e) {
612  super.processFocusEvent(e);
613

642  private class FocusLostHandler implements Runnable, Serializable {

645  if (fb == JFormattedTextField.COMMIT ||
646  fb == JFormattedTextField.COMMIT_OR_REVERT) {

648  JFormattedTextField.this.commitEdit();
