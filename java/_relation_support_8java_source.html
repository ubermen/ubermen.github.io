723  // Checks if the role exists and is readable according to the relation
724  // type.
725  //
726  // This method is called in getRole() above.
727  // It is also called in the Relation Service getRole() method.
728  // It is also called in getRolesInt() below (used for getRoles() above

730  //
731  // Depending on parameters reflecting its use (either in the scope of
732  // getting a single role or of getting several roles), will return:
733  // - in case of success:
734  // - for single role retrieval, the ArrayList of ObjectNames being the
735  // role value
736  // - for multi-role retrieval, the Role object itself
737  // - in case of failure (except critical exceptions):
738  // - for single role retrieval, if role does not exist or is not
739  // readable, an RoleNotFoundException exception is raised
740  // - for multi-role retrieval, a RoleUnresolved object
741  //
742  // -param roleName name of role to be retrieved
743  // -param relationServCallFlg true if call from the Relation Service; this
744  // will happen if the current RelationSupport object has been created by

746  // possible.
747  // -param relationServ reference to Relation Service object, if object
748  // created by Relation Service.
749  // -param multiRoleFlg true if getting the role in the scope of a
750  // multiple retrieval.
751  //
752  // -return:
753  // - for single role retrieval (multiRoleFlg false):
754  // - ArrayList of ObjectName objects, value of role with given name, if
755  // the role can be retrieved
756  // - raise a RoleNotFoundException exception else
757  // - for multi-role retrieval (multiRoleFlg true):
758  // - the Role object for given role name if role can be retrieved
759  // - a RoleUnresolved object with problem.
760  //
761  // -exception IllegalArgumentException if null parameter
762  // -exception RoleNotFoundException if multiRoleFlg is false and:
763  // - there is no role with given name

793  if (role == null) {
794  pbType = RoleStatus.NO_ROLE_WITH_NAME;
795 
796  } else {
797  // Checks if the role is readable
798  Integer status;
799 
800  if (relationServCallFlg) {
801 
802  // Call from the Relation Service, so direct access to it,
803  // avoiding MBean Server
804  // Shall not throw a RelationTypeNotFoundException

816  // via the MBean Server.

818  params[0] = roleName;
819  params[1] = myRelTypeName;
