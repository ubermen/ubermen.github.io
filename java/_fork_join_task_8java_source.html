214  /*
215  * See the internal documentation of class ForkJoinPool for a
216  * general implementation overview. ForkJoinTasks are mainly
217  * responsible for maintaining their "status" field amidst relays
218  * to methods in ForkJoinWorkerThread and ForkJoinPool.
219  *
220  * The methods of this class are more-or-less layered into
221  * (1) basic status maintenance
222  * (2) execution and awaiting completion
223  * (3) user-level methods that additionally report results.
224  * This is sometimes hard to see because this file orders exported
225  * methods in a way that flows well in javadocs.
226  */

228  /*
229  * The status field holds run control status bits packed into a
230  * single int to minimize footprint and to ensure atomicity (via
231  * CAS). Status is initially zero, and takes on nonnegative
232  * values until completed, upon which status (anded with
233  * DONE_MASK) holds value NORMAL, CANCELLED, or EXCEPTIONAL. Tasks
234  * undergoing blocking waits by other threads have the SIGNAL bit
235  * set. Completion of a stolen task with SIGNAL set awakens any
236  * waiters via notifyAll. Even though suboptimal for some
237  * purposes, we use basic builtin wait/notify to take advantage of
238  * "monitor inflation" in JVMs that we would otherwise need to
239  * emulate to avoid adding further per-task bookkeeping overhead.
240  * We want these monitors to be "fat", i.e., not use biasing or
241  * thin-lock techniques, so use some odd coding idioms that tend
242  * to avoid them, mainly by arranging that every synchronized
243  * block performs a wait, notifyAll or both.
244  *
245  * These control bits occupy only (some of) the upper half (16
246  * bits) of status field. The lower bits are used for user-defined
247  * tags.
248  */
249

459  final int recordExceptionalCompletion(Throwable ex) {
460  int s;
461  if ((s = status) >= 0) {
462  int h = System.identityHashCode(this);
463  final ReentrantLock lock = exceptionTableLock;
464  lock.lock();

475  break;

563  private Throwable getThrowableException() {
564  if ((status & DONE_MASK) != EXCEPTIONAL)
565  return null;
566  int h = System.identityHashCode(this);
567  ExceptionNode e;
568  final ReentrantLock lock = exceptionTableLock;
569  lock.lock();
570  try {
571  expungeStaleExceptions();
572  ExceptionNode[] t = exceptionTable;
573  e = t[h & (t.length - 1)];
574  while (e != null && e.get() != this)
575  e = e.next;
576  } finally {
577  lock.unlock();
578  }
579  Throwable ex;
580  if (e == null || (ex = e.ex) == null)
581  return null;
582  if (e.thrower != Thread.currentThread().getId()) {
583  Class<? extends Throwable> ec = ex.getClass();
584  try {
585  Constructor<?> noArgCtor = null;
586  Constructor<?>[] cs = ec.getConstructors();// public ctors only
587  for (int i = 0; i < cs.length; ++i) {
588  Constructor<?> c = cs[i];
589  Class<?>[] ps = c.getParameterTypes();
590  if (ps.length == 0)

592  else if (ps.length == 1 && ps[0] == Throwable.class) {
593  Throwable wx = (Throwable)c.newInstance(ex);
594  return (wx == null) ? ex : wx;
595  }
596  }
597  if (noArgCtor != null) {
598  Throwable wx = (Throwable)(noArgCtor.newInstance());
599  if (wx != null) {
600  wx.initCause(ex);
601  return wx;
602  }
603  }
604  } catch (Exception ignore) {
605  }
606  }
607  return ex;
608  }

1024  public final V get(long timeout, TimeUnit unit)
1025  throws InterruptedException, ExecutionException, TimeoutException {
1026  int s;
1027  long nanos = unit.toNanos(timeout);
1028  if (Thread.interrupted())
1029  throw new InterruptedException();
1030  if ((s = status) >= 0 && nanos > 0L) {
1031  long d = System.nanoTime() + nanos;
1032  long deadline = (d == 0L) ? 1L : d; // avoid 0
1033  Thread t = Thread.currentThread();
1034  if (t instanceof ForkJoinWorkerThread) {
1035  ForkJoinWorkerThread wt = (ForkJoinWorkerThread)t;
1036  s = wt.pool.awaitJoin(wt.workQueue, this, deadline);
1037  }
1038  else if ((s = ((this instanceof CountedCompleter) ?
1039  ForkJoinPool.common.externalHelpComplete(
1040  (CountedCompleter<?>)this, 0) :
1041  ForkJoinPool.common.tryExternalUnpush(this) ?
1042  doExec() : 0)) >= 0) {
1043  long ns, ms; // measure in nanosecs, but wait in millisecs
1044  while ((s = status) >= 0 &&
1045  (ns = deadline - System.nanoTime()) > 0L) {
1046  if ((ms = TimeUnit.NANOSECONDS.toMillis(ns)) > 0L &&
1047  U.compareAndSwapInt(this, STATUS, s, s | SIGNAL)) {
1048  synchronized (this) {
1049  if (status >= 0)
1050  wait(ms); // OK to throw InterruptedException
1051  else
1052  notifyAll();
1053  }
1054  }
1055  }
1056  }
1057  }
1058  if (s >= 0)
1059  s = status;
1060  if ((s &= DONE_MASK) != NORMAL) {
1061  Throwable ex;
1062  if (s == CANCELLED)
1063  throw new CancellationException();
1064  if (s != EXCEPTIONAL)
1065  throw new TimeoutException();
1066  if ((ex = getThrowableException()) != null)
1067  throw new ExecutionException(ex);
1068  }
1069  return getRawResult();
1070  }
