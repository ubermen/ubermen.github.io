475  public Dimension minimumLayoutSize(Container c) {
476  // Calculate width.
477  int width = 22;
478 
479  if (frame.isClosable()) {
480  width += 19;
481  }
482  if (frame.isMaximizable()) {
483  width += 19;
484  }
485  if (frame.isIconifiable()) {
486  width += 19;
487  }
488 
489  FontMetrics fm = frame.getFontMetrics(getFont());
490  String frameTitle = frame.getTitle();
491  int title_w = frameTitle != null ? SwingUtilities2.stringWidth(
492  frame, fm, frameTitle) : 0;
493  int title_length = frameTitle != null ? frameTitle.length() : 0;
494 
495  // Leave room for three characters in the title.
496  if (title_length > 3) {
497  int subtitle_w = SwingUtilities2.stringWidth(
498  frame, fm, frameTitle.substring(0, 3) + "...");
499  width += (title_w < subtitle_w) ? title_w : subtitle_w;
500  } else {
501  width += title_w;
502  }
503 
504  // Calculate height.
505  Icon icon = frame.getFrameIcon();
506  int fontHeight = fm.getHeight();
507  fontHeight += 2;
508  int iconHeight = 0;
509  if (icon != null) {
510  // SystemMenuBar forces the icon to be 16x16 or less.
511  iconHeight = Math.min(icon.getIconHeight(), 16);
512  }
513  iconHeight += 2;
514 
515  int height = Math.max( fontHeight, iconHeight );
516 
517  Dimension dim = new Dimension(width, height);
518 
519  // Take into account the border insets if any.
520  if (getBorder() != null) {
521  Insets insets = getBorder().getBorderInsets(c);
522  dim.height += insets.top + insets.bottom;
523  dim.width += insets.left + insets.right;
524  }
525  return dim;
526  }
