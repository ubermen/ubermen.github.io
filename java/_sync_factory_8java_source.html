466  private static void parseProperties(Properties p) {
467 
468  ProviderImpl impl = null;
469  String key = null;

472  for (Enumeration<?> e = p.propertyNames(); e.hasMoreElements();) {

477

484  // no property index has been set.

492  impl.setClassname(key);

550  public static SyncProvider getInstance(String providerID)
551  throws SyncFactoryException {
552 
553  if(providerID == null) {
554  throw new SyncFactoryException("The providerID cannot be null");
555  }

557  initMapIfNecessary(); // populate HashTable
558  initJNDIContext(); // check JNDI context for any additional bindings
559 
560  ProviderImpl impl = (ProviderImpl) implementations.get(providerID);
561 
562  if (impl == null) {
563  // Requested SyncProvider is unavailable. Return default provider.
564  return new com.sun.rowset.providers.RIOptimisticProvider();
565  }
566 
567  try {
568  ReflectUtil.checkPackageAccess(providerID);
569  } catch (java.security.AccessControlException e) {
570  SyncFactoryException sfe = new SyncFactoryException();
571  sfe.initCause(e);
572  throw sfe;
573  }
574 
575  // Attempt to invoke classname from registered SyncProvider list
576  Class<?> c = null;
577  try {
578  ClassLoader cl = Thread.currentThread().getContextClassLoader();
579 
586  c = Class.forName(providerID, true, cl);
587 
588  if (c != null) {
589  return (SyncProvider) c.newInstance();
590  } else {
591  return new com.sun.rowset.providers.RIOptimisticProvider();
592  }
593 
594  } catch (IllegalAccessException e) {
595  throw new SyncFactoryException("IllegalAccessException: " + e.getMessage());

791  private static void enumerateBindings(NamingEnumeration<?> bindings,
792  Properties properties) throws NamingException {
793 
794  boolean syncProviderObj = false; // move to parameters ?
795 
796  try {
797  Binding bd = null;
798  Object elementObj = null;
799  String element = null;
800  while (bindings.hasMore()) {
801  bd = (Binding) bindings.next();
802  element = bd.getName();
803  elementObj = bd.getObject();
804 
805  if (!(ic.lookup(element) instanceof Context)) {
806  // skip directories/sub-contexts
807  if (ic.lookup(element) instanceof SyncProvider) {
808  syncProviderObj = true;
809  }
810  }
811 
812  if (syncProviderObj) {
813  SyncProvider sync = (SyncProvider) elementObj;
814  properties.put(SyncFactory.ROWSET_SYNC_PROVIDER,
815  sync.getProviderID());
816  syncProviderObj = false; // reset
817  }
818 
819  }
820  } catch (javax.naming.NotContextException e) {
821  bindings.next();
822  // Re-entrant call into method
823  enumerateBindings(bindings, properties);
