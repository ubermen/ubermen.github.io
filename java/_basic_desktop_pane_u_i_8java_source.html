347  public void actionPerformed(ActionEvent e) {
348  JDesktopPane dp = (JDesktopPane)e.getSource();
349  String key = getName();
350 
351  if (CLOSE == key || MAXIMIZE == key || MINIMIZE == key ||
352  RESTORE == key) {
353  setState(dp, key);
354  }
355  else if (ESCAPE == key) {
356  if (sourceFrame == dp.getSelectedFrame() &&
357  focusOwner != null) {
358  focusOwner.requestFocus();
359  }
360  moving = false;
361  resizing = false;
362  sourceFrame = null;
363  focusOwner = null;
364  }
365  else if (MOVE == key || RESIZE == key) {
366  sourceFrame = dp.getSelectedFrame();
367  if (sourceFrame == null) {
368  return;
369  }
370  moving = (key == MOVE) ? true : false;
371  resizing = (key == RESIZE) ? true : false;
372 
373  focusOwner = KeyboardFocusManager.
374  getCurrentKeyboardFocusManager().getFocusOwner();
375  if (!SwingUtilities.isDescendingFrom(focusOwner, sourceFrame)) {
376  focusOwner = null;
377  }
378  sourceFrame.requestFocus();
379  }
380  else if (LEFT == key ||
381  RIGHT == key ||
382  UP == key ||
383  DOWN == key ||
384  SHRINK_RIGHT == key ||
385  SHRINK_LEFT == key ||
386  SHRINK_UP == key ||
387  SHRINK_DOWN == key) {
388  JInternalFrame c = dp.getSelectedFrame();
389  if (sourceFrame == null || c != sourceFrame ||
390  KeyboardFocusManager.
391  getCurrentKeyboardFocusManager().getFocusOwner() !=
392  sourceFrame) {
393  return;
394  }
395  Insets minOnScreenInsets =
396  UIManager.getInsets("Desktop.minOnScreenInsets");
397  Dimension size = c.getSize();
398  Dimension minSize = c.getMinimumSize();
399  int dpWidth = dp.getWidth();
400  int dpHeight = dp.getHeight();
401  int delta;
402  Point loc = c.getLocation();
403  if (LEFT == key) {
404  if (moving) {
405  c.setLocation(
406  loc.x + size.width - MOVE_RESIZE_INCREMENT <
407  minOnScreenInsets.right ?
408  -size.width + minOnScreenInsets.right :
409  loc.x - MOVE_RESIZE_INCREMENT,
410  loc.y);
411  } else if (resizing) {
412  c.setLocation(loc.x - MOVE_RESIZE_INCREMENT, loc.y);
413  c.setSize(size.width + MOVE_RESIZE_INCREMENT,
414  size.height);
415  }
416  } else if (RIGHT == key) {
417  if (moving) {
418  c.setLocation(
419  loc.x + MOVE_RESIZE_INCREMENT >
420  dpWidth - minOnScreenInsets.left ?
421  dpWidth - minOnScreenInsets.left :
422  loc.x + MOVE_RESIZE_INCREMENT,
423  loc.y);
424  } else if (resizing) {
425  c.setSize(size.width + MOVE_RESIZE_INCREMENT,
426  size.height);
427  }
428  } else if (UP == key) {
429  if (moving) {
430  c.setLocation(loc.x,
431  loc.y + size.height - MOVE_RESIZE_INCREMENT <
432  minOnScreenInsets.bottom ?
433  -size.height +
434  minOnScreenInsets.bottom :
435  loc.y - MOVE_RESIZE_INCREMENT);
436  } else if (resizing) {
437  c.setLocation(loc.x, loc.y - MOVE_RESIZE_INCREMENT);
438  c.setSize(size.width,
439  size.height + MOVE_RESIZE_INCREMENT);
440  }
441  } else if (DOWN == key) {
442  if (moving) {
443  c.setLocation(loc.x,
444  loc.y + MOVE_RESIZE_INCREMENT >
445  dpHeight - minOnScreenInsets.top ?
446  dpHeight - minOnScreenInsets.top :
447  loc.y + MOVE_RESIZE_INCREMENT);
448  } else if (resizing) {
449  c.setSize(size.width,
450  size.height + MOVE_RESIZE_INCREMENT);
451  }
452  } else if (SHRINK_LEFT == key && resizing) {
453  // Make sure we don't resize less than minimum size.
454  if (minSize.width < (size.width - MOVE_RESIZE_INCREMENT)) {
455  delta = MOVE_RESIZE_INCREMENT;
456  } else {
457  delta = size.width - minSize.width;
458  }
459 
460  // Ensure that we keep the internal frame on the desktop.
461  if (loc.x + size.width - delta < minOnScreenInsets.left) {
462  delta = loc.x + size.width - minOnScreenInsets.left;
463  }
464  c.setSize(size.width - delta, size.height);
465  } else if (SHRINK_RIGHT == key && resizing) {
466  // Make sure we don't resize less than minimum size.
467  if (minSize.width < (size.width - MOVE_RESIZE_INCREMENT)) {
468  delta = MOVE_RESIZE_INCREMENT;
469  } else {
470  delta = size.width - minSize.width;
471  }
472 
473  // Ensure that we keep the internal frame on the desktop.
474  if (loc.x + delta > dpWidth - minOnScreenInsets.right) {
475  delta = (dpWidth - minOnScreenInsets.right) - loc.x;
476  }
477 
478  c.setLocation(loc.x + delta, loc.y);
479  c.setSize(size.width - delta, size.height);
480  } else if (SHRINK_UP == key && resizing) {
481  // Make sure we don't resize less than minimum size.
482  if (minSize.height <
483  (size.height - MOVE_RESIZE_INCREMENT)) {
484  delta = MOVE_RESIZE_INCREMENT;
485  } else {
486  delta = size.height - minSize.height;
487  }
488 
489  // Ensure that we keep the internal frame on the desktop.
490  if (loc.y + size.height - delta <
491  minOnScreenInsets.bottom) {
492  delta = loc.y + size.height - minOnScreenInsets.bottom;
493  }
494 
495  c.setSize(size.width, size.height - delta);
496  } else if (SHRINK_DOWN == key && resizing) {
497  // Make sure we don't resize less than minimum size.
498  if (minSize.height <
499  (size.height - MOVE_RESIZE_INCREMENT)) {
500  delta = MOVE_RESIZE_INCREMENT;
501  } else {
502  delta = size.height - minSize.height;
503  }
504 
505  // Ensure that we keep the internal frame on the desktop.
506  if (loc.y + delta > dpHeight - minOnScreenInsets.top) {
507  delta = (dpHeight - minOnScreenInsets.top) - loc.y;
508  }
509 
510  c.setLocation(loc.x, loc.y + delta);
511  c.setSize(size.width, size.height - delta);
512  }
513  }
514  else if (NEXT_FRAME == key || PREVIOUS_FRAME == key) {
515  dp.selectFrame((key == NEXT_FRAME) ? true : false);
516  }
517  else if (NAVIGATE_NEXT == key ||
518  NAVIGATE_PREVIOUS == key) {
519  boolean moveForward = true;
520  if (NAVIGATE_PREVIOUS == key) {
521  moveForward = false;
522  }
523  Container cycleRoot = dp.getFocusCycleRootAncestor();
524 
525  if (cycleRoot != null) {
526  FocusTraversalPolicy policy =
527  cycleRoot.getFocusTraversalPolicy();
528  if (policy != null && policy instanceof
529  SortingFocusTraversalPolicy) {
530  SortingFocusTraversalPolicy sPolicy =
531  (SortingFocusTraversalPolicy)policy;
532  boolean idc = sPolicy.getImplicitDownCycleTraversal();
533  try {
534  sPolicy.setImplicitDownCycleTraversal(false);
535  if (moveForward) {
536  KeyboardFocusManager.
537  getCurrentKeyboardFocusManager().
538  focusNextComponent(dp);
539  } else {
540  KeyboardFocusManager.
541  getCurrentKeyboardFocusManager().
542  focusPreviousComponent(dp);
543  }
544  } finally {
545  sPolicy.setImplicitDownCycleTraversal(idc);
546  }
547  }
548  }
549  }

552  private void setState(JDesktopPane dp, String state) {
553  if (state == CLOSE) {
554  JInternalFrame f = dp.getSelectedFrame();
555  if (f == null) {
556  return;
557  }
558  f.doDefaultCloseAction();
559  } else if (state == MAXIMIZE) {
560  // maximize the selected frame
561  JInternalFrame f = dp.getSelectedFrame();
562  if (f == null) {
563  return;
564  }
565  if (!f.isMaximum()) {
566  if (f.isIcon()) {
567  try {
568  f.setIcon(false);
569  f.setMaximum(true);
570  } catch (PropertyVetoException pve) {}
571  } else {
572  try {
573  f.setMaximum(true);
574  } catch (PropertyVetoException pve) {
575  }
576  }
577  }
578  } else if (state == MINIMIZE) {
579  // minimize the selected frame
580  JInternalFrame f = dp.getSelectedFrame();
581  if (f == null) {
582  return;
583  }
584  if (!f.isIcon()) {
585  try {
586  f.setIcon(true);
587  } catch (PropertyVetoException pve) {
588  }
589  }
590  } else if (state == RESTORE) {
591  // restore the selected minimized or maximized frame
592  JInternalFrame f = dp.getSelectedFrame();
593  if (f == null) {
594  return;
595  }
596  try {
597  if (f.isIcon()) {
598  f.setIcon(false);
599  } else if (f.isMaximum()) {
600  f.setMaximum(false);
601  }
602  f.setSelected(true);
603  } catch (PropertyVetoException pve) {
604  }
605  }
