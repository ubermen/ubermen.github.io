150  private static char[] lookUpTable(String src, int index, Locale locale, boolean bLowerCasing) {
151  HashSet<Entry> set = entryTable.get(new Integer(src.codePointAt(index)));
152  char[] ret = null;
153 
154  if (set != null) {
155  Iterator<Entry> iter = set.iterator();
156  String currentLang = locale.getLanguage();
157  while (iter.hasNext()) {
158  Entry entry = iter.next();
159  String conditionLang = entry.getLanguage();
160  if (((conditionLang == null) || (conditionLang.equals(currentLang))) &&
161  isConditionMet(src, index, locale, entry.getCondition())) {
162  ret = bLowerCasing ? entry.getLowerCase() : entry.getUpperCase();
163  if (conditionLang != null) {
164  break;
165  }
166  }
167  }
168  }

206  BreakIterator wordBoundary = BreakIterator.getWordInstance(locale);
207  wordBoundary.setText(src);
208  int ch;
209

212  i -= Character.charCount(ch)) {
213 
214  ch = src.codePointBefore(i);
215  if (isCased(ch)) {

220  (i < len) && !wordBoundary.isBoundary(i);
