387  protected void setThumbBounds(int x, int y, int width, int height)
388  {
389  /* If the thumbs bounds haven't changed, we're done.
390  */
391  if ((thumbRect.x == x) &&
392  (thumbRect.y == y) &&
393  (thumbRect.width == width) &&
394  (thumbRect.height == height)) {
395  return;
396  }
397 
398  /* Update thumbRect, and repaint the union of x,y,w,h and
399  * the old thumbRect.
400  */
401  int minX = Math.min(x, thumbRect.x);
402  int minY = Math.min(y, thumbRect.y);
403  int maxX = Math.max(x + width, thumbRect.x + thumbRect.width);
404  int maxY = Math.max(y + height, thumbRect.y + thumbRect.height);
405 
406  thumbRect.setBounds(x, y, width, height);
407  scrollbar.repaint(minX, minY, (maxX - minX)+1, (maxY - minY)+1);
408  }

427  {

431  boolean becameFlush = temp == false && isFreeStanding == true;
432  boolean becameNormal = temp == true && isFreeStanding == false;
