936  private void addItem(File directory) {
937 
938  if(directory == null) {
939  return;
940  }
941 
942  boolean useShellFolder = FilePane.usesShellFolder(chooser);
943 
944  directories.clear();
945 
946  File[] baseFolders = (useShellFolder)
947  ? (File[]) ShellFolder.get("fileChooserComboBoxFolders")
948  : fsv.getRoots();
949  directories.addAll(Arrays.asList(baseFolders));
950 
951  // Get the canonical (full) path. This has the side
952  // benefit of removing extraneous chars from the path,
953  // for example /foo/bar/ becomes /foo/bar
954  File canonical;
955  try {
956  canonical = ShellFolder.getNormalizedFile(directory);
957  } catch (IOException e) {
958  // Maybe drive is not ready. Can't abort here.
959  canonical = directory;
960  }
961 
962  // create File instances of each directory leading up to the top
963  try {
964  File sf = useShellFolder ? ShellFolder.getShellFolder(canonical)
965  : canonical;
966  File f = sf;
967  Vector<File> path = new Vector<File>(10);
968  do {
969  path.addElement(f);
970  } while ((f = f.getParentFile()) != null);
971 
972  int pathCount = path.size();
973  // Insert chain at appropriate place in vector
974  for (int i = 0; i < pathCount; i++) {
975  f = path.get(i);
976  if (directories.contains(f)) {
977  int topIndex = directories.indexOf(f);
978  for (int j = i-1; j >= 0; j--) {
979  directories.insertElementAt(path.get(j), topIndex+i-j);
980  }
981  break;
982  }
983  }
984  calculateDepths();
985  setSelectedItem(sf);
986  } catch (FileNotFoundException ex) {
987  calculateDepths();
988  }

997  if (parent != null) {
998  for (int j = i-1; j >= 0; j--) {
999  if (parent.equals(directories.get(j))) {
1000  depths[i] = depths[j] + 1;
1001  break;
1002  }
1003  }
1004  }
