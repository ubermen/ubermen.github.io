461  public static synchronized MBeanServer getPlatformMBeanServer() {
462  SecurityManager sm = System.getSecurityManager();
463  if (sm != null) {
464  Permission perm = new MBeanServerPermission("createMBeanServer");
465  sm.checkPermission(perm);
466  }
467 
468  if (platformMBeanServer == null) {
469  platformMBeanServer = MBeanServerFactory.createMBeanServer();
470  for (PlatformComponent pc : PlatformComponent.values()) {
471  List<? extends PlatformManagedObject> list =
472  pc.getMXBeans(pc.getMXBeanInterface());
473  for (PlatformManagedObject o : list) {
474  // Each PlatformComponent represents one management
475  // interface. Some MXBean may extend another one.
476  // The MXBean instances for one platform component
477  // (returned by pc.getMXBeans()) might be also
478  // the MXBean instances for another platform component.
479  // e.g. com.sun.management.GarbageCollectorMXBean
480  //
481  // So need to check if an MXBean instance is registered
482  // before registering into the platform MBeanServer
483  if (!platformMBeanServer.isRegistered(o.getObjectName())) {
484  addMXBean(platformMBeanServer, o);
485  }
486  }
487  }
488  HashMap<ObjectName, DynamicMBean> dynmbeans =
489  ManagementFactoryHelper.getPlatformDynamicMBeans();
490  for (Map.Entry<ObjectName, DynamicMBean> e : dynmbeans.entrySet()) {
491  addDynamicMBean(platformMBeanServer, e.getValue(), e.getKey());
492  }
493  for (final PlatformManagedObject o :
494  ExtendedPlatformComponent.getMXBeans()) {
495  if (!platformMBeanServer.isRegistered(o.getObjectName())) {
496  addMXBean(platformMBeanServer, o);
497  }
498  }
499  }
500  return platformMBeanServer;
501  }

593  public static <T> T
594  newPlatformMXBeanProxy(MBeanServerConnection connection,
595  String mxbeanName,
596  Class<T> mxbeanInterface)
597  throws java.io.IOException {
598 
599  // Only allow MXBean interfaces from rt.jar loaded by the
600  // bootstrap class loader
601  final Class<?> cls = mxbeanInterface;
602  ClassLoader loader =
603  AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
604  public ClassLoader run() {
605  return cls.getClassLoader();
606  }
607  });
608  if (!sun.misc.VM.isSystemDomainLoader(loader)) {
609  throw new IllegalArgumentException(mxbeanName +
610  " is not a platform MXBean");
611  }
612 
613  try {
614  final ObjectName objName = new ObjectName(mxbeanName);
615  // skip the isInstanceOf check for LoggingMXBean
616  String intfName = mxbeanInterface.getName();
617  if (!connection.isInstanceOf(objName, intfName)) {
618  throw new IllegalArgumentException(mxbeanName +
619  " is not an instance of " + mxbeanInterface);
620  }
621 
622  final Class[] interfaces;
623  // check if the registered MBean is a notification emitter
624  boolean emitter = connection.isInstanceOf(objName, NOTIF_EMITTER);
625 
626  // create an MXBean proxy
627  return JMX.newMXBeanProxy(connection, objName, mxbeanInterface,
628  emitter);
629  } catch (InstanceNotFoundException|MalformedObjectNameException e) {
630  throw new IllegalArgumentException(e);
631  }
632  }

852  private static void addMXBean(final MBeanServer mbs, final PlatformManagedObject pmo) {
853  // Make DynamicMBean out of MXBean by wrapping it with a StandardMBean
854  try {
855  AccessController.doPrivileged(new PrivilegedExceptionAction<Void>() {
856  public Void run() throws InstanceAlreadyExistsException,
857  MBeanRegistrationException,
858  NotCompliantMBeanException {
859  final DynamicMBean dmbean;
860  if (pmo instanceof DynamicMBean) {
861  dmbean = DynamicMBean.class.cast(pmo);
862  } else if (pmo instanceof NotificationEmitter) {
863  dmbean = new StandardEmitterMBean(pmo, null, true, (NotificationEmitter) pmo);
864  } else {
865  dmbean = new StandardMBean(pmo, null, true);
866  }
867 
868  mbs.registerMBean(dmbean, pmo.getObjectName());
869  return null;
870  }
871  });
872  } catch (PrivilegedActionException e) {
873  throw new RuntimeException(e.getException());
874  }
875  }
