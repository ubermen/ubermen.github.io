1574  static private boolean checkfpx(InputStream is) throws IOException {
1575 
1576  /* Test for FlashPix image data in Microsoft Structured Storage format.
1577  * In general, should do this with calls to an SS implementation.
1578  * Lacking that, need to dig via offsets to get to the FlashPix
1579  * ClassID. Details:
1580  *
1581  * Offset to Fpx ClsID from beginning of stream should be:
1582  *
1583  * FpxClsidOffset = rootEntryOffset + clsidOffset
1584  *
1585  * where: clsidOffset = 0x50.
1586  * rootEntryOffset = headerSize + sectorSize*sectDirStart
1587  * + 128*rootEntryDirectory
1588  *
1589  * where: headerSize = 0x200 (always)
1590  * sectorSize = 2 raised to power of uSectorShift,
1591  * which is found in the header at
1592  * offset 0x1E.
1593  * sectDirStart = found in the header at offset 0x30.
1594  * rootEntryDirectory = in general, should search for
1595  * directory labelled as root.
1596  * We will assume value of 0 (i.e.,
1597  * rootEntry is in first directory)
1598  */
1599 
1600  // Mark the stream so we can reset it. 0x100 is enough for the first
1601  // few reads, but the mark will have to be reset and set again once
1602  // the offset to the root directory entry is computed. That offset
1603  // can be very large and isn't know until the stream has been read from
1604  is.mark(0x100);
1605 
1606  // Get the byte ordering located at 0x1E. 0xFE is Intel,
1607  // 0xFF is other
1608  long toSkip = (long)0x1C;
1609  long posn;
1610 
1611  if ((posn = skipForward(is, toSkip)) < toSkip) {
1612  is.reset();
1613  return false;
1614  }
1615 
1616  int c[] = new int[16];
1617  if (readBytes(c, 2, is) < 0) {
1618  is.reset();
1619  return false;
1620  }
1621 
1622  int byteOrder = c[0];
1623 
1624  posn+=2;
1625  int uSectorShift;
1626  if (readBytes(c, 2, is) < 0) {
1627  is.reset();
1628  return false;
1629  }
1630 
1631  if(byteOrder == 0xFE) {
1632  uSectorShift = c[0];
1633  uSectorShift += c[1] << 8;
1634  }
1635  else {
1636  uSectorShift = c[0] << 8;
1637  uSectorShift += c[1];
1638  }
1639 
1640  posn += 2;
1641  toSkip = (long)0x30 - posn;
1642  long skipped = 0;
1643  if ((skipped = skipForward(is, toSkip)) < toSkip) {
1644  is.reset();
1645  return false;
1646  }
1647  posn += skipped;
1648 
1649  if (readBytes(c, 4, is) < 0) {
1650  is.reset();
1651  return false;
1652  }
1653 
1654  int sectDirStart;
1655  if(byteOrder == 0xFE) {
1656  sectDirStart = c[0];
1657  sectDirStart += c[1] << 8;
1658  sectDirStart += c[2] << 16;
1659  sectDirStart += c[3] << 24;
1660  } else {
1661  sectDirStart = c[0] << 24;
1662  sectDirStart += c[1] << 16;
1663  sectDirStart += c[2] << 8;
1664  sectDirStart += c[3];
1665  }
1666  posn += 4;
1667  is.reset(); // Reset back to the beginning
1668 
1669  toSkip = 0x200L + (long)(1<<uSectorShift)*sectDirStart + 0x50L;
1670 
1671  // Sanity check!
1672  if (toSkip < 0) {
1673  return false;
1674  }

1678  * This skip can be fairly long, at least 0x4c650 in at least
