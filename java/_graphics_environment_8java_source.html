165  String nm = System.getProperty("java.awt.headless");
166 
167  if (nm == null) {
168  /* No need to ask for DISPLAY when run in a browser */
169  if (System.getProperty("javaplugin.version") != null) {
170  headless = defaultHeadless = Boolean.FALSE;
171  } else {

173  if (osName.contains("OS X") && "sun.awt.HToolkit".equals(
