827  if (e.getSource() == list) {
828  return;
829  }
830  if (!SwingUtilities.isLeftMouseButton(e) || !comboBox.isEnabled())
831  return;
832

834  Component comp = comboBox.getEditor().getEditorComponent();
835  if ((!(comp instanceof JComponent)) || ((JComponent)comp).isRequestFocusEnabled()) {
836  comp.requestFocus();

838  }

845  public void mouseReleased(MouseEvent e) {
846  if (e.getSource() == list) {
847  if (list.getModel().getSize() > 0) {
848  // JList mouse listener
849  if (comboBox.getSelectedIndex() == list.getSelectedIndex()) {
850  comboBox.getEditor().setItem(list.getSelectedValue());
851  }
852  comboBox.setSelectedIndex(list.getSelectedIndex());
853  }
854  comboBox.setPopupVisible(false);
855  // workaround for cancelling an edited item (bug 4530953)
856  if (comboBox.isEditable() && comboBox.getEditor() != null) {
857  comboBox.configureEditor(comboBox.getEditor(),
858  comboBox.getSelectedItem());
859  }
860  return;
861  }
862  // JComboBox mouse listener
863  Component source = (Component)e.getSource();
864  Dimension size = source.getSize();
865  Rectangle bounds = new Rectangle( 0, 0, size.width - 1, size.height - 1 );
866  if ( !bounds.contains( e.getPoint() ) ) {
867  MouseEvent newEvent = convertMouseEvent( e );
868  Point location = newEvent.getPoint();
869  Rectangle r = new Rectangle();
870  list.computeVisibleRect( r );
871  if ( r.contains( location ) ) {
872  if (comboBox.getSelectedIndex() == list.getSelectedIndex()) {
873  comboBox.getEditor().setItem(list.getSelectedValue());
874  }
875  comboBox.setSelectedIndex(list.getSelectedIndex());
876  }
877  comboBox.setPopupVisible(false);
878  }
879  hasEntered = false;
880  stopAutoScrolling();
881  }

904  public void mouseDragged( MouseEvent e ) {
905  if (e.getSource() == list) {
906  return;
907  }
908  if ( isVisible() ) {
909  MouseEvent newEvent = convertMouseEvent( e );
910  Rectangle r = new Rectangle();
911  list.computeVisibleRect( r );
912 
913  if ( newEvent.getPoint().y >= r.y && newEvent.getPoint().y <= r.y + r.height - 1 ) {
914  hasEntered = true;
915  if ( isAutoScrolling ) {
916  stopAutoScrolling();
917  }
918  Point location = newEvent.getPoint();
919  if ( r.contains( location ) ) {
920  updateListBoxSelectionForEvent( newEvent, false );
921  }
922  }
923  else {
924  if ( hasEntered ) {
925  int directionToScroll = newEvent.getPoint().y < r.y ? SCROLL_UP : SCROLL_DOWN;
926  if ( isAutoScrolling && scrollDirection != directionToScroll ) {
927  stopAutoScrolling();
928  startAutoScrolling( directionToScroll );
929  }
930  else if ( !isAutoScrolling ) {
931  startAutoScrolling( directionToScroll );
932  }
933  }
934  else {
935  if ( e.getPoint().y < 0 ) {
936  hasEntered = true;
937  startAutoScrolling( SCROLL_UP );
938  }
939  }
940  }
941  }
