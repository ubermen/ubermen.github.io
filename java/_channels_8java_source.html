374  int bytesRead = 0;

376  while (totalRead < len) {

378  TRANSFER_SIZE);
379  if (buf.length < bytesToRead)

381  if ((totalRead > 0) && !(in.available() > 0))
382  break; // block at most once
383  try {

391  else

393  dst.put(buf, 0, bytesRead);
