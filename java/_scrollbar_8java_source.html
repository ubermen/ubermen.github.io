888  if (maxMinusMin > Integer.MAX_VALUE) {

890  maximum = minimum + (int) maxMinusMin;
891  }
892  if (visible > (int) maxMinusMin) {
893  visible = (int) maxMinusMin;
894  }
895  if (visible < 1) {
896  visible = 1;
