1094  //validate the method arguments
1095  if (region == null || prefix == null) {
1096  throw new IllegalArgumentException(
1097  "Neither Region nor Prefix may be null");
1098  }

1102  if (styles == null) {
1103  styles = new LinkedList<LazyStyle>();
1104  styles.add(new LazyStyle(prefix));
1105  m.put(region, styles);
1106  } else {
1107  //iterate over all the current styles and see if this prefix has
1108  //already been registered. If not, then register it.
1109  for (LazyStyle s : styles) {

1141  SynthStyle getStyle(JComponent comp, Region r) {
1142  //validate method arguments
1143  if (comp == null || r == null) {
1144  throw new IllegalArgumentException(
1145  "Neither comp nor r may be null");
1146  }
1147 
1148  //if there are no lazy styles registered for the region r, then return
1149  //the default style
1150  List<LazyStyle> styles = m.get(r);
1151  if (styles == null || styles.size() == 0) {
1152  return defaultStyle;
1153  }
1154 
1155  //Look for the best SynthStyle for this component/region pair.
1156  LazyStyle foundStyle = null;
1157  for (LazyStyle s : styles) {
1158  if (s.matches(comp)) {
1159  //replace the foundStyle if foundStyle is null, or
1160  //if the new style "s" is more specific (ie, its path was
1161  //longer), or if the foundStyle was "simple" and the new style
1162  //was not (ie: the foundStyle was for something like Button and
1163  //the new style was for something like "MyButton", hence, being
1164  //more specific.) In all cases, favor the most specific style
1165  //found.
1166  if (foundStyle == null ||
1167  (foundStyle.parts.length < s.parts.length) ||
1168  (foundStyle.parts.length == s.parts.length
1169  && foundStyle.simple && !s.simple)) {
1170  foundStyle = s;
1171  }
1172  }
1173  }
1174 
1175  //return the style, if found, or the default style if not found
1176  return foundStyle == null ? defaultStyle : foundStyle.getStyle(comp, r);
