85  def get_pandas_df(self, hql, parameters=None):
86  """
87  Get a pandas dataframe from a sql query.
88  """
89  import pandas
90  cursor = self.get_cursor()
91  try:
92  cursor.execute(self._strip_sql(hql), parameters)
93  data = cursor.fetchall()
94  except DatabaseError as e:
95  raise PrestoException(e[0]['message'])
96  column_descriptions = cursor.description
97  if data:
98  df = pandas.DataFrame(data)
